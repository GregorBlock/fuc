033public class Controller {
034        // the input file, stdin by default
035        static InputStream input = System.in;
036
037        // variables which hold our components,
038        // initially "null", will be assigned if
039        // component plugins are found.
040        static Lexer lexer = null;
041        static Parser parser = null;
042        static SemanticAnalyser analyzer = null;
043        static IntermediateCodeGenerator irgen = null;
044        static Backend backend = null;
045
046        // several veriables which are affected by command line options
047        static boolean disableVisualization = false;
048        static String outname = null;
049        static String loglevel = null;
050        static boolean jit = false;
051
052        // our error report log for the parser
053        static ReportLogImpl errlog = new ReportLogImpl();
054
055        // visualization service loaders
056        private static ServiceLoader<TokenStreamVisualization> tokenVisuService;
057        private static ServiceLoader<ASTVisualization> ASTVisuService;
058        private static ServiceLoader<TACVisualization> TACVisuService;
059
060        // load component plugins
061        static void loadPlugins() {
062
063                // create loaders for component plugins
064                ServiceLoader<Lexer> lexerService = ServiceLoader.load(Lexer.class);
065                ServiceLoader<Parser> parserService = ServiceLoader.load(Parser.class);
066                ServiceLoader<SemanticAnalyser> analyzerService = ServiceLoader.load(SemanticAnalyser.class);
067                ServiceLoader<IntermediateCodeGenerator> irgenService = ServiceLoader.load(IntermediateCodeGenerator.class);
068                ServiceLoader<Backend> backendService = ServiceLoader.load(Backend.class);
069
070                tokenVisuService = ServiceLoader.load(TokenStreamVisualization.class);
071                ASTVisuService = ServiceLoader.load(ASTVisualization.class);
072                TACVisuService = ServiceLoader.load(TACVisualization.class);
073
074                // lexer:
075                for (Lexer l : lexerService) {
076                        System.err.print("  Plugin found (lexer): '" + l.getClass() + "'");
077                        // if still null, assign and mention that we use this plugin on
078                        // stderr
079                        if (lexer == null) {
080                                System.err.println("   <- USED");
081                                lexer = l;
082                        } else {
083                                System.err.println();
084                        }
085                }
086
087                // load component plugins
088                // parser:
089                for (Parser p : parserService) {
090                        System.err.print("  Plugin found (parser): '" + p.getClass() + "'");
091                        // if still null, assign and mention that we use this plugin on
092                        // stderr
093                        if (parser == null) {
094                                System.err.println("   <- USED");
095                                parser = p;
096                        } else {
097                                System.err.println();
098                        }
099                }
100                
101                // load component plugins
102                // analyzer:
103                for (SemanticAnalyser a : analyzerService) {
104                        System.err.print("  Plugin found (analyzer): '" + a.getClass() + "'");
105                        // if still null, assign and mention that we use this plugin on
106                        // stderr
107                        if (analyzer == null) {
108                                System.err.println("   <- USED");
109                                analyzer = a;
110                        } else {
111                                System.err.println();
112                        }
113                }
114
115                // load component plugins
116                // intermediate code generator:
117                for (IntermediateCodeGenerator i : irgenService) {
118                        System.err.print("  Plugin found (IRGen): '" + i.getClass() + "'");
119                        // if still null, assign and mention that we use this plugin on
120                        // stderr
121                        if (irgen == null) {
122                                System.err.println("   <- USED");
123                                irgen = i;
124                        } else {
125                                System.err.println();
126                        }
127                }
128
129                // load component plugins
130                // backend:
131                for (Backend b : backendService) {
132                        System.err.print("  Plugin found (backend): '" + b.getClass() + "'");
133                        // if still null, assign and mention that we use this plugin on
134                        // stderr
135                        if (backend == null) {
136                                System.err.println("   <- USED");
137                                backend = b;
138                        } else {
139                                System.err.println();
140                        }
141                }
142
143                // check that we have a working trio of lexer, parser and backend
144                if (lexer == null) {
145                        System.err.println("ERROR: no lexer plugin found!");
146                }
147                if (parser == null) {
148                        System.err.println("ERROR: no parser plugin found!");
149                }
150                if (analyzer == null) {
151                        System.err.println("ERROR: no analyzer plugin found!");
152                }
153                if (irgen == null) {
154                        System.err.println("ERROR: no IRGen plugin found!");
155                }
156                if (backend == null) {
157                        System.err.println("ERROR: no backend plugin found!");
158                }
159                if (lexer == null || parser == null || irgen == null || backend == null) {
160                        System.exit(1);
161                }
162
163                System.err.println("  Token Stream Visualization found:");
164                for (TokenStreamVisualization tokenvisu : tokenVisuService) {
165                        System.err.println("\t" + tokenvisu.getClass().getName());
166                }
167
168                System.err.println("  AST Visualization found:");
169                for (ASTVisualization astvisu : ASTVisuService) {
170                        System.err.println("\t" + astvisu.getClass().getName());
171                }
172
173                System.err.println("  TAC Visualization found:");
174                for (TACVisualization tacvisu : TACVisuService) {
175                        System.err.println("\t" + tacvisu.getClass().getName());
176                }
177        }
178
179        public static void main(String[] args) throws IOException, IntermediateCodeGeneratorException, BackendException,
180                        InstantiationException, IllegalAccessException {
181                System.err.println("SWP Compiler v0.0\n");
182
183                boolean expect_outname = false;
184                boolean expect_loglevel = false;
185
186                // parser CMD args:
187                // if set to true, further arguments may be options,
188                // will be false after encounering -- argument
189                boolean may_be_option = true;
190                for (String arg : args) {
191
192                        // basically "OPTARG" strings work via setting a boolean when the
193                        // flag is read,
194                        // the flag is checked for the next argument then after one
195                        // iteration.
196                        // once all arguments for the option are read, the flag is unset
197                        // again.
198                        // example: -o output.ll
199                        // <- reads "-o" and sets expect_outname to true
200                        // <- reads name and sets expect_outname to false_
201                        if (expect_outname) {
202                                outname = arg;
203                                expect_outname = false;
204                        } else if (expect_loglevel) {
205                                loglevel = arg;
206                                expect_loglevel = false;
207                        }
208                        // set file
209                        else if (!may_be_option || arg.charAt(0) != '-') {
210                                if (input != System.in) { // already set!
211                                        System.err.println("ERROR: only none or one input file allowed!");
212                                        System.exit(2);
213                                } else {
214                                        if (arg.equals("-")) {
215                                                ; // already set to stdin
216                                        } else {
217                                                input = new FileInputStream(arg);
218                                        }
219                                }
220                                continue; // it's no option, so we should continue with next arg
221                        }
222                        // -- seperator argument
223                        else if (arg.equals("--")) {
224                                may_be_option = false;
225                        }
226                        // -V disbales visualization passes
227                        else if (arg.equals("-V") || arg.equals("--no-visualization")) {
228                                disableVisualization = true;
229                        } else if (arg.equals("-j") || arg.equals("--jit")) {
230                                jit = true;
231                        } else if (arg.equals("-o")) {
232                                expect_outname = true;
233                        } else if (arg.equals("-l") || arg.equals("--loglevel")) {
234                                expect_loglevel = true;
235                        }
236
237                        // help option, just displays usage summary
238                        if (arg.equals("-h") || arg.equals("--help") || arg.equals("-?")) {
239                                System.out
240                                                .println("Usage: java -cp plugun1.jar:plugin2.jar:...:Controller.jar [options] [--] [input-filename]");
241                                System.out.println();
242                                System.out.println("  input-filename may be '-' for stdin or a relative or absolute path to a file.");
243                                System.out.println();
244                                System.out
245                                                .println("  '--' is an optional separator between options and the input-file, useful if path to the input file begins with character '-'.");
246                                System.out.println();
247                                System.out.println();
248                                System.out.println("  arguments may be any of:");
249                                System.out.println("     -h/-?/--help            emits this help message");
250                                System.out.println("     -o <output>             output filename, only used by llvm");
251                                System.out.println("     -j --jit                just in time compile (not yet supported)");
252                                System.out.println("     -l --loglevel <level>   set the loglevel. allowed values are:");
253                                System.out.println("                               ALL DEBUG ERROR FATAL INFO OFF TRACE WARN");
254                                System.out
255                                                .println("     -V --no-visualization   disables the automatic visualization pass if visualization plugins are found");
256                                System.out.println();
257                                System.exit(0);
258                        }
259                }
260
261                // set the logger level
262                if (loglevel != null) {
263                        loglevel = loglevel.toUpperCase();
264                        Level level = Level.toLevel(loglevel);
265                        if (level.toString().equals(loglevel)) {
266                                Logger.getRootLogger().setLevel(level);
267                        } else {
268                                System.out.println("unknown loglevel see -h for help");
269                                System.exit(0);
270                        }
271                }
272
273                // try to find a lexer, parser and backend
274                // this call assigns the variables lexer,parser,backend
275                // or fails if any one components is missing.
276                loadPlugins();
277
278                // copy the input stream
279                ByteArrayOutputStream baos = new ByteArrayOutputStream();
280                byte[] buffer = new byte[1024];
281                int len;
282                while ((len = input.read(buffer)) > -1) {
283                        baos.write(buffer, 0, len);
284                }
285                baos.flush();
286
287                input = new ByteArrayInputStream(baos.toByteArray());
288
289                // use the components now to compile our file
290                // lexer...
291                lexer.setSourceStream(input);
292
293                if (!disableVisualization) {
294                        for (TokenStreamVisualization tokenvisu : tokenVisuService) {
295                                input = new ByteArrayInputStream(baos.toByteArray());
296                                Lexer vlexer = lexer.getClass().newInstance();
297                                vlexer.setSourceStream(input);
298                                tokenvisu.visualizeTokenStream(vlexer);
299                        }
300                }
301
302                // parser...
303                parser.setLexer(lexer);
304                parser.setReportLog(errlog);
305                AST ast = parser.getParsedAST();
306
307                // in case of parser errors: abort and display errors
308                if (errlog.hasErrors()) {
309                        System.err.println("ERROR: compile failed due to errors:");
310                        for (LogEntry e : errlog.getEntries()) {
311                                System.err.println(e.getMessage());
312                        }
313                        System.exit(1);
314                }
315
316                if (!disableVisualization) {
317                        for (ASTVisualization astvisu : ASTVisuService) {
318                                astvisu.visualizeAST(ast);
319                        }
320                }
321
322                // analyzer
323                analyzer.setReportLog(errlog);
324                ast = analyzer.analyse(ast);
325                
326                // in case of analyzer errors: abort and display errors
327                if (errlog.hasErrors()) {
328                        System.err.println("ERROR: compile failed due to errors:");
329                        for (LogEntry e : errlog.getEntries()) {
330                                System.err.println(e.getMessage());
331                        }
332                        System.exit(1);
333                } else if (errlog.hasWarnings()) {
334                    System.err.println("WARNINGS:");
335                        for (LogEntry e : errlog.getEntries()) {
336                                System.err.println(e.getMessage());
337                        }
338                }
339                
340                // IR gen...
341                List<Quadruple> tac = irgen.generateIntermediateCode(ast);
342
343                if (!disableVisualization) {
344                        for (TACVisualization tacvisu : TACVisuService) {
345                                tacvisu.visualizeTAC(tac);
346                        }
347                }
348
349                // backend...
350                Map<String, InputStream> targets = backend.generateTargetCode("", tac);
351
352                // output..
353                Set<String> outputFilenames = targets.keySet();
354                // for now, we always print everything on stdout.
355                for (String outFile : outputFilenames) {
356                        OutputStream output;
357                        if (outFile.equals(".ll")) {
358                                if (outname != null) {
359                                        output = new FileOutputStream(outname);
360                                } else {
361                                        outname = "<<stdout>>";
362                                        output = System.out;
363                                }
364                        } else // java bytecode output
365                        {
366                                output = new FileOutputStream(outFile);
367                        }
368                        System.err.println("//code: " + outname);
369                        // input stream on the code
370                        InputStream is = targets.get(outFile);
371
372                        // copied and modified from stackoverflow.com
373                        // http://stackoverflow.com/questions/1574837/connecting-an-input-stream-to-an-outputstream
374                        byte[] buf = new byte[4096];
375                        int bytesRead;
376                        while ((bytesRead = is.read(buf)) != -1) {
377                                output.write(buf, 0, bytesRead);
378                        }
379                }
380        }
