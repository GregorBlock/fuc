032public class Controller {
033        // the input file, stdin by default
034        static InputStream input = System.in;
035
036        // variables which hold our components,
037        // initially "null", will be assigned if
038        // component plugins are found.
039        static Lexer lexer = null;
040        static Parser parser = null;
041        static IntermediateCodeGenerator irgen = null;
042        static Backend backend = null;
043
044        // several veriables which are affected by command line options
045        static boolean disableVisualization = false;
046        static String outname = null;
047        static String loglevel = null;
048        static boolean jit = false;
049
050        // our error report log for the parser
051        static ReportLogImpl errlog = new ReportLogImpl();
052
053        // visualization service loaders
054        private static ServiceLoader<TokenStreamVisualization> tokenVisuService;
055        private static ServiceLoader<ASTVisualization> ASTVisuService;
056        private static ServiceLoader<TACVisualization> TACVisuService;
057
058        // load component plugins
059        static void loadPlugins() {
060
061                // create loaders for component plugins
062                ServiceLoader<Lexer> lexerService = ServiceLoader.load(Lexer.class);
063                ServiceLoader<Parser> parserService = ServiceLoader.load(Parser.class);
064                ServiceLoader<IntermediateCodeGenerator> irgenService = ServiceLoader.load(IntermediateCodeGenerator.class);
065                ServiceLoader<Backend> backendService = ServiceLoader.load(Backend.class);
066
067                tokenVisuService = ServiceLoader.load(TokenStreamVisualization.class);
068                ASTVisuService = ServiceLoader.load(ASTVisualization.class);
069                TACVisuService = ServiceLoader.load(TACVisualization.class);
070
071                // lexer:
072                for (Lexer l : lexerService) {
073                        System.err.print("  Plugin found (lexer): '" + l.getClass() + "'");
074                        // if still null, assign and mention that we use this plugin on
075                        // stderr
076                        if (lexer == null) {
077                                System.err.println("   <- USED");
078                                lexer = l;
079                        } else {
080                                System.err.println();
081                        }
082                }
083
084                // load component plugins
085                // parser:
086                for (Parser p : parserService) {
087                        System.err.print("  Plugin found (parser): '" + p.getClass() + "'");
088                        // if still null, assign and mention that we use this plugin on
089                        // stderr
090                        if (parser == null) {
091                                System.err.println("   <- USED");
092                                parser = p;
093                        } else {
094                                System.err.println();
095                        }
096                }
097
098                // load component plugins
099                // intermediate code generator:
100                for (IntermediateCodeGenerator i : irgenService) {
101                        System.err.print("  Plugin found (IRGen): '" + i.getClass() + "'");
102                        // if still null, assign and mention that we use this plugin on
103                        // stderr
104                        if (irgen == null) {
105                                System.err.println("   <- USED");
106                                irgen = i;
107                        } else {
108                                System.err.println();
109                        }
110                }
111
112                // load component plugins
113                // backend:
114                for (Backend b : backendService) {
115                        System.err.print("  Plugin found (backend): '" + b.getClass() + "'");
116                        // if still null, assign and mention that we use this plugin on
117                        // stderr
118                        if (backend == null) {
119                                System.err.println("   <- USED");
120                                backend = b;
121                        } else {
122                                System.err.println();
123                        }
124                }
125
126                // check that we have a working trio of lexer, parser and backend
127                if (lexer == null) {
128                        System.err.println("ERROR: no lexer plugin found!");
129                }
130                if (parser == null) {
131                        System.err.println("ERROR: no parser plugin found!");
132                }
133                if (irgen == null) {
134                        System.err.println("ERROR: no IRGen plugin found!");
135                }
136                if (backend == null) {
137                        System.err.println("ERROR: no backend plugin found!");
138                }
139                if (lexer == null || parser == null || irgen == null || backend == null) {
140                        System.exit(1);
141                }
142
143                System.err.println("  Token Stream Visualization found:");
144                for (TokenStreamVisualization tokenvisu : tokenVisuService) {
145                        System.err.println("\t" + tokenvisu.getClass().getName());
146                }
147
148                System.err.println("  AST Visualization found:");
149                for (ASTVisualization astvisu : ASTVisuService) {
150                        System.err.println("\t" + astvisu.getClass().getName());
151                }
152
153                System.err.println("  TAC Visualization found:");
154                for (TACVisualization tacvisu : TACVisuService) {
155                        System.err.println("\t" + tacvisu.getClass().getName());
156                }
157        }
158
159        public static void main(String[] args) throws IOException, IntermediateCodeGeneratorException, BackendException,
160                        InstantiationException, IllegalAccessException {
161                System.err.println("SWP Compiler v0.0\n");
162
163                boolean expect_outname = false;
164                boolean expect_loglevel = false;
165
166                // parser CMD args:
167                // if set to true, further arguments may be options,
168                // will be false after encounering -- argument
169                boolean may_be_option = true;
170                for (String arg : args) {
171
172                        // basically "OPTARG" strings work via setting a boolean when the
173                        // flag is read,
174                        // the flag is checked for the next argument then after one
175                        // iteration.
176                        // once all arguments for the option are read, the flag is unset
177                        // again.
178                        // example: -o output.ll
179                        // <- reads "-o" and sets expect_outname to true
180                        // <- reads name and sets expect_outname to false_
181                        if (expect_outname) {
182                                outname = arg;
183                                expect_outname = false;
184                        } else if (expect_loglevel) {
185                                loglevel = arg;
186                                expect_loglevel = false;
187                        }
188                        // set file
189                        else if (!may_be_option || arg.charAt(0) != '-') {
190                                if (input != System.in) { // already set!
191                                        System.err.println("ERROR: only none or one input file allowed!");
192                                        System.exit(2);
193                                } else {
194                                        if (arg.equals("-")) {
195                                                ; // already set to stdin
196                                        } else {
197                                                input = new FileInputStream(arg);
198                                        }
199                                }
200                                continue; // it's no option, so we should continue with next arg
201                        }
202                        // -- seperator argument
203                        else if (arg.equals("--")) {
204                                may_be_option = false;
205                        }
206                        // -V disbales visualization passes
207                        else if (arg.equals("-V") || arg.equals("--no-visualization")) {
208                                disableVisualization = true;
209                        } else if (arg.equals("-j") || arg.equals("--jit")) {
210                                jit = true;
211                        } else if (arg.equals("-o")) {
212                                expect_outname = true;
213                        } else if (arg.equals("-l") || arg.equals("--loglevel")) {
214                                expect_loglevel = true;
215                        }
216
217                        // help option, just displays usage summary
218                        if (arg.equals("-h") || arg.equals("--help") || arg.equals("-?")) {
219                                System.out
220                                                .println("Usage: java -cp plugun1.jar:plugin2.jar:...:Controller.jar [options] [--] [input-filename]");
221                                System.out.println();
222                                System.out.println("  input-filename may be '-' for stdin or a relative or absolute path to a file.");
223                                System.out.println();
224                                System.out
225                                                .println("  '--' is an optional separator between options and the input-file, useful if path to the input file begins with character '-'.");
226                                System.out.println();
227                                System.out.println();
228                                System.out.println("  arguments may be any of:");
229                                System.out.println("     -h/-?/--help            emits this help message");
230                                System.out.println("     -o <output>             output filename, only used by llvm");
231                                System.out.println("     -j --jit                just in time compile (not yet supported)");
232                                System.out.println("     -l --loglevel <level>   set the loglevel. allowed values are:");
233                                System.out.println("                               ALL DEBUG ERROR FATAL INFO OFF TRACE WARN");
234                                System.out
235                                                .println("     -V --no-visualization   disables the automatic visualization pass if visualization plugins are found");
236                                System.out.println();
237                                System.exit(0);
238                        }
239                }
240
241                // set the logger level
242                if (loglevel != null) {
243                        loglevel = loglevel.toUpperCase();
244                        Level level = Level.toLevel(loglevel);
245                        if (level.toString().equals(loglevel)) {
246                                Logger.getRootLogger().setLevel(level);
247                        } else {
248                                System.out.println("unknown loglevel see -h for help");
249                                System.exit(0);
250                        }
251                }
252
253                // try to find a lexer, parser and backend
254                // this call assigns the variables lexer,parser,backend
255                // or fails if any one components is missing.
256                loadPlugins();
257
258                // copy the input stream
259                ByteArrayOutputStream baos = new ByteArrayOutputStream();
260                byte[] buffer = new byte[1024];
261                int len;
262                while ((len = input.read(buffer)) > -1) {
263                        baos.write(buffer, 0, len);
264                }
265                baos.flush();
266
267                input = new ByteArrayInputStream(baos.toByteArray());
268
269                // use the components now to compile our file
270                // lexer...
271                lexer.setSourceStream(input);
272
273                if (!disableVisualization) {
274                        for (TokenStreamVisualization tokenvisu : tokenVisuService) {
275                                input = new ByteArrayInputStream(baos.toByteArray());
276                                Lexer vlexer = lexer.getClass().newInstance();
277                                vlexer.setSourceStream(input);
278                                tokenvisu.visualizeTokenStream(vlexer);
279                        }
280                }
281
282                // parser...
283                parser.setLexer(lexer);
284                parser.setReportLog(errlog);
285                AST ast = parser.getParsedAST();
286
287                // in case of parser errors: abort and display errors
288                if (errlog.hasErrors()) {
289                        System.err.println("ERROR: compile failed due to errors:");
290                        for (LogEntry e : errlog.getEntries()) {
291                                System.err.println(e.getMessage());
292                        }
293                        System.exit(1);
294                }
295
296                if (!disableVisualization) {
297                        for (ASTVisualization astvisu : ASTVisuService) {
298                                astvisu.visualizeAST(ast);
299                        }
300                }
301
302                // IR gen...
303                List<Quadruple> tac = irgen.generateIntermediateCode(ast);
304
305                if (!disableVisualization) {
306                        for (TACVisualization tacvisu : TACVisuService) {
307                                tacvisu.visualizeTAC(tac);
308                        }
309                }
310
311                // backend...
312                Map<String, InputStream> targets = backend.generateTargetCode("", tac);
313
314                // output..
315                Set<String> outputFilenames = targets.keySet();
316                // for now, we always print everything on stdout.
317                for (String outFile : outputFilenames) {
318                        OutputStream output;
319                        if (outFile.equals(".ll")) {
320                                if (outname != null) {
321                                        output = new FileOutputStream(outname);
322                                } else {
323                                        outname = "<<stdout>>";
324                                        output = System.out;
325                                }
326                        } else // java bytecode output
327                        {
328                                output = new FileOutputStream(outFile);
329                        }
330                        System.err.println("//code: " + outname);
331                        // input stream on the code
332                        InputStream is = targets.get(outFile);
333
334                        // copied and modified from stackoverflow.com
335                        // http://stackoverflow.com/questions/1574837/connecting-an-input-stream-to-an-outputstream
336                        byte[] buf = new byte[4096];
337                        int bytesRead;
338                        while ((bytesRead = is.read(buf)) != -1) {
339                                output.write(buf, 0, bytesRead);
340                        }
341                }
342        }
