001package swp_compiler_ss13.common.report;
002
003import java.util.List;
004
005import swp_compiler_ss13.common.lexer.Token;
006
007/**
008 * ReportLog provides convenient way to communicate problems with source code
009 * processed by the compiler modules.
010 */
011public interface ReportLog {
012        /**
013         * reportWarning adds a new warning to the ReportLog. A warning does not
014         * force the compiler to stop proceeding. 
015         * 
016         * @param type distinguishes the warning in machine readable format 
017         * @param tokens is a list of tokens containing the warning cause and 
018         *        identify the position in code
019         * @param message describes the warning in human readable manner
020         */
021        public void reportWarning(ReportType type, List<Token> tokens, String message);
022    
023        /**
024         * reportError adds a new error to the ReportLog. A error force the 
025         * compiler to stop proceeding. 
026         * 
027         * @param type distinguishes the error in machine readable format 
028         * @param tokens is a list of tokens containing the error cause and 
029         *        identify the position in code
030         * @param message describes the error in human readable manner
031         */
032        public void reportError(ReportType type, List<Token> tokens, String message);
033}
