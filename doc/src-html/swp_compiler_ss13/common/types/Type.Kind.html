001package swp_compiler_ss13.common.types;
002
003/**
004 * Base Type for type definitions.
005 *
006 * @author "Frank Zechert", "Danny Maasch"
007 * @version 1
008 * @see <a target="_top" href="https://github.com/swp-uebersetzerbau-ss13/common/wiki/Types">Types Wiki</a>
009 * @see <a target="_top" href="https://github.com/swp-uebersetzerbau-ss13/common/issues/8">Types Issue Tracker</a>
010 */
011public abstract class Type
012{
013        public enum Kind
014        {
015                /* Primitive types */
016                LONG,
017                DOUBLE,
018                BOOLEAN,
019                STRING,
020
021                /* Derived types */
022                ARRAY,
023                STRUCT
024        }
025
026        /**
027         * Width of this type in bytes.
028         */
029        protected Long width;
030
031        /**
032         * Kind of this type.
033         */
034        protected final Kind kind;
035
036        protected Type(Kind kind)
037        {
038                this.kind = kind;
039    }
040
041        /**
042         * Returns the width of this type in bytes.
043         * @return Number of bytes for this type.
044         */
045        public Long getWidth()
046        {
047                return this.width;
048        }
049
050        /**
051         * Returns the kind of this type.
052         * @return Kind of this type.
053         */
054        public Kind getKind()
055        {
056                return this.kind;
057        }
058
059        /**
060         * Get the name of the type.
061         * @return The name of the type.
062         */
063        public abstract String getTypeName();
064
065        /**
066         * Returns a nice string representation of this type
067         * @return String representation of this type
068         */
069        public abstract String toString();
070}
