001package swp_compiler_ss13.common.ast.nodes.binary;
002
003import swp_compiler_ss13.common.ast.nodes.ExpressionNode;
004import swp_compiler_ss13.common.ast.nodes.StatementNode;
005import swp_compiler_ss13.common.ast.nodes.marynary.BlockNode;
006
007/**
008 * The node to represent a loop of any kind
009 * 
010 * @author "Frank Zechert"
011 * @version 1
012 */
013public interface LoopNode extends StatementNode
014{
015        /**
016         * Set the block that is the loop's body.
017         * 
018         * @param block
019         *            the loop's body.
020         */
021        public void setLoopBody(BlockNode block);
022
023        /**
024         * Get the block that is the loop's body.
025         * 
026         * @return the loop's body.
027         */
028        public BlockNode getLoopBody();
029
030        /**
031         * Set the loop condition. The loop will run until the condition is false.
032         * 
033         * @param condition
034         *            The loop condition.
035         */
036        public void setCondition(ExpressionNode condition);
037
038        /**
039         * Returns the condition of this loop. The loop will run until the condition
040         * is false.
041         * 
042         * @return The loop conditoin.
043         */
044        public ExpressionNode getCondition();
045}
