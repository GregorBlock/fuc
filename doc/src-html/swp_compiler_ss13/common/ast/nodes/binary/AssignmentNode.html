001package swp_compiler_ss13.common.ast.nodes.binary;
002
003import swp_compiler_ss13.common.ast.nodes.ExpressionNode;
004import swp_compiler_ss13.common.ast.nodes.IdentifierNode;
005
006/**
007 * The node to represent an assignment.
008 * 
009 * @author "Frank Zechert"
010 * @version 1
011 */
012public interface AssignmentNode extends ExpressionNode
013{
014        /**
015         * The left value of this node. The left node is an identifier (variable).
016         * The left value is the target of the assignment.
017         * 
018         * @return The left value of this node.
019         */
020        public IdentifierNode getLeftValue();
021
022        /**
023         * Set the left value of the assignment. The left node is an identifier
024         * (variable). The left value is the target of the assignment.
025         * 
026         * @param identifier
027         *            The identifier to set as the left value.
028         */
029        public void setLeftValue(IdentifierNode identifier);
030
031        /**
032         * The right value of this node. The right value is an expression of any kind.
033         * The right value is the value to assign to the left value.
034         * 
035         * @return the identifier of this assignment.
036         */
037        public ExpressionNode getRightValue();
038
039        /**
040         * Set the right value of this node. The right value is an expression of any
041         * kind. The right value is the value to assign to the left value.
042         * 
043         * @param node
044         *            The statement to use as the right value.
045         */
046        public void setRightValue(ExpressionNode node);
047}
