001package swp_compiler_ss13.common.ast.nodes.unary;
002
003import swp_compiler_ss13.common.ast.ASTNode;
004import swp_compiler_ss13.common.types.Type;
005
006/**
007 * The node to represent a declaration
008 * 
009 * @author "Frank Zechert"
010 * @version 1
011 */
012public interface DeclarationNode extends ASTNode
013{
014        /**
015         * Get the identifier that is being declared
016         * 
017         * @return the identifier name.
018         */
019        public String getIdentifier();
020
021        /**
022         * Get the type of this declaration.
023         * 
024         * @return The type of this declaration.
025         */
026        public Type getType();
027
028        /**
029         * Set the identifier name that is being declared.
030         * 
031         * @param identifier
032         *            the identifier name.
033         */
034        public void setIdentifier(String identifier);
035
036        /**
037         * Set the type of this identifier.
038         * 
039         * @param type
040         *            The type of this identifier.
041         */
042        public void setType(Type type);
043}
