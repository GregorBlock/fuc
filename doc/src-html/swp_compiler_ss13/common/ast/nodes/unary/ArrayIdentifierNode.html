001package swp_compiler_ss13.common.ast.nodes.unary;
002
003import swp_compiler_ss13.common.ast.nodes.ExpressionNode;
004import swp_compiler_ss13.common.ast.nodes.IdentifierNode;
005
006/**
007 * The node to reference an array identifier. The index defines the index to
008 * access. The identifier itself is another IdentifierNode. This IdentifierNode
009 * can be a BasicIdentifierNode to access the first array level. It can be
010 * another ArrayIdentifierNode to access a multidimensional array or a
011 * StructIdentifierNode to access an array inside a struct.
012 * 
013 * @author "Frank Zechert"
014 * 
015 */
016public interface ArrayIdentifierNode extends IdentifierNode
017{
018        /**
019         * Set the {@link ExpressionNode} that yields the index
020         * to access the array.
021         * 
022         * @param index
023         *            The array index.
024         */
025        public void setIndexNode(ExpressionNode index);
026
027        /**
028         * Get the {@link ExpressionNode} that yields the index
029         * to access the array.
030         * 
031         * @return The array index expression node.
032         */
033        public ExpressionNode getIndexNode();
034
035        /**
036         * Set the identifier to access. This is the array type itself.
037         * 
038         * @param node
039         *            The array to access.
040         */
041        public void setIdentifierNode(IdentifierNode node);
042
043        /**
044         * Get the array to access.
045         * 
046         * @return The array to access.
047         */
048        public IdentifierNode getIdentifierNode();
049}
