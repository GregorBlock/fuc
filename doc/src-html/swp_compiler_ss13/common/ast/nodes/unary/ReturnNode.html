001package swp_compiler_ss13.common.ast.nodes.unary;
002
003import swp_compiler_ss13.common.ast.nodes.IdentifierNode;
004import swp_compiler_ss13.common.ast.nodes.StatementNode;
005
006/**
007 * The node to represent a return statement.
008 * 
009 * @author "Frank Zechert"
010 * @version 1
011 */
012public interface ReturnNode extends StatementNode
013{
014        /**
015         * Set the optional value this return statement returns.
016         * 
017         * @param identifier
018         *            the identifier to return.
019         */
020        public void setRightValue(IdentifierNode identifier);
021
022        /**
023         * Get the optional value this return statement returns or null if none.
024         * 
025         * @return identifier to return or null if none.
026         */
027        public IdentifierNode getRightValue();
028}
