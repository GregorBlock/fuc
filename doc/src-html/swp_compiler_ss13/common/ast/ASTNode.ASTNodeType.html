001package swp_compiler_ss13.common.ast;
002
003import java.util.Iterator;
004import java.util.List;
005import java.util.Map;
006
007import swp_compiler_ss13.common.lexer.Token;
008
009/**
010 * A node in the AST. Every node in the AST is of the type ASTNode. Though there
011 * are a lot of specialized nodes.
012 * 
013 * @author "Frank Zechert, Tilman Krauss"
014 * @version 1
015 */
016public interface ASTNode
017{
018        /**
019         * This enumeration is used to assign node types to the specialized nodes.
020         * This enumerations enables you to use a switch case statement to
021         * distinguish between the nodes instead of using the instanceof operator in
022         * a lot of if statements.
023         * 
024         * @author "Frank Zechert"
025         * @version 1
026         */
027        public enum ASTNodeType
028        {
029                // leaf nodes
030                BasicIdentifierNode,
031                BreakNode,
032                LiteralNode,
033
034                // unary nodes
035                ArithmeticUnaryExpressionNode,
036                ArrayIdentifierNode,
037                DeclarationNode,
038                LogicUnaryExpressionNode,
039                PrintNode,
040                ReturnNode,
041                StructIdentifierNode,
042
043                // binary nodes
044                ArithmeticBinaryExpressionNode,
045                AssignmentNode,
046                DoWhileNode,
047                LogicBinaryExpressionNode,
048                RelationExpressionNode,
049                WhileNode,
050
051                // ternay nodes
052                BranchNode,
053
054                // m-ary n-ary nodes
055                BlockNode
056        }
057
058        /**
059         * Return the type of this node. This is one of the ASTNodeType enumeration
060         * constants. Each node needs to return the correct enumeration constant.
061         * 
062         * @return the type of the node.
063         */
064        public ASTNodeType getNodeType();
065
066        /**
067         * Return the number of nodes that are in this node's subtree. This is the
068         * number of nodes following this node. The number of nodes contains this
069         * node as well.
070         * 
071         * @return the number of nodes.
072         */
073        public Integer getNumberOfNodes();
074
075        /**
076         * Get the Depth-First-Search Left-To-Right (DFSLTR) iterator for the
077         * subtree this node is the root of.
078         * 
079         * @return the DFSLTR iterator.
080         */
081        public Iterator<ASTNode> getDFSLTRNodeIterator();
082
083        /**
084         * Get all the children of this node. These are the nodes directly following
085         * on the next level of the tree which parent is this node.
086         * 
087         * @return the children of this node.
088         */
089        public List<ASTNode> getChildren();
090
091        /**
092         * Get the parent node of this node. If this node is the root node and no
093         * parent exist, this should return null.
094         * 
095         * @return the parent node or null if no parent exists (this is a root
096         *         node).
097         */
098        public ASTNode getParentNode();
099
100        /**
101         * Set the parent node of this node. Root nodes do not have a parent. The
102         * parent of a root node is null.
103         * 
104         * @param node
105         *            the node to set as the parent.
106         */
107        public void setParentNode(ASTNode node);
108        
109        /**
110         * Computes the Range of tokens, whose are represented by this nodes subtree.
111         * Following properties must hold, if n1, n2 are two nodes and c1,c2 the corresponding coverages:
112         * c1 includes at least all elements of c2 <=> t2 is in the subtree of t2 or t2 is t1
113         * c1 includes all elements of c2 and c2 includes all elements of c1 <=> t1 is t2
114         * Let c' be the concatenation of c1 and c2. c' exists as a valid coverage of a node t', where c1 is the first
115         * part of the list and c2 the second part if and only if t1 and t2 is in the subtree of t' 
116         * and t1 represents a piece of source code which occurs before the code of t2. 
117         * @return tokens, whose are in the subtree of this node.
118         */
119        public List<Token> coverage();
120        
121        /**
122         * Sets the value of attribute for the semantic analyser
123         * 
124         * @param attribute
125         *            the attribute of the semantic analysis
126         * @param value
127         *            the value of the attribute
128         */
129        public void setAttributeValue(Object attribute, Object value);
130        
131        /**
132         * Get the value of the attribute for the semantic analyser
133         * 
134         * @param attribute
135         *            the attribute of the semantic analysis
136         * 
137         * @return the value of the attribute
138         */
139        public Object getAttributeValue(Object attribute);
140        
141        /**
142         * Computes the map of the attributes set by {@link #setAttributeValue}
143         * 
144         * @return map with attributes of the semantic analyser, which can be visualised.
145         */
146        public Map<?,?> getAttributeValues();
147}
