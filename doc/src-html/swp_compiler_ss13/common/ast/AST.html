001package swp_compiler_ss13.common.ast;
002
003import java.util.Iterator;
004
005import swp_compiler_ss13.common.parser.SymbolTable;
006import swp_compiler_ss13.common.ast.nodes.marynary.BlockNode;
007
008/**
009 * The abstract syntax tree (AST).
010 * 
011 * @author "Frank Zechert"
012 * @version 1
013 */
014public interface AST
015{
016        /**
017         * Returns the root node of the AST. The root node is always a BlockNode for
018         * correct ASTs.
019         * 
020         * @return the root node of the AST.
021         */
022        public BlockNode getRootNode();
023
024        /**
025         * Set the root node of the AST.
026         * 
027         * @param rootNode
028         *            The root node.
029         */
030        public void setRootNode(BlockNode rootNode);
031
032        /**
033         * Get the number of nodes the complete tree contains.
034         * 
035         * @return the number of nodes in this AST.
036         */
037        public Integer getNumberOfNodes();
038
039        /**
040         * Gets an iterator for this AST. The iterator will traverse the tree in
041         * Depth First Search from Left to Right (DFSLTR).
042         * 
043         * @return the DFSLTR iterator for this tree.
044         */
045        public Iterator<ASTNode> getDFSLTRIterator();
046
047        /**
048         * Get the symbol table associated with the complete program. This is the
049         * symbol table that contains all program scopes. The symbol table returned
050         * is most likely the symbol table of the root node.
051         * 
052         * @return The root symbol table.
053         */
054        public SymbolTable getRootSymbolTable();
055}
