001package swp_compiler_ss13.common.backend;
002
003/**
004 * Signals that an error occurred in the Backend <br><br>
005 * 
006 * Note: No analysis of the TAC is performed in the backend. The
007 * <code>BackendException</code> is only thrown if there are errors because incorrect code is
008 * passed to the backend.
009 * 
010 * @author "Jens V. Fischer"
011 */
012public class BackendException extends Exception {
013
014        /**
015         * serial version uid
016         */
017        private static final long serialVersionUID = 1234746139461254784L;
018
019        /**
020         * Constructs an <code>BackendException</code> with <code>null</code> as its error detail message.
021         */
022        public BackendException() {
023        }
024
025        /**
026         * Constructs an <code>BackendException</code> with the specified detail message.
027         * 
028         * @param message
029         *            The detail message (which is saved for later retrieval by the
030         *            Throwable.getMessage() method)
031         */
032        public BackendException(String message) {
033                super(message);
034        }
035
036        /**
037         * Constructs an <code>BackendException</code> with the specified detail message and
038         * cause. Note that the detail message associated with cause is not
039         * automatically incorporated into this exception's detail message
040         * 
041         * @param message
042         *            The detail message (which is saved for later retrieval by the
043         *            Throwable.getMessage() method)
044         * @param cause
045         *            The cause (which is saved for later retrieval by the
046         *            Throwable.getCause() method). (A null value is permitted, and
047         *            indicates that the cause is nonexistent or unknown.)
048         */
049        public BackendException(String message, Throwable cause) {
050                super(message, cause);
051        }
052
053        /**
054         * Constructs an <code>BackendException</code> with the specified cause and a detail
055         * message of (cause==null ? null : cause.toString()) (which typically
056         * contains the class and detail message of cause).
057         * 
058         * @param cause
059         *            The cause (which is saved for later retrieval by the
060         *            Throwable.getCause() method). (A null value is permitted, and
061         *            indicates that the cause is nonexistent or unknown.)
062         */
063        public BackendException(Throwable cause) {
064                super(cause);
065        }
066
067}
