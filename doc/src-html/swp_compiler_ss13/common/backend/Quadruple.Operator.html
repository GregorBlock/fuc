001package swp_compiler_ss13.common.backend;
002
003public interface Quadruple
004{
005        Operator getOperator();
006        String   getArgument1();
007        String   getArgument2();
008        String   getResult();
009
010        static final String EmptyArgument = "!";
011
012        /* Should print the Quadruple as a String */
013        String toString();
014
015        public enum Operator
016        {
017                /* Variable declaration */
018                DECLARE_LONG,
019                DECLARE_DOUBLE,
020                DECLARE_BOOLEAN,
021                DECLARE_STRING,
022                DECLARE_REFERENCE,
023                DECLARE_ARRAY,
024
025                /* Type conversion */
026                LONG_TO_DOUBLE,
027                DOUBLE_TO_LONG,
028
029                /* Unindexed copy */
030                ASSIGN_LONG,
031                ASSIGN_DOUBLE,
032                ASSIGN_BOOLEAN,
033                ASSIGN_STRING,
034
035                /* Indexed copy */
036                ARRAY_GET_LONG,
037                ARRAY_GET_DOUBLE,
038                ARRAY_GET_BOOLEAN,
039                ARRAY_GET_STRING,
040                ARRAY_GET_ARRAY,
041                ARRAY_GET_REFERENCE,
042                ARRAY_SET_LONG,
043                ARRAY_SET_DOUBLE,
044                ARRAY_SET_BOOLEAN,
045                ARRAY_SET_STRING,
046                ARRAY_SET_ARRAY,
047
048                /* Arithmetic */
049                ADD_LONG,
050                ADD_DOUBLE,
051                SUB_LONG,
052                SUB_DOUBLE,
053                MUL_LONG,
054                MUL_DOUBLE,
055                DIV_LONG,
056                DIV_DOUBLE,
057                NOT_BOOLEAN,
058                OR_BOOLEAN,
059                AND_BOOLEAN,
060
061                /* Compare */
062                COMPARE_LONG_E,
063                COMPARE_LONG_G,
064                COMPARE_LONG_L,
065                COMPARE_LONG_GE,
066                COMPARE_LONG_LE,
067                COMPARE_DOUBLE_E,
068                COMPARE_DOUBLE_G,
069                COMPARE_DOUBLE_L,
070                COMPARE_DOUBLE_GE,
071                COMPARE_DOUBLE_LE,
072
073                /* Control flow */
074                RETURN,
075                LABEL,
076                BRANCH,
077
078                /* I/O */
079                PRINT_BOOLEAN,
080                PRINT_LONG,
081                PRINT_DOUBLE,
082                PRINT_STRING
083        }
084}
