001package swp_compiler_ss13.common.backend;
002
003import java.io.InputStream;
004import java.util.List;
005import java.util.Map;
006
007public interface Backend {
008        /**
009         * Generate target code for the provided TAC. For naming of output files the
010         * baseFileName has to be provided. The backend decide the exact filename of the
011         * output files inclusive the file extension by the keys of the return map.
012         * 
013         * @param baseFileName
014         *            of the input source file without file extension
015         * @param tac
016         *            generated by frontend
017         * @return map whose keys are the output file names
018         * @throws BackendException
019         *             for errors in backend processing
020         */
021        Map<String, InputStream> generateTargetCode(String baseFileName, List<Quadruple> tac) throws BackendException;
022}
