253        private static String loadExample(String progName) {
254                String userDir = System.getProperty("user.dir");
255                Path path;
256
257                /* handling different paths depending on from where the tests are called*/
258                if (userDir.endsWith("fuc/code") || userDir.endsWith("crosstesting"))
259                        path = FileSystems.getDefault().getPath("common", "examples", progName);
260                else if (userDir.endsWith("fuc" + File.separator + "code/test") || userDir.endsWith("crosstesting" + File.separator + "crosstest") )
261                        path = FileSystems.getDefault().getPath("..", "common", "examples", progName);
262                else {
263                   String regex = ".*fuc\\" + File.separator + "code\\" + File.separator + "[^\\" + File.separator + "]*";
264                   if (userDir.matches(regex)) /* for paths directly under fuc\code */
265                   path = FileSystems.getDefault().getPath("..", "common", "examples", progName);
266                else /* will fail */
267                        path = FileSystems.getDefault().getPath(progName);
268                }
269                try {
270                        return new Scanner(path, "UTF-8").useDelimiter("\\A").next();
271                } catch (IOException e) {
272                        throw new RuntimeException("file '" + progName + "' not found at '" +
273                                        path.toAbsolutePath().normalize() + "'");
274                } catch (NoSuchElementException n) {
275                        return "";
276                }
277        }
