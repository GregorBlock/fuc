001package swp_compiler_ss13.common.lexer;
002
003/**
004 * Token are typed sequences of input characters with a specific meaning for a 
005 * grammar.
006 * 
007 * @author "Damla Durmaz", "Ferhat Beyaz", "Sebastian Barthel"
008 * @version 3
009 * @see <a target="_top"
010 *      href="https://github.com/swp-uebersetzerbau-ss13/common/issues/3</a>
011 */
012public interface Token {
013        /**
014         * @return string readed by lexer for this token
015         */
016        String getValue();
017        
018        /**
019         * @return type of token
020         */
021        TokenType getTokenType();
022        
023        /**
024         * @return line of code in source file
025         */
026        Integer getLine();
027        
028        /**
029         * @return column of code in source file
030         */
031        Integer getColumn();
032}
