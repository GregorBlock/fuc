001package swp_compiler_ss13.common.lexer;
002
003import java.io.InputStream;
004
005/**
006 * Proposed interface for lexer
007 * 
008 * @author "Damla Durmaz", "Ferhat Beyaz", "Sebastian Barthel"
009 * @version 2
010 * @see <a target="_top" href="https://github.com/swp-uebersetzerbau-ss13/common/issues/3</a>
011 */
012public interface Lexer {
013        
014        /**
015         * Defines the stream for program source. The stream has to be ready for
016         * reading. Normally the lexer will read the complete stream till it ends,
017         * but this is not guaranteed.
018         * 
019         * After every call to this method, the lexer behaves as if after the first
020         * call.
021         * 
022         * @param stream
023         */
024        public void setSourceStream(InputStream stream);
025        
026        /** 
027         * As long as not all characters are tokenized, it returns a token with
028         * token.getTokenType() != TokenType.EOF
029         * 
030         * If there are no characters left for tokenization it always returns a
031         * EOF-token.
032         * 
033         * For all minimal sequences of characters which are not matched by a
034         * token definition, the lexer returns these as NOT_A_TOKEN-tokens, eg:
035         * 3 + $$$$ 4 2w
036         * would result in:
037         * <NUM, '3'> <PLUS, '+'> <NOT_A_TOKEN, '$$$$'> <NUM, '4'> <NOT_A_TOKEN, '2w'>
038         * 
039         * Note that whitespaces are ignored but are essentially for token split. 
040         * This means '2w' may not be parsed as <NUM, '2'> <ID, 'w'>.
041         * 
042         * @return a token instance
043         */
044        public Token getNextToken();
045}
