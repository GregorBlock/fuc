001package swp_compiler_ss13.fuc.ir;
002
003import java.util.Stack;
004
005import swp_compiler_ss13.common.ast.ASTNode;
006import swp_compiler_ss13.common.ast.nodes.binary.AssignmentNode;
007import swp_compiler_ss13.common.ast.nodes.unary.StructIdentifierNode;
008import swp_compiler_ss13.common.types.Type;
009import swp_compiler_ss13.common.types.derived.ArrayType;
010import swp_compiler_ss13.common.types.derived.Member;
011import swp_compiler_ss13.common.types.derived.StructType;
012import swp_compiler_ss13.common.types.primitive.PrimitiveType;
013
014/**
015 * This utility class contains helper methods for structs.
016 * 
017 * @author "Frank Zechert"
018 * @author "Danny Maasch"
019 * @author kaworu
020 * @version 3
021 */
022public class StructHelper {
023
024        /**
025         * Check if the given structIdentifierNode is used as the target of an
026         * assignment. The target of an assignment is the left hand side value
027         * (assignment: target = source)
028         * 
029         * @param node
030         *            The node to check
031         * @return true if the node is target of an assignment.
032         */
033        public static boolean isTargetOfAssignment(StructIdentifierNode node) {
034                if (node.getParentNode() instanceof AssignmentNode) {
035                        AssignmentNode parent = (AssignmentNode) node.getParentNode();
036                        if (parent.getLeftValue() == node) {
037                                return true;
038                        }
039                }
040                return false;
041        }
042
043        /**
044         * Get the base type of the given derived type. The base type of an
045         * ArrayType is the inner most type.
046         * 
047         * @param type
048         *            The type to get the base type for.
049         * @param node
050         *            The node
051         * @return An array with the type and the node
052         */
053        public static Object[] getInternalBaseType(Type type, ASTNode node) {
054                Stack<String> fieldAccess = new Stack<String>();
055                while (!(type instanceof PrimitiveType)) {
056                        if (!(node instanceof StructIdentifierNode)) {
057                                while (!fieldAccess.empty()) {
058                                        String field = fieldAccess.pop();
059                                        Member[] members = ((StructType) type).members();
060                                        for (Member member : members) {
061                                                if (member.getName().equals(field)) {
062                                                        type = member.getType();
063                                                }
064                                        }
065                                }
066                        }
067                        if (type instanceof ArrayType) {
068                                Object[] retA = ArrayHelper.getInternalBaseType(type, node);
069                                type = (Type) retA[0];
070                                node = (ASTNode) retA[1];
071                        }
072                        if (type instanceof StructType) {
073                                if (node instanceof StructIdentifierNode) {
074                                        String field = ((StructIdentifierNode) node).getFieldName();
075                                        node = ((StructIdentifierNode) node).getIdentifierNode();
076                                        fieldAccess.push(field);
077                                }
078                                else {
079                                        break;
080                                }
081                        }
082                }
083                return new Object[] { type, node };
084        }
085
086        /**
087         * Get the base type of the given derived type. The base type of an
088         * ArrayType is the inner most type.
089         * 
090         * @param type
091         *            The type to get the base type for.
092         * @param node
093         *            The node
094         * @return The base type
095         */
096        public static Type getBaseType(Type type, ASTNode node) {
097                node = getOuterStruct(node);
098                return (Type) getInternalBaseType(type, node)[0];
099        }
100
101        /**
102         * Get the outer most struct identifier node
103         * 
104         * @param node
105         *            the node to use
106         * @return the outer most struct identifier node
107         */
108        public static ASTNode getOuterStruct(ASTNode node) {
109                while (node.getParentNode() instanceof StructIdentifierNode) {
110                        node = node.getParentNode();
111                }
112                return node;
113        }
114
115        /**
116         * Check if the given node is the outer most dimension of a struct.
117         * 
118         * @param node
119         *            The node to check
120         * @return true if the given node is the outer most dimension of a struct.
121         */
122        public static boolean isOuterMostDimension(StructIdentifierNode node) {
123                return !(node.getParentNode() instanceof StructIdentifierNode);
124        }
125
126        /**
127         * Check if the ArrayType has a field named index
128         * 
129         * @param type
130         *            the type to check
131         * @param index
132         *            the fieldname of the struct
133         * @return true if the field exists
134         */
135        public static boolean fieldExists(StructType type, String index) {
136                Member[] members = type.members();
137                for (Member member : members) {
138                        if (member.getName().equals(index)) {
139                                return true;
140                        }
141                }
142                return false;
143        }
144}
