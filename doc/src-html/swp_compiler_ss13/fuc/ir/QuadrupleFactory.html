036        /**
037         * Create one or more quadruples to represent the declaration.
038         * 
039         * @param name
040         *            The name of the variable to be declared.
041         * @param type
042         *            Type of the variable to be declared.
043         * @return A List of Quadruples that represent the declaration
044         * @throws IntermediateCodeGeneratorException
045         *             Declaration of the given type is unsupported
046         */
047        public static List<Quadruple> declare(String name, Type type)
048                        throws IntermediateCodeGeneratorException {
049                List<Quadruple> declaration = new LinkedList<>();
050                switch (type.getKind()) {
051                case ARRAY:
052                        declaration.addAll(QuadrupleFactory.declareArray(name, (ArrayType) type));
053                        break;
054                case BOOLEAN:
055                        declaration.add(new QuadrupleImpl(Operator.DECLARE_BOOLEAN, name));
056                        break;
057                case DOUBLE:
058                        declaration.add(new QuadrupleImpl(Operator.DECLARE_DOUBLE, name));
059                        break;
060                case LONG:
061                        declaration.add(new QuadrupleImpl(Operator.DECLARE_LONG, name));
062                        break;
063                case STRING:
064                        declaration.add(new QuadrupleImpl(Operator.DECLARE_STRING, name));
065                        break;
066                case STRUCT:
067                        declaration.addAll(QuadrupleFactory.declareStruct(name, (StructType) type));
068                        break;
069                default:
070                        String error = String.format(
071                                        "Variable declaration for a variable of type %s is not supported!",
072                                        type.toString());
073                        QuadrupleFactory.logger.fatal(error);
074                        throw new IntermediateCodeGeneratorException(error);
075                }
076
077                return declaration;
078        }

080        /**
081         * Declare a struct
082         * 
083         * @param name
084         *            the name of the struct
085         * @param type
086         *            the type of the struct
087         * @return Quadruples to declare the given struct
088         * @throws IntermediateCodeGeneratorException
089         *             Error in struct declaration
090         */
091        private static List<Quadruple> declareStruct(String name, StructType type)
092                        throws IntermediateCodeGeneratorException {
093
094                List<Quadruple> decl = new LinkedList<>();
095                Member[] members = type.members();
096                String memberCount = "#" + members.length;
097                decl.add(new QuadrupleImpl(Operator.DECLARE_STRUCT, memberCount, name));
098                for (Member member : members) {
099                        decl.addAll(QuadrupleFactory.declare(member.getName(), member.getType()));
100                }
101                return decl;
102
103        }

105        /**
106         * Declare an array
107         * 
108         * @param name
109         *            the name of the array
110         * @param type
111         *            the type of the array
112         * @return the Quadruples to declare the given array
113         * @throws IntermediateCodeGeneratorException
114         *             Error in array declaration
115         */
116        private static List<Quadruple> declareArray(String name, ArrayType type)
117                        throws IntermediateCodeGeneratorException {
118
119                List<Quadruple> decl = new LinkedList<>();
120                String length = "#" + type.getLength();
121                Quadruple level = new QuadrupleImpl(Operator.DECLARE_ARRAY, length, name);
122                decl.add(level);
123                decl.addAll(QuadrupleFactory.declare("!", type.getInnerType()));
124                return decl;
125
126        }

176        /**
177         * Create a quadruple for an arithmetic binary operation
178         * 
179         * @param op
180         *            The operator for the operation
181         * @param type
182         *            The type of the left and right values
183         * @param left
184         *            the left value
185         * @param right
186         *            the right value
187         * @param result
188         *            the target value
189         * @return the quadruple for the given operation
190         * @throws IntermediateCodeGeneratorException
191         *             Unsupported operator or type
192         */
193        public static Quadruple arithmeticBinary(BinaryOperator op, Type type, String left,
194                        String right, String result) throws IntermediateCodeGeneratorException {
195                switch (type.getKind()) {
196                case DOUBLE:
197                        switch (op) {
198                        case ADDITION:
199                                return new QuadrupleImpl(Operator.ADD_DOUBLE, left, right, result);
200                        case DIVISION:
201                                return new QuadrupleImpl(Operator.DIV_DOUBLE, left, right, result);
202                        case MULTIPLICATION:
203                                return new QuadrupleImpl(Operator.MUL_DOUBLE, left, right, result);
204                        case SUBSTRACTION:
205                                return new QuadrupleImpl(Operator.SUB_DOUBLE, left, right, result);
206                        default:
207                                break;
208                        }
209                        break;
210                case LONG:
211                        switch (op) {
212                        case ADDITION:
213                                return new QuadrupleImpl(Operator.ADD_LONG, left, right, result);
214                        case DIVISION:
215                                return new QuadrupleImpl(Operator.DIV_LONG, left, right, result);
216                        case MULTIPLICATION:
217                                return new QuadrupleImpl(Operator.MUL_LONG, left, right, result);
218                        case SUBSTRACTION:
219                                return new QuadrupleImpl(Operator.SUB_LONG, left, right, result);
220                        default:
221                                break;
222                        }
223                        break;
224                default:
225                        break;
226
227                }
228                throw new IntermediateCodeGeneratorException("The type " + type.toString()
229                                + " and Operator " + op.toString()
230                                + " is not supported for arithmetic binary expressions");
231        }

262        /**
263         * Create a Quadruple for a relation expresion
264         * 
265         * @param operator
266         *            The operator
267         * @param v1
268         *            the left value
269         * @param v2
270         *            the right value
271         * @param res
272         *            the result target
273         * @param type
274         *            the type
275         * @return the Quadruple for the operation
276         * @throws IntermediateCodeGeneratorException
277         *             Unsupported type or operator
278         */
279        public static Quadruple relation(BinaryOperator operator, String v1, String v2, String res,
280                        Type type) throws IntermediateCodeGeneratorException {
281                switch (operator) {
282                case EQUAL:
283                case INEQUAL:
284                        switch (type.getKind()) {
285                        case DOUBLE:
286                                return new QuadrupleImpl(Operator.COMPARE_DOUBLE_E, v1, v2, res);
287                        case LONG:
288                                return new QuadrupleImpl(Operator.COMPARE_LONG_E, v1, v2, res);
289                        default:
290                                break;
291                        }
292                        break;
293                case GREATERTHAN:
294                        switch (type.getKind()) {
295                        case DOUBLE:
296                                return new QuadrupleImpl(Operator.COMPARE_DOUBLE_G, v1, v2, res);
297                        case LONG:
298                                return new QuadrupleImpl(Operator.COMPARE_LONG_G, v1, v2, res);
299                        default:
300                                break;
301                        }
302                        break;
303                case GREATERTHANEQUAL:
304                        switch (type.getKind()) {
305                        case DOUBLE:
306                                return new QuadrupleImpl(Operator.COMPARE_DOUBLE_GE, v1, v2, res);
307                        case LONG:
308                                return new QuadrupleImpl(Operator.COMPARE_LONG_GE, v1, v2, res);
309                        default:
310                                break;
311                        }
312                        break;
313                case LESSTHAN:
314                        switch (type.getKind()) {
315                        case DOUBLE:
316                                return new QuadrupleImpl(Operator.COMPARE_DOUBLE_L, v1, v2, res);
317                        case LONG:
318                                return new QuadrupleImpl(Operator.COMPARE_LONG_L, v1, v2, res);
319                        default:
320                                break;
321                        }
322                        break;
323                case LESSTHANEQUAL:
324                        switch (type.getKind()) {
325                        case DOUBLE:
326                                return new QuadrupleImpl(Operator.COMPARE_DOUBLE_LE, v1, v2, res);
327                        case LONG:
328                                return new QuadrupleImpl(Operator.COMPARE_LONG_LE, v1, v2, res);
329                        default:
330                                break;
331                        }
332                        break;
333                default:
334                        break;
335                }
336                throw new IntermediateCodeGeneratorException("The type " + type.toString()
337                                + " and operator " + operator.toString()
338                                + " is not supported for relation expressions");
339        }
