001package swp_compiler_ss13.fuc.ir;
002
003import java.util.HashSet;
004import java.util.List;
005import java.util.Set;
006import java.util.Stack;
007
008import org.apache.log4j.Logger;
009
010import swp_compiler_ss13.common.backend.Quadruple;
011import swp_compiler_ss13.common.ir.IntermediateCodeGeneratorException;
012import swp_compiler_ss13.common.parser.SymbolTable;
013import swp_compiler_ss13.common.types.Type;
014import swp_compiler_ss13.fuc.ir.data.IntermediateResult;
015import swp_compiler_ss13.fuc.ir.data.LoggingLinkedList;
016import swp_compiler_ss13.fuc.ir.data.LoggingStack;
017
018/**
019 * This objects holds the current state of the IntermediateCodeGenerator.
020 * 
021 * @author "Frank Zechert"
022 * @author "Danny Maasch"
023 * @author kaworu
024 * @version 3
025 */
026public class GeneratorState {
027
028        /**
029         * The logger instance
030         */
031        private static Logger logger = Logger.getLogger(GeneratorState.class);
032
033        /**
034         * Prefix for label names
035         */
036        private static final String LABEL_PREFIX = "label";
037
038        /**
039         * A list to store the generated intermediate code.
040         */
041        private final List<Quadruple> intermediateCode;
042
043        /**
044         * The stack for intermediate results.
045         */
046        private final Stack<IntermediateResult> intermediateResults;
047
048        /**
049         * The stack for symbol tables.
050         */
051        private final Stack<SymbolTable> symbolTables;
052
053        /**
054         * The stack for break labels.
055         */
056        private final Stack<String> breakLabels;
057
058        /**
059         * Set of all identifier names used.
060         */
061        private final Set<String> usedIdentifierNames;
062
063        /**
064         * Number of next free label
065         */
066        private int label;
067
068        /**
069         * Create a new instance of the GeneratorState class.
070         */
071        public GeneratorState() {
072                this.intermediateCode = new LoggingLinkedList<Quadruple>();
073                this.intermediateResults = new LoggingStack<IntermediateResult>();
074                this.symbolTables = new Stack<SymbolTable>();
075                this.breakLabels = new Stack<String>();
076                this.usedIdentifierNames = new HashSet<String>();
077                this.label = 0;
078        }
079
080        /**
081         * Reset the generator state object to the state of initializing.
082         */
083        public void reset() {
084                this.intermediateCode.clear();
085                this.intermediateResults.clear();
086                this.symbolTables.clear();
087                this.breakLabels.clear();
088                this.usedIdentifierNames.clear();
089                this.label = 0;
090        }
091
092        /**
093         * Return the intermediate code that was created up to the current state.
094         * 
095         * @return The intermediate code (three address code - tac).
096         */
097        public List<Quadruple> getIntermediateCode() {
098                return this.intermediateCode;
099        }
100
101        /**
102         * Pop an intermediate result from the stack of intermediate results.
103         * 
104         * @return The popped intermediate result.
105         */
106        public IntermediateResult popIntermediateResult() {
107                return this.intermediateResults.pop();
108        }
109
110        /**
111         * Push the intermediate result to the stack of intermediate results.
112         * 
113         * @param result
114         *            The intermediate result to push to the stack.
115         */
116        public void pushIntermediateResult(IntermediateResult result) {
117                this.intermediateResults.push(result);
118        }
119
120        /**
121         * Push the intermediate result to the stack of intermediate results.
122         * 
123         * @param type
124         *            The type of the intermediate result.
125         * @param value
126         *            The value of the intermediate result.
127         */
128        public void pushIntermediateResult(Type type, String value) {
129                this.pushIntermediateResult(new IntermediateResult(type, value));
130        }
131
132        /**
133         * Pop a symbol table from the stack of symbol tables.
134         * 
135         * @return the popped symbol table.
136         */
137        public SymbolTable popSymbolTable() {
138                return this.symbolTables.pop();
139        }
140
141        /**
142         * Push a symbol table to the stack of symbol tables.
143         * 
144         * @param symbolTable
145         *            the symbol table to push
146         */
147        public void pushSymbolTable(SymbolTable symbolTable) {
148                this.symbolTables.push(symbolTable);
149        }
150
151        /**
152         * Get the current symbol table from the stack of symbol tables.
153         * 
154         * @return symbolTable the symbol table currently on top of the stack.
155         */
156        public SymbolTable getCurrentSymbolTable() {
157                return this.symbolTables.peek();
158        }
159
160        /**
161         * Save a new identifier and rename it if needed.
162         * 
163         * @param identifierName
164         *            The identifier to save
165         * @param type
166         *            The type of the identifier
167         * @return The renamed identifier
168         */
169        public String saveIdentifier(String identifierName, Type type) {
170                SymbolTable currentSymbolTable = this.getCurrentSymbolTable();
171
172                if (this.usedIdentifierNames.contains(identifierName)) {
173                        // the identifier name was already used
174                        // assign a new name to this identifier
175                        String newName = currentSymbolTable.getNextFreeTemporary();
176                        currentSymbolTable.putTemporary(newName, type);
177                        currentSymbolTable.setIdentifierAlias(identifierName, newName);
178                        identifierName = newName;
179                }
180
181                this.usedIdentifierNames.add(identifierName);
182                return identifierName;
183        }
184
185        /**
186         * Load the given identifier and return its actual name (if renaming was done)
187         * 
188         * @param id
189         *            The identifier name to load
190         * @return The actual name of the identifier
191         * @throws IntermediateCodeGeneratorException
192         *             Identifier was not found
193         */
194        public String loadIdentifier(String id) throws IntermediateCodeGeneratorException {
195                String name = this.getCurrentSymbolTable().getIdentifierAlias(id);
196                if (name == null) {
197                        GeneratorState.logger.fatal("Undeclared variable found: " + id);
198                        throw new IntermediateCodeGeneratorException("Undeclared variable found: " + id);
199                }
200                return name;
201        }
202
203        /**
204         * Create a new label name
205         * 
206         * @return the new label
207         */
208        public String createNewLabel() {
209                return GeneratorState.LABEL_PREFIX + (this.label++);
210        }
211
212        /**
213         * Push a new label to the break label stack
214         * 
215         * @param label
216         *            The label to push
217         */
218        public void pushBreakLabel(String label) {
219                this.breakLabels.push(label);
220        }
221
222        /**
223         * Pop a label from the break label stack
224         * 
225         * @return The popped label
226         */
227        public String popBreakLabel() {
228                return this.breakLabels.pop();
229        }
230
231        /**
232         * Add Intermediate Code (Quadruples)
233         * 
234         * @param quadruples
235         *            the quadruples to add
236         */
237        public void addIntermediateCode(List<Quadruple> quadruples) {
238                this.intermediateCode.addAll(quadruples);
239        }
240
241        /**
242         * Add Intermediate Code (Quadruple)
243         * 
244         * @param quadruple
245         *            the quadruple to add
246         */
247        public void addIntermediateCode(Quadruple quadruple) {
248                this.intermediateCode.add(quadruple);
249        }
250
251        /**
252         * Get the current break label
253         * 
254         * @return The label to use to jump out of the current loop
255         */
256        public String getCurrentBreakLabel() {
257                return this.breakLabels.peek();
258        }
259
260        /**
261         * Get the next free temporary identifier
262         * 
263         * @param type
264         *            The type of the temporary identifier
265         * @return The name of the temporary
266         */
267        public String nextTemporaryIdentifier(Type type) {
268                String id = this.getCurrentSymbolTable().getNextFreeTemporary();
269                this.getCurrentSymbolTable().putTemporary(id, type);
270                this.usedIdentifierNames.add(id);
271                return id;
272        }
273
274}
