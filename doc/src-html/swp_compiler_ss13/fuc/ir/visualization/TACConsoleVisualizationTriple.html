001package swp_compiler_ss13.fuc.ir.visualization;
002
003import java.util.HashMap;
004import java.util.List;
005import java.util.Map;
006
007import swp_compiler_ss13.common.backend.Quadruple;
008import swp_compiler_ss13.common.backend.Quadruple.Operator;
009import swp_compiler_ss13.common.visualization.TACVisualization;
010
011/**
012 * Visualization for Three Address Code
013 * 
014 * @author "Frank Zechert"
015 * @version 1
016 */
017public class TACConsoleVisualizationTriple implements TACVisualization {
018
019        @Override
020        public void visualizeTAC(List<Quadruple> tac) {
021
022                System.out.println("Printing the TAC in Triple format.");
023                System.out.println("Empty Argument Symbol is \"" + Quadruple.EmptyArgument + "\"");
024                System.out.println("");
025
026                String format = "%03d: (%s | %s | %s)";
027                Map<String, Integer> indirectAddresses = new HashMap<>();
028
029                int line = 0;
030
031                for (Quadruple quadruple : tac) {
032                        String operator = quadruple.getOperator().toString();
033                        String argument1 = quadruple.getArgument1();
034                        String argument2 = quadruple.getArgument2();
035                        String result = quadruple.getResult();
036                        Operator openum = quadruple.getOperator();
037
038                        if (openum == Operator.DECLARE_BOOLEAN ||
039                                        openum == Operator.DECLARE_DOUBLE ||
040                                        openum == Operator.DECLARE_LONG ||
041                                        openum == Operator.DECLARE_STRING) {
042                                continue;
043                        }
044
045                        if (openum == Operator.ASSIGN_BOOLEAN ||
046                                        openum == Operator.ASSIGN_DOUBLE ||
047                                        openum == Operator.ASSIGN_LONG ||
048                                        openum == Operator.ASSIGN_STRING) {
049                                indirectAddresses.put(result, indirectAddresses.get(argument1));
050                                continue;
051                        }
052
053                        if (!argument1.startsWith("#") && indirectAddresses.get(argument1) != null) {
054                                argument1 = "(" + indirectAddresses.get(argument1) + ")";
055                        }
056                        if (!argument2.startsWith("#") && indirectAddresses.get(argument2) != null) {
057                                argument2 = "(" + indirectAddresses.get(argument2) + ")";
058                        }
059
060                        indirectAddresses.put(result, line);
061
062                        System.out.println(String.format(format, line, operator, argument1, argument2));
063                        line++;
064                }
065        }
066}
