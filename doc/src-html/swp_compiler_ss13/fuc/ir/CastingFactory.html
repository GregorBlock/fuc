001package swp_compiler_ss13.fuc.ir;
002
003import org.apache.log4j.Logger;
004
005import swp_compiler_ss13.common.backend.Quadruple;
006import swp_compiler_ss13.common.backend.Quadruple.Operator;
007import swp_compiler_ss13.common.ir.IntermediateCodeGeneratorException;
008import swp_compiler_ss13.common.types.Type;
009import swp_compiler_ss13.common.types.Type.Kind;
010import swp_compiler_ss13.common.types.derived.DerivedType;
011import swp_compiler_ss13.fuc.ir.data.IntermediateResult;
012import swp_compiler_ss13.fuc.ir.data.QuadrupleImpl;
013
014/**
015 * Factory for all casting tasks
016 * 
017 * @author "Frank Zechert"
018 * @author "Danny Maasch"
019 * @author kaworu
020 * @version 3
021 */
022public class CastingFactory {
023
024        /**
025         * the logger
026         */
027        private static Logger logger = Logger.getLogger(CastingFactory.class);
028
029        /**
030         * Check if a cast is needed between a and b
031         * 
032         * @param type1
033         *            IntermediateResult a
034         * @param type2
035         *            IntermediateResult b
036         * @return true, if a and b are not of the same type
037         * @throws IntermediateCodeGeneratorException
038         *             a and b could not be checked
039         */
040        public static boolean isCastNeeded(IntermediateResult type1, IntermediateResult type2)
041                        throws IntermediateCodeGeneratorException {
042                return CastingFactory.isCastNeeded(type1.getType(), type2.getType());
043        }
044
045        /**
046         * Check if a cast is needed between a and b
047         * 
048         * @param type1
049         *            Type a
050         * @param type2
051         *            Type b
052         * @return true, if a and b are not of the same type
053         * @throws IntermediateCodeGeneratorException
054         *             a and b could not be checked
055         */
056        public static boolean isCastNeeded(Type type1, Type type2)
057                        throws IntermediateCodeGeneratorException {
058
059                if (isPrimitive(type1) && isPrimitive(type2)) {
060                        return type1.getKind() != type2.getKind();
061                }
062                return !type1.toString().equals(type2.toString());
063        }
064
065        /**
066         * Check if the given type is a simple type (not derived).
067         * 
068         * @param type
069         *            The type to check
070         * @return true if the type is simple, false if the type is derived
071         */
072        public static boolean isPrimitive(Type type) {
073                return !(type instanceof DerivedType);
074        }
075
076        /**
077         * Create a cast from long to double
078         * 
079         * @param leftValue
080         *            The long value to cast to double
081         * @param castedLeft
082         *            The target double identifier
083         * @return The Quadruple for the cast
084         */
085        public static Quadruple longToDouble(String leftValue, String castedLeft) {
086                if (leftValue.startsWith("#")) {
087                        leftValue = leftValue + ".0";
088                        return new QuadrupleImpl(Operator.ASSIGN_DOUBLE, leftValue, castedLeft);
089                }
090                return new QuadrupleImpl(Operator.LONG_TO_DOUBLE, leftValue, castedLeft);
091        }
092
093        /**
094         * Create a cast from double to long
095         * 
096         * @param leftValue
097         *            The double value to cast to long
098         * @param castedLeft
099         *            The target long identifier
100         * @return The Quadruple for the cast
101         */
102        public static Quadruple doubleToLong(String leftValue, String castedLeft) {
103                if (leftValue.startsWith("#")) {
104                        int decimalpoint = leftValue.lastIndexOf('.');
105                        if (decimalpoint >= 0) {
106                                leftValue = leftValue.substring(0, decimalpoint);
107                        }
108                        return new QuadrupleImpl(Operator.ASSIGN_LONG, leftValue, castedLeft);
109                }
110                return new QuadrupleImpl(Operator.DOUBLE_TO_LONG, leftValue, castedLeft);
111        }
112
113        /**
114         * Check if the value is numeric (long or double)
115         * 
116         * @param value
117         *            The value to check
118         * @return true if the type is numeric
119         */
120        public static boolean isNumeric(IntermediateResult value) {
121                return CastingFactory.isNumeric(value.getType());
122        }
123
124        /**
125         * Check if the type is numeric (long or double)
126         * 
127         * @param type
128         *            The type to check
129         * @return true if the type is numeric
130         */
131        public static boolean isNumeric(Type type) {
132                if ((type.getKind() == Kind.LONG) || (type.getKind() == Kind.DOUBLE)) {
133                        return true;
134                }
135                return false;
136        }
137
138        /**
139         * Cast the given identifier to string
140         * 
141         * @param type
142         *            The type to cast from
143         * @param value
144         *            The value to cast
145         * @param tmp
146         *            The destination value
147         * @return The Quadruple representing the cast
148         * @throws IntermediateCodeGeneratorException
149         *             Cast not possible
150         */
151        public static Quadruple castToString(Type type, String value, String tmp)
152                        throws IntermediateCodeGeneratorException {
153                switch (type.getKind()) {
154                case BOOLEAN:
155                        if (value.equals("#TRUE")) {
156                                return new QuadrupleImpl(Operator.ASSIGN_STRING, "#\"true\"", tmp);
157                        }
158                        if (value.equals("#FALSE")) {
159                                return new QuadrupleImpl(Operator.ASSIGN_STRING, "#\"false\"", tmp);
160                        }
161                        return new QuadrupleImpl(Operator.BOOLEAN_TO_STRING, value, tmp);
162                case DOUBLE:
163                        if (value.startsWith("#")) {
164                                return new QuadrupleImpl(Operator.ASSIGN_STRING, "#\"" + value.substring(1) + "\"", tmp);
165                        }
166                        return new QuadrupleImpl(Operator.DOUBLE_TO_STRING, value, tmp);
167                case LONG:
168                        if (value.startsWith("#")) {
169                                return new QuadrupleImpl(Operator.ASSIGN_STRING, "#\"" + value.substring(1) + "\"", tmp);
170                        }
171                        return new QuadrupleImpl(Operator.LONG_TO_STRING, value, tmp);
172                default:
173                        break;
174                }
175                String error = "can not cast from " + type.toString() + " to string";
176                CastingFactory.logger.fatal(error);
177                throw new IntermediateCodeGeneratorException(error);
178        }
179
180}
