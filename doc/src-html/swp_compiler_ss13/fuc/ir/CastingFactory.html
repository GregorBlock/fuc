001package swp_compiler_ss13.fuc.ir;
002
003import org.apache.log4j.Logger;
004
005import swp_compiler_ss13.common.backend.Quadruple;
006import swp_compiler_ss13.common.ir.IntermediateCodeGeneratorException;
007import swp_compiler_ss13.common.types.Type;
008import swp_compiler_ss13.common.types.Type.Kind;
009import swp_compiler_ss13.common.types.primitive.PrimitiveType;
010
011/**
012 * This factory adds castings where needed to the IR Code. Castings can be
013 * disabled by setting the property -Dfuc.irgen.castings=false
014 * 
015 * @author "Frank Zechert"
016 * @version 2
017 */
018public class CastingFactory {
019
020        /**
021         * Indicates whether casting should be enabled or disabled. By default
022         * casting is enabled. To disable casting set -Dfuc.irgen.castings=false
023         */
024        static final boolean castingEnabled = Boolean.parseBoolean(System.getProperty("fuc.irgen.castings", "true"));
025
026        /**
027         * The logger
028         */
029        static final Logger logger = Logger.getLogger(CastingFactory.class);
030
031        /**
032         * Check if a cast is needed between type1 and type2
033         * 
034         * @param type1
035         *            Type 1 to check
036         * @param type2
037         *            Type 2 to check
038         * @return true if a cast is needed between type1 and type2
039         * @throws IntermediateCodeGeneratorException
040         *             An error occurred
041         */
042        public static boolean isCastNeeded(Type type1, Type type2) throws IntermediateCodeGeneratorException {
043                if (!castingEnabled) {
044                        return false;
045                }
046                if (isPrimitive(type1) && isPrimitive(type2)) {
047                        return type1.getKind() != type2.getKind();
048                }
049
050                String err = "CastingFactory.isCastNeeded is not implemented for non-primitive types. Types given: %s and %s";
051                String errf = String.format(err, type1.toString(), type2.toString());
052                logger.fatal(errf);
053                throw new IntermediateCodeGeneratorException(errf, new UnsupportedOperationException());
054        }
055
056        /**
057         * Check if a cast is needed between the types of result1 and result2
058         * 
059         * @param result1
060         *            Result 1 to check
061         * @param result2
062         *            Result 2 to check
063         * @return true if a cast is needed between the types of result1 and result2
064         * @throws IntermediateCodeGeneratorException
065         *             An error occurred
066         */
067        public static boolean isCastNeeded(IntermediateResult result1, IntermediateResult result2)
068                        throws IntermediateCodeGeneratorException {
069                return isCastNeeded(result1.getType(), result2.getType());
070        }
071
072        /**
073         * Checks whether the given type is a primitive type.
074         * 
075         * @param type
076         *            The type to check
077         * @return True if the given type is a primitive type (Boolean, Double,
078         *         Long, String).
079         */
080        public static boolean isPrimitive(Type type) {
081                return (type instanceof PrimitiveType);
082        }
083
084        /**
085         * Create and return a quadruple to cast fromId of type fromType to toId of
086         * type toType
087         * 
088         * @param fromType
089         *            The type to cast from
090         * @param fromId
091         *            The id to cast from
092         * @param toType
093         *            The type to cast to
094         * @param toId
095         *            The id to cast to
096         * @return The Quadruple that represents the cast.
097         * @throws IntermediateCodeGeneratorException
098         *             Something went wrong
099         */
100        public static Quadruple createCast(Type fromType, String fromId, Type toType, String toId)
101                        throws IntermediateCodeGeneratorException {
102                switch (fromType.getKind()) {
103                case DOUBLE:
104                        switch (toType.getKind()) {
105                        case LONG:
106                                return new QuadrupleImpl(Quadruple.Operator.DOUBLE_TO_LONG, fromId, Quadruple.EmptyArgument, toId);
107                        default:
108                                break;
109                        }
110                case LONG:
111                        switch (toType.getKind()) {
112                        case DOUBLE:
113                                return new QuadrupleImpl(Quadruple.Operator.LONG_TO_DOUBLE, fromId, Quadruple.EmptyArgument, toId);
114                        default:
115                                break;
116                        }
117                default:
118                        break;
119                }
120
121                String err = "Can not create a cast from %s of type %s to %s of type %s.";
122                String errf = String.format(err, fromId, fromType, toId, toType);
123                throw new IntermediateCodeGeneratorException(errf, new UnsupportedOperationException());
124        }
125
126        /**
127         * Check if the given result is of a numeric type (double or long)
128         * 
129         * @param result
130         *            the result to check
131         * @return true if the type of the result is double or long
132         */
133        public static boolean isNumeric(IntermediateResult result) {
134                return isNumeric(result.getType());
135        }
136
137        /**
138         * Check if the given type is numeric (double or long)
139         * 
140         * @param type
141         *            the type to check
142         * @return true if the type is numeric
143         */
144        private static boolean isNumeric(Type type) {
145                if (type.getKind() == Kind.LONG) {
146                        return true;
147                }
148                if (type.getKind() == Kind.DOUBLE) {
149                        return true;
150                }
151                return false;
152        }
153}
