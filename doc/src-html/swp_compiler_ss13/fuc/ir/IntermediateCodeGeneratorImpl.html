113        @Override
114        public List<Quadruple> generateIntermediateCode(AST ast)
115                        throws IntermediateCodeGeneratorException {
116                if (ast == null) {
117                        logger.fatal("The argument ast in generateIntermediateCode(AST ast) can not be null!");
118                        throw new IntermediateCodeGeneratorException(
119                                        "The argument ast in generateIntermediateCode(AST ast) can not be null!");
120                }
121
122                ASTNode program = ast.getRootNode();
123                if (program == null) {
124                        logger.fatal("The ast given in generateIntermediateCode(AST ast) must have a root node!");
125                        throw new IntermediateCodeGeneratorException(
126                                        "The ast given in generateIntermediateCode(AST ast) must have a root node");
127                }
128                this.reset();
129                this.callProcessing(program);
130                return this.irCode;
131        }

133        /**
134         * call the method that handles the node in the AST
135         * 
136         * @param node
137         *            The node to handle
138         * @throws IntermediateCodeGeneratorException
139         *             An error occurred
140         */
141        private void callProcessing(ASTNode node)
142                        throws IntermediateCodeGeneratorException {
143                logger.debug("Processing next node: " + node.getNodeType().toString()
144                                + " with " + (node.getNumberOfNodes() - 1) + " subnodes");
145                switch (node.getNodeType()) {
146                case ArithmeticBinaryExpressionNode:
147                        this.processArithmeticBinaryExpressionNode((ArithmeticBinaryExpressionNode) node);
148                        break;
149                case ArithmeticUnaryExpressionNode:
150                        this.processArithmeticUnaryExpressionNode((ArithmeticUnaryExpressionNode) node);
151                        break;
152                case ArrayIdentifierNode:
153                        this.processArrayIdentifierNode((ArrayIdentifierNode) node);
154                        break;
155                case AssignmentNode:
156                        this.processAssignmentNode((AssignmentNode) node);
157                        break;
158                case BasicIdentifierNode:
159                        this.processBasicIdentifierNode((BasicIdentifierNode) node);
160                        break;
161                case BlockNode:
162                        this.processBlockNode((BlockNode) node);
163                        break;
164                case BranchNode:
165                        this.processBranchNode((BranchNode) node);
166                        break;
167                case BreakNode:
168                        this.processBreakNode((BreakNode) node);
169                        break;
170                case DeclarationNode:
171                        this.processDeclarationNode((DeclarationNode) node);
172                        break;
173                case DoWhileNode:
174                        this.processDoWhileNode((DoWhileNode) node);
175                        break;
176                case LiteralNode:
177                        this.processLiteralNode((LiteralNode) node);
178                        break;
179                case LogicBinaryExpressionNode:
180                        this.processLogicBinaryExpressionNode((LogicBinaryExpressionNode) node);
181                        break;
182                case LogicUnaryExpressionNode:
183                        this.processLogicUnaryExpressionNode((LogicUnaryExpressionNode) node);
184                        break;
185                case PrintNode:
186                        this.processPrintNode((PrintNode) node);
187                        break;
188                case ReturnNode:
189                        this.processReturnNode((ReturnNode) node);
190                        break;
191                case StructIdentifierNode:
192                        this.processStructIdentifierNode((StructIdentifierNode) node);
193                        break;
194                case WhileNode:
195                        this.processWhileNode((WhileNode) node);
196                        break;
197                case RelationExpressionNode:
198                        this.processRelationExpressionNode((RelationExpressionNode) node);
199                        break;
200                default:
201                        throw new IntermediateCodeGeneratorException("Unknown node type: "
202                                        + node.getNodeType().toString());
203                }
204        }

206        /**
207         * Process a relationexpression node
208         * 
209         * @param node
210         *            the node to process
211         * @throws IntermediateCodeGeneratorException
212         *             Something went wrong
213         */
214        private void processRelationExpressionNode(RelationExpressionNode node)
215                        throws IntermediateCodeGeneratorException {
216                BinaryOperator operator = node.getOperator();
217
218                ExpressionNode leftNode = node.getLeftValue();
219                ExpressionNode rightNode = node.getRightValue();
220
221                this.callProcessing(leftNode);
222                this.callProcessing(rightNode);
223
224                IntermediateResult rightResult = this.intermediateResults.pop();
225                IntermediateResult leftResult = this.intermediateResults.pop();
226
227                boolean castNeeded = CastingFactory.isCastNeeded(leftResult,
228                                rightResult);
229                String castedleft = leftResult.getValue();
230                String castedright = rightResult.getValue();
231                Type type = leftResult.getType();
232                if (castNeeded) {
233                        if (CastingFactory.isNumeric(leftResult)
234                                        && CastingFactory.isNumeric(rightResult)) {
235                                type = new DoubleType();
236                                if (leftResult.getType().getKind() == Kind.LONG) {
237                                        castedleft = this.createAndSaveTemporaryIdentifier(type);
238                                        this.irCode.add(CastingFactory.createCast(leftResult.getType(),
239                                                        leftResult.getValue(), type, castedleft));
240                                }
241                                if (rightResult.getType().getKind() == Kind.LONG) {
242                                        this.irCode.add(CastingFactory.createCast(rightResult.getType(),
243                                                        rightResult.getValue(), type, castedright));
244                                }
245                        } else {
246                                String err = String
247                                                .format("unsupported types %s %s and %s %s for relation expression",
248                                                                leftResult.getType(), leftResult.getValue(),
249                                                                rightResult.getType(), rightResult.getValue());
250                                logger.fatal(err);
251                                throw new IntermediateCodeGeneratorException(err);
252                        }
253                }
254
255                String result = this.createAndSaveTemporaryIdentifier(new BooleanType());
256
257                switch (operator) {
258                case EQUAL:
259                        this.irCode.add(QuadrupleFactory.relationEqual(castedleft, castedright,
260                                        result, type));
261                        break;
262                case GREATERTHAN:
263                        this.irCode.add(QuadrupleFactory.relationGreater(castedleft,
264                                        castedright, result, type));
265                        break;
266                case GREATERTHANEQUAL:
267                        this.irCode.add(QuadrupleFactory.relationGreaterEqual(castedleft,
268                                        castedright, result, type));
269                        break;
270                case INEQUAL:
271                        String tmp = this.createAndSaveTemporaryIdentifier(new BooleanType());
272                        this.irCode.add(QuadrupleFactory.relationEqual(castedleft, castedright,
273                                        tmp, type));
274                        this.irCode.add(QuadrupleFactory.booleanArithmetic(
275                                        UnaryOperator.LOGICAL_NEGATE, tmp, result));
276                        break;
277                case LESSTHAN:
278                        this.irCode.add(QuadrupleFactory.relationLess(castedleft, castedright,
279                                        result, type));
280                        break;
281                case LESSTHANEQUAL:
282                        this.irCode.add(QuadrupleFactory.relationLessEqual(castedleft,
283                                        castedright, result, type));
284                        break;
285                default:
286                        String err = "BinaryOperator %s is not allowed within a RelationExpressionNode";
287                        String errf = String.format(err, operator);
288                        logger.fatal(errf);
289                        throw new IntermediateCodeGeneratorException(errf);
290                }
291
292                this.intermediateResults.push(new IntermediateResult(result,
293                                new BooleanType()));
294        }

296        /**
297         * Process a block node and generate the needed IR Code
298         * 
299         * @param node
300         *            The block node to process
301         * @throws IntermediateCodeGeneratorException
302         *             an error occurred while process the node
303         */
304        private void processBlockNode(BlockNode node)
305                        throws IntermediateCodeGeneratorException {
306                // push current symbol table
307                this.currentSymbolTable.push(node.getSymbolTable());
308
309                // get declarations
310                Iterator<DeclarationNode> declIterator = node.getDeclarationIterator();
311                while (declIterator.hasNext()) {
312                        this.callProcessing(declIterator.next());
313                }
314
315                // get statements
316                Iterator<StatementNode> statementIterator = node.getStatementIterator();
317                while (statementIterator.hasNext()) {
318                        StatementNode statement = statementIterator.next();
319                        this.callProcessing(statement);
320                }
321
322                // pop the symbol scope
323                this.currentSymbolTable.pop();
324        }

408                        throw new IntermediateCodeGeneratorException(err);
409                }
410
411                String temp = this.createAndSaveTemporaryIdentifier(new BooleanType());
412                this.irCode.add(QuadrupleFactory.unaryNot(result.getValue(), temp));
413                this.intermediateResults
414                                .push(new IntermediateResult(temp, new BooleanType()));

418         * Process a logicbinaryexpression node
419         * 
420         * @param node
421         *            the node to process
422         * @throws IntermediateCodeGeneratorException
423         *             Something went wrong
424         */
425        private void processLogicBinaryExpressionNode(LogicBinaryExpressionNode node)
426                        throws IntermediateCodeGeneratorException {
427                this.callProcessing(node.getLeftValue());
428                this.callProcessing(node.getRightValue());
429
430                IntermediateResult rightResult = this.intermediateResults.pop();
431                IntermediateResult leftResult = this.intermediateResults.pop();
432
433                if (leftResult.getType().getKind() != Kind.BOOLEAN
434                                || rightResult.getType().getKind() != Kind.BOOLEAN) {
435                        String err = "Binary logic operation is not supported for "
436                                        + leftResult.getType() + " and " + rightResult.getType();
437                        logger.fatal(err);
438                        throw new IntermediateCodeGeneratorException(err);
439                }
440
441                String temp = this.createAndSaveTemporaryIdentifier(new BooleanType());
442                this.irCode.add(QuadrupleFactory.booleanArithmetic(node.getOperator(),
443                                leftResult.getValue(), rightResult.getValue(), temp));
444                this.intermediateResults
445                                .push(new IntermediateResult(temp, new BooleanType()));
446        }

448        /**
449         * Process a literal node
450         * 
451         * @param node
452         *            the node to process
453         * @throws IntermediateCodeGeneratorException
454         *             Something went wrong
455         */
456        private void processLiteralNode(LiteralNode node)
457                        throws IntermediateCodeGeneratorException {
458                String literal = node.getLiteral();
459                Type type = node.getLiteralType();
460                switch (type.getKind()) {
461                case DOUBLE:
462                case LONG:
463                case BOOLEAN:
464                        // Literal of type Long or Double needs to start with a # to mark it
465                        // as a constant
466                        this.intermediateResults.push(new IntermediateResult("#"
467                                        + literal.toUpperCase(), type));
468                        break;
469                case STRING:
470                        // Literal of type String needs to be in " and start with a #
471                        // If the lexer already gives a string in " and " do not add them
472                        if (literal.startsWith("\"") && literal.endsWith("\"")
473                                        && !literal.endsWith("\\\"")) {
474                                literal = literal.substring(1, literal.length() - 1);
475                        }
476                        // Make C-Style escapings, only if they are not escaped already
477                        literal = this.escapeString(literal, "\"", "\\\"");
478                        literal = this.escapeString(literal, "\n", "\\n");
479                        literal = this.escapeString(literal, "\r", "\\r");
480                        literal = this.escapeString(literal, "\t", "\\t");
481                        literal = this.escapeString(literal, "\0", "\\0");
482                        this.intermediateResults.push(new IntermediateResult("#\"" + literal
483                                        + "\"", type));
484                        break;
485                default:
486                        // Literal of other types are not defined yet.
487                        throw new IntermediateCodeGeneratorException(
488                                        "Literal node of type " + node.getLiteralType().toString()
489                                                        + " is not supported");
490                }
491        }

493        /**
494         * Escape special char sequences but ignore them if they are already escaped
495         * 
496         * @param literal
497         *            The string to escape
498         * @param search
499         *            The char sequence to replace
500         * @param replace
501         *            The char sequence to use as replacement
502         * @return The escaped string
503         */
504        private String escapeString(String literal, String search, String replace) {
505                int fromIndex = 0;
506                int pos = 0;
507                while ((pos = literal.indexOf(search, fromIndex)) >= 0) {
508                        if (pos > 0 && literal.charAt(pos - 1) == '\\') {
509                                fromIndex = pos + 1;
510                                continue;
511                        }
512                        fromIndex = pos + replace.length();
513                        if (pos > 0) {
514                                literal = literal.substring(0, pos) + replace
515                                                + literal.substring(pos + 1);
516                        } else {
517                                literal = replace + literal.substring(pos + 1);
518                        }
519                }
520                return literal;
521        }

567        /**
568         * Process a branch node
569         * 
570         * @param node
571         *            the node to process
572         * @throws IntermediateCodeGeneratorException
573         *             Something went wrong
574         */
575        private void processBranchNode(BranchNode node)
576                        throws IntermediateCodeGeneratorException {
577                ExpressionNode condition = node.getCondition();
578                StatementNode onTrue = node.getStatementNodeOnTrue();
579                StatementNode onFalse = node.getStatementNodeOnFalse();
580
581                this.callProcessing(condition);
582                IntermediateResult conditionResult = this.intermediateResults.pop();
583                if (conditionResult.getType().getKind() != Kind.BOOLEAN) {
584                        String err = "A condition is not of type boolean but of unsupported type "
585                                        + conditionResult.getType();
586                        logger.fatal(err);
587                        throw new IntermediateCodeGeneratorException(err);
588                }
589
590                String trueLabel = this.createNewLabel();
591                String falseLabel = this.createNewLabel();
592                String endLabel = this.createNewLabel();
593
594                this.irCode.add(QuadrupleFactory.branch(conditionResult.getValue(),
595                                trueLabel, falseLabel));
596                this.irCode.add(QuadrupleFactory.label(trueLabel));
597                this.callProcessing(onTrue);
598                this.irCode.add(QuadrupleFactory.jump(endLabel));
599                this.irCode.add(QuadrupleFactory.label(falseLabel));
600
601                if (onFalse != null) {
602                        this.callProcessing(onFalse);
603                }
604
605                this.irCode.add(QuadrupleFactory.label(endLabel));
606
607        }

609        /**
610         * Process a basicidentifier node
611         * 
612         * @param node
613         *            the node to process
614         * @throws IntermediateCodeGeneratorException
615         *             Something went wrong
616         */
617        private void processBasicIdentifierNode(BasicIdentifierNode node)
618                        throws IntermediateCodeGeneratorException {
619                // A basic identifier can be pushed to the stack of results immediately
620                String identifier = node.getIdentifier();
621                Type identifierType = this.currentSymbolTable.peek().lookupType(identifier);
622
623                // arrays get assigned with their full type but we want the base type in
624                // the center of it
625                while (identifierType instanceof ArrayType) {
626                        identifierType = ((ArrayType) identifierType).getInnerType();
627                }
628
629                String actualIdentifier = this.loadIdentifier(identifier);
630                this.intermediateResults.push(new IntermediateResult(actualIdentifier,
631                                identifierType));
632        }

634        /**
635         * Process a assignment node
636         * 
637         * @param node
638         *            the node to process
639         * @throws IntermediateCodeGeneratorException
640         *             Something went wrong
641         */
642        private void processAssignmentNode(AssignmentNode node)
643                        throws IntermediateCodeGeneratorException {
644                // the id to assign the value to
645                IdentifierNode id = node.getLeftValue();
646
647                this.arrayAssignment = true;
648                this.arrayLevel = null;
649                this.callProcessing(id);
650                Integer toArrayIndex = this.arrayLevel;
651                this.arrayLevel = null;
652
653                IntermediateResult toIntermediate = this.intermediateResults.pop();
654
655                // the id to assign the value to is a basic identifier.
656                // no need to resolve an array or struct etc.
657                StatementNode value = node.getRightValue();
658
659                // process the right hand expression
660                this.callProcessing(value);
661                Integer fromArrayIndex = this.arrayLevel;
662                this.arrayLevel = null;
663                this.arrayAssignment = false;
664
665                // the result of the right hand expression
666                IntermediateResult fromIntermediate = this.intermediateResults.pop();
667
668                // get the name of the id and resolve it from our saved structures
669                String toId = toIntermediate.getValue();
670
671                // the type of the id. If necessary the value of the right hand
672                // expression
673                // needs to be casted to this type.
674                Type toType = toIntermediate.getType();
675
676                // check if the cast is needed
677                boolean castNeeded = CastingFactory.isCastNeeded(toType,
678                                fromIntermediate.getType());
679
680                // get the name and id of the source
681                String fromId = fromIntermediate.getValue();
682                Type fromType = fromIntermediate.getType();
683
684                if (fromArrayIndex != null && toArrayIndex != null) {
685                        // assign array to array
686                        String temporary = this.createAndSaveTemporaryIdentifier(fromType);
687                        this.irCode.addAll(QuadrupleFactory.assign(fromType, fromId, temporary,
688                                        fromArrayIndex, null));
689                        if (castNeeded) {
690                                String temporary2 = this.createAndSaveTemporaryIdentifier(toType);
691                                this.irCode.add(CastingFactory.createCast(fromType, temporary,
692                                                toType, temporary2));
693                                this.irCode.addAll(QuadrupleFactory.assign(toType, temporary2, toId,
694                                                null, toArrayIndex));
695                                toId = temporary2;
696                        } else {
697                                this.irCode.addAll(QuadrupleFactory.assign(toType, temporary, toId,
698                                                null, toArrayIndex));
699                                toId = temporary;
700                        }
701                } else {
702                        if (castNeeded) {
703                                if (toArrayIndex == null) {
704                                        // assign array to variable
705                                        String temporary = this.createAndSaveTemporaryIdentifier(fromType);
706                                        this.irCode.addAll(QuadrupleFactory.assign(fromType, fromId,
707                                                        temporary, fromArrayIndex, toArrayIndex));
708                                        this.irCode.add(CastingFactory.createCast(fromType, temporary,
709                                                        toType, toId));
710                                        toId = temporary;
711                                } else {
712                                        // assign variable to array
713                                        String temporary = this.createAndSaveTemporaryIdentifier(toType);
714                                        this.irCode.add(CastingFactory.createCast(fromType, fromId,
715                                                        toType, temporary));
716                                        this.irCode.addAll(QuadrupleFactory.assign(toType, temporary,
717                                                        toId, fromArrayIndex, toArrayIndex));
718                                        toId = temporary;
719                                }
720                        } else {
721                                this.irCode.addAll(QuadrupleFactory.assign(toType, fromId, toId,
722                                                fromArrayIndex, toArrayIndex));
723                        }
724                }
725                this.intermediateResults.push(new IntermediateResult(toId, toType));
726        }

728        /**
729         * Process a arrayidentifier node
730         * 
731         * @param node
732         *            the node to process
733         * @throws IntermediateCodeGeneratorException
734         *             Something went wrong
735         */
736        private void processArrayIdentifierNode(ArrayIdentifierNode node)
737                        throws IntermediateCodeGeneratorException {
738
739                IdentifierNode innerNode = node.getIdentifierNode();
740
741                boolean outerArray = false;
742                if (this.arrayLevel == null) {
743                        this.arrayLevel = node.getIndex();
744                        outerArray = true;
745                }
746
747                this.callProcessing(innerNode);
748
749                // array assignments are special so don't push the reference of the
750                // outermost array
751                if (!this.arrayAssignment || !outerArray) {
752                        IntermediateResult innerResult = this.intermediateResults.pop();
753                        Type nodeType = innerResult.getType();
754                        String tmpIdentifier;
755
756                        if (outerArray) {
757                                tmpIdentifier = this.createAndSaveTemporaryIdentifier(nodeType);
758                        } else {
759                                tmpIdentifier = this.createAndSaveTemporaryIdentifierReference(nodeType);
760                        }
761                        this.currentSymbolTable.peek().lookupType(tmpIdentifier);
762
763                        // FIXME just looks at the outermost type so this works
764                        Type arrayType = new ArrayType(new StringType(0L), 0);
765                        if (outerArray) {
766                                arrayType = nodeType;
767                        }
768                        this.irCode.add(QuadrupleFactory.arrayGet(arrayType,
769                                        innerResult.getValue(), "#" + node.getIndex(),
770                                        tmpIdentifier));
771                        this.intermediateResults.push(new IntermediateResult(tmpIdentifier,
772                                        innerResult.getType()));
773                }
774        }

776        /**
777         * Process a arithmeticunaryexpression node
778         * 
779         * @param node
780         *            the node to process
781         * @throws IntermediateCodeGeneratorException
782         *             Something went wrong
783         */
784        private void processArithmeticUnaryExpressionNode(
785                        ArithmeticUnaryExpressionNode node)
786                        throws IntermediateCodeGeneratorException {
787                ExpressionNode rightNode = node.getRightValue();
788
789                // process the right hand value
790                this.callProcessing(rightNode);
791
792                IntermediateResult rightIntermediate = this.intermediateResults.pop();
793
794                String temp = this.createAndSaveTemporaryIdentifier(rightIntermediate
795                                .getType());
796                this.irCode.add(QuadrupleFactory.unaryMinus(rightIntermediate.getType(),
797                                rightIntermediate.getValue(), temp));
798                this.intermediateResults.push(new IntermediateResult(temp, rightIntermediate
799                                .getType()));
800        }

802        /**
803         * Binary Arithmetic Operation
804         * 
805         * @param node
806         *            the arithmetic binary expression node
807         * @throws IntermediateCodeGeneratorException
808         *             something went wrong
809         */
810        private void processArithmeticBinaryExpressionNode(
811                        ArithmeticBinaryExpressionNode node)
812                        throws IntermediateCodeGeneratorException {
813
814                // process the left and right value first
815                this.callProcessing(node.getLeftValue());
816                this.callProcessing(node.getRightValue());
817
818                // get the left and right value
819                IntermediateResult right = this.intermediateResults.pop();
820                IntermediateResult left = this.intermediateResults.pop();
821
822                // check if a cast is needed
823                boolean castNeeded = CastingFactory.isCastNeeded(right, left);
824
825                if (!castNeeded) {
826                        // no cast is needed
827                        // either both values are of type LONG or of type DOUBLE
828                        if (left.getType() instanceof LongType) {
829                                // both values are of type LONG
830                                String temp = this.createAndSaveTemporaryIdentifier(new LongType());
831                                this.irCode.add(QuadrupleFactory.longArithmeticBinaryOperation(
832                                                node.getOperator(), left.getValue(), right.getValue(),
833                                                temp));
834
835                                this.intermediateResults.push(new IntermediateResult(temp,
836                                                new LongType()));
837                        } else if (left.getType() instanceof DoubleType) {
838                                // both values are of type DOUBLE
839                                String temp = this.createAndSaveTemporaryIdentifier(new DoubleType());
840                                this.irCode.add(QuadrupleFactory.doubleArithmeticBinaryOperation(
841                                                node.getOperator(), left.getValue(), right.getValue(),
842                                                temp));
843
844                                this.intermediateResults.push(new IntermediateResult(temp,
845                                                new DoubleType()));
846                        } else {
847                                // this is an unsupported combination of types
848                                String err = "Arithmetic Binary Expression with arguments of types %s and %s is not supported";
849                                String errf = String.format(err, left.getType().toString(),
850                                                right.getType().toString());
851                                logger.fatal(errf);
852                                throw new IntermediateCodeGeneratorException(errf,
853                                                new UnsupportedOperationException());
854                        }
855                } else {
856                        // A cast is needed.
857                        // Only LONG and DOUBLE types are valid for arithmetic operations
858                        // We will always cast to type DOUBLE as it is more precise than
859                        // LONG
860                        // and nothing will be lost during the conversion
861                        if (left.getType() instanceof LongType) {
862                                // the left value is of type LONG, so it needs to be casted
863                                String temp = this.createAndSaveTemporaryIdentifier(new DoubleType());
864                                String temp2 = this.createAndSaveTemporaryIdentifier(new DoubleType());
865                                this.irCode.add(CastingFactory.createCast(left.getType(),
866                                                left.getValue(), new DoubleType(), temp));
867                                this.irCode.add(QuadrupleFactory.doubleArithmeticBinaryOperation(
868                                                node.getOperator(), temp, right.getValue(), temp2));
869
870                                this.intermediateResults.push(new IntermediateResult(temp2,
871                                                new DoubleType()));
872                        } else if (right.getType() instanceof LongType) {
873                                // the right value is of type LONG, so it needs to be casted
874                                String temp = this.createAndSaveTemporaryIdentifier(new DoubleType());
875                                String temp2 = this.createAndSaveTemporaryIdentifier(new DoubleType());
876                                this.irCode.add(CastingFactory.createCast(right.getType(),
877                                                right.getValue(), new DoubleType(), temp));
878                                this.irCode.add(QuadrupleFactory.doubleArithmeticBinaryOperation(
879                                                node.getOperator(), left.getValue(), temp, temp2));
880
881                                this.intermediateResults.push(new IntermediateResult(temp2,
882                                                new DoubleType()));
883                        } else {
884                                // this combinations of types is not supported
885                                String err = "Arithmetic Binary Expression with arguments of types %s and %s is not supported";
886                                String errf = String.format(err, left.getType().toString(),
887                                                right.getType().toString());
888                                logger.fatal(errf);
889                                throw new IntermediateCodeGeneratorException(errf,
890                                                new UnsupportedOperationException());
891                        }
892                }
893
894        }

896        /**
897         * Save the given identifier to the interal store of variables
898         * 
899         * @param identifier
900         *            The name of the variable
901         * @param type
902         *            the type of the variable
903         * @return The renamed name of the variable that was stored
904         * @throws IntermediateCodeGeneratorException
905         *             an exception occurred
906         */
907        private String saveIdentifier(String identifier, Type type)
908                        throws IntermediateCodeGeneratorException {
909                if (!this.usedNames.contains(identifier)) {
910                        // an identifier with this name was not yet used.
911                        // it does not need to be renamed to stick to SSA
912                        this.usedNames.add(identifier);
913                        this.irCode.addAll(QuadrupleFactory.declaration(identifier, type));
914                        return identifier;
915                } else {
916                        // rename is required to keep single static assignment
917                        String newName = this.currentSymbolTable.peek().getNextFreeTemporary();
918                        this.currentSymbolTable.peek().putTemporary(newName, type);
919                        this.usedNames.add(newName);
920                        this.currentSymbolTable.peek().setIdentifierAlias(identifier, newName);
921                        this.irCode.addAll(QuadrupleFactory.declaration(newName, type));
922                        return newName;
923                }
924        }

926        /**
927         * Create a new temporary value and save it to the internal store of
928         * variables
929         * 
930         * @param type
931         *            The type of the new variable
932         * @return The name of the new variable
933         * @throws IntermediateCodeGeneratorException
934         *             An error occurred
935         */
936        private String createAndSaveTemporaryIdentifier(Type type)
937                        throws IntermediateCodeGeneratorException {
938                // Condition: getNextFreeTemporary never returns a name that was already
939                // used in the IR until now
940                // (globally unique names)
941                String id = this.currentSymbolTable.peek().getNextFreeTemporary();
942                this.currentSymbolTable.peek().putTemporary(id, type);
943                this.usedNames.add(id);
944                this.irCode.addAll(QuadrupleFactory.declaration(id, type));
945                return id;
946        }

949         * Create a new temporary variable reference and save it to the internal
950         * store of variables
951         * 
952         * @param type
953         *            The type of the new variable reference
954         * @return The name of the new variable
955         * @throws IntermediateCodeGeneratorException
956         *             An error occurred
957         */
958        private String createAndSaveTemporaryIdentifierReference(Type type)
959                        throws IntermediateCodeGeneratorException {
960                String id = this.currentSymbolTable.peek().getNextFreeTemporary();

962                this.usedNames.add(id);
963                this.irCode.add(new QuadrupleImpl(Operator.DECLARE_REFERENCE,
964                                Quadruple.EmptyArgument, Quadruple.EmptyArgument, id));
965                return id;

969         * Load the given identifier and return its actual name (if renaming was
970         * done)
971         * 
972         * @param id
973         *            The identifier name to load
974         * @return The actual name of the identifier
975         * @throws IntermediateCodeGeneratorException
976         *             Identifier was not found
977         */
978        private String loadIdentifier(String id)
979                        throws IntermediateCodeGeneratorException {
980                String name = this.currentSymbolTable.peek().getIdentifierAlias(id);
981                if (name == null) {
982                        logger.fatal("Undeclared variable found: " + id);
983                        throw new IntermediateCodeGeneratorException(
984                                        "Undeclared variable found: " + id);
985                }
986                return name;
987        }
