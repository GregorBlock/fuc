001package swp_compiler_ss13.fuc.ir.components;
002
003import swp_compiler_ss13.common.ast.nodes.unary.DeclarationNode;
004import swp_compiler_ss13.common.ir.IntermediateCodeGeneratorException;
005import swp_compiler_ss13.common.types.Type;
006import swp_compiler_ss13.fuc.ir.GeneratorExecutor;
007import swp_compiler_ss13.fuc.ir.GeneratorState;
008import swp_compiler_ss13.fuc.ir.QuadrupleFactory;
009
010/**
011 * Processor for a declaration node
012 * 
013 * @author "Frank Zechert"
014 * @author "Danny Maasch"
015 * @author kaworu
016 * @version 3
017 */
018public class DeclarationNodeProcessor extends NodeProcessor {
019
020        /**
021         * Create a new instance of the declaration node processor.
022         * 
023         * @param state
024         *            the state of the intermediate code generator
025         * @param executor
026         *            the intermediate code generator executor
027         */
028        public DeclarationNodeProcessor(GeneratorState state, GeneratorExecutor executor) {
029                super(state, executor);
030        }
031
032        /**
033         * Process the declaration node.
034         * 
035         * @param declaration
036         *            the declaration node to process.
037         * @throws IntermediateCodeGeneratorException
038         *             An error occurred while processing the node.
039         */
040        public void processDeclarationNode(DeclarationNode declaration)
041                        throws IntermediateCodeGeneratorException {
042                String declaredName = declaration.getIdentifier();
043                Type declaredType = declaration.getType();
044
045                // Ensure ssa (single static assignment) by avoiding
046                // all identifier names that have been used already
047                String ssaName = this.state.saveIdentifier(declaredName, declaredType);
048
049                this.state.addIntermediateCode(QuadrupleFactory.declare(ssaName, declaredType));
050        }
051}
