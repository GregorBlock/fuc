001package swp_compiler_ss13.fuc.ir.components.statements;
002
003import swp_compiler_ss13.common.ast.nodes.IdentifierNode;
004import swp_compiler_ss13.common.ast.nodes.unary.PrintNode;
005import swp_compiler_ss13.common.ast.nodes.unary.ReturnNode;
006import swp_compiler_ss13.common.ir.IntermediateCodeGeneratorException;
007import swp_compiler_ss13.common.types.Type.Kind;
008import swp_compiler_ss13.common.types.primitive.LongType;
009import swp_compiler_ss13.common.types.primitive.StringType;
010import swp_compiler_ss13.fuc.ir.CastingFactory;
011import swp_compiler_ss13.fuc.ir.GeneratorExecutor;
012import swp_compiler_ss13.fuc.ir.GeneratorState;
013import swp_compiler_ss13.fuc.ir.QuadrupleFactory;
014import swp_compiler_ss13.fuc.ir.components.NodeProcessor;
015import swp_compiler_ss13.fuc.ir.data.IntermediateResult;
016
017/**
018 * Processor for an output node
019 * 
020 * @author "Frank Zechert"
021 * @author "Danny Maasch"
022 * @author kaworu
023 * @version 3
024 */
025public class OutputNodeProcessor extends NodeProcessor {
026
027        /**
028         * Create a new instance of the output node processor.
029         * 
030         * @param state
031         *            the state of the intermediate code generator
032         * @param executor
033         *            the intermediate code generator executor
034         */
035        public OutputNodeProcessor(GeneratorState state, GeneratorExecutor executor) {
036                super(state, executor);
037        }
038
039        /**
040         * Process the print node.
041         * 
042         * @param node
043         *            the print node to process.
044         * @throws IntermediateCodeGeneratorException
045         *             An error occurred while processing the node.
046         */
047        public void processPrintNode(PrintNode node) throws IntermediateCodeGeneratorException {
048                IntermediateResult result = this.executor.process(node.getRightValue());
049
050                String variable = result.getValue();
051                result.getType();
052
053                if (result.getType().getKind() != Kind.STRING) {
054                        // if variable is not of type string cast it to string!
055                        String tmp = this.state.nextTemporaryIdentifier(new StringType(255L));
056                        this.state.addIntermediateCode(QuadrupleFactory.declare(tmp, new StringType(255L)));
057
058                        this.state.addIntermediateCode(CastingFactory.castToString(result.getType(),
059                                        result.getValue(), tmp));
060                        variable = tmp;
061                        new StringType(255L);
062                }
063
064                this.state.addIntermediateCode(QuadrupleFactory.print(variable));
065        }
066
067        /**
068         * Process the return node.
069         * 
070         * @param node
071         *            the return node to process.
072         * @throws IntermediateCodeGeneratorException
073         *             An error occurred while processing the node.
074         */
075        public void processReturnNode(ReturnNode node) throws IntermediateCodeGeneratorException {
076                IdentifierNode right = node.getRightValue();
077                if (right != null) {
078                        IntermediateResult intermediateResult = this.executor.process(right);
079                        if (intermediateResult.getType().getKind() == Kind.LONG) {
080                                // long can be used instantly
081                                this.state.addIntermediateCode(QuadrupleFactory.exit(intermediateResult.getValue()));
082                        }
083                        else if (intermediateResult.getType().getKind() == Kind.DOUBLE) {
084                                // double needs cast to long
085                                logger.warn("Return value needed a cast from double to long. Precision may be lost.");
086                                String tmp = this.state.nextTemporaryIdentifier(new LongType());
087                                this.state.addIntermediateCode(QuadrupleFactory.declare(tmp, new LongType()));
088                                this.state.addIntermediateCode(CastingFactory.doubleToLong(intermediateResult.getValue(), tmp));
089                                this.state.addIntermediateCode(QuadrupleFactory.exit(tmp));
090                        }
091                        else {
092                                // other types can not be return
093                                String err = "A value of type " + intermediateResult.getType().toString()
094                                                + " can not be used as a return value!";
095                                logger.fatal(err);
096                                throw new IntermediateCodeGeneratorException(err);
097                        }
098                }
099                else {
100                        logger.warn("Return value was not specified. Using 0 as default return value.");
101                        this.state.addIntermediateCode(QuadrupleFactory.exit("#0"));
102                }
103        }
104}
