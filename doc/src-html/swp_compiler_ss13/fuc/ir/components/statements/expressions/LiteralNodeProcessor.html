001package swp_compiler_ss13.fuc.ir.components.statements.expressions;
002
003import swp_compiler_ss13.common.ast.nodes.leaf.LiteralNode;
004import swp_compiler_ss13.common.ir.IntermediateCodeGeneratorException;
005import swp_compiler_ss13.common.types.Type;
006import swp_compiler_ss13.fuc.ir.GeneratorExecutor;
007import swp_compiler_ss13.fuc.ir.GeneratorState;
008import swp_compiler_ss13.fuc.ir.components.NodeProcessor;
009
010/**
011 * Processor for a literal node
012 * 
013 * @author "Frank Zechert"
014 * @author "Danny Maasch"
015 * @author kaworu
016 * @version 3
017 */
018public class LiteralNodeProcessor extends NodeProcessor {
019
020        /**
021         * Create a new instance of the literal node processor.
022         * 
023         * @param state
024         *            the state of the intermediate code generator
025         * @param executor
026         *            the intermediate code generator executor
027         */
028        public LiteralNodeProcessor(GeneratorState state, GeneratorExecutor executor) {
029                super(state, executor);
030        }
031
032        /**
033         * Process the literal node.
034         * 
035         * @param literal
036         *            the literal node to process.
037         * @throws IntermediateCodeGeneratorException
038         *             An error occurred while processing the node.
039         */
040        public void processLiteralNode(LiteralNode literal) throws IntermediateCodeGeneratorException {
041                String literalValue = literal.getLiteral();
042                Type literalType = literal.getLiteralType();
043
044                switch (literalType.getKind()) {
045                        case BOOLEAN:
046                                // convert true and false to uppercase
047                                this.state.pushIntermediateResult(literalType, "#" + literalValue.toUpperCase());
048                                break;
049                        case DOUBLE:
050                        case LONG:
051                                this.state.pushIntermediateResult(literalType, "#" + literalValue);
052                                break;
053                        case STRING:
054                                // escape the string
055                                literalValue = this.escape(literalValue);
056                                // prepend and append a quote
057                                this.state.pushIntermediateResult(literalType, "#\"" + literalValue + "\"");
058                                break;
059                        default:
060                                // literals of other types (array, struct) are not supported
061                                String err = String.format("A literal of Type %s is not supported",
062                                                literalType.toString());
063                                NodeProcessor.logger.fatal(err);
064                                throw new IntermediateCodeGeneratorException(err);
065
066                }
067        }
068
069        /**
070         * Escape a string with c-style escapes.
071         * 
072         * @param literal
073         *            The string to escape.
074         * @return The c-style escaped string.
075         */
076        private String escape(String literal) {
077                // if the string starts and end with double quotes
078                // remove the quotes. We will add them again later and
079                // do not want to have them twice.
080                // Condition:
081                // The literal starts with " and ends with "
082                // and the last " is not an escaped quote (\")
083                if (literal.startsWith("\"") && literal.endsWith("\"") && !literal.endsWith("\\\"")) {
084                        literal = literal.substring(1, literal.length() - 1);
085                }
086
087                // Make C-Style escapings, only if they are not escaped already
088                literal = this.escapeIfNeeded(literal, "\"", "\\\"");
089                literal = this.escapeIfNeeded(literal, "\n", "\\n");
090                literal = this.escapeIfNeeded(literal, "\r", "\\r");
091                literal = this.escapeIfNeeded(literal, "\t", "\\t");
092                literal = this.escapeIfNeeded(literal, "\0", "\\0");
093                return literal;
094        }
095
096        /**
097         * Replace the search string with the replace string inside the literal string. Only replace if
098         * the substring was not already replaced to prevent double replacement.
099         * 
100         * @param literal
101         *            The literal to replace substring inside.
102         * @param search
103         *            The substring to replace.
104         * @param replace
105         *            The replacement to replace matches with.
106         * @return The string where all occurrences of search are replace by replace.
107         */
108        private String escapeIfNeeded(String literal, String search, String replace) {
109                int fromIndex = 0;
110                int pos = 0;
111                while ((pos = literal.indexOf(search, fromIndex)) >= 0) {
112                        // for every occurrence of the search string
113                        // check if the occurrence is already escaped
114                        if ((pos > 0) && (literal.charAt(pos - 1) == '\\')) {
115                                // skip this occurrence as it is already escaped
116                                fromIndex = pos + 1;
117                                continue;
118                        }
119                        fromIndex = pos + replace.length();
120                        if (pos > 0) {
121                                literal = literal.substring(0, pos) + replace + literal.substring(pos + 1);
122                        }
123                        else {
124                                literal = replace + literal.substring(pos + 1);
125                        }
126                }
127                return literal;
128        }
129}
