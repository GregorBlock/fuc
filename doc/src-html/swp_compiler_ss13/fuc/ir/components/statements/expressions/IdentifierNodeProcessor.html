001package swp_compiler_ss13.fuc.ir.components.statements.expressions;
002
003import swp_compiler_ss13.common.ast.nodes.leaf.BasicIdentifierNode;
004import swp_compiler_ss13.common.ast.nodes.unary.ArrayIdentifierNode;
005import swp_compiler_ss13.common.ir.IntermediateCodeGeneratorException;
006import swp_compiler_ss13.common.types.Type;
007import swp_compiler_ss13.fuc.ir.ArrayHelper;
008import swp_compiler_ss13.fuc.ir.GeneratorExecutor;
009import swp_compiler_ss13.fuc.ir.GeneratorState;
010import swp_compiler_ss13.fuc.ir.QuadrupleFactory;
011import swp_compiler_ss13.fuc.ir.components.NodeProcessor;
012import swp_compiler_ss13.fuc.ir.data.IntermediateResult;
013
014/**
015 * Processor for a block node
016 * 
017 * @author "Frank Zechert"
018 * @author "Danny Maasch"
019 * @author kaworu
020 * @version 3
021 */
022public class IdentifierNodeProcessor extends NodeProcessor {
023
024        /**
025         * Create a new instance of the block node processor.
026         * 
027         * @param state
028         *            the state of the intermediate code generator
029         * @param executor
030         *            the intermediate code generator executor
031         */
032        public IdentifierNodeProcessor(GeneratorState state, GeneratorExecutor executor) {
033                super(state, executor);
034        }
035
036        /**
037         * Process the basic identifier node.
038         * 
039         * @param node
040         *            the basic identifier node to process.
041         * @throws IntermediateCodeGeneratorException
042         *             The given identifier is undeclared
043         */
044
045        public void processBasicIdentifier(BasicIdentifierNode node)
046                        throws IntermediateCodeGeneratorException {
047                // A basic identifier can be pushed to the stack of results immediately
048                String identifier = node.getIdentifier();
049                Type identifierType = this.state.getCurrentSymbolTable().lookupType(identifier);
050
051                // if the identifier was renamed due to single static assignment
052                // get the renamed identifier name
053                String actualIdentifier = this.state.loadIdentifier(identifier);
054                this.state.pushIntermediateResult(identifierType, actualIdentifier);
055        }
056
057        /**
058         * Process the array identifier node.
059         * 
060         * @param node
061         *            the array identifier node to process.
062         * @throws IntermediateCodeGeneratorException
063         *             The given identifier is undeclared
064         */
065
066        public void processArrayIdentifier(ArrayIdentifierNode node)
067                        throws IntermediateCodeGeneratorException {
068
069                IntermediateResult index = this.executor.process(node.getIndexNode());
070                String indexValue = index.getValue();
071
072                IntermediateResult base = this.executor.process(node.getIdentifierNode());
073                String baseValue = base.getValue();
074                Type arrayType = base.getType();
075
076                Type baseType = ArrayHelper.getBaseType(arrayType);
077                boolean isOuterMostDimension = ArrayHelper.isOuterMostDimension(node);
078                if (isOuterMostDimension) {
079                        // we are at the outermost dimension of the array
080                        boolean isTargetOfAssignment = ArrayHelper.isTargetOfAssignment(node);
081                        if (isTargetOfAssignment) {
082                                // the value of the array is not used, but a new value
083                                // is assigned to this array.
084                                // now we need to use one of the array_set_{TYPE} methods
085                                // but this needs to be done by the assignmentNodeProcessor
086                                // because we do not know the value that is assigned to us
087                                // already
088                                // we need to push the intermediate result back to the stack
089                                // of intermediate results to make it available in
090                                // assignmentNodeProcessor
091                                this.logger
092                                                .debug("Pushing back intermediate results, beacuse this array is on the target side of an assignment");
093                                this.state.pushIntermediateResult(index);
094                                this.state.pushIntermediateResult(base);
095                        }
096                        else {
097                                // the value of the array is used somewhere
098                                // now we need to use one of the array_get_{TYPE} methods
099                                String tmp = this.state.nextTemporaryIdentifier(baseType);
100                                this.state.addIntermediateCode(QuadrupleFactory.declare(tmp, baseType));
101                                this.state.addIntermediateCode(QuadrupleFactory.arrayGetType(baseType, baseValue, indexValue, tmp));
102                                this.state.pushIntermediateResult(baseType, tmp);
103                        }
104                }
105                else {
106                        // we are not at the outermost dimension of the array
107                        // use ARRAY_GET_REFERENCE to get a reference to this dimension
108
109                        // XXX: baseType is actually incorrect here. But there is no
110                        // XXX: ReferenceType in the interface.
111                        // XXX: this will work, because no body checks this type later on.
112                        String tmp = this.state.nextTemporaryIdentifier(baseType);
113                        this.state.addIntermediateCode(QuadrupleFactory.declareReference(tmp));
114                        this.state.addIntermediateCode(QuadrupleFactory.arrayReference(baseValue, indexValue, tmp));
115                        this.state.pushIntermediateResult(arrayType, tmp);
116                }
117        }
118}
