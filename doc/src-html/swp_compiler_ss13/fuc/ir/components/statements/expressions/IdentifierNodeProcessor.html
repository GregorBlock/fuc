017/**
018 * Processor for an identifier node
019 * 
020 * @author "Frank Zechert"
021 * @author "Danny Maasch"
022 * @author kaworu
023 * @version 3
024 */
025public class IdentifierNodeProcessor extends NodeProcessor {
026
027        /**
028         * Create a new instance of the block node processor.
029         * 
030         * @param state
031         *            the state of the intermediate code generator
032         * @param executor
033         *            the intermediate code generator executor
034         */
035        public IdentifierNodeProcessor(GeneratorState state, GeneratorExecutor executor) {
036                super(state, executor);
037        }
038
039        /**
040         * Process the basic identifier node.
041         * 
042         * @param node
043         *            the basic identifier node to process.
044         * @throws IntermediateCodeGeneratorException
045         *             The given identifier is undeclared
046         */
047
048        public void processBasicIdentifier(BasicIdentifierNode node)
049                        throws IntermediateCodeGeneratorException {
050                // A basic identifier can be pushed to the stack of results immediately
051                String identifier = node.getIdentifier();
052                Type identifierType = this.state.getCurrentSymbolTable().lookupType(identifier);
053
054                // if the identifier was renamed due to single static assignment
055                // get the renamed identifier name
056                String actualIdentifier = this.state.loadIdentifier(identifier);
057                this.state.pushIntermediateResult(identifierType, actualIdentifier);
058        }
059
060        /**
061         * Process the array identifier node.
062         * 
063         * @param node
064         *            the array identifier node to process.
065         * @throws IntermediateCodeGeneratorException
066         *             The given identifier is undeclared
067         */
068
069        public void processArrayIdentifier(ArrayIdentifierNode node)
070                        throws IntermediateCodeGeneratorException {
071
072                IntermediateResult index = this.executor.process(node.getIndexNode());
073                String indexValue = index.getValue();
074
075                IntermediateResult base = this.executor.process(node.getIdentifierNode());
076                String baseValue = base.getValue();
077                Type arrayType = base.getType();
078
079                Type baseType = ArrayHelper.getBaseType(arrayType, node);
080                boolean isOuterMostDimension = ArrayHelper.isOuterMostDimension(node);
081                boolean isTargetOfAssignment = ArrayHelper.isTargetOfAssignment(node);
082                boolean isNeededAsReference = ArrayHelper.isNeededAsReference(node);
083                if (isOuterMostDimension) {
084                        // we are at the outermost dimension of the array
085                        if (isTargetOfAssignment) {
086                                // the value of the array is not used, but a new value
087                                // is assigned to this array.
088                                // now we need to use one of the array_set_{TYPE} methods
089                                // but this needs to be done by the assignmentNodeProcessor
090                                // because we do not know the value that is assigned to us
091                                // already
092                                // we need to push the intermediate result back to the stack
093                                // of intermediate results to make it available in
094                                // assignmentNodeProcessor
095                                NodeProcessor.logger
096                                                .debug("Pushing back intermediate results, beacuse this array is on the target side of an assignment");
097                                this.state.pushIntermediateResult(index);
098                                this.state.pushIntermediateResult(base);
099                        }
100                        else {
101                                // the value of the array is used somewhere
102                                // now we need to use one of the array_get_{TYPE} methods
103                                if (isNeededAsReference) {
104                                        // reference is needed for structs in arrays in arrays in
105                                        // structs etc.
106
107                                        String tmp = this.state.nextTemporaryIdentifier(baseType);
108                                        this.state.addIntermediateCode(QuadrupleFactory.declareReference(tmp));
109                                        this.state.addIntermediateCode(QuadrupleFactory.arrayReference(baseValue, indexValue, tmp));
110                                        this.state.pushIntermediateResult(baseType, tmp);
111                                }
112                                else {
113                                        String tmp = this.state.nextTemporaryIdentifier(baseType);
114                                        this.state.addIntermediateCode(QuadrupleFactory.declare(tmp, baseType));
115                                        this.state.addIntermediateCode(QuadrupleFactory.arrayGetType(baseType, baseValue, indexValue, tmp));
116                                        this.state.pushIntermediateResult(baseType, tmp);
117                                }
118                        }
119                }
120                else {
121                        // we are not at the outermost dimension of the array
122                        // use ARRAY_GET_REFERENCE to get a reference to this dimension
123
124                        // XXX: baseType is actually incorrect here. But there is no
125                        // XXX: ReferenceType in the interface.
126                        // XXX: this will work, because no body checks this type later on.
127                        String tmp = this.state.nextTemporaryIdentifier(baseType);
128                        this.state.addIntermediateCode(QuadrupleFactory.declareReference(tmp));
129                        this.state.addIntermediateCode(QuadrupleFactory.arrayReference(baseValue, indexValue, tmp));
130                        this.state.pushIntermediateResult(arrayType, tmp);
131                }
132        }
133
134        /**
135         * Process the struct identifier node.
136         * 
137         * @param node
138         *            the struct identifier node to process.
139         * @throws IntermediateCodeGeneratorException
140         *             The given identifier is undeclared
141         */
142
143        public void processStructIdentifier(StructIdentifierNode node)
144                        throws IntermediateCodeGeneratorException {
145
146                String indexValue = node.getFieldName();
147
148                IntermediateResult base = this.executor.process(node.getIdentifierNode());
149                String baseValue = base.getValue();
150                Type structType = base.getType();
151
152                // if (!StructHelper.fieldExists((StructType) structType, indexValue)) {
153                // String err = "the field name " + indexValue +
154                // " is undeclared in the struct " + structType;
155                // logger.fatal(err);
156                // throw new IntermediateCodeGeneratorException(err);
157                // }
158
159                Type baseType = StructHelper.getBaseType(structType, node);
160                boolean isOuterMostDimension = StructHelper.isOuterMostDimension(node);
161                if (isOuterMostDimension) {
162                        // we are at the outermost dimension of the struct
163                        boolean isTargetOfAssignment = StructHelper.isTargetOfAssignment(node);
164                        if (isTargetOfAssignment) {
165                                // the value of the struct is not used, but a new value
166                                // is assigned to this struct.
167                                // now we need to use one of the struct_set_{TYPE} methods
168                                // but this needs to be done by the assignmentNodeProcessor
169                                // because we do not know the value that is assigned to us
170                                // already
171                                // we need to push the intermediate result back to the stack
172                                // of intermediate results to make it available in
173                                // assignmentNodeProcessor
174                                NodeProcessor.logger
175                                                .debug("Pushing back intermediate results, beacuse this struct is on the target side of an assignment");
176                                this.state.pushIntermediateResult(new IntermediateResult(new StringType(255L), indexValue));
177                                this.state.pushIntermediateResult(base);
178                        }
179                        else {
180                                // the value of the struct is used somewhere
181                                // now we need to use one of the struct_get_{TYPE} methods
182                                String tmp = this.state.nextTemporaryIdentifier(baseType);
183                                this.state.addIntermediateCode(QuadrupleFactory.declare(tmp, baseType));
184                                this.state.addIntermediateCode(QuadrupleFactory.structGetType(baseType, baseValue, indexValue, tmp));
185                                this.state.pushIntermediateResult(baseType, tmp);
186                        }
187                }
188                else {
189                        // we are not at the outermost dimension of the struct
190                        // use STRUCT_GET_REFERENCE to get a reference to this dimension
191
192                        // XXX: baseType is actually incorrect here. But there is no
193                        // XXX: ReferenceType in the interface.
194                        // XXX: this will work, because no body checks this type later on.
195                        String tmp = this.state.nextTemporaryIdentifier(baseType);
196                        this.state.addIntermediateCode(QuadrupleFactory.declareReference(tmp));
197                        this.state.addIntermediateCode(QuadrupleFactory.structReference(baseValue, indexValue, tmp));
198                        this.state.pushIntermediateResult(structType, tmp);
199                }
200        }
