001package swp_compiler_ss13.fuc.ir.components.statements;
002
003import swp_compiler_ss13.common.ast.nodes.ExpressionNode;
004import swp_compiler_ss13.common.ast.nodes.StatementNode;
005import swp_compiler_ss13.common.ast.nodes.ternary.BranchNode;
006import swp_compiler_ss13.common.backend.Quadruple;
007import swp_compiler_ss13.common.ir.IntermediateCodeGeneratorException;
008import swp_compiler_ss13.common.types.Type.Kind;
009import swp_compiler_ss13.fuc.ir.GeneratorExecutor;
010import swp_compiler_ss13.fuc.ir.GeneratorState;
011import swp_compiler_ss13.fuc.ir.QuadrupleFactory;
012import swp_compiler_ss13.fuc.ir.components.NodeProcessor;
013import swp_compiler_ss13.fuc.ir.data.IntermediateResult;
014
015/**
016 * Processor for a branch node
017 * 
018 * @author "Frank Zechert"
019 * @author "Danny Maasch"
020 * @author kaworu
021 * @version 3
022 */
023public class BranchNodeProcessor extends NodeProcessor {
024
025        /**
026         * Create a new instance of the branch node processor.
027         * 
028         * @param state
029         *            the state of the intermediate code generator
030         * @param executor
031         *            the intermediate code generator executor
032         */
033        public BranchNodeProcessor(GeneratorState state, GeneratorExecutor executor) {
034                super(state, executor);
035        }
036
037        /**
038         * Process the branch node.
039         * 
040         * @param branch
041         *            the branch node to process.
042         * @throws IntermediateCodeGeneratorException
043         *             An error occurred while processing the node.
044         */
045        public void processBranchNode(BranchNode branch) throws IntermediateCodeGeneratorException {
046                ExpressionNode condition = branch.getCondition();
047                StatementNode onTrue = branch.getStatementNodeOnTrue();
048                StatementNode onFalse = branch.getStatementNodeOnFalse();
049
050                IntermediateResult conditionResult = this.executor.process(condition);
051                if (conditionResult.getType().getKind() != Kind.BOOLEAN) {
052                        String err = "A condition is not of type boolean but of unsupported type "
053                                        + conditionResult.getType();
054                        NodeProcessor.logger.fatal(err);
055                        throw new IntermediateCodeGeneratorException(err);
056                }
057
058                String trueLabel = this.state.createNewLabel();
059                String falseLabel = this.state.createNewLabel();
060                String endLabel = this.state.createNewLabel();
061
062                String conditionValue = conditionResult.getValue();
063                Quadruple branchQuadruple = QuadrupleFactory.branch(conditionValue, trueLabel, falseLabel);
064                this.state.addIntermediateCode(branchQuadruple);
065                this.state.addIntermediateCode(QuadrupleFactory.label(trueLabel));
066                this.executor.processWithoutResult(onTrue);
067                this.state.addIntermediateCode(QuadrupleFactory.jump(endLabel));
068                this.state.addIntermediateCode(QuadrupleFactory.label(falseLabel));
069
070                if (onFalse != null) {
071                        this.executor.processWithoutResult(onFalse);
072                }
073
074                this.state.addIntermediateCode(QuadrupleFactory.label(endLabel));
075        }
076}
