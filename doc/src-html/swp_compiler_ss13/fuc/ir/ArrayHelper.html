001package swp_compiler_ss13.fuc.ir;
002
003import swp_compiler_ss13.common.ast.ASTNode;
004import swp_compiler_ss13.common.ast.nodes.binary.AssignmentNode;
005import swp_compiler_ss13.common.ast.nodes.leaf.BasicIdentifierNode;
006import swp_compiler_ss13.common.ast.nodes.unary.ArrayIdentifierNode;
007import swp_compiler_ss13.common.ast.nodes.unary.StructIdentifierNode;
008import swp_compiler_ss13.common.types.Type;
009import swp_compiler_ss13.common.types.derived.ArrayType;
010import swp_compiler_ss13.common.types.derived.StructType;
011import swp_compiler_ss13.common.types.primitive.PrimitiveType;
012
013/**
014 * This utility class contains helper methods for arrays.
015 * 
016 * @author "Frank Zechert"
017 * @author "Danny Maasch"
018 * @author kaworu
019 * @version 3
020 */
021public class ArrayHelper {
022
023        /**
024         * Check if the given arrayIdentifierNode is used as the target of an
025         * assignment. The target of an assignment is the left hand side value
026         * (assignment: target = source)
027         * 
028         * @param node
029         *            The node to check
030         * @return true if the node is target of an assignment.
031         */
032        public static boolean isTargetOfAssignment(ArrayIdentifierNode node) {
033                if (node.getParentNode() instanceof AssignmentNode) {
034                        AssignmentNode parent = (AssignmentNode) node.getParentNode();
035                        if (parent.getLeftValue() == node) {
036                                return true;
037                        }
038                }
039                return false;
040        }
041
042        /**
043         * Get the base type of the given derived type. The base type of an
044         * ArrayType is the inner most type.
045         * 
046         * @param type
047         *            The type to get the base type for.
048         * @param node
049         *            The node to traverse
050         * @return An array with the type and the node after resolution
051         */
052        public static Object[] getInternalBaseType(Type type, ASTNode node) {
053                // if an array is used, the inner type is needed
054                while (!(type instanceof PrimitiveType)) {
055                        while (type instanceof ArrayType) {
056                                type = ((ArrayType) type).getInnerType();
057                                node = ((ArrayIdentifierNode) node).getIdentifierNode();
058                        }
059                        if (node instanceof BasicIdentifierNode) {
060                                break;
061                        }
062                        if (type instanceof StructType) {
063                                Object[] retS = StructHelper.getInternalBaseType(type, node);
064                                type = (Type) retS[0];
065                                node = (ASTNode) retS[1];
066                        }
067                }
068                return new Object[] { type, node };
069        }
070
071        /**
072         * Get the base type of the given derived type. The base type of an
073         * ArrayType is the inner most type.
074         * 
075         * @param type
076         *            The type to get the base type for.
077         * @param node
078         *            The node to traverse
079         * @return An array with the type and the node after resolution
080         */
081        public static Type getBaseType(Type type, ASTNode node) {
082                node = getOuterArray(node);
083                return (Type) ArrayHelper.getInternalBaseType(type, node)[0];
084        }
085
086        /**
087         * Get the outer most array identifier node
088         * 
089         * @param node
090         *            the node to use
091         * @return the outer most array identifier node
092         */
093        public static ASTNode getOuterArray(ASTNode node) {
094                while (node.getParentNode() instanceof ArrayIdentifierNode) {
095                        node = node.getParentNode();
096                }
097                return node;
098        }
099
100        /**
101         * Check if the given node is the outer most dimension of an array.
102         * 
103         * @param node
104         *            The node to check
105         * @return true if the given node is the outer most dimension of an array.
106         */
107        public static boolean isOuterMostDimension(ArrayIdentifierNode node) {
108                return !(node.getParentNode() instanceof ArrayIdentifierNode);
109        }
110
111        /**
112         * Check if a reference is needed to this array instead of a type
113         * 
114         * @param node
115         * @return
116         */
117        public static boolean isNeededAsReference(ArrayIdentifierNode node) {
118                return (node.getParentNode() instanceof StructIdentifierNode);
119        }
120}
