001package swp_compiler_ss13.fuc.ir.data;
002
003import org.apache.log4j.Logger;
004
005import swp_compiler_ss13.common.types.Type;
006
007/**
008 * An intermediate result. Intermediate results are created when nodes of the AST are processed.
009 * Intermediate results are immutable.
010 * 
011 * @author "Frank Zechert"
012 * @author "Danny Maasch"
013 * @author kaworu
014 * @version 3
015 */
016public class IntermediateResult {
017
018        /**
019         * The logger for this class
020         */
021        private static Logger logger = Logger.getLogger(IntermediateResult.class);
022
023        /**
024         * The type of the intermediate result.
025         */
026        private final Type type;
027
028        /**
029         * The value of the intermediate result.
030         */
031        private final String value;
032
033        /**
034         * Create a new instance of IntermediateResult class with the given initial values.
035         * 
036         * @param type
037         *            The type of the intermediate result.
038         * @param value
039         *            The value of the intermediate result.
040         */
041        public IntermediateResult(Type type, String value) {
042                this.type = type;
043                this.value = value;
044
045                if (this.type == null) {
046                        IntermediateResult.logger
047                                        .warn("The type of an intermediate result should not be null!");
048                }
049        }
050
051        /**
052         * Get the type of this intermediate result.
053         * 
054         * @return the type.
055         */
056        public Type getType() {
057                return this.type;
058        }
059
060        /**
061         * Get the value of this intermediate result.
062         * 
063         * @return the value.
064         */
065        public String getValue() {
066                return this.value;
067        }
068
069        /**
070         * {@inheritDoc}
071         */
072        @Override
073        public String toString() {
074                // make sure to avoid null pointer exception
075                if (this.type != null) {
076                        return "IntermediateResult: " + this.type.toString() + " " + this.value;
077                }
078                return "IntermediateResult: (unknown type) " + this.value;
079        }
080}
