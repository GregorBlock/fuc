001package swp_compiler_ss13.fuc.ir.data;
002
003import java.util.LinkedList;
004
005import org.apache.log4j.Logger;
006
007/**
008 * A normal linked list extended to log calls to the add-method with Log4J.
009 * 
010 * @author "Frank Zechert"
011 * @author "Danny Maasch"
012 * @author kaworu
013 * @version 3
014 * @param <E>
015 *            The class of the list elements
016 */
017public class LoggingLinkedList<E> extends LinkedList<E> {
018
019        /**
020         * The serial version UID of this class.
021         */
022        private static final long serialVersionUID = 6844805057880276057L;
023
024        /**
025         * The logger to log the information to.
026         */
027        private static Logger logger = Logger.getLogger(LoggingLinkedList.class);
028
029        /**
030         * Create a new instance of the LogigngList class.
031         */
032        public LoggingLinkedList() {
033                super();
034        }
035
036        @Override
037        public boolean add(E item) {
038                LoggingLinkedList.logger.debug("Adding " + item.toString());
039                return super.add(item);
040        }
041}
