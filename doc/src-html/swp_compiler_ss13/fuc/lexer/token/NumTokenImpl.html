001package swp_compiler_ss13.fuc.lexer.token;
002
003import swp_compiler_ss13.common.lexer.NumToken;
004import swp_compiler_ss13.common.lexer.TokenType;
005
006/**
007 * Implementation of the interface {@link NumToken}
008 * 
009 * @author "Ho, Tay Phuong", "Thomas Benndorf"
010 * 
011 */
012public class NumTokenImpl implements NumToken {
013
014        private final String value;
015        private final TokenType type;
016        private final Integer line;
017        private final Integer column;
018
019        /**
020         * constructor
021         * 
022         * @param value
023         *            , string read by lexer for this token
024         * @param type
025         *            , type of token
026         * @param line
027         *            , line of code in source file
028         * @param column
029         *            , column of code in source file
030         */
031        public NumTokenImpl(String value, TokenType type, Integer line,
032                        Integer column) {
033
034                this.value = value;
035                this.type = type;
036                this.line = line;
037                this.column = column;
038        }
039
040        /**
041         * @return string representation of value read by lexer for this token
042         */
043        @Override
044        public String getValue() {
045                return this.value;
046        }
047
048        /**
049         * @return type of token
050         */
051        @Override
052        public TokenType getTokenType() {
053                return this.type;
054        }
055
056        /**
057         * @return line of code in source file
058         */
059        @Override
060        public Integer getLine() {
061                return this.line;
062        }
063
064        /**
065         * @return column of code in source file
066         */
067        @Override
068        public Integer getColumn() {
069                return this.column;
070        }
071
072        /**
073         * Method parses the {@link String} value of the token into a {@link Long}
074         * value
075         * 
076         * @return converted {@link Long} of value read by lexer for this token
077         */
078        @Override
079        public Long getLongValue() {
080                try {
081                        if (this.value.contains("e") || this.value.contains("E")) {
082                                String[] longValueParts = this.value.split("(e|E)");
083                                return Math.round(Double.parseDouble(longValueParts[0])
084                                                * Math.pow(10, Double.parseDouble(longValueParts[1])));
085                        } else {
086                                return Long.parseLong(this.value);
087                        }
088                } catch (Exception e) {
089                        return null;
090                }
091        }
092}
