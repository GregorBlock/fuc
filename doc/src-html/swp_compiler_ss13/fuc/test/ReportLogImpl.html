001package swp_compiler_ss13.fuc.test;
002
003import swp_compiler_ss13.common.lexer.Token;
004import swp_compiler_ss13.common.report.ReportLog;
005import swp_compiler_ss13.common.report.ReportType;
006
007import java.util.ArrayList;
008import java.util.List;
009
010/**
011 * Implementation of the ReportLog interface for use in testing.
012 */
013public class ReportLogImpl implements ReportLog, Cloneable {
014        List<ReportType> errorList = new ArrayList<>();
015        List<ReportType> warningList = new ArrayList<>();
016
017
018        /**
019         * Report a warning.
020         *
021         * @param type distinguishes the warning in machine readable format
022         * @param tokens is a list of tokens containing the warning cause and
023         *        identify the position in code
024         * @param message describes the warning in human readable manner
025         */
026        @Override
027        public void reportWarning(ReportType type, List<Token> tokens,
028                                                          String message) {
029                warningList.add(type);
030        }
031
032        /**
033         *  Report an error.
034         *
035         * @param type distinguishes the error in machine readable format
036         * @param tokens is a list of tokens containing the error cause and
037         *        identify the position in code
038         * @param message describes the error in human readable manner
039         */
040        @Override
041        public void reportError(ReportType type, List<Token> tokens, String message) {
042                errorList.add(type);
043        }
044
045        /**
046         * Test if the ReprotLog contains errors.
047         *
048         * @return true if the ReportLog contains errors.
049         */
050        public boolean hasErrors(){
051                return ! errorList.isEmpty();
052        }
053
054        /**
055         * Test if the ReprotLog contains warnings.
056         *
057         * @return true if the ReportLog contains warning.
058         */
059        public boolean hasWarnings(){
060                return ! warningList.isEmpty();
061        }
062
063        /**
064         * Test if the ReprotLog contains warnings or errors.
065         *
066         * @return true if the ReportLog contains warning or erros.
067         */
068        public boolean hasEntries(){
069                return ! errorList.isEmpty();
070        }
071
072        /**
073         * Get the list of errors.
074         * @return the list of errors
075         */
076        public List<ReportType> getErrors() {
077                return errorList;
078        }
079
080        /**
081         * Get the list of warnings.
082         * @return the list of warnings
083         */
084        public List<ReportType> getWarnings() {
085                return warningList;
086        }
087
088        /**
089         * Get a list of errors and warnings.
090         * @return list of errors and warnings
091         */
092        public List<ReportType> getEntries(){
093                List<ReportType> reportTypes = new ArrayList<>(warningList);
094                reportTypes.addAll(errorList);
095                return reportTypes;
096        }
097
098        /**
099         * Clone the ReportLog.
100         *
101         * @return a clone of the ReportLog
102         * @throws CloneNotSupportedException
103         */
104        @Override
105        protected ReportLogImpl clone() {
106                List<ReportType> errorList = new ArrayList<>();
107                List<ReportType> warningList = new ArrayList<>();
108                for ( ReportType reportType : this.errorList)
109                        errorList.add(reportType);
110                for (ReportType reportType : this.warningList)
111                        warningList.add(reportType);
112                ReportLogImpl reportLog = new ReportLogImpl();
113                reportLog.errorList = errorList;
114                reportLog.warningList = warningList;
115
116                return reportLog;
117        }
118}
