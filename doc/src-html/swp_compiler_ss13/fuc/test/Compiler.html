001package swp_compiler_ss13.fuc.test;
002
003import swp_compiler_ss13.common.ast.AST;
004import swp_compiler_ss13.common.backend.BackendException;
005import swp_compiler_ss13.common.backend.Quadruple;
006import swp_compiler_ss13.common.ir.IntermediateCodeGeneratorException;
007import swp_compiler_ss13.fuc.backend.LLVMBackend;
008import swp_compiler_ss13.fuc.ir.IntermediateCodeGeneratorImpl;
009import swp_compiler_ss13.fuc.lexer.LexerImpl;
010import swp_compiler_ss13.fuc.parser.ParserImpl;
011import swp_compiler_ss13.fuc.semantic_analyser.SemanticAnalyser;
012
013import java.io.ByteArrayInputStream;
014import java.io.IOException;
015import java.io.InputStream;
016import java.util.List;
017import java.util.Map;
018
019public class Compiler {
020        private LexerImpl lexer;
021        private ParserImpl parser;
022        private SemanticAnalyser analyser;
023        private IntermediateCodeGeneratorImpl irgen;
024        private LLVMBackend backend;
025        private ReportLogImpl errlog;
026        protected ReportLogImpl errlogAfterParser;
027        protected ReportLogImpl errlogAfterAnalyzer;
028
029        public Compiler() {
030                this.lexer = new LexerImpl();
031                this.parser = new ParserImpl();
032                this.analyser = new SemanticAnalyser();
033                this.irgen = new IntermediateCodeGeneratorImpl();
034                this.backend = new LLVMBackend();
035        }
036
037        public Compiler(LexerImpl lexer, ParserImpl parser, SemanticAnalyser analyser, IntermediateCodeGeneratorImpl irgen, LLVMBackend backend) {
038                this.lexer = lexer;
039                this.parser = parser;
040                this.analyser = analyser;
041                this.irgen = irgen;
042                this.backend = backend;
043        }
044
045        protected InputStream compile(String prog) throws BackendException,
046                        IntermediateCodeGeneratorException, IOException, InterruptedException {
047                errlog = new ReportLogImpl();
048
049                lexer.setSourceStream(new ByteArrayInputStream(prog.getBytes("UTF-8")));
050
051                parser.setLexer(lexer);
052                parser.setReportLog(errlog);
053                AST ast = parser.getParsedAST();
054                errlogAfterParser = errlog.clone();
055                if (errlog.hasErrors())
056                        return null;
057
058                analyser.setReportLog(errlog);
059                AST ast2 = analyser.analyse(ast);
060                errlogAfterAnalyzer = errlog.clone();
061                if (errlog.hasErrors())
062                        return null;
063
064                List<Quadruple> tac = irgen.generateIntermediateCode(ast2);
065
066                Map<String, InputStream> targets = backend.generateTargetCode("prog", tac);
067
068                return targets.get(targets.keySet().iterator().next());
069        }
070
071        public ReportLogImpl getErrlog() {
072                return errlog;
073        }
074
075        public ReportLogImpl getErrlogAfterParser() {
076                return errlogAfterParser;
077        }
078
079        public ReportLogImpl getErrlogAfterAnalyzer() {
080                return errlogAfterAnalyzer;
081        }
082
083}
