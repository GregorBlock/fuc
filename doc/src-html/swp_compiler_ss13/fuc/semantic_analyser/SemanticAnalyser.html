045        enum Attribute {
046
047                /**
048                 * num, basic, array,...
049                 */
050                TYPE,
051                /**
052                 * <code>"1"</code> - identifier is initialized,<br/>
053                 * <code>"0"</code> - identifier is not initialized
054                 *
055                 * @see ExpressionNode
056                 */
057                INITIALIZATION_STATUS,
058                /**
059                 * name of identifier
060                 *
061                 * @see IdentifierNode
062                 */
063                IDENTIFIER,
064                CAN_BREAK,
065                TYPE_CHECK,
066                CODE_STATE
067        }
068        private static final String IS_INITIALIZED = "1";
069        private static final String NO_ATTRIBUTE_VALUE = "undefined";
070        private static final String CAN_BREAK = "true";
071        private static final String TYPE_MISMATCH = "type mismatch";
072        private static final String DEAD_CODE = "dead";
073        private ReportLog errorLog;
074        private Map<ASTNode, Map<Attribute, String>> attributes;
075        /**
076         * Contains all initialized identifiers. As soon it has assigned it will be
077         * added.
078         */
079        private Map<SymbolTable, Set<String>> initializedIdentifiers;

081        public SemanticAnalyser() {
082                this.attributes = new HashMap<>();
083                this.initializedIdentifiers = new HashMap<>();

086        public SemanticAnalyser(ReportLog log) {
087                this.attributes = new HashMap<>();
088                this.initializedIdentifiers = new HashMap<>();

092        protected Map<SymbolTable, Set<String>> copy(Map<SymbolTable, Set<String>> m) {
093                Map<SymbolTable, Set<String>> c = new HashMap<>(m);
094
095                for (SymbolTable x : m.keySet()) {
096                        Set<String> i = new HashSet<>();
097
098                        for (String y : m.get(x)) {
099                                i.add(y);
100                        }
101
102                        c.put(x, i);
103                }
104
105                return c;
106        }

108        @Override
109        public void setReportLog(ReportLog log) {
110                errorLog = log;
111        }

113        @Override
114        public AST analyse(AST ast) {
115                assert (errorLog != null);
116
117                attributes.clear();
118                initializedIdentifiers.clear();
119
120                logger.debug("Analyzing ... please stand by!");
121                traverse(ast.getRootNode(), ast.getRootSymbolTable());
122
123                return ast;
124        }
125
126        protected void traverse(ASTNode node, SymbolTable table) {
127                logger.debug("traverse: " + node);
128
129                switch (node.getNodeType()) {
130                        case BasicIdentifierNode:
131                                logger.trace("handle BasicIdentifierNode");
132                                this.handleNode((BasicIdentifierNode) node, table);
133                                break;
134                        case BreakNode:
135                                this.handleNode((BreakNode) node, table);
136                                break;
137                        case LiteralNode:
138                                logger.trace("handle LiteralNode");
139                                this.handleNode((LiteralNode) node, table);
140                                break;
141                        case ArithmeticUnaryExpressionNode:
142                                logger.trace("handle ArithmeticUnaryExpressionNode");
143                                this.handleNode((ArithmeticUnaryExpressionNode) node, table);
144                                break;
145                        case ArrayIdentifierNode:
146                                break;
147                        case DeclarationNode:
148                                break;
149                        case LogicUnaryExpressionNode:
150                                handleNode((LogicUnaryExpressionNode) node, table);
151                                break;
152                        case PrintNode:
153                                handleNode((PrintNode) node, table);
154                                break;
155                        case ReturnNode:
156                                logger.trace("handle ReturnNode");
157                                this.handleNode((ReturnNode) node, table);
158                                break;
159                        case StructIdentifierNode:
160                                break;
161                        case ArithmeticBinaryExpressionNode:
162                                logger.trace("handle ArithmeticBinaryExpressionNode");
163                                this.handleNode((ArithmeticBinaryExpressionNode) node, table);
164                                break;
165                        case AssignmentNode:
166                                logger.trace("handle AssignmentNode");
167                                this.handleNode((AssignmentNode) node, table);
168                                break;
169                        case DoWhileNode:
170                                handleNode((DoWhileNode) node, table);
171                                break;
172                        case LogicBinaryExpressionNode:
173                                handleNode((LogicBinaryExpressionNode) node, table);
174                                break;
175                        case RelationExpressionNode:
176                                handleNode((RelationExpressionNode) node, table);
177                                break;
178                        case WhileNode:
179                                handleNode((WhileNode) node, table);
180                                break;
181                        case BranchNode:
182                                handleNode((BranchNode) node, table);
183                                break;
184                        case BlockNode:
185                                logger.trace("handle BlockNode");
186                                this.handleNode((BlockNode) node, table);
187                                break;
188                        default:
189                                throw new IllegalArgumentException("unknown ASTNodeType");
190                }
191        }

193        protected Type.Kind getType(ASTNode node) {
194                String attr = getAttribute(node, Attribute.TYPE);
195                if (!attr.equals(NO_ATTRIBUTE_VALUE)) {
196                        return Type.Kind.valueOf(attr);
197                } else {
198                        return null;
199                }
200        }
201
202        protected Type.Kind leastUpperBoundType(ASTNode left, ASTNode right) {
203                TreeSet<Type.Kind> types = new TreeSet<>();
204                Type.Kind leftType = getType(left);
205                Type.Kind rightType = getType(right);
206                
207                if (leftType == null || rightType == null) {
208                        return null;
209                }
210                
211                types.add(leftType);
212                types.add(rightType);
213
214                if (types.size() == 1) {
215                        return types.first();
216                }
217
218                if (types.contains(Type.Kind.LONG) && types.contains(Type.Kind.DOUBLE)) {
219                        return Type.Kind.DOUBLE;
220                }
221
222                return null;
223        }

225        /**
226         * Check if k is in (LONG, DOUBLE).
227         *
228         * @param k
229         * @return
230         */
231        protected boolean isNumeric(Type.Kind k) {
232                return k == Type.Kind.DOUBLE || k == Type.Kind.LONG;
233        }
234
235        protected boolean isBool(Type.Kind k) {
236                return k == Type.Kind.BOOLEAN;
237        }

239        /*
240         * Loops
241         */
242        protected void checkLoopNode(LoopNode node, SymbolTable table) {
243                if (!hasAttribute(node.getCondition(), Attribute.TYPE, Type.Kind.BOOLEAN.name())) {
244                        errorLog.reportError(ReportType.TYPE_MISMATCH, node.coverage(), "The condition must be of type bool.");
245                }
246        }
247
248        protected void handleNode(DoWhileNode node, SymbolTable table) {
249                setAttribute(node, Attribute.CAN_BREAK, CAN_BREAK);
250                traverse(node.getLoopBody(), table);
251                traverse(node.getCondition(), table);
252
253                checkLoopNode(node, table);
254        }
255
256        protected void handleNode(WhileNode node, SymbolTable table) {
257                setAttribute(node, Attribute.CAN_BREAK, CAN_BREAK);
258                traverse(node.getCondition(), table);
259
260                /*
261                 * Ignore all initialization after leaving the body's scope,
262                 * because the body may not be entered. But inside the body
263                 * the initializations are valid.
264                 * TODO: Better checking!
265                 */
266                Map<SymbolTable, Set<String>> before = copy(initializedIdentifiers);
267                traverse(node.getLoopBody(), table);
268                initializedIdentifiers = before;
269
270                checkLoopNode(node, table);
271        }
272
273        protected void handleNode(BreakNode node, SymbolTable table) {
274                if (!hasAttribute(node.getParentNode(), Attribute.CAN_BREAK, CAN_BREAK)) {
275                        errorLog.reportError(ReportType.UNDEFINED, node.coverage(), "Break can only be used in a loop.");
276                }
277        }
278
279        /*
280         * Branch node
281         */
282        protected List<SymbolTable> getSymbolTableChain(SymbolTable table) {
283                List<SymbolTable> chain = new LinkedList<>();
284
285                do {
286                        chain.add(table);
287                        table = table.getParentSymbolTable();
288                } while (table != null);
289
290                return chain;
291        }
292
293        protected void handleNode(BranchNode node, SymbolTable table) {
294                traverse(node.getCondition(), table);
295
296                Map<SymbolTable, Set<String>> beforeTrue = copy(initializedIdentifiers);
297                List<SymbolTable> beforeTrueTableChain = getSymbolTableChain(table);
298                traverse(node.getStatementNodeOnTrue(), table);
299
300                if (node.getStatementNodeOnFalse() != null) {
301                        Map<SymbolTable, Set<String>> beforeFalse = copy(initializedIdentifiers);
302                        initializedIdentifiers = beforeTrue;
303                        traverse(node.getStatementNodeOnFalse(), table);
304
305                        /*
306                         * Remove all initializations that do not occur in both branches.
307                         */
308                        for (SymbolTable s : beforeTrueTableChain) {
309                                if (initializedIdentifiers.containsKey(s)) {
310                                        /*
311                                         * initializedIdentifiers contains the state after the
312                                         * else branch, beforeFalse contains the state after
313                                         * the true branch.
314                                         * 
315                                         * Remove all from else, that were not also assigent
316                                         * in the true branch.
317                                         */
318                                        logger.debug(s + ": f:" + initializedIdentifiers.get(s) + ",  t:" + beforeFalse.get(s));
319                                        initializedIdentifiers.get(s).retainAll(beforeFalse.get(s));
320                                }
321                        }
322                } else {
323                        logger.debug("No else branch, reset initialization.");
324                        initializedIdentifiers = beforeTrue;
325                }
326
327                if (!hasAttribute(node.getCondition(), Attribute.TYPE, Type.Kind.BOOLEAN.name())) {
328                        errorLog.reportError(ReportType.TYPE_MISMATCH, node.coverage(), "The condition must be of type bool.");
329                }
330        }

332        /*
333         * Unary and binary expressions.
334         */
335        protected void binaryExpression(BinaryExpressionNode node, SymbolTable table) {
336                /*
337                 * Left sub tree
338                 */
339                ExpressionNode left = node.getLeftValue();
340                traverse(left, table);
341
342                /*
343                 * right sub tree
344                 */
345                ExpressionNode right = node.getRightValue();
346                traverse(right, table);
347
348                Type.Kind type = leastUpperBoundType(left, right);
349
350                if (hasAttribute(left, Attribute.TYPE_CHECK, TYPE_MISMATCH)
351                        || hasAttribute(right, Attribute.TYPE_CHECK, TYPE_MISMATCH)) {
352                        setAttribute(node, Attribute.TYPE_CHECK, TYPE_MISMATCH);
353                } else {
354                        if (type == null) {
355                                setAttribute(node, Attribute.TYPE_CHECK, TYPE_MISMATCH);
356                                errorLog.reportError(ReportType.TYPE_MISMATCH, node.coverage(),
357                                        "No implicit cast (no upper bound) defined for " + getType(left) + " and " + getType(right) + ".");
358                        } else {
359                                setAttribute(node, Attribute.TYPE, type.name());
360                        }
361                }
362        }
363
364        protected void unaryExpression(UnaryExpressionNode node, SymbolTable table) {
365                ExpressionNode expression = node.getRightValue();
366                traverse(expression, table);
367
368                if (hasAttribute(expression, Attribute.TYPE_CHECK, TYPE_MISMATCH)) {
369                        setAttribute(node, Attribute.TYPE_CHECK, TYPE_MISMATCH);
370                } else {
371                        setAttribute(node, Attribute.TYPE, getAttribute(expression, Attribute.TYPE));
372                }
373        }
374
375        /*
376         * Arithmetic
377         */
378        protected void handleNode(ArithmeticBinaryExpressionNode node, SymbolTable table) {
379                binaryExpression(node, table);
380
381                if (!hasAttribute(node, Attribute.TYPE_CHECK, TYPE_MISMATCH)) {
382                        Type.Kind type = getType(node);
383
384                        if (!isNumeric(type) /*|| (type == Type.Kind.STRING && node.getOperator() == BinaryExpressionNode.BinaryOperator.ADDITION)*/) {
385                                setAttribute(node, Attribute.TYPE_CHECK, TYPE_MISMATCH);
386                                errorLog.reportError(ReportType.TYPE_MISMATCH, node.coverage(),
387                                        "Operator " + node.getOperator().name() + " is not defined for " + getType(node) + ".");
388                        }
389                }
390        }
391
392        protected void handleNode(ArithmeticUnaryExpressionNode node, SymbolTable table) {
393                unaryExpression(node, table);
394
395                if (hasAttribute(node, Attribute.TYPE_CHECK, TYPE_MISMATCH)) {
396                        if (!isNumeric(getType(node))) {
397                                setAttribute(node, Attribute.TYPE_CHECK, TYPE_MISMATCH);
398                                errorLog.reportError(ReportType.TYPE_MISMATCH, node.coverage(),
399                                        "Operation " + node.getOperator() + " is not defined for " + getType(node) + ".");
400                        }
401                }
402        }
403
404        protected void handleNode(RelationExpressionNode node, SymbolTable table) {
405                binaryExpression(node, table);
406
407                if (!hasAttribute(node, Attribute.TYPE_CHECK, TYPE_MISMATCH)) {
408                        Type.Kind type = getType(node);
409
410                        BinaryExpressionNode.BinaryOperator op = node.getOperator();
411
412                        if (!isNumeric(type) && (op != BinaryExpressionNode.BinaryOperator.EQUAL || op != BinaryExpressionNode.BinaryOperator.INEQUAL)) {
413                                setAttribute(node, Attribute.TYPE_CHECK, TYPE_MISMATCH);
414                                errorLog.reportError(ReportType.TYPE_MISMATCH, node.coverage(),
415                                        "Operator " + node.getOperator() + " expects numeric operands.");
416                        } else {
417                                /*
418                                 * A ReleationExpression results in a boolean.
419                                 */
420                                setAttribute(node, Attribute.TYPE, Type.Kind.BOOLEAN.name());
421                        }
422                }
423        }

425        /*
426         * Bool
427         */
428        protected void handleNode(LogicBinaryExpressionNode node, SymbolTable table) {
429                binaryExpression(node, table);
430
431                if (!hasAttribute(node, Attribute.TYPE_CHECK, TYPE_MISMATCH)) {
432                        if (!isBool(getType(node))) {
433                                setAttribute(node, Attribute.TYPE_CHECK, TYPE_MISMATCH);
434                                errorLog.reportError(ReportType.TYPE_MISMATCH, node.coverage(),
435                                        "Operator " + node.getOperator() + " expects boolean operands");
436                        }
437                }
438        }
439
440        protected void handleNode(LogicUnaryExpressionNode node, SymbolTable table) {
441                unaryExpression(node, table);
442
443                if (!hasAttribute(node, Attribute.TYPE_CHECK, TYPE_MISMATCH)) {
444                        if (!isBool(getType(node))) {
445                                setAttribute(node, Attribute.TYPE_CHECK, TYPE_MISMATCH);
446                                errorLog.reportError(ReportType.TYPE_MISMATCH, node.coverage(),
447                                        "Operator " + node.getOperator() + " expects boolean operand.");
448                        }
449                }
450        }

452        /*
453         * Block
454         */
455        protected void handleNode(BlockNode node, SymbolTable table) {
456                if (hasAttribute(node.getParentNode(), Attribute.CAN_BREAK, CAN_BREAK)) {
457                        setAttribute(node, Attribute.CAN_BREAK, CAN_BREAK);
458                }
459
460                SymbolTable blockScope = node.getSymbolTable();
461
462                for (StatementNode child : node.getStatementList()) {
463                        if (hasAttribute(node, Attribute.CODE_STATE, DEAD_CODE)) {
464                                errorLog.reportError(ReportType.UNDEFINED, child.coverage(),
465                                        "Unreachable statement, see previous “return” in block.");
466                        }
467
468                        this.traverse(child, blockScope);
469                }
470        }
471
472        protected void handleNode(AssignmentNode node, SymbolTable table) {
473                traverse(node.getLeftValue(), table);
474                traverse(node.getRightValue(), table);
475
476                if (hasAttribute(node.getLeftValue(), Attribute.TYPE_CHECK, TYPE_MISMATCH)
477                        || hasAttribute(node.getRightValue(), Attribute.TYPE_CHECK, TYPE_MISMATCH)) {
478                        setAttribute(node, Attribute.TYPE_CHECK, TYPE_MISMATCH);
479                } else if (getType(node.getLeftValue())
480                        != leastUpperBoundType(node.getLeftValue(), node.getRightValue())) {
481
482                        setAttribute(node, Attribute.TYPE_CHECK, TYPE_MISMATCH);
483                        errorLog.reportError(ReportType.TYPE_MISMATCH, node.coverage(),
484                                "Expected " + getType(node.getLeftValue())
485                                + " found " + getType(node.getRightValue()));
486                } else {
487                        markIdentifierAsInitialized(table, getAttribute(node.getLeftValue(), Attribute.IDENTIFIER));
488                        setAttribute(node, Attribute.TYPE, getAttribute(node.getLeftValue(), Attribute.TYPE));
489
490                        logger.debug("Assignment: " + getAttribute(node.getLeftValue(), Attribute.IDENTIFIER)
491                                + ":" + getType(node.getLeftValue()) + " := " + getType(node.getRightValue()));
492                }
493        }

495        /*
496         * Leaf nodes
497         */
498        protected void handleNode(LiteralNode node, SymbolTable table) {
499                setAttribute(node, Attribute.INITIALIZATION_STATUS, IS_INITIALIZED);
500                setAttribute(node, Attribute.TYPE, node.getLiteralType().getKind().name());
501        }
502
503        protected void handleNode(BasicIdentifierNode node, SymbolTable table) {
504                String identifier = node.getIdentifier();
505                boolean initialzed = isInitialized(table, identifier);
506                Type t = table.lookupType(node.getIdentifier());
507
508                logger.debug("BasicIdentifierNode: identifier=" + identifier + ", initialized=" + initialzed + ", type=" + t);
509
510                if (t == null) {
511                        setAttribute(node, Attribute.TYPE_CHECK, TYPE_MISMATCH);
512                        errorLog.reportError(ReportType.UNDECLARED_VARIABLE_USAGE, node.coverage(),
513                                "Identifier “" + identifier + "” has not been declared.");
514
515                        return;
516                }
517
518                setAttribute(node, Attribute.IDENTIFIER, identifier);
519                setAttribute(node, Attribute.TYPE, t.getKind().name());
520
521                /*
522                 * checks
523                 */
524                boolean reportInitialization = false;
525
526                if (node.getParentNode() instanceof AssignmentNode) {
527                        AssignmentNode p = (AssignmentNode) node.getParentNode();
528                        reportInitialization = p.getLeftValue() != node;
529                } else if (node.getParentNode().getNodeType() != ASTNode.ASTNodeType.AssignmentNode) {
530                        reportInitialization = true;
531                }
532
533                if (reportInitialization && !initialzed) {
534                        errorLog.reportWarning(ReportType.UNDEFINED, node.coverage(),
535                                "Variable “" + identifier + "” may be used without initialization.");
536                }
537        }
538
539        protected void handleNode(ReturnNode node, SymbolTable table) {
540                IdentifierNode identifier = node.getRightValue();
541
542                if (identifier != null) {
543                        traverse(identifier, table);
544
545                        if (!getAttribute(identifier, Attribute.TYPE).equals(Type.Kind.LONG.name())) {
546                                errorLog.reportError(ReportType.TYPE_MISMATCH, node.coverage(),
547                                        "Only variables of type long can be returned.");
548                        }
549                }
550
551                setAttribute(node.getParentNode(), Attribute.CODE_STATE, DEAD_CODE);
552        }
553        
554        protected void handleNode(PrintNode node, SymbolTable table) {
555                IdentifierNode identifier = node.getRightValue();
556                traverse(identifier, table);
557        }
558
559        private boolean isInitialized(SymbolTable table, String identifier) {
560                SymbolTable declarationTable = table.getDeclaringSymbolTable(identifier);
561                Set<String> identifiers = this.initializedIdentifiers.get(declarationTable);
562
563                return identifiers != null && identifiers.contains(identifier);
564        }
565
566        protected void markIdentifierAsInitialized(SymbolTable table, String identifier) {
567                SymbolTable declarationTable = table.getDeclaringSymbolTable(identifier);
568
569                if (!initializedIdentifiers.containsKey(declarationTable)) {
570                        initializedIdentifiers.put(declarationTable, new HashSet<String>());
571                }
572
573                initializedIdentifiers.get(declarationTable).add(identifier);
574        }
575
576        protected String getAttribute(ASTNode node, Attribute attribute) {
577                Map<Attribute, String> nodeMap = this.attributes.get(node);
578                if (nodeMap == null) {
579                        return NO_ATTRIBUTE_VALUE;
580                }
581                String value = nodeMap.get(attribute);
582                return value == null ? NO_ATTRIBUTE_VALUE : value;
583        }
584
585        protected void setAttribute(ASTNode node, Attribute attribute, String value) {
586                if (!this.attributes.containsKey(node)) {
587                        this.attributes.put(node, new HashMap<Attribute, String>());
588                }
589                this.attributes.get(node).put(attribute, value);
590        }
591
592        protected boolean hasAttribute(ASTNode node, Attribute attribute, String value) {
593                return getAttribute(node, attribute).equals(value);
594        }
