001package swp_compiler_ss13.fuc.backend;
002
003import swp_compiler_ss13.common.backend.Backend;
004import swp_compiler_ss13.common.backend.BackendException;
005import swp_compiler_ss13.common.backend.Quadruple;
006import swp_compiler_ss13.common.types.Type;
007
008import java.io.ByteArrayInputStream;
009import java.io.ByteArrayOutputStream;
010import java.io.InputStream;
011import java.io.PrintWriter;
012import java.io.BufferedReader;
013import java.io.InputStreamReader;
014import java.io.IOException;
015import java.util.HashMap;
016import java.util.List;
017import java.util.LinkedList;
018import java.util.Map;
019
020/**
021 * This implements a backend for LLVM IR.
022 *
023 */
024public class LLVMBackend implements Backend
025{
026        /**
027         * Contains the preamble for any generated
028         * LLVM IR program; this preamble contains
029         * declarations of stdlib C functions used
030         * (e.g. printf for output to stdout),
031         * declarations of the external C++ functions
032         * used for runtime exceptions, the list
033         * of currently implemented (custom) exceptions
034         * and LLVM IR functions used by generated LLVM
035         * IR code for various other purposes, such
036         * as conversion functions and functions
037         * used to safely envelop volatile LLVM IR
038         * instructions that may lead to segfaults
039         * for bad arguments, throwing exceptions instead.
040         *
041         */
042        public final String llvm_preamble;
043        /**
044         * Contains the default catch for all thrown exceptions,
045         * so that instead of crashing with verbosity dependent
046         * on both the operating system and the respective
047         * implementation of the C++ standard library, information
048         * about the exception causing the program crash is
049         * given in a uniform fashion.
050         * This is achieved by appending the code for this
051         * default catch to the main function after it has been
052         * generated.
053         *
054         */
055        public final String llvm_uncaught;
056
057        /**
058         * Creates a new <code>LLVMBackend</code> instance
059         * and sets up all necessary variables.
060         *
061         */
062        public LLVMBackend() {
063                /* Load the preamble containing
064                   functionality used in generated IR code */
065                StringBuilder out = new StringBuilder();
066                try {
067                        BufferedReader reader = new BufferedReader(
068                                new InputStreamReader(
069                                        this.getClass().getResourceAsStream("preamble.ll")));
070                        String line;
071                        while((line = reader.readLine()) != null) {
072                                out.append(line + "\n");
073                        }
074                }
075                catch(IOException e) { }
076
077                this.llvm_preamble = out.toString();
078
079                /* Load the exception handler for uncaught
080                   exceptions. */
081                out = new StringBuilder();
082                try {
083                        BufferedReader reader = new BufferedReader(
084                                new InputStreamReader(
085                                        this.getClass().getResourceAsStream("uncaught.ll")));
086                        String line;
087                        while((line = reader.readLine()) != null) {
088                                out.append("  " + line + "\n");
089                        }
090                }
091                catch(IOException e) { }
092
093                this.llvm_uncaught = "\n" + out.toString();
094        }
095
096        // indicates whether we are currently parsing an array declaration,
097        //  (which is the only stateful parse operation)
098        private boolean array_mode = false; 
099        private List<Integer> array_sizes = new LinkedList<Integer>();
100        private String  array_name = "";
101
102        /**
103         * This function generates LLVM IR code for
104         * given three address code.
105         *
106         * @param tac the three address code
107         * @return the generated LLVM IR code.
108         */
109        @Override
110        public Map<String, InputStream> generateTargetCode(String baseFileName, List<Quadruple> tac) throws BackendException {
111                ByteArrayOutputStream outStream = new ByteArrayOutputStream();
112                PrintWriter out = new PrintWriter(outStream);
113
114                Module m = new Module(out);
115
116                /* Write preamble */
117                out.println(this.llvm_preamble);
118
119                /* Write begin for main function */
120                out.println("define i64 @main() {");
121
122                for(Quadruple q : tac)
123                {
124                        if(array_mode) {
125                                if(q.getOperator() == Quadruple.Operator.DECLARE_ARRAY) {
126                                        int size = Integer.parseInt(q.getArgument1().substring(1));
127                                        assert(size>=0);
128                                        array_sizes.add(size);
129                                }
130                                else {
131                                        Type.Kind final_type = null;
132                                        switch(q.getOperator()) {
133                                        case DECLARE_DOUBLE:
134                                                final_type = Type.Kind.DOUBLE;
135                                                break;
136                                        case DECLARE_LONG:
137                                                final_type = Type.Kind.LONG;
138                                                break;
139                                        case DECLARE_BOOLEAN:
140                                                final_type = Type.Kind.BOOLEAN;
141                                                break;
142                                        case DECLARE_STRING:
143                                                final_type = Type.Kind.STRING;
144                                                break;
145                                        default:
146                                                throw new BackendException("LOLWUT!?! Final Array Type unknown!!!");
147                                        }
148                                        m.addArrayDeclare(final_type,array_sizes,array_name);
149                                        array_sizes=new LinkedList();
150                                        array_name="";
151                                        array_mode=false;
152                                }
153                                continue;
154                        }
155
156                        switch(q.getOperator())
157                        {
158                                /* Variable declaration */
159                                case DECLARE_LONG:
160                                        m.addPrimitiveDeclare(
161                                                Type.Kind.LONG,
162                                                q.getResult(),
163                                                q.getArgument1());
164                                        break;
165                                case DECLARE_DOUBLE:
166                                        m.addPrimitiveDeclare(
167                                                Type.Kind.DOUBLE,
168                                                q.getResult(),
169                                                q.getArgument1());
170                                        break;
171                                case DECLARE_BOOLEAN:
172                                        m.addPrimitiveDeclare(
173                                                Type.Kind.BOOLEAN,
174                                                q.getResult(),
175                                                Module.toIRBoolean(q.getArgument1()));
176                                        break;
177                                case DECLARE_STRING:
178                                        m.addPrimitiveDeclare(
179                                                Type.Kind.STRING,
180                                                q.getResult(),
181                                                Module.toIRString(q.getArgument1()));
182                                        break;
183
184                                /* Arrays */
185                                case DECLARE_ARRAY:
186                                        int size = Integer.parseInt(q.getArgument1().substring(1));
187                                        assert(size>=0);
188                                        array_sizes.add(size);
189                                        array_mode=true;
190                                        array_name=q.getResult();
191                                        break;
192
193                                /* Type conversion */
194                                case LONG_TO_DOUBLE:
195                                        m.addPrimitiveConversion(
196                                                Type.Kind.LONG,
197                                                q.getArgument1(),
198                                                Type.Kind.DOUBLE,
199                                                q.getResult());
200                                        break;
201                                case DOUBLE_TO_LONG:
202                                        m.addPrimitiveConversion(
203                                                Type.Kind.DOUBLE,
204                                                q.getArgument1(),
205                                                Type.Kind.LONG,
206                                                q.getResult());
207                                        break;
208
209                                /* Unindexed copy */
210                                case ASSIGN_LONG:
211                                        m.addPrimitiveAssign(
212                                                Type.Kind.LONG,
213                                                q.getResult(),
214                                                q.getArgument1());
215                                        break;
216                                case ASSIGN_DOUBLE:
217                                        m.addPrimitiveAssign(
218                                                Type.Kind.DOUBLE,
219                                                q.getResult(),
220                                                q.getArgument1());
221                                        break;
222                                case ASSIGN_BOOLEAN:
223                                        m.addPrimitiveAssign(
224                                                Type.Kind.BOOLEAN,
225                                                q.getResult(),
226                                                Module.toIRBoolean(q.getArgument1()));
227                                        break;
228                                case ASSIGN_STRING:
229                                        m.addPrimitiveAssign(
230                                                Type.Kind.STRING,
231                                                q.getResult(),
232                                                Module.toIRString(q.getArgument1()));
233                                        break;
234
235                                /* Indexed Copy */
236                                // types are determined from context, since we have to store
237                                //  the full array type anyways
238                                case ARRAY_GET_LONG:
239                                case ARRAY_GET_DOUBLE:
240                                case ARRAY_GET_BOOLEAN:
241                                case ARRAY_GET_STRING:
242                                case ARRAY_GET_REFERENCE:
243                                        m.addArrayGet(
244                                                q.getArgument1(),
245                                                Integer.parseInt(q.getArgument2().substring(1)),
246                                                q.getResult());
247                                        break;
248                                case ARRAY_SET_LONG:
249                                case ARRAY_SET_DOUBLE:
250                                case ARRAY_SET_BOOLEAN:
251                                case ARRAY_SET_STRING:
252                                        m.addArraySet(
253                                                q.getArgument1(),
254                                                Integer.parseInt(q.getArgument2().substring(1)),
255                                                q.getResult());
256                                        break;
257                                
258                                /* References */
259                                // reference declarations are superfluous and do not generate any code,
260                                //  nor provide information which is not clear by context.
261                                case DECLARE_REFERENCE:
262                                        m.addNewReference(q.getResult());
263                                        break;
264
265                                /* Arithmetic */
266                                case ADD_LONG:
267                                        m.addPrimitiveBinaryInstruction(
268                                                q.getOperator(),
269                                                Type.Kind.LONG,
270                                                Type.Kind.LONG,
271                                                q.getArgument1(),
272                                                q.getArgument2(),
273                                                q.getResult());
274                                        break;
275                                case ADD_DOUBLE:
276                                        m.addPrimitiveBinaryInstruction(
277                                                q.getOperator(),
278                                                Type.Kind.DOUBLE,
279                                                Type.Kind.DOUBLE,
280                                                q.getArgument1(),
281                                                q.getArgument2(),
282                                                q.getResult());
283                                        break;
284                                case SUB_LONG:
285                                        m.addPrimitiveBinaryInstruction(
286                                                q.getOperator(),
287                                                Type.Kind.LONG,
288                                                Type.Kind.LONG,
289                                                q.getArgument1(),
290                                                q.getArgument2(),
291                                                q.getResult());
292                                        break;
293                                case SUB_DOUBLE:
294                                        m.addPrimitiveBinaryInstruction(
295                                                q.getOperator(),
296                                                Type.Kind.DOUBLE,
297                                                Type.Kind.DOUBLE,
298                                                q.getArgument1(),
299                                                q.getArgument2(),
300                                                q.getResult());
301                                        break;
302                                case MUL_LONG:
303                                        m.addPrimitiveBinaryInstruction(
304                                                q.getOperator(),
305                                                Type.Kind.LONG,
306                                                Type.Kind.LONG,
307                                                q.getArgument1(),
308                                                q.getArgument2(),
309                                                q.getResult());
310                                        break;
311                                case MUL_DOUBLE:
312                                        m.addPrimitiveBinaryInstruction(
313                                                q.getOperator(),
314                                                Type.Kind.DOUBLE,
315                                                Type.Kind.DOUBLE,
316                                                q.getArgument1(),
317                                                q.getArgument2(),
318                                                q.getResult());
319                                        break;
320                                case DIV_LONG:
321                                        m.addPrimitiveBinaryCall(
322                                                q.getOperator(),
323                                                Type.Kind.LONG,
324                                                Type.Kind.LONG,
325                                                q.getArgument1(),
326                                                q.getArgument2(),
327                                                q.getResult());
328                                        break;
329                                case DIV_DOUBLE:
330                                        m.addPrimitiveBinaryCall(
331                                                q.getOperator(),
332                                                Type.Kind.DOUBLE,
333                                                Type.Kind.DOUBLE,
334                                                q.getArgument1(),
335                                                q.getArgument2(),
336                                                q.getResult());
337                                        break;
338                                case NOT_BOOLEAN:
339                                        m.addBooleanNot(Module.toIRBoolean(q.getArgument1()), q.getResult());
340                                        break;
341                                case OR_BOOLEAN:
342                                        m.addPrimitiveBinaryInstruction(
343                                                        q.getOperator(),
344                                                        Type.Kind.BOOLEAN,
345                                                        Type.Kind.BOOLEAN,
346                                                        Module.toIRBoolean(q.getArgument1()),
347                                                        Module.toIRBoolean(q.getArgument2()),
348                                                        q.getResult());
349                                        break;
350                                case AND_BOOLEAN:
351                                        m.addPrimitiveBinaryInstruction(
352                                                        q.getOperator(),
353                                                        Type.Kind.BOOLEAN,
354                                                        Type.Kind.BOOLEAN,
355                                                        Module.toIRBoolean(q.getArgument1()),
356                                                        Module.toIRBoolean(q.getArgument2()),
357                                                        q.getResult());
358                                        break;
359
360                                /* Comparisons */
361                                case COMPARE_LONG_E:
362                                m.addPrimitiveBinaryInstruction(
363                                                q.getOperator(),
364                                                Type.Kind.BOOLEAN,
365                                                Type.Kind.LONG,
366                                                q.getArgument1(),
367                                                q.getArgument2(),
368                                                q.getResult());
369                                break;
370                                case COMPARE_LONG_G:
371                                        m.addPrimitiveBinaryInstruction(
372                                                        q.getOperator(),
373                                                        Type.Kind.LONG,
374                                                        Type.Kind.BOOLEAN,
375                                                        q.getArgument1(),
376                                                        q.getArgument2(),
377                                                        q.getResult());
378                                        break;
379                                case COMPARE_LONG_L:
380                                        m.addPrimitiveBinaryInstruction(
381                                                        q.getOperator(),
382                                                        Type.Kind.BOOLEAN,
383                                                        Type.Kind.LONG,
384                                                        q.getArgument1(),
385                                                        q.getArgument2(),
386                                                        q.getResult());
387                                        break;
388                                case COMPARE_LONG_GE:
389                                        m.addPrimitiveBinaryInstruction(
390                                                        q.getOperator(),
391                                                        Type.Kind.BOOLEAN,
392                                                        Type.Kind.LONG,
393                                                        q.getArgument1(),
394                                                        q.getArgument2(),
395                                                        q.getResult());
396                                        break;
397                                case COMPARE_LONG_LE:
398                                        m.addPrimitiveBinaryInstruction(
399                                                        q.getOperator(),
400                                                        Type.Kind.BOOLEAN,
401                                                        Type.Kind.LONG,
402                                                        q.getArgument1(),
403                                                        q.getArgument2(),
404                                                        q.getResult());
405                                        break;
406                                case COMPARE_DOUBLE_E:
407                                        m.addPrimitiveBinaryInstruction(
408                                                        q.getOperator(),
409                                                        Type.Kind.BOOLEAN,
410                                                        Type.Kind.DOUBLE,
411                                                        q.getArgument1(),
412                                                        q.getArgument2(),
413                                                        q.getResult());
414                                        break;
415                                case COMPARE_DOUBLE_G:
416                                        m.addPrimitiveBinaryInstruction(
417                                                        q.getOperator(),
418                                                        Type.Kind.BOOLEAN,
419                                                        Type.Kind.DOUBLE,
420                                                        q.getArgument1(),
421                                                        q.getArgument2(),
422                                                        q.getResult());
423                                        break;
424                                case COMPARE_DOUBLE_L:
425                                        m.addPrimitiveBinaryInstruction(
426                                                        q.getOperator(),
427                                                        Type.Kind.BOOLEAN,
428                                                        Type.Kind.DOUBLE,
429                                                        q.getArgument1(),
430                                                        q.getArgument2(),
431                                                        q.getResult());
432                                        break;
433                                case COMPARE_DOUBLE_GE:
434                                        m.addPrimitiveBinaryInstruction(
435                                                        q.getOperator(),
436                                                        Type.Kind.BOOLEAN,
437                                                        Type.Kind.DOUBLE,
438                                                        q.getArgument1(),
439                                                        q.getArgument2(),
440                                                        q.getResult());
441                                        break;
442                                case COMPARE_DOUBLE_LE:
443                                        m.addPrimitiveBinaryInstruction(
444                                                        q.getOperator(),
445                                                        Type.Kind.BOOLEAN,
446                                                        Type.Kind.DOUBLE,
447                                                        q.getArgument1(),
448                                                        q.getArgument2(),
449                                                        q.getResult());
450                                        break;
451
452                                /* Control flow */
453                                case RETURN:
454                                        m.addMainReturn(q.getArgument1());
455                                        break;
456                                case LABEL:
457                                        m.addLabel(q.getArgument1());
458                                        break;
459                                case BRANCH:
460                                        m.addBranch(q.getArgument1(),
461                                                        q.getArgument2(),
462                                                        q.getResult());
463                                        break;
464
465                                /* Print */
466                                case PRINT_LONG:
467                                        m.addPrint(q.getArgument1(), Type.Kind.LONG);
468                                        break;
469                                case PRINT_DOUBLE:
470                                        m.addPrint(q.getArgument1(), Type.Kind.DOUBLE);
471                                        break;
472                                case PRINT_BOOLEAN:
473                                        m.addPrint(Module.toIRBoolean(q.getArgument1()), Type.Kind.BOOLEAN);
474                                        break;
475                                case PRINT_STRING:
476                                        m.addPrint(Module.toIRString(q.getArgument1()), Type.Kind.STRING);
477                                        break;
478                        }
479                }
480
481                /* Write return 0 if last operation is not a return */
482                if(tac.size() == 0 || tac.get(tac.size() - 1).getOperator() != Quadruple.Operator.RETURN)
483                {
484                        out.println("  ret i64 0");
485                }
486
487                /* Write handler for uncaught exceptions */
488                out.print(this.llvm_uncaught);
489
490                /* Write end for main function */
491                out.println("}");
492
493                /* Finish writing */
494                out.close();
495
496                /* Convert written target code to readable input format */
497                Map<String,InputStream> map = new HashMap<String,InputStream>();
498                map.put(baseFileName+".ll", new ByteArrayInputStream(outStream.toByteArray()));
499
500                /* Return the map containing the readable target code */
501                return map;
502        }
503}
