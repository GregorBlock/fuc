020/**
021 * This implements a backend for LLVM IR.
022 *
023 */
024public class LLVMBackend implements Backend
025{
026        /**
027         * Contains the preamble for any generated
028         * LLVM IR program; this preamble contains
029         * declarations of stdlib C functions used
030         * (e.g. printf for output to stdout),
031         * declarations of the external C++ functions
032         * used for runtime exceptions, the list
033         * of currently implemented (custom) exceptions
034         * and LLVM IR functions used by generated LLVM
035         * IR code for various other purposes, such
036         * as conversion functions and functions
037         * used to safely envelop volatile LLVM IR
038         * instructions that may lead to segfaults
039         * for bad arguments, throwing exceptions instead.
040         *
041         */
042        public final String llvm_preamble;
043        /**
044         * Contains the default catch for all thrown exceptions,
045         * so that instead of crashing with verbosity dependent
046         * on both the operating system and the respective
047         * implementation of the C++ standard library, information
048         * about the exception causing the program crash is
049         * given in a uniform fashion.
050         * This is achieved by appending the code for this
051         * default catch to the main function after it has been
052         * generated.
053         *
054         */
055        public final String llvm_uncaught;
056
057        /**
058         * Creates a new <code>LLVMBackend</code> instance
059         * and sets up all necessary variables.
060         *
061         */
062        public LLVMBackend() {
063                /* Load the preamble containing
064                   functionality used in generated IR code */
065                StringBuilder out = new StringBuilder();
066                try {
067                        BufferedReader reader = new BufferedReader(
068                                new InputStreamReader(
069                                        this.getClass().getResourceAsStream("preamble.ll")));
070                        String line;
071                        while((line = reader.readLine()) != null) {
072                                out.append(line + "\n");
073                        }
074                }
075                catch(IOException e) { }
076
077                this.llvm_preamble = out.toString();
078
079                /* Load the exception handler for uncaught
080                   exceptions. */
081                out = new StringBuilder();
082                try {
083                        BufferedReader reader = new BufferedReader(
084                                new InputStreamReader(
085                                        this.getClass().getResourceAsStream("uncaught.ll")));
086                        String line;
087                        while((line = reader.readLine()) != null) {
088                                out.append("  " + line + "\n");
089                        }
090                }
091                catch(IOException e) { }
092
093                this.llvm_uncaught = "\n" + out.toString();
094        }
095
096        // indicates whether we are currently parsing an array declaration,
097        //  (which is the only stateful parse operation)
098        private boolean array_mode = false; 
099        private List<Integer> array_sizes = new LinkedList<Integer>();
100        private String  array_name = "";
101
102        /**
103         * This is a pass the backend performs on the TAC
104         * to ensure it is correct before generating any code.
105         * This pass in particular ensures that all basic blocks
106         * are terminated by a valid Quadruple, which is either
107         * a <code>Quadruple.Operator.RETURN</code> or a <code>Quadruple.Operator.BRANCH</code>,
108         * by adding them where necessary.
109         *
110         * @param tac the program in TAC format to perform the pass on
111         * @return the program in TAC format after the pass has been performed
112         */
113        private List<Quadruple> basicBlocksTerminatedPass(List<Quadruple> tac) {
114
115                /* Whenever a label is not preceded by a terminator instruction,
116                 * add a branching instruction to the label before it. */
117                for(int i = 0; i < tac.size(); i += 1) {
118                        Quadruple q = tac.get(i);
119                        Quadruple.Operator op = q.getOperator();
120                        if((op == Quadruple.Operator.LABEL)) {
121                                Quadruple.Operator last_op = tac.get(i-1).getOperator();
122                                if(! ((last_op == Quadruple.Operator.RETURN) ||
123                                      (last_op == Quadruple.Operator.BRANCH))) {
124                                        tac.add(i, new QuadrupleImpl(
125                                                        Quadruple.Operator.BRANCH,
126                                                        q.getArgument1(),
127                                                        Quadruple.EmptyArgument,
128                                                        Quadruple.EmptyArgument));
129                                        i += 1;
130                                }
131                        }
132                }
133
134                /* If the program is not terminated by a proper return,
135                 * add such a return with the default value of 0 (indicating
136                 * successful termination). */
137                if(tac.size() == 0 || tac.get(tac.size() - 1).getOperator() != Quadruple.Operator.RETURN) {
138                        tac.add(new QuadrupleImpl(Quadruple.Operator.RETURN,
139                                                  "#0",
140                                                  Quadruple.EmptyArgument,
141                                                  Quadruple.EmptyArgument));
142                }
143
144                return tac;
145        }
146
147        /**
148         * This function generates LLVM IR code for
149         * given three address code.
150         *
151         * @param tac the three address code
152         * @return the generated LLVM IR code.
153         */
154        @Override
155        public Map<String, InputStream> generateTargetCode(String baseFileName, List<Quadruple> tac) throws BackendException {
156                ByteArrayOutputStream outStream = new ByteArrayOutputStream();
157                PrintWriter out = new PrintWriter(outStream);
158
159                /* Run passes on the three address code to ensure it is correct. */
160                tac = basicBlocksTerminatedPass(tac);
161
162                Module m = new Module(out);
163
164                /* Write preamble */
165                out.println(this.llvm_preamble);
166
167                /* Write begin for main function */
168                out.println("define i64 @main() {");
169
170                for(Quadruple q : tac)
171                {
172                        if(array_mode) {
173                                if(q.getOperator() == Quadruple.Operator.DECLARE_ARRAY) {
174                                        int size = Integer.parseInt(q.getArgument1().substring(1));
175                                        assert(size>=0);
176                                        array_sizes.add(size);
177                                }
178                                else {
179                                        Type.Kind final_type = null;
180                                        switch(q.getOperator()) {
181                                        case DECLARE_DOUBLE:
182                                                final_type = Type.Kind.DOUBLE;
183                                                break;
184                                        case DECLARE_LONG:
185                                                final_type = Type.Kind.LONG;
186                                                break;
187                                        case DECLARE_BOOLEAN:
188                                                final_type = Type.Kind.BOOLEAN;
189                                                break;
190                                        case DECLARE_STRING:
191                                                final_type = Type.Kind.STRING;
192                                                break;
193                                        default:
194                                                throw new BackendException("LOLWUT!?! Final Array Type unknown!!!");
195                                        }
196                                        m.addArrayDeclare(final_type,array_sizes,array_name);
197                                        array_sizes=new LinkedList();
198                                        array_name="";
199                                        array_mode=false;
200                                }
201                                continue;
202                        }
203
204                        switch(q.getOperator())
205                        {
206                                /* Variable declaration */
207                                case DECLARE_LONG:
208                                        m.addPrimitiveDeclare(
209                                                Type.Kind.LONG,
210                                                q.getResult(),
211                                                q.getArgument1());
212                                        break;
213                                case DECLARE_DOUBLE:
214                                        m.addPrimitiveDeclare(
215                                                Type.Kind.DOUBLE,
216                                                q.getResult(),
217                                                q.getArgument1());
218                                        break;
219                                case DECLARE_BOOLEAN:
220                                        m.addPrimitiveDeclare(
221                                                Type.Kind.BOOLEAN,
222                                                q.getResult(),
223                                                Module.toIRBoolean(q.getArgument1()));
224                                        break;
225                                case DECLARE_STRING:
226                                        m.addPrimitiveDeclare(
227                                                Type.Kind.STRING,
228                                                q.getResult(),
229                                                Module.toIRString(q.getArgument1()));
230                                        break;
231
232                                /* Arrays */
233                                case DECLARE_ARRAY:
234                                        int size = Integer.parseInt(q.getArgument1().substring(1));
235                                        assert(size>=0);
236                                        array_sizes.add(size);
237                                        array_mode=true;
238                                        array_name=q.getResult();
239                                        break;
240
241                                /* Type conversion */
242                                case LONG_TO_DOUBLE:
243                                        m.addPrimitiveConversion(
244                                                Type.Kind.LONG,
245                                                q.getArgument1(),
246                                                Type.Kind.DOUBLE,
247                                                q.getResult());
248                                        break;
249                                case DOUBLE_TO_LONG:
250                                        m.addPrimitiveConversion(
251                                                Type.Kind.DOUBLE,
252                                                q.getArgument1(),
253                                                Type.Kind.LONG,
254                                                q.getResult());
255                                        break;
256
257                                /* Unindexed copy */
258                                case ASSIGN_LONG:
259                                        m.addPrimitiveAssign(
260                                                Type.Kind.LONG,
261                                                q.getResult(),
262                                                q.getArgument1());
263                                        break;
264                                case ASSIGN_DOUBLE:
265                                        m.addPrimitiveAssign(
266                                                Type.Kind.DOUBLE,
267                                                q.getResult(),
268                                                q.getArgument1());
269                                        break;
270                                case ASSIGN_BOOLEAN:
271                                        m.addPrimitiveAssign(
272                                                Type.Kind.BOOLEAN,
273                                                q.getResult(),
274                                                Module.toIRBoolean(q.getArgument1()));
275                                        break;
276                                case ASSIGN_STRING:
277                                        m.addPrimitiveAssign(
278                                                Type.Kind.STRING,
279                                                q.getResult(),
280                                                Module.toIRString(q.getArgument1()));
281                                        break;
282
283                                /* Indexed Copy */
284                                // types are determined from context, since we have to store
285                                //  the full array type anyways
286                                case ARRAY_GET_LONG:
287                                case ARRAY_GET_DOUBLE:
288                                case ARRAY_GET_BOOLEAN:
289                                case ARRAY_GET_STRING:
290                                case ARRAY_GET_REFERENCE:
291                                        m.addArrayGet(
292                                                q.getArgument1(),
293                                                Integer.parseInt(q.getArgument2().substring(1)),
294                                                q.getResult());
295                                        break;
296                                case ARRAY_SET_LONG:
297                                case ARRAY_SET_DOUBLE:
298                                case ARRAY_SET_BOOLEAN:
299                                case ARRAY_SET_STRING:
300                                        m.addArraySet(
301                                                q.getArgument1(),
302                                                Integer.parseInt(q.getArgument2().substring(1)),
303                                                q.getResult());
304                                        break;
305                                
306                                /* References */
307                                // reference declarations are superfluous and do not generate any code,
308                                //  nor provide information which is not clear by context.
309                                case DECLARE_REFERENCE:
310                                        m.addNewReference(q.getResult());
311                                        break;
312
313                                /* Arithmetic */
314                                case ADD_LONG:
315                                        m.addPrimitiveBinaryInstruction(
316                                                q.getOperator(),
317                                                Type.Kind.LONG,
318                                                Type.Kind.LONG,
319                                                q.getArgument1(),
320                                                q.getArgument2(),
321                                                q.getResult());
322                                        break;
323                                case ADD_DOUBLE:
324                                        m.addPrimitiveBinaryInstruction(
325                                                q.getOperator(),
326                                                Type.Kind.DOUBLE,
327                                                Type.Kind.DOUBLE,
328                                                q.getArgument1(),
329                                                q.getArgument2(),
330                                                q.getResult());
331                                        break;
332                                case SUB_LONG:
333                                        m.addPrimitiveBinaryInstruction(
334                                                q.getOperator(),
335                                                Type.Kind.LONG,
336                                                Type.Kind.LONG,
337                                                q.getArgument1(),
338                                                q.getArgument2(),
339                                                q.getResult());
340                                        break;
341                                case SUB_DOUBLE:
342                                        m.addPrimitiveBinaryInstruction(
343                                                q.getOperator(),
344                                                Type.Kind.DOUBLE,
345                                                Type.Kind.DOUBLE,
346                                                q.getArgument1(),
347                                                q.getArgument2(),
348                                                q.getResult());
349                                        break;
350                                case MUL_LONG:
351                                        m.addPrimitiveBinaryInstruction(
352                                                q.getOperator(),
353                                                Type.Kind.LONG,
354                                                Type.Kind.LONG,
355                                                q.getArgument1(),
356                                                q.getArgument2(),
357                                                q.getResult());
358                                        break;
359                                case MUL_DOUBLE:
360                                        m.addPrimitiveBinaryInstruction(
361                                                q.getOperator(),
362                                                Type.Kind.DOUBLE,
363                                                Type.Kind.DOUBLE,
364                                                q.getArgument1(),
365                                                q.getArgument2(),
366                                                q.getResult());
367                                        break;
368                                case DIV_LONG:
369                                        m.addPrimitiveBinaryCall(
370                                                q.getOperator(),
371                                                Type.Kind.LONG,
372                                                Type.Kind.LONG,
373                                                q.getArgument1(),
374                                                q.getArgument2(),
375                                                q.getResult());
376                                        break;
377                                case DIV_DOUBLE:
378                                        m.addPrimitiveBinaryCall(
379                                                q.getOperator(),
380                                                Type.Kind.DOUBLE,
381                                                Type.Kind.DOUBLE,
382                                                q.getArgument1(),
383                                                q.getArgument2(),
384                                                q.getResult());
385                                        break;
386                                case NOT_BOOLEAN:
387                                        m.addBooleanNot(Module.toIRBoolean(q.getArgument1()), q.getResult());
388                                        break;
389                                case OR_BOOLEAN:
390                                        m.addPrimitiveBinaryInstruction(
391                                                        q.getOperator(),
392                                                        Type.Kind.BOOLEAN,
393                                                        Type.Kind.BOOLEAN,
394                                                        Module.toIRBoolean(q.getArgument1()),
395                                                        Module.toIRBoolean(q.getArgument2()),
396                                                        q.getResult());
397                                        break;
398                                case AND_BOOLEAN:
399                                        m.addPrimitiveBinaryInstruction(
400                                                        q.getOperator(),
401                                                        Type.Kind.BOOLEAN,
402                                                        Type.Kind.BOOLEAN,
403                                                        Module.toIRBoolean(q.getArgument1()),
404                                                        Module.toIRBoolean(q.getArgument2()),
405                                                        q.getResult());
406                                        break;
407
408                                /* Comparisons */
409                                case COMPARE_LONG_E:
410                                m.addPrimitiveBinaryInstruction(
411                                                q.getOperator(),
412                                                Type.Kind.BOOLEAN,
413                                                Type.Kind.LONG,
414                                                q.getArgument1(),
415                                                q.getArgument2(),
416                                                q.getResult());
417                                break;
418                                case COMPARE_LONG_G:
419                                        m.addPrimitiveBinaryInstruction(
420                                                        q.getOperator(),
421                                                        Type.Kind.LONG,
422                                                        Type.Kind.BOOLEAN,
423                                                        q.getArgument1(),
424                                                        q.getArgument2(),
425                                                        q.getResult());
426                                        break;
427                                case COMPARE_LONG_L:
428                                        m.addPrimitiveBinaryInstruction(
429                                                        q.getOperator(),
430                                                        Type.Kind.BOOLEAN,
431                                                        Type.Kind.LONG,
432                                                        q.getArgument1(),
433                                                        q.getArgument2(),
434                                                        q.getResult());
435                                        break;
436                                case COMPARE_LONG_GE:
437                                        m.addPrimitiveBinaryInstruction(
438                                                        q.getOperator(),
439                                                        Type.Kind.BOOLEAN,
440                                                        Type.Kind.LONG,
441                                                        q.getArgument1(),
442                                                        q.getArgument2(),
443                                                        q.getResult());
444                                        break;
445                                case COMPARE_LONG_LE:
446                                        m.addPrimitiveBinaryInstruction(
447                                                        q.getOperator(),
448                                                        Type.Kind.BOOLEAN,
449                                                        Type.Kind.LONG,
450                                                        q.getArgument1(),
451                                                        q.getArgument2(),
452                                                        q.getResult());
453                                        break;
454                                case COMPARE_DOUBLE_E:
455                                        m.addPrimitiveBinaryInstruction(
456                                                        q.getOperator(),
457                                                        Type.Kind.BOOLEAN,
458                                                        Type.Kind.DOUBLE,
459                                                        q.getArgument1(),
460                                                        q.getArgument2(),
461                                                        q.getResult());
462                                        break;
463                                case COMPARE_DOUBLE_G:
464                                        m.addPrimitiveBinaryInstruction(
465                                                        q.getOperator(),
466                                                        Type.Kind.BOOLEAN,
467                                                        Type.Kind.DOUBLE,
468                                                        q.getArgument1(),
469                                                        q.getArgument2(),
470                                                        q.getResult());
471                                        break;
472                                case COMPARE_DOUBLE_L:
473                                        m.addPrimitiveBinaryInstruction(
474                                                        q.getOperator(),
475                                                        Type.Kind.BOOLEAN,
476                                                        Type.Kind.DOUBLE,
477                                                        q.getArgument1(),
478                                                        q.getArgument2(),
479                                                        q.getResult());
480                                        break;
481                                case COMPARE_DOUBLE_GE:
482                                        m.addPrimitiveBinaryInstruction(
483                                                        q.getOperator(),
484                                                        Type.Kind.BOOLEAN,
485                                                        Type.Kind.DOUBLE,
486                                                        q.getArgument1(),
487                                                        q.getArgument2(),
488                                                        q.getResult());
489                                        break;
490                                case COMPARE_DOUBLE_LE:
491                                        m.addPrimitiveBinaryInstruction(
492                                                        q.getOperator(),
493                                                        Type.Kind.BOOLEAN,
494                                                        Type.Kind.DOUBLE,
495                                                        q.getArgument1(),
496                                                        q.getArgument2(),
497                                                        q.getResult());
498                                        break;
499
500                                /* Control flow */
501                                case RETURN:
502                                        m.addMainReturn(q.getArgument1());
503                                        break;
504                                case LABEL:
505                                        m.addLabel(q.getArgument1());
506                                        break;
507                                case BRANCH:
508                                        m.addBranch(q.getArgument1(),
509                                                        q.getArgument2(),
510                                                        q.getResult());
511                                        break;
512
513                                /* Print */
514                                case PRINT_LONG:
515                                        m.addPrint(q.getArgument1(), Type.Kind.LONG);
516                                        break;
517                                case PRINT_DOUBLE:
518                                        m.addPrint(q.getArgument1(), Type.Kind.DOUBLE);
519                                        break;
520                                case PRINT_BOOLEAN:
521                                        m.addPrint(Module.toIRBoolean(q.getArgument1()), Type.Kind.BOOLEAN);
522                                        break;
523                                case PRINT_STRING:
524                                        m.addPrint(Module.toIRString(q.getArgument1()), Type.Kind.STRING);
525                                        break;
526                        }
527                }
528
529                /* Write handler for uncaught exceptions */
530                out.print(this.llvm_uncaught);
531
532                /* Write end for main function */
533                out.println("}");
534
535                /* Finish writing */
536                out.close();
537
538                /* Convert written target code to readable input format */
539                Map<String,InputStream> map = new HashMap<String,InputStream>();
540                map.put(baseFileName+".ll", new ByteArrayInputStream(outStream.toByteArray()));
541
542                /* Return the map containing the readable target code */
543                return map;
544        }
