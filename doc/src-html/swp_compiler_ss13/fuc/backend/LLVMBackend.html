023/**
024 * This implements a backend for LLVM IR.
025 *
026 */
027public class LLVMBackend implements Backend
028{
029        /**
030         * Contains the preamble for any generated
031         * LLVM IR program; this preamble contains
032         * declarations of stdlib C functions used
033         * (e.g. printf for output to stdout),
034         * declarations of the external C++ functions
035         * used for runtime exceptions, the list
036         * of currently implemented (custom) exceptions
037         * and LLVM IR functions used by generated LLVM
038         * IR code for various other purposes, such
039         * as conversion functions and functions
040         * used to safely envelop volatile LLVM IR
041         * instructions that may lead to segfaults
042         * for bad arguments, throwing exceptions instead.
043         *
044         */
045        public final String llvm_preamble;
046        /**
047         * Contains the default catch for all thrown exceptions,
048         * so that instead of crashing with verbosity dependent
049         * on both the operating system and the respective
050         * implementation of the C++ standard library, information
051         * about the exception causing the program crash is
052         * given in a uniform fashion.
053         * This is achieved by appending the code for this
054         * default catch to the main function after it has been
055         * generated.
056         *
057         */
058        public final String llvm_uncaught;
059
060        /**
061         * Creates a new <code>LLVMBackend</code> instance
062         * and sets up all necessary variables.
063         *
064         */
065        public LLVMBackend() {
066                /* Load the preamble containing
067                   functionality used in generated IR code */
068                StringBuilder out = new StringBuilder();
069                try {
070                        BufferedReader reader = new BufferedReader(
071                                new InputStreamReader(
072                                        this.getClass().getResourceAsStream("preamble.ll")));
073                        String line;
074                        while((line = reader.readLine()) != null) {
075                                out.append(line + "\n");
076                        }
077                }
078                catch(IOException e) { }
079
080                this.llvm_preamble = out.toString();
081
082                /* Load the exception handler for uncaught
083                   exceptions. */
084                out = new StringBuilder();
085                try {
086                        BufferedReader reader = new BufferedReader(
087                                new InputStreamReader(
088                                        this.getClass().getResourceAsStream("uncaught.ll")));
089                        String line;
090                        while((line = reader.readLine()) != null) {
091                                out.append("  " + line + "\n");
092                        }
093                }
094                catch(IOException e) { }
095
096                this.llvm_uncaught = "\n" + out.toString();
097        }

099        /**
100         * This is a pass the backend performs on the TAC
101         * to ensure it is correct before generating any code.
102         * This pass in particular ensures that all basic blocks
103         * are terminated by a valid Quadruple, which is either
104         * a <code>Quadruple.Operator.RETURN</code> or a <code>Quadruple.Operator.BRANCH</code>,
105         * by adding them where necessary.
106         *
107         * @param tac the program in TAC format to perform the pass on
108         * @return the program in TAC format after the pass has been performed
109         */
110        private List<Quadruple> basicBlocksTerminatedPass(List<Quadruple> tac) {
111
112                /* Whenever a label is not preceded by a terminator instruction,
113                 * add a branching instruction to the label before it. */
114                for(int i = 0; i < tac.size(); i += 1) {
115                        Quadruple q = tac.get(i);
116                        Quadruple.Operator op = q.getOperator();
117                        if((op == Quadruple.Operator.LABEL)) {
118                                Quadruple.Operator last_op = tac.get(i-1).getOperator();
119                                if(! ((last_op == Quadruple.Operator.RETURN) ||
120                                      (last_op == Quadruple.Operator.BRANCH))) {
121                                        tac.add(i, new QuadrupleImpl(
122                                                        Quadruple.Operator.BRANCH,
123                                                        q.getArgument1(),
124                                                        Quadruple.EmptyArgument,
125                                                        Quadruple.EmptyArgument));
126                                        i += 1;
127                                }
128                        }
129                }
130
131                /* If the program is not terminated by a proper return,
132                 * add such a return with the default value of 0 (indicating
133                 * successful termination). */
134                if(tac.size() == 0 || tac.get(tac.size() - 1).getOperator() != Quadruple.Operator.RETURN) {
135                        tac.add(new QuadrupleImpl(Quadruple.Operator.RETURN,
136                                                  "#0",
137                                                  Quadruple.EmptyArgument,
138                                                  Quadruple.EmptyArgument));
139                }
140
141                return tac;
142        }

144        private LLVMBackendArrayType parseArrayDeclaration(Iterator<Quadruple> tacIterator, int size) throws BackendException {
145                boolean done = false;
146                Type type = null;
147                List<Integer> dimensions = new LinkedList<Integer>();
148                dimensions.add(size);
149
150                while(!done && tacIterator.hasNext()) {
151                        Quadruple q = tacIterator.next();
152
153                        if(q.getOperator() == Quadruple.Operator.DECLARE_ARRAY) {
154                                size = Integer.parseInt(q.getArgument1().substring(1));
155                                assert(size>=0);
156                                dimensions.add(size);
157                        }
158                        else {
159                                switch(q.getOperator()) {
160                                        case DECLARE_DOUBLE:
161                                                type = new DoubleType();
162                                                break;
163                                        case DECLARE_LONG:
164                                                type = new LongType();
165                                                break;
166                                        case DECLARE_BOOLEAN:
167                                                type = new BooleanType();
168                                                break;
169                                        case DECLARE_STRING:
170                                                type = new StringType(0L);
171                                                break;
172                                        case DECLARE_STRUCT:
173                                                size = Integer.parseInt(q.getArgument1().substring(1));
174                                                assert(size >= 0);
175                                                type = this.parseStructDeclaration(tacIterator, size);
176                                                break;
177                                        default:
178                                                throw new BackendException("Unknown type for array declaration");
179                                }
180
181                                done = true;
182                        }
183                }
184
185                return new LLVMBackendArrayType(dimensions, type);
186        }
187
188        private LLVMBackendStructType parseStructDeclaration(Iterator<Quadruple> tacIterator, int size) throws BackendException {
189                int i = 0;
190                List<Member> members = new LinkedList<Member>();
191
192                while((i < size) && tacIterator.hasNext()) {
193                        Quadruple q = tacIterator.next();
194                        int memberSize = 0;
195
196                        switch(q.getOperator()) {
197                                case DECLARE_DOUBLE:
198                                        members.add(new Member(q.getResult(), new DoubleType()));
199                                        break;
200                                case DECLARE_LONG:
201                                        members.add(new Member(q.getResult(), new LongType()));
202                                        break;
203                                case DECLARE_BOOLEAN:
204                                        members.add(new Member(q.getResult(), new BooleanType()));
205                                        break;
206                                case DECLARE_STRING:
207                                        members.add(new Member(q.getResult(), new StringType(0L)));
208                                        break;
209                                case DECLARE_ARRAY:
210                                        memberSize = Integer.parseInt(q.getArgument1().substring(1));
211                                        assert(size>=0);
212                                        members.add(new Member(q.getResult(), this.parseArrayDeclaration(tacIterator, memberSize)));
213                                        break;
214                                case DECLARE_STRUCT:
215                                        memberSize = Integer.parseInt(q.getArgument1().substring(1));
216                                        assert(size>=0);
217                                        members.add(new Member(q.getResult(), this.parseStructDeclaration(tacIterator, memberSize)));
218                                        break;
219                                default:
220                                        break;
221                        }
222
223                        i += 1;
224                }
225
226                return new LLVMBackendStructType(members);
227        }

229        /**
230         * This function generates LLVM IR code for
231         * given three address code.
232         *
233         * @param tac the three address code
234         * @return the generated LLVM IR code.
235         */
236        @Override
237        public Map<String, InputStream> generateTargetCode(String baseFileName, List<Quadruple> tac) throws BackendException {
238                ByteArrayOutputStream outStream = new ByteArrayOutputStream();
239                PrintWriter out = new PrintWriter(outStream);
240
241                /* Run passes on the three address code to ensure it is correct. */
242                tac = basicBlocksTerminatedPass(tac);
243
244                /* Write preamble */
245                out.println(this.llvm_preamble);
246
247                Iterator<Quadruple> tacIterator = tac.iterator();
248
249                Function main = new Function(
250                        "main",
251                        Type.Kind.LONG,
252                        new LinkedList<Map.Entry<String,Type.Kind>>());
253
254                /* Generate LLVM IR code corresponding
255                 * to the TAC given as input */
256                while(tacIterator.hasNext())
257                {
258                        Quadruple q = tacIterator.next();
259                        int size = 0;
260
261                        switch(q.getOperator())
262                        {
263                                /* Variable declaration */
264                                case DECLARE_LONG:
265                                case DECLARE_DOUBLE:
266                                        main.addPrimitiveDeclare(
267                                                Module.toArgument1TypeKind(q.getOperator()),
268                                                q.getResult(),
269                                                q.getArgument1());
270                                        break;
271                                case DECLARE_BOOLEAN:
272                                        main.addPrimitiveDeclare(
273                                                Type.Kind.BOOLEAN,
274                                                q.getResult(),
275                                                Module.toIRBoolean(q.getArgument1()));
276                                        break;
277                                case DECLARE_STRING:
278                                        main.addPrimitiveDeclare(
279                                                Type.Kind.STRING,
280                                                q.getResult(),
281                                                Module.toIRString(q.getArgument1()));
282                                        break;
283                                case DECLARE_REFERENCE:
284                                        /* Reference declarations are only needed to keep the syntax
285                                         * style of the TAC homogenous, they hold no semantic relevance
286                                         * (as it is a meta-type) and thus get simply ignored. */
287                                        break;
288                                case DECLARE_ARRAY:
289                                        size = Integer.parseInt(q.getArgument1().substring(1));
290                                        assert(size>=0);
291                                        LLVMBackendArrayType array = this.parseArrayDeclaration(tacIterator, size);
292                                        main.addDerivedDeclare(array, q.getResult());
293                                        break;
294                                case DECLARE_STRUCT:
295                                        size = Integer.parseInt(q.getArgument1().substring(1));
296                                        assert(size>=0);
297                                        LLVMBackendStructType struct = this.parseStructDeclaration(tacIterator, size);
298                                        main.addDerivedDeclare(struct, q.getResult());
299                                        break;
300
301                                /* Type conversion */
302                                case LONG_TO_DOUBLE:
303                                case DOUBLE_TO_LONG:
304                                case BOOLEAN_TO_STRING:
305                                case LONG_TO_STRING:
306                                case DOUBLE_TO_STRING:
307                                        main.addPrimitiveConversion(
308                                                Module.toArgument1TypeKind(q.getOperator()),
309                                                q.getArgument1(),
310                                                Module.toResultTypeKind(q.getOperator()),
311                                                q.getResult());
312                                        break;
313
314                                /* Unindexed copy */
315                                case ASSIGN_LONG:
316                                case ASSIGN_DOUBLE:
317                                        main.addPrimitiveAssign(
318                                                Module.toArgument1TypeKind(q.getOperator()),
319                                                q.getResult(),
320                                                q.getArgument1());
321                                        break;
322                                case ASSIGN_BOOLEAN:
323                                        main.addPrimitiveAssign(
324                                                Type.Kind.BOOLEAN,
325                                                q.getResult(),
326                                                Module.toIRBoolean(q.getArgument1()));
327                                        break;
328                                case ASSIGN_STRING:
329                                        main.addPrimitiveAssign(
330                                                Type.Kind.STRING,
331                                                q.getResult(),
332                                                Module.toIRString(q.getArgument1()));
333                                        break;
334
335                                /* Indexed Copy */
336                                case ARRAY_GET_REFERENCE:
337                                case STRUCT_GET_REFERENCE:
338                                        main.addReferenceDerivedGet(
339                                                Module.toArgument1TypeKind(q.getOperator()),
340                                                q.getArgument1(),
341                                                q.getArgument2(),
342                                                q.getResult());
343                                        break;
344                                case ARRAY_GET_LONG:
345                                case ARRAY_GET_DOUBLE:
346                                case ARRAY_GET_BOOLEAN:
347                                case ARRAY_GET_STRING:
348                                case STRUCT_GET_LONG:
349                                case STRUCT_GET_DOUBLE:
350                                case STRUCT_GET_BOOLEAN:
351                                case STRUCT_GET_STRING:
352                                        main.addPrimitiveDerivedSetOrGet(
353                                                Module.toArgument1TypeKind(q.getOperator()),
354                                                q.getArgument1(),
355                                                q.getArgument2(),
356                                                Module.toResultTypeKind(q.getOperator()),
357                                                q.getResult(),
358                                                false);
359                                        break;
360                                case ARRAY_SET_LONG:
361                                case ARRAY_SET_DOUBLE:
362                                case ARRAY_SET_BOOLEAN:
363                                case ARRAY_SET_STRING:
364                                case STRUCT_SET_LONG:
365                                case STRUCT_SET_DOUBLE:
366                                case STRUCT_SET_BOOLEAN:
367                                case STRUCT_SET_STRING:
368                                        main.addPrimitiveDerivedSetOrGet(
369                                                Module.toArgument1TypeKind(q.getOperator()),
370                                                q.getArgument1(),
371                                                q.getArgument2(),
372                                                Module.toResultTypeKind(q.getOperator()),
373                                                q.getResult(),
374                                                true);
375                                        break;
376
377                                /* Arithmetic */
378                                case ADD_LONG:
379                                case ADD_DOUBLE:
380                                case SUB_LONG:
381                                case SUB_DOUBLE:
382                                case MUL_LONG:
383                                case MUL_DOUBLE:
384                                        main.addPrimitiveBinaryInstruction(
385                                                q.getOperator(),
386                                                Module.toResultTypeKind(q.getOperator()),
387                                                Module.toArgument1TypeKind(q.getOperator()),
388                                                q.getArgument1(),
389                                                q.getArgument2(),
390                                                q.getResult());
391                                        break;
392                                case DIV_LONG:
393                                case DIV_DOUBLE:
394                                        main.addPrimitiveBinaryCall(
395                                                q.getOperator(),
396                                                Module.toResultTypeKind(q.getOperator()),
397                                                Module.toArgument1TypeKind(q.getOperator()),
398                                                q.getArgument1(),
399                                                q.getArgument2(),
400                                                q.getResult());
401                                        break;
402                                case NOT_BOOLEAN:
403                                        main.addBooleanNot(Module.toIRBoolean(q.getArgument1()), q.getResult());
404                                        break;
405                                case OR_BOOLEAN:
406                                case AND_BOOLEAN:
407                                        main.addPrimitiveBinaryInstruction(
408                                                        q.getOperator(),
409                                                        Type.Kind.BOOLEAN,
410                                                        Type.Kind.BOOLEAN,
411                                                        Module.toIRBoolean(q.getArgument1()),
412                                                        Module.toIRBoolean(q.getArgument2()),
413                                                        q.getResult());
414                                        break;
415                                case CONCAT_STRING:
416                                        main.addPrimitiveBinaryCall(
417                                                q.getOperator(),
418                                                Type.Kind.STRING,
419                                                Type.Kind.STRING,
420                                                Module.toIRString(q.getArgument1()),
421                                                Module.toIRString(q.getArgument2()),
422                                                q.getResult());
423                                        break;
424
425                                /* Comparisons */
426                                case COMPARE_LONG_E:
427                                case COMPARE_LONG_G:
428                                case COMPARE_LONG_L:
429                                case COMPARE_LONG_GE:
430                                case COMPARE_LONG_LE:
431                                        main.addPrimitiveBinaryInstruction(
432                                                q.getOperator(),
433                                                Type.Kind.BOOLEAN,
434                                                Type.Kind.LONG,
435                                                q.getArgument1(),
436                                                q.getArgument2(),
437                                                q.getResult());
438                                        break;
439
440                                case COMPARE_DOUBLE_E:
441                                case COMPARE_DOUBLE_G:
442                                case COMPARE_DOUBLE_L:
443                                case COMPARE_DOUBLE_GE:
444                                case COMPARE_DOUBLE_LE:
445                                        main.addPrimitiveBinaryInstruction(
446                                                q.getOperator(),
447                                                Type.Kind.BOOLEAN,
448                                                Type.Kind.DOUBLE,
449                                                q.getArgument1(),
450                                                q.getArgument2(),
451                                                q.getResult());
452                                        break;
453
454                                /* Control flow */
455                                case RETURN:
456                                        main.addReturn(q.getArgument1());
457                                        break;
458                                case LABEL:
459                                        main.addLabel(q.getArgument1());
460                                        break;
461                                case BRANCH:
462                                        main.addBranch(q.getArgument1(),
463                                                        q.getArgument2(),
464                                                        q.getResult());
465                                        break;
466
467                                /* Print */
468                                case PRINT_STRING:
469                                        main.addPrint(Module.toIRString(q.getArgument1()), Type.Kind.STRING);
470                                        break;
471                        }
472                }
473
474                /* Write the generated LLVM IR code */
475                out.print(main.getCode());
476
477                /* Write handler for uncaught exceptions */
478                out.print(this.llvm_uncaught);
479
480                /* Write end for main function */
481                out.println("}");
482
483                /* Finish writing */
484                out.close();
485
486                /* Convert written target code to readable input format */
487                Map<String,InputStream> map = new HashMap<String,InputStream>();
488                map.put(baseFileName + ".ll", new ByteArrayInputStream(outStream.toByteArray()));
489
490                /* Return the map containing the readable target code */
491                return map;
492        }
