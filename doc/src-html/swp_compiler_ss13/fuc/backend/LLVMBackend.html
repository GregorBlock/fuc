001package swp_compiler_ss13.fuc.backend;
002
003import swp_compiler_ss13.common.backend.Backend;
004import swp_compiler_ss13.common.backend.BackendException;
005import swp_compiler_ss13.common.backend.Quadruple;
006import swp_compiler_ss13.common.types.Type;
007
008import java.io.ByteArrayInputStream;
009import java.io.ByteArrayOutputStream;
010import java.io.InputStream;
011import java.io.PrintWriter;
012import java.io.BufferedReader;
013import java.io.InputStreamReader;
014import java.io.IOException;
015import java.util.HashMap;
016import java.util.List;
017import java.util.Map;
018
019/**
020 * This implements a backend for LLVM IR.
021 *
022 */
023public class LLVMBackend implements Backend
024{
025        /**
026         * Contains the preamble for any generated
027         * LLVM IR program; this preamble contains
028         * declarations of stdlib C functions used
029         * (e.g. printf for output to stdout),
030         * declarations of the external C++ functions
031         * used for runtime exceptions, the list
032         * of currently implemented (custom) exceptions
033         * and LLVM IR functions used by generated LLVM
034         * IR code for various other purposes, such
035         * as conversion functions and functions
036         * used to safely envelop volatile LLVM IR
037         * instructions that may lead to segfaults
038         * for bad arguments, throwing exceptions instead.
039         *
040         */
041        public final String llvm_preamble;
042        /**
043         * Contains the default catch for all thrown exceptions,
044         * so that instead of crashing with verbosity dependent
045         * on both the operating system and the respective
046         * implementation of the C++ standard library, information
047         * about the exception causing the program crash is
048         * given in a uniform fashion.
049         * This is achieved by appending the code for this
050         * default catch to the main function after it has been
051         * generated.
052         *
053         */
054        public final String llvm_uncaught;
055
056        /**
057         * Creates a new <code>LLVMBackend</code> instance
058         * and sets up all necessary variables.
059         *
060         */
061        public LLVMBackend() {
062                /* Load the preamble containing
063                   functionality used in generated IR code */
064                StringBuilder out = new StringBuilder();
065                try {
066                        BufferedReader reader = new BufferedReader(
067                                new InputStreamReader(
068                                        this.getClass().getResourceAsStream("preamble.ll")));
069                        String line;
070                        while((line = reader.readLine()) != null) {
071                                out.append(line + "\n");
072                        }
073                }
074                catch(IOException e) { }
075
076                this.llvm_preamble = out.toString();
077
078                /* Load the exception handler for uncaught
079                   exceptions. */
080                out = new StringBuilder();
081                try {
082                        BufferedReader reader = new BufferedReader(
083                                new InputStreamReader(
084                                        this.getClass().getResourceAsStream("uncaught.ll")));
085                        String line;
086                        while((line = reader.readLine()) != null) {
087                                out.append("  " + line + "\n");
088                        }
089                }
090                catch(IOException e) { }
091
092                this.llvm_uncaught = "\n" + out.toString();
093        }
094
095        /**
096         * This function generates LLVM IR code for
097         * given three address code.
098         *
099         * @param tac the three address code
100         * @return the generated LLVM IR code.
101         */
102        @Override
103        public Map<String, InputStream> generateTargetCode(String baseFileName, List<Quadruple> tac) throws BackendException {
104                ByteArrayOutputStream outStream = new ByteArrayOutputStream();
105                PrintWriter out = new PrintWriter(outStream);
106
107                Module m = new Module(out);
108
109                /* Write preamble */
110                out.println(this.llvm_preamble);
111
112                /* Write begin for main function */
113                out.println("define i64 @main() {");
114
115                for(Quadruple q : tac)
116                {
117                        switch(q.getOperator())
118                        {
119                                /* Variable declaration */
120                                case DECLARE_LONG:
121                                        m.addPrimitiveDeclare(
122                                                Type.Kind.LONG,
123                                                q.getResult(),
124                                                q.getArgument1());
125                                        break;
126                                case DECLARE_DOUBLE:
127                                        m.addPrimitiveDeclare(
128                                                Type.Kind.DOUBLE,
129                                                q.getResult(),
130                                                q.getArgument1());
131                                        break;
132                                case DECLARE_BOOLEAN:
133                                        m.addPrimitiveDeclare(
134                                                Type.Kind.BOOLEAN,
135                                                q.getResult(),
136                                                Module.toIRBoolean(q.getArgument1()));
137                                        break;
138                                case DECLARE_STRING:
139                                        m.addPrimitiveDeclare(
140                                                Type.Kind.STRING,
141                                                q.getResult(),
142                                                Module.toIRString(q.getArgument1()));
143                                        break;
144
145                                /* Type conversion */
146                                case LONG_TO_DOUBLE:
147                                        m.addPrimitiveConversion(
148                                                Type.Kind.LONG,
149                                                q.getArgument1(),
150                                                Type.Kind.DOUBLE,
151                                                q.getResult());
152                                        break;
153                                case DOUBLE_TO_LONG:
154                                        m.addPrimitiveConversion(
155                                                Type.Kind.DOUBLE,
156                                                q.getArgument1(),
157                                                Type.Kind.LONG,
158                                                q.getResult());
159                                        break;
160
161                                /* Unindexed copy */
162                                case ASSIGN_LONG:
163                                        m.addPrimitiveAssign(
164                                                Type.Kind.LONG,
165                                                q.getResult(),
166                                                q.getArgument1());
167                                        break;
168                                case ASSIGN_DOUBLE:
169                                        m.addPrimitiveAssign(
170                                                Type.Kind.DOUBLE,
171                                                q.getResult(),
172                                                q.getArgument1());
173                                        break;
174                                case ASSIGN_BOOLEAN:
175                                        m.addPrimitiveAssign(
176                                                Type.Kind.BOOLEAN,
177                                                q.getResult(),
178                                                Module.toIRBoolean(q.getArgument1()));
179                                        break;
180                                case ASSIGN_STRING:
181                                        m.addPrimitiveAssign(
182                                                Type.Kind.STRING,
183                                                q.getResult(),
184                                                Module.toIRString(q.getArgument1()));
185                                        break;
186
187                                /* Arithmetic */
188                                case ADD_LONG:
189                                        m.addPrimitiveBinaryInstruction(
190                                                q.getOperator(),
191                                                Type.Kind.LONG,
192                                                Type.Kind.LONG,
193                                                q.getArgument1(),
194                                                q.getArgument2(),
195                                                q.getResult());
196                                        break;
197                                case ADD_DOUBLE:
198                                        m.addPrimitiveBinaryInstruction(
199                                                q.getOperator(),
200                                                Type.Kind.DOUBLE,
201                                                Type.Kind.DOUBLE,
202                                                q.getArgument1(),
203                                                q.getArgument2(),
204                                                q.getResult());
205                                        break;
206                                case SUB_LONG:
207                                        m.addPrimitiveBinaryInstruction(
208                                                q.getOperator(),
209                                                Type.Kind.LONG,
210                                                Type.Kind.LONG,
211                                                q.getArgument1(),
212                                                q.getArgument2(),
213                                                q.getResult());
214                                        break;
215                                case SUB_DOUBLE:
216                                        m.addPrimitiveBinaryInstruction(
217                                                q.getOperator(),
218                                                Type.Kind.DOUBLE,
219                                                Type.Kind.DOUBLE,
220                                                q.getArgument1(),
221                                                q.getArgument2(),
222                                                q.getResult());
223                                        break;
224                                case MUL_LONG:
225                                        m.addPrimitiveBinaryInstruction(
226                                                q.getOperator(),
227                                                Type.Kind.LONG,
228                                                Type.Kind.LONG,
229                                                q.getArgument1(),
230                                                q.getArgument2(),
231                                                q.getResult());
232                                        break;
233                                case MUL_DOUBLE:
234                                        m.addPrimitiveBinaryInstruction(
235                                                q.getOperator(),
236                                                Type.Kind.DOUBLE,
237                                                Type.Kind.DOUBLE,
238                                                q.getArgument1(),
239                                                q.getArgument2(),
240                                                q.getResult());
241                                        break;
242                                case DIV_LONG:
243                                        m.addPrimitiveBinaryCall(
244                                                q.getOperator(),
245                                                Type.Kind.LONG,
246                                                Type.Kind.LONG,
247                                                q.getArgument1(),
248                                                q.getArgument2(),
249                                                q.getResult());
250                                        break;
251                                case DIV_DOUBLE:
252                                        m.addPrimitiveBinaryCall(
253                                                q.getOperator(),
254                                                Type.Kind.DOUBLE,
255                                                Type.Kind.DOUBLE,
256                                                q.getArgument1(),
257                                                q.getArgument2(),
258                                                q.getResult());
259                                        break;
260                                case NOT_BOOLEAN:
261                                        m.addBooleanNot(Module.toIRBoolean(q.getArgument1()), q.getResult());
262                                        break;
263                                case OR_BOOLEAN:
264                                        m.addPrimitiveBinaryInstruction(
265                                                        q.getOperator(),
266                                                        Type.Kind.BOOLEAN,
267                                                        Type.Kind.BOOLEAN,
268                                                        Module.toIRBoolean(q.getArgument1()),
269                                                        Module.toIRBoolean(q.getArgument2()),
270                                                        q.getResult());
271                                        break;
272                                case AND_BOOLEAN:
273                                        m.addPrimitiveBinaryInstruction(
274                                                        q.getOperator(),
275                                                        Type.Kind.BOOLEAN,
276                                                        Type.Kind.BOOLEAN,
277                                                        Module.toIRBoolean(q.getArgument1()),
278                                                        Module.toIRBoolean(q.getArgument2()),
279                                                        q.getResult());
280                                        break;
281
282                                /* Comparisons */
283                                case COMPARE_LONG_E:
284                                m.addPrimitiveBinaryInstruction(
285                                                q.getOperator(),
286                                                Type.Kind.BOOLEAN,
287                                                Type.Kind.LONG,
288                                                q.getArgument1(),
289                                                q.getArgument2(),
290                                                q.getResult());
291                                break;
292                                case COMPARE_LONG_G:
293                                        m.addPrimitiveBinaryInstruction(
294                                                        q.getOperator(),
295                                                        Type.Kind.LONG,
296                                                        Type.Kind.BOOLEAN,
297                                                        q.getArgument1(),
298                                                        q.getArgument2(),
299                                                        q.getResult());
300                                        break;
301                                case COMPARE_LONG_L:
302                                        m.addPrimitiveBinaryInstruction(
303                                                        q.getOperator(),
304                                                        Type.Kind.BOOLEAN,
305                                                        Type.Kind.LONG,
306                                                        q.getArgument1(),
307                                                        q.getArgument2(),
308                                                        q.getResult());
309                                        break;
310                                case COMPARE_LONG_GE:
311                                        m.addPrimitiveBinaryInstruction(
312                                                        q.getOperator(),
313                                                        Type.Kind.BOOLEAN,
314                                                        Type.Kind.LONG,
315                                                        q.getArgument1(),
316                                                        q.getArgument2(),
317                                                        q.getResult());
318                                        break;
319                                case COMPARE_LONG_LE:
320                                        m.addPrimitiveBinaryInstruction(
321                                                        q.getOperator(),
322                                                        Type.Kind.BOOLEAN,
323                                                        Type.Kind.LONG,
324                                                        q.getArgument1(),
325                                                        q.getArgument2(),
326                                                        q.getResult());
327                                        break;
328                                case COMPARE_DOUBLE_E:
329                                        m.addPrimitiveBinaryInstruction(
330                                                        q.getOperator(),
331                                                        Type.Kind.BOOLEAN,
332                                                        Type.Kind.DOUBLE,
333                                                        q.getArgument1(),
334                                                        q.getArgument2(),
335                                                        q.getResult());
336                                        break;
337                                case COMPARE_DOUBLE_G:
338                                        m.addPrimitiveBinaryInstruction(
339                                                        q.getOperator(),
340                                                        Type.Kind.BOOLEAN,
341                                                        Type.Kind.DOUBLE,
342                                                        q.getArgument1(),
343                                                        q.getArgument2(),
344                                                        q.getResult());
345                                        break;
346                                case COMPARE_DOUBLE_L:
347                                        m.addPrimitiveBinaryInstruction(
348                                                        q.getOperator(),
349                                                        Type.Kind.BOOLEAN,
350                                                        Type.Kind.DOUBLE,
351                                                        q.getArgument1(),
352                                                        q.getArgument2(),
353                                                        q.getResult());
354                                        break;
355                                case COMPARE_DOUBLE_GE:
356                                        m.addPrimitiveBinaryInstruction(
357                                                        q.getOperator(),
358                                                        Type.Kind.BOOLEAN,
359                                                        Type.Kind.DOUBLE,
360                                                        q.getArgument1(),
361                                                        q.getArgument2(),
362                                                        q.getResult());
363                                        break;
364                                case COMPARE_DOUBLE_LE:
365                                        m.addPrimitiveBinaryInstruction(
366                                                        q.getOperator(),
367                                                        Type.Kind.BOOLEAN,
368                                                        Type.Kind.DOUBLE,
369                                                        q.getArgument1(),
370                                                        q.getArgument2(),
371                                                        q.getResult());
372                                        break;
373
374                                /* Control flow */
375                                case RETURN:
376                                        m.addMainReturn(q.getArgument1());
377                                        break;
378                                case LABEL:
379                                        m.addLabel(q.getArgument1());
380                                        break;
381                                case BRANCH:
382                                        m.addBranch(q.getArgument1(),
383                                                        q.getArgument2(),
384                                                        q.getResult());
385                                        break;
386
387                                /* Print */
388                                case PRINT_LONG:
389                                        m.addPrint(q.getArgument1(), Type.Kind.LONG);
390                                        break;
391                                case PRINT_DOUBLE:
392                                        m.addPrint(q.getArgument1(), Type.Kind.DOUBLE);
393                                        break;
394                                case PRINT_BOOLEAN:
395                                        m.addPrint(Module.toIRBoolean(q.getArgument1()), Type.Kind.BOOLEAN);
396                                        break;
397                                case PRINT_STRING:
398                                        m.addPrint(Module.toIRString(q.getArgument1()), Type.Kind.STRING);
399                                        break;
400                        }
401                }
402
403                /* Write return 0 if last operation is not a return */
404                if(tac.size() == 0 || tac.get(tac.size() - 1).getOperator() != Quadruple.Operator.RETURN)
405                {
406                        out.println("  ret i64 0");
407                }
408
409                /* Write handler for uncaught exceptions */
410                out.print(this.llvm_uncaught);
411
412                /* Write end for main function */
413                out.println("}");
414
415                /* Finish writing */
416                out.close();
417
418                /* Convert written target code to readable input format */
419                Map<String,InputStream> map = new HashMap<String,InputStream>();
420                map.put(baseFileName+".ll", new ByteArrayInputStream(outStream.toByteArray()));
421
422                /* Return the map containing the readable target code */
423                return map;
424        }
425}
