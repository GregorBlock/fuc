015/**
016 * This class allows for the generation of an LLVM IR module.
017 * Each methode that begins with "add" generates LLVM IR code
018 * and writes it to the <code>PrintWriter</code> <code>out</code>.
019 *
020 */
021public class Module
022{
023        /**
024         * A list of all the string literals that
025         * exist in this module where each element
026         * describes a string literal's number with its
027         * position in the list and that string literal's
028         * length with the value of the element.
029         *
030         */
031        private ArrayList<Integer> stringLiterals;
032
033        /**
034         * This maps a variable's name (not its
035         * identifier, which is %{name}) to the
036         * number of times it has been used in
037         * the module already.
038         * This is necessary because LLVM's IR
039         * allows only for static single assignment
040         * (SSA), wich means we have to store the
041         * variable's actual value as the destination
042         * of a static pointer and generate a new
043         * 'use' variable (%{name}.{use_number})
044         * for every time the variable is read
045         * from or written to and route the pointer
046         * access through that use variable.
047         *
048         */
049        private Map<String,Integer> variableUseCount;
050        private Set<String> variableIsReference;
051
052        /**
053         * The <code>PrintWriter</code> which
054         * the generated LLVM IR is written with.
055         *
056         */
057        private PrintWriter out;
058
059        /**
060         * Creates a new <code>Module</code> instance.
061         *
062         * @param out <code>PrintWriter</code> used
063         * to write the LLVM IR
064         */
065        public Module(PrintWriter out)
066        {
067                reset(out);
068        }
069
070        /**
071         * Completely resets this <code>Module</code>,
072         * so it can be reused.
073         *
074         * @param out <code>PrintWriter</code> used
075         * to write the LLVM IR
076         */
077        public void reset(PrintWriter out)
078        {
079                stringLiterals = new ArrayList<Integer>();
080                variableUseCount = new HashMap<String,Integer>();
081                variableIsReference = new HashSet<String>();
082
083                /* Add fake temporary variable */
084                variableUseCount.put(".tmp", 0);
085
086                this.out = out;
087        }

089        /**
090         * Prefix the LLVM IR with two spaces
091         * (as it may only exist inside the main
092         * function for now) and write it with <code>out</code>
093         *
094         * @param code a <code>String</code> value
095         */
096        private void gen(String code)
097        {
098                out.println("  " + code);
099        }

101        /**
102         * Get a string literal's identifier by its
103         * id (i.e. its position in the list of string literals).
104         *
105         * @param id the string literal's id
106         * @return the string literal's identifier

108        private static String getStringLiteralIdentifier(int id)

110                return "%.string_" + String.valueOf(id);

113        /**
114         * Get a string literal's type by its
115         * id (i.e. [i8 * {length}], where the length
116         * is the string literal's length).
117         *
118         * @param id an <code>int</code> value
119         * @return a <code>String</code> value

123                return "[" + String.valueOf(stringLiterals.get(id)) + " x i8]";

126        /**
127         * Get the LLVM IR type corresponding to
128         * a type from the three address code.
129         *
130         * @param type the type from the three address code
131         * @return the corresponding LLVM IR type
132         */
133        private String getIRType(Kind type) {
134                return getIRType(type,false);
135        }
136        private String getIRType(Kind type, boolean as_ptr)
137        {
138                String irType = "";
139                switch(type)
140                {
141                        case LONG:
142                                /* Longs are 64bit signed integers */
143                                irType = "i64";
144                                break;
145                        case DOUBLE:
146                                /* Doubles are 64bit IEEE floats */
147                                irType = "double";
148                                break;
149                        case BOOLEAN:
150                                /* Booleans are 8bit signed integers */
151                                irType = "i1";
152                                break;
153                        case STRING:
154                                /* Strings are each a pointer to a string literal
155                                   (which itself is a pointer to a memory segment
156                                   of signed integers).*/
157                                irType = "i8*";
158                                break;
159                }
160
161                return irType + (as_ptr ? "*" : "");
162        }
163
164        /**
165         * Get the LLVM IR type corresponding to
166         * an *array* type from the three address code.
167         *
168         * @param type the final type of the array, eg: double[][] --> double
169         * @param sizes a list of sizes of the array dimensions, eg:
170         *          double x[5][10][7] --> [5,10,7]
171         * @return the corresponding LLVM IR type for the array
172         */
173        private String getIRAType(Kind type, List<Integer> dimensions) {
174                return getIRAType(type,dimensions,false);
175        }
176        private String getIRAType(Kind type, List<Integer> dimensions, boolean as_ptr)
177        {
178                String irType = "";
179                // write dimensions
180                for (int d : dimensions)
181                        irType = irType + "[" + d + " x ";
182                // write actual type
183                irType = irType + getIRType(type);
184                // write closing brakets
185                for (int d : dimensions)
186                        irType += "]";
187                irType.trim();
188                return irType + (as_ptr ? "*" : "");
189        }

195        /**
196         * Gets the LLVM IR instruction for a binary TAC operator
197         *
198         * @param operator the binary TAC operator
199         * @return the corresponding LLVM IR instruction
200         */
201        private String getIRBinaryInstruction(Quadruple.Operator operator)
202        {
203                String irInst = "";
204
205                switch(operator)
206                {
207                        /* Arithmetic */
208                        case ADD_LONG:
209                                irInst = "add";
210                                break;
211                        case ADD_DOUBLE:
212                                irInst = "fadd";
213                                break;
214                        case SUB_LONG:
215                                irInst = "sub";
216                                break;
217                        case SUB_DOUBLE:
218                                irInst = "fsub";
219                                break;
220                        case MUL_LONG:
221                                irInst = "mul";
222                                break;
223                        case MUL_DOUBLE:
224                                irInst = "fmul";
225                                break;
226                        case DIV_LONG:
227                                irInst = "sdiv";
228                                break;
229                        case DIV_DOUBLE:
230                                irInst = "fdiv";
231                                break;
232
233                        /* Boolean Arithmetic */
234                        case OR_BOOLEAN:
235                                irInst = "or";
236                                break;
237                        case AND_BOOLEAN:
238                                irInst = "and";
239                                break;
240
241                        /* Comparisons */
242                        case COMPARE_LONG_E:
243                                irInst = " icmp eq";
244                                break;
245                        case COMPARE_LONG_G:
246                                irInst = " icmp sgt";
247                        break;
248                        case COMPARE_LONG_L:
249                                irInst = " icmp slt";
250                                break;
251                        case COMPARE_LONG_GE:
252                                irInst = " icmp sge";
253                        break;
254                        case COMPARE_LONG_LE:
255                                irInst = " icmp sle";
256                        break;
257
258                        case COMPARE_DOUBLE_E:
259                                irInst = " fcmp oeq";
260                        break;
261                        case COMPARE_DOUBLE_G:
262                                irInst = " fcmp ogt";
263                        break;
264                        case COMPARE_DOUBLE_L:
265                                irInst = " fcmp olt";
266                        break;
267                        case COMPARE_DOUBLE_GE:
268                                irInst = " fcmp oge";
269                        break;
270                        case COMPARE_DOUBLE_LE:
271                                irInst = " fcmp ole";
272                        break;
273                }
274
275                return irInst;
276        }

302        /**
303         * Gets a unique use identifier for
304         * a variable; no two calls will return
305         * the same use identifier.
306         *
307         * @param variable the variable's name
308         * @return a free use identifier for the variable
309         * @exception BackendException if an error occurs
310         */
311        private String getUseIdentifierForVariable(String variable) throws BackendException {
312                int ssa_suffix = 0;
313                try {
314                        ssa_suffix = variableUseCount.get(variable);
315                } catch (NullPointerException e) {
316                        throw new BackendException("Use of undeclared variable");
317                }
318                variableUseCount.put(variable, ssa_suffix + 1);
319                return "%" + variable + "." + String.valueOf(ssa_suffix);
320        }

322        /**
323         * Convert a three adress code boolean
324         * to a LLVM IR boolean.
325         * All other public functions expect booleans
326         * to be in the LLVM IR format.
327         *
328         * @param bool a TAC boolean
329         * @return the converted LLVM IR boolean
330         */
331        public static String toIRBoolean(String bool)
332        {
333                if(bool.equals("#FALSE"))
334                {
335                        return "#0";
336                }
337                else if(bool.equals("#TRUE"))
338                {
339                        return "#1";
340                }
341                else
342                {
343                        return bool;
344                }
345        }

347        /**
348         * Convert a three adress code string
349         * to a LLVM IR string.
350         * All other public functions expect strings
351         * to be in the LLVM IR format.
352         *
353         * @param str a TAC string
354         * @return the converted LLVM IR string
355         */
356        public static String toIRString(String str)
357        {
358                String irString = "";
359
360                /* Unescape special characters */
361                irString = str.replace("\\\"", "\"").
362                        replace("\\r", "\r").
363                        replace("\\n", "\n").
364                        replace("\\t", "\t").
365                        replace("\\0", "\0");
366
367                return irString;
368        }

370        /**
371         * Generates a new string literal from a <code>String</code>
372         * value and returns its id (i.e. its position in the
373         * list of string literals <code>stringLiterals</code>).
374         *
375         * @param literal the string to use
376         * @return the new string literal's id
377         */
378        private Integer addStringLiteral(String literal) throws BackendException
379        {
380                int length = 0;
381
382                try
383                {
384                        literal = literal.substring(1, literal.length() - 1);
385                        byte[] utf8 = literal.getBytes("UTF-8");
386                        literal = Arrays.toString(utf8).replaceAll("(-?)([0-9]+)(,?)","i8 $1$2$3");
387
388                        if(utf8.length > 0)
389                        {
390                                literal = literal.replace("]",", i8 0]");
391                        }
392                        else
393                        {
394                                literal = "[i8 0]";
395                        }
396
397                        length = utf8.length + 1;
398                }
399                catch(UnsupportedEncodingException e)
400                {
401                        throw new BackendException("LLVM backend cannot handle strings without utf8 support");
402                }
403
404                int id = stringLiterals.size();
405                stringLiterals.add(length);
406
407                String type = getStringLiteralType(id);
408
409                String identifier = getStringLiteralIdentifier(id);
410
411                gen(identifier + " = alloca " + type);
412                gen("store " + type + " " + literal + ", " + type + "* " + identifier);
413
414                return id;
415        }

417        /**
418         * Sets a variable (which must be of the string primitive type)
419         * to contain a pointer to a string literal.
420         *
421         * @param variable the variable's name
422         * @param literalID the string literal's id
423         * @return the used "use identifier" for variable
424         * @exception BackendException if an error occurs
425         */
426        private String addLoadStringLiteral(String variable, int literalID) throws BackendException {
427                String variableIdentifier = "%" + variable;
428                String variableUseIdentifier = getUseIdentifierForVariable(variable);
429                String literalIdentifier = getStringLiteralIdentifier(literalID);
430                String literalType = getStringLiteralType(literalID);
431
432                gen(variableUseIdentifier + " = getelementptr " + literalType + "* " + literalIdentifier + ", i64 0, i64 0");
433                gen("store i8* " + variableUseIdentifier + ", i8** " + variableIdentifier);
434                return variableUseIdentifier;
435        }

488        /**
489         * Assigns either a constant value, or the value of
490         * one variable (<code>src</code>) to another variable
491         * (<code>dst</code>). Only assignments for identical
492         * types are allowed.
493         *
494         * @param type the type of the assignment
495         * @param dst the destination variable's name
496         * @param src the source constant or the source variable's name
497         * @exception BackendException if an error occurs
498         */
499        public void addPrimitiveAssign(Kind type, String dst, String src) throws BackendException {
500                boolean constantSrc = false;
501                if(src.charAt(0) == '#')
502                {
503                        src = src.substring(1);
504                        constantSrc = true;
505                }
506
507                String irType = getIRType(type);
508                String dstIdentifier = "%" + dst;
509
510                if(constantSrc)
511                {
512                        if(type == Kind.STRING)
513                        {
514                                int id = addStringLiteral(src);
515                                addLoadStringLiteral(dst, id);
516                        }
517                        else
518                        {
519                                gen("store " + irType + " " + src + ", " + irType + "* " + dstIdentifier);
520                        }
521                }
522                else
523                {
524                        String srcUseIdentifier = getUseIdentifierForVariable(src);
525                        String srcIdentifier = "%" + src;
526                        gen(srcUseIdentifier + " = load " + irType + "* " + srcIdentifier);
527                        gen("store " + irType + " " + srcUseIdentifier + ", " + irType + "* " + dstIdentifier);
528                }
529        }

531        /** 
532         *  Conditionally loads a constant or an identifier. If given a constant, just the leading # character
533         *  is removed. If given a variable (anything without leading # character),
534         *  it is being loaded and the temporary register name is returned.
535         *      
536         *      @param constantOrIdentifier Name of a constant or identifier.
537         *  @returns name of usable primitive
538         */
539        private String cdl(String constantOrIdentifier, Kind type) throws BackendException {
540                if(constantOrIdentifier.charAt(0) == '#')
541                        // constant
542                        if(type==Kind.STRING)
543                        {
544                                int literalID = addStringLiteral(constantOrIdentifier.substring(1));
545                                String literalIdentifier = getStringLiteralIdentifier(literalID);
546                                String literalType = getStringLiteralType(literalID);
547                                String variableUseIdentifier = getUseIdentifierForVariable(".tmp");
548                                gen(variableUseIdentifier + " = getelementptr " + literalType + "* " + literalIdentifier + ", i64 0, i64 0");
549
550                                return variableUseIdentifier;
551                        }
552                        else
553                                return constantOrIdentifier.substring(1);
554                else {
555                        // identifier
556                        String id = getUseIdentifierForVariable(constantOrIdentifier);
557                        gen(id + " = load " + getIRType(type,true) + " %"+constantOrIdentifier);
558                        return id;
559                }
560        }

562        public void addArrayGet(String arrayName, int idx, String dst) throws BackendException {
563                ArrayType arType = arNamespace.get(arrayName);
564                String arrayTypeStr = getIRAType(arType.storage_type,arType.dimensions);
565                String realsource;
566                // if source is a reference, we have to consider multiple assignments to it due to SSA
567                if(variableIsReference.contains(arrayName))
568                        realsource = "%" + arrayName + "." + (variableUseCount.get(arrayName)-1);
569                else
570                    realsource = "%" + arrayName;
571                //  if the array has more than one level, dst must be a reference
572                if(arType.dimensions.size() > 1) {
573                        // reference mode (returns a reference)
574                        // in reference mode, we just store the getelementptr result into dst
575                        gen(getUseIdentifierForVariable(dst) + " = getelementptr " + arrayTypeStr + "* " + realsource + ", i64 0" + ", i64 " + idx);
576                        // it is important to note, that references's types are JUST like array types: pointers to arrays
577                        //  the next time we do a GET_ARRAY_* instruction, we need the the encapsulated array type.
578                        //  eg: if we dereference [2 x [4 x i64]] here, we need [4 x i64] next time we use the reference.
579                        // effectively, after each ARRAY_GET_REFERENCE operation we have to remove the first dimension from
580                        //  the array dimensions list:
581                        
582                        // copy old list (shallow is okay, we never change the Integers)
583                        ArrayType newArrayType = new ArrayType(new LinkedList<Integer>(arType.dimensions),
584                                                               arType.storage_type);
585                        newArrayType.dimensions.remove(0);
586                        arNamespace.put(dst, newArrayType);
587                }
588                else {
589                        // direct access
590                        // construct getelementptr instruction
591                        String ptrId = getUseIdentifierForVariable(".tmp");
592                        gen(ptrId + " = getelementptr " + arrayTypeStr + "* " + realsource + ", i64 0" + ", i64 " + idx);
593                        // retrieve value from pointer via stack load instruction
594                        String storageTypeStr = getIRType(arType.storage_type);
595                        String valId = getUseIdentifierForVariable(".tmp");
596                        gen(valId + " = load " + storageTypeStr + "* " + ptrId);
597                        gen("store "+storageTypeStr+" "+valId+", "+storageTypeStr+"* %"+dst);
598                }
599        }

601        public void addArraySet(String arrayName, int idx, String src) throws BackendException {
602                // if source is a reference, we have to consider multiple assignments to it due to SSA
603                String realarray;
604                if(variableIsReference.contains(arrayName))
605                        realarray = "%" + arrayName + "." + (variableUseCount.get(arrayName)-1);
606                else
607                realarray = "%" + arrayName;
608                ArrayType arType = arNamespace.get(arrayName);
609                if(arType.dimensions.size() > 1)
610                        throw new BackendException("you cannot use a multidimensional array with an ARRAY_SET_* operation");
611                // construct getelementptr instruction
612                String ptrId = getUseIdentifierForVariable(".tmp");
613                String arrayTypeStr = getIRAType(arType.storage_type,arType.dimensions);
614                gen(ptrId + " = getelementptr " + arrayTypeStr + "* " + realarray + ", i64 0" + ", i64 " + idx);
615                // store value
616                String storageTypeStr = getIRType(arType.storage_type);
617                gen("store "+storageTypeStr+" "+cdl(src,arType.storage_type)+", "+storageTypeStr+"* "+ptrId);                           
618        }

620        static public Kind QuadTypeToKind(Quadruple.Operator op) throws BackendException {
621                switch (op) {
622                case DECLARE_DOUBLE:
623                case ARRAY_GET_DOUBLE:
624                case ARRAY_SET_DOUBLE:
625                        return Kind.DOUBLE;
626                case DECLARE_BOOLEAN:
627                case ARRAY_GET_BOOLEAN:
628                case ARRAY_SET_BOOLEAN:
629                        return Kind.BOOLEAN;
630                case DECLARE_LONG:
631                case ARRAY_GET_LONG:
632                case ARRAY_SET_LONG:
633                        return Kind.LONG;
634                }
635                throw new BackendException("QuadTypeToKind: unimplemented OP -> unknown type");
636        }

638        /**
639         * Assigns the value of one variable (<code>src</code>)
640         * to another variable (<code>dst</code>), where their types
641         * must only be convertible, not identical.
642         *
643         * @param srcType the source variable's type
644         * @param src the source variable's name
645         * @param dstType the destination variable's type
646         * @param dst the destination variable's name
647         * @exception BackendException if an error occurs
648         */
649        public void addPrimitiveConversion(Kind srcType,
650                                           String src,
651                                           Kind dstType,
652                                           String dst) throws BackendException {
653                String srcUseIdentifier = getUseIdentifierForVariable(src);
654                String dstUseIdentifier = getUseIdentifierForVariable(dst);
655                String srcIdentifier = "%" + src;
656                String dstIdentifier = "%" + dst;
657
658                if((srcType == Kind.LONG) && (dstType == Kind.DOUBLE))
659                {
660                        gen(srcUseIdentifier + " = load i64* " + srcIdentifier);
661                        gen(dstUseIdentifier + " = sitofp i64 " + srcUseIdentifier + " to double");
662                        gen("store double " + dstUseIdentifier + ", double* " + dstIdentifier);
663                }
664                else if((srcType == Kind.DOUBLE) && (dstType == Kind.LONG))
665                {
666                        gen(srcUseIdentifier + " = load double* " + srcIdentifier);
667                        gen(dstUseIdentifier + " = fptosi double " + srcUseIdentifier + " to i64");
668                        gen("store i64 " + dstUseIdentifier + ", i64* " + dstIdentifier);
669                }
670        }

672        /**
673         * Adds a generic binary operation of two sources - each can
674         * either be a constant or a variable - the result
675         * of which will be stored in a variable (<code>dst</code>).
676         * All types must be identical.
677         *
678         * @param op the binary operation to add
679         * @param resultType  the result type of the binary operation
680         * @param argumentType the argument type of the binary operation
681         * @param lhs the constant or name of the variable on the left hand side
682         * @param rhs the constant or name of the variable on the right hand side
683         * @param dst the destination variable's name
684         * @exception BackendException if an error occurs
685         */
686        public void addPrimitiveBinaryInstruction(Quadruple.Operator op,
687                                                  Kind resultType,
688                                                  Kind argumentType,
689                                                  String lhs,
690                                                  String rhs,
691                                                  String dst) throws BackendException {
692                String irArgumentType = getIRType(argumentType);
693                String irResultType = getIRType(resultType);
694                String irInst = getIRBinaryInstruction(op);
695
696                if(lhs.charAt(0) == '#')
697                {
698                        lhs = lhs.substring(1);
699                }
700                else
701                {
702                        String lhsIdentifier = "%" + lhs;
703                        lhs = getUseIdentifierForVariable(lhs);
704                        gen(lhs + " = load " + irArgumentType + "* " + lhsIdentifier);
705                }
706
707                if(rhs.charAt(0) == '#')
708                {
709                        rhs = rhs.substring(1);
710                }
711                else
712                {
713                        String rhsIdentifier = "%" + rhs;
714                        rhs = getUseIdentifierForVariable(rhs);
715                        gen(rhs + " = load " + irArgumentType + "* " + rhsIdentifier);
716                }
717
718                String dstUseIdentifier = getUseIdentifierForVariable(dst);
719                String dstIdentifier = "%" + dst;
720
721                gen(dstUseIdentifier + " = " + irInst + " " + irArgumentType + " " + lhs + ", " + rhs);
722                gen("store " + irResultType + " " + dstUseIdentifier + ", " + irResultType + "* " + dstIdentifier);
723        }

725        /**
726         * Adds a call to a argument-homogenous binary functions
727         * where each argument can either be a constant or a variable;
728         * the result of the function will be stored in a variable (<code>dst</code>).
729         * Argument types must both be equal, but may differ
730         * from the result type.
731         *
732         * @param op the binary operation to add
733         * @param resultType the type of the binary operation's result
734         * @param argumentType the type of the binary operation's arguments
735         * @param lhs the constant or name of the variable on the left hand side
736         * @param rhs the constant or name of the variable on the right hand side
737         * @param dst the destination variable's name
738         * @exception BackendException if an error occurs
739         */
740        public void addPrimitiveBinaryCall(Quadruple.Operator op,
741                                           Kind resultType,
742                                           Kind argumentType,
743                                           String lhs,
744                                           String rhs,
745                                           String dst) throws BackendException {
746                String irArgumentType = getIRType(argumentType);
747                String irResultType = getIRType(resultType);
748                String irCall = getIRBinaryCall(op);
749
750                if(lhs.charAt(0) == '#')
751                {
752                        lhs = lhs.substring(1);
753                }
754                else
755                {
756                        String lhsIdentifier = "%" + lhs;
757                        lhs = getUseIdentifierForVariable(lhs);
758                        gen(lhs + " = load " + irArgumentType + "* " + lhsIdentifier);
759                }
760
761                if(rhs.charAt(0) == '#')
762                {
763                        rhs = rhs.substring(1);
764
765                        if((op == Quadruple.Operator.DIV_LONG) &&
766                           (Long.valueOf(rhs).equals(Long.valueOf(0))))
767                        {
768                                throw new BackendException("Division by zero");
769                        }
770
771                        if((op == Quadruple.Operator.DIV_DOUBLE) &&
772                           ((Double.valueOf(rhs).equals(Double.valueOf(0.0))) ||
773                            (Double.valueOf(rhs).equals(Double.valueOf(-0.0)))))
774                        {
775                                throw new BackendException("Division by zero");
776                        }
777                }
778                else
779                {
780                        String rhsIdentifier = "%" + rhs;
781                        rhs = getUseIdentifierForVariable(rhs);
782                        gen(rhs + " = load " + irArgumentType + "* " + rhsIdentifier);
783                }
784
785                String dstUseIdentifier = getUseIdentifierForVariable(dst);
786                String dstIdentifier = "%" + dst;
787
788                gen(dstUseIdentifier + " = invoke " + irResultType + " (" + irArgumentType + ", " + irArgumentType + ")* " +
789                    "@" + irCall + "(" + irArgumentType + " " + lhs + ", " + irArgumentType + " " + rhs + ") to label " + dstUseIdentifier + ".ok unwind label %UncaughtException");
790                gen(dstUseIdentifier.substring(1, dstUseIdentifier.length()) + ".ok:");
791                gen("store " + irResultType + " " + dstUseIdentifier + ", " + irResultType + "* " + dstIdentifier);
792        }

860         * Adds a branching instruction, which may either
861         * be unconditional (e.g. branch immediately) or
862         * depend on a boolean variable 'condition' (e.g. branch to
863         * one of two target labels depening on the boolean
864         * variable's value at runtime).
865         *
866         * @param target1 target label for unconditional branch, target
867         *                for the condition being true for conditional branch
868         * @param target2 <code>Quadruple.EmptyArgument</code> for unconditional
869         *                branch, target for the condition being false for
870         *                the conditional branch
871         * @param condition <code>Quadruple.EmptyArgument</code> for unconditional
872         *                  branch, the condition being tested for conditional branch
873         * @exception BackendException if an error occurs
874         */
875        public void addBranch(String target1, String target2, String condition) throws BackendException {
876                /* conditional branch */
877                if (!target2.equals(Quadruple.EmptyArgument)){
878                        String irType = getIRType(Kind.BOOLEAN);
879                        String conditionUseIdentifier = getUseIdentifierForVariable(condition);
880                        gen(conditionUseIdentifier + " = load " + irType + "* %" + condition);
881                        gen("br " + irType + " " + conditionUseIdentifier + ", label %" + target1 + ", label %"+ target2);
882                }
883                else {
884                        gen("br label %" + target1);
885                }
886        }

888        /**
889         * Adds a call to the printf function for
890         * the value, which may be either a constant
891         * or the name of a variable.
892         *
893         * @param value the value to be printed
894         * @param type the type of the value (boolean, long, double, string)
895         * @exception BackendException if an error occurs
896         */
897        public void addPrint(String value, Kind type) throws BackendException {
898                String irType = getIRType(type);
899                boolean constantSrc = false;
900
901                /* value is constant */
902                if(value.charAt(0) == '#')
903                {
904                        value = value.substring(1);
905                        constantSrc = true;
906                }
907                /* value is identifier */
908                else
909                {
910                        String valueIdentifier = "%" + value;
911                        value = getUseIdentifierForVariable(value);
912                        gen(value + " = load " + irType + "* " + valueIdentifier);
913                }
914
915                String temporaryIdentifier = "";
916
917                switch (type) {
918                        case BOOLEAN:
919                                temporaryIdentifier = getUseIdentifierForVariable(".tmp");
920                                gen(temporaryIdentifier + " = call i8* (" + irType + ")* @btoa(" + irType + " " + value + ")");
921                                break;
922                        case LONG:
923                                temporaryIdentifier = getUseIdentifierForVariable(".tmp");
924                                gen(temporaryIdentifier + " = call i8* (" + irType + ")* @ltoa(" + irType + " " + value + ")");
925                                break;
926                        case DOUBLE:
927                                temporaryIdentifier = getUseIdentifierForVariable(".tmp");
928                                gen(temporaryIdentifier + " = call i8* (" + irType + ")* @dtoa(" + irType + " " + value + ")");
929                                break;
930                        case STRING:
931                                if(constantSrc)
932                                {
933                                        temporaryIdentifier = getUseIdentifierForVariable(".tmp");
934                                        int id = addStringLiteral(value);
935                                        String literalIdentifier = getStringLiteralIdentifier(id);
936                                        String literalType = getStringLiteralType(id);
937
938                                        gen(temporaryIdentifier + " = getelementptr " + literalType + "* " + literalIdentifier + ", i64 0, i64 0");
939                                }
940                                else
941                                {
942                                        temporaryIdentifier = value;
943                                }
944                                break;
945                }
946
947                gen("call i32 (i8*, ...)* @printf(i8* " + temporaryIdentifier + ")");
948        }
