019/**
020 * This class allows for the generation of an LLVM IR module.
021 * Each methode that begins with "add" generates LLVM IR code
022 * and writes it to the <code>PrintWriter</code> <code>out</code>.
023 *
024 */
025public class Module
026{
027        /**
028         * Get a string literal's identifier by its
029         * id (i.e. its position in the list of string literals).
030         *
031         * @param id the string literal's id
032         * @return the string literal's identifier
033         */
034        public static String getStringLiteralIdentifier(int id)
035        {
036                return "%.string_" + String.valueOf(id);
037        }
038        /**
039         * Get the LLVM IR type corresponding to
040         * a type from the three address code.
041         *
042         * @param type the type from the three address code
043         * @return the corresponding LLVM IR type
044         */
045        public static String getIRType(Kind type) {
046                return Module.getIRType(type,false);
047        }
048        public static String getIRType(Kind type, boolean as_ptr)
049        {
050                String irType = "";
051                switch(type)
052                {
053                        case LONG:
054                                /* Longs are 64bit signed integers */
055                                irType = "i64";
056                                break;
057                        case DOUBLE:
058                                /* Doubles are 64bit IEEE floats */
059                                irType = "double";
060                                break;
061                        case BOOLEAN:
062                                /* Booleans are 8bit signed integers */
063                                irType = "i1";
064                                break;
065                        case STRING:
066                                /* Strings are each a pointer to a string literal
067                                   (which itself is a pointer to a memory segment
068                                   of signed 8bit integers in UTF-8 encoding).*/
069                                irType = "i8*";
070                                break;
071                }
072
073                return irType + (as_ptr ? "*" : "");
074        }
075
076        /**
077         * Get the LLVM IR type corresponding to
078         * an *array* type from the three address code.
079         *
080         * @param type the final type of the array, eg: double[][] --> double
081         * @param sizes a list of sizes of the array dimensions, eg:
082         *          double x[5][10][7] --> [5,10,7]
083         * @return the corresponding LLVM IR type for the array
084         */
085        public static String getIRAType(Type type, List<Integer> dimensions) {
086                return Module.getIRAType(type,dimensions,false);
087        }
088        public static String getIRAType(Type type, List<Integer> dimensions, boolean as_ptr)
089        {
090                String irType = "";
091                /* write dimensions */
092                for (int d : dimensions)
093                        irType = irType + "[" + d + " x ";
094                /* write actual type */
095                if(type instanceof PrimitiveType) {
096                        irType = irType + Module.getIRType(type.getKind());
097                }
098                else if(type instanceof LLVMBackendStructType) {
099                        irType += Module.getIRSType(((LLVMBackendStructType) type).getMembers());
100                }
101                /* write closing brakets */
102                for (int d : dimensions)
103                        irType += "]";
104                irType.trim();
105                return irType + (as_ptr ? "*" : "");
106        }
107
108        public static String getIRSType(List<Member> members) {
109                String irType = "{ ";
110
111                for(Member m: members) {
112                        Type type = m.getType();
113
114                        if(type instanceof PrimitiveType) {
115                                irType += Module.getIRType(type.getKind());
116                        }
117                        else if(type instanceof LLVMBackendArrayType) {
118                                LLVMBackendArrayType array = (LLVMBackendArrayType) type;
119                                irType += Module.getIRAType(array.getStorageType(), array.getDimensions());
120                        }
121                        else if(type instanceof LLVMBackendStructType) {
122                                LLVMBackendStructType struct = (LLVMBackendStructType) type;
123                                irType += Module.getIRSType(struct.getMembers());
124                        }
125
126                        irType += ", ";
127                }
128
129                irType = irType.substring(0, irType.length() - 2);
130
131                return irType + " }";
132        }

134        /**
135         * Gets the LLVM IR instruction for a binary TAC operator
136         *
137         * @param operator the binary TAC operator
138         * @return the corresponding LLVM IR instruction
139         */
140        public static String getIRBinaryInstruction(Quadruple.Operator operator)
141        {
142                String irInst = "";
143
144                switch(operator)
145                {
146                        /* Arithmetic */
147                        case ADD_LONG:
148                                irInst = "add";
149                                break;
150                        case ADD_DOUBLE:
151                                irInst = "fadd";
152                                break;
153                        case SUB_LONG:
154                                irInst = "sub";
155                                break;
156                        case SUB_DOUBLE:
157                                irInst = "fsub";
158                                break;
159                        case MUL_LONG:
160                                irInst = "mul";
161                                break;
162                        case MUL_DOUBLE:
163                                irInst = "fmul";
164                                break;
165                        case DIV_LONG:
166                                irInst = "sdiv";
167                                break;
168                        case DIV_DOUBLE:
169                                irInst = "fdiv";
170                                break;
171
172                        /* Boolean Arithmetic */
173                        case OR_BOOLEAN:
174                                irInst = "or";
175                                break;
176                        case AND_BOOLEAN:
177                                irInst = "and";
178                                break;
179
180                        /* Comparisons */
181                        case COMPARE_LONG_E:
182                                irInst = " icmp eq";
183                                break;
184                        case COMPARE_LONG_G:
185                                irInst = " icmp sgt";
186                        break;
187                        case COMPARE_LONG_L:
188                                irInst = " icmp slt";
189                                break;
190                        case COMPARE_LONG_GE:
191                                irInst = " icmp sge";
192                        break;
193                        case COMPARE_LONG_LE:
194                                irInst = " icmp sle";
195                        break;
196
197                        case COMPARE_DOUBLE_E:
198                                irInst = " fcmp oeq";
199                        break;
200                        case COMPARE_DOUBLE_G:
201                                irInst = " fcmp ogt";
202                        break;
203                        case COMPARE_DOUBLE_L:
204                                irInst = " fcmp olt";
205                        break;
206                        case COMPARE_DOUBLE_GE:
207                                irInst = " fcmp oge";
208                        break;
209                        case COMPARE_DOUBLE_LE:
210                                irInst = " fcmp ole";
211                        break;
212                }
213
214                return irInst;
215        }

244        /**
245         * Convert a three adress code boolean
246         * to a LLVM IR boolean.
247         * All other public functions expect booleans
248         * to be in the LLVM IR format.
249         *
250         * @param bool a TAC boolean
251         * @return the converted LLVM IR boolean
252         */
253        public static String toIRBoolean(String bool)
254        {
255                if(bool.equals("#FALSE"))
256                {
257                        return "#0";
258                }
259                else if(bool.equals("#TRUE"))
260                {
261                        return "#1";
262                }
263                else
264                {
265                        return bool;
266                }
267        }
268
269        /**
270         * Convert a three adress code string
271         * to a LLVM IR string.
272         * All other public functions expect strings
273         * to be in the LLVM IR format.
274         *
275         * @param str a TAC string
276         * @return the converted LLVM IR string
277         */
278        public static String toIRString(String str)
279        {
280                String irString = "";
281
282                /* Unescape special characters */
283                irString = str.replace("\\\"", "\"").
284                        replace("\\r", "\r").
285                        replace("\\n", "\n").
286                        replace("\\t", "\t").
287                        replace("\\0", "\0");
288
289                return irString;
290        }
