001package swp_compiler_ss13.fuc.backend;
002
003import swp_compiler_ss13.common.backend.*;
004import swp_compiler_ss13.common.types.*;
005import swp_compiler_ss13.common.types.primitive.*;
006import swp_compiler_ss13.common.types.derived.*;
007
008import java.util.*;
009
010/**
011 * Corresponds to an array in three address code
012 * and extends the normal <code>ArrayType</code>
013 * with convenience functions for the LLVMBackend.
014 *
015 */
016public class LLVMBackendArrayType extends ArrayType
017{
018        /**
019         * Creates a new backend array type.
020         *
021         * @param dimensions the dimensions of this array type
022         * @param type the type of the array type's elements
023         */
024        public LLVMBackendArrayType(List<Integer> dimensions, Type type) {
025                super(type, dimensions.get(0));
026                assert((type instanceof PrimitiveType) ||
027                       (type instanceof StructType));
028
029                dimensions.remove(0);
030
031                if(dimensions.size() > 0) {
032                        this.type = new LLVMBackendArrayType(dimensions, type);
033                        this.width = this.type.getWidth() * this.length;
034                }
035        }
036
037        /**
038         * Get the dimensions.
039         *
040         * @return the dimensions
041         */
042        public List<Integer> getDimensions() {
043                List<Integer> dimensions = new LinkedList<Integer>();
044                dimensions.add(this.length);
045
046                if(this.type instanceof LLVMBackendArrayType) {
047                        dimensions.addAll(((LLVMBackendArrayType) this.type).getDimensions());
048                }
049
050                return dimensions;
051        }
052
053        /**
054         * Get the type of the elements.
055         *
056         * @return the elements' type
057         */
058        public Type getStorageType() {
059                if(this.type instanceof LLVMBackendArrayType) {
060                        return ((LLVMBackendArrayType) this.type).getStorageType();
061                }
062                else {
063                        return this.type;
064                }
065        }
066}
