001package swp_compiler_ss13.fuc.backend;
002
003import swp_compiler_ss13.common.backend.*;
004import swp_compiler_ss13.common.types.*;
005import swp_compiler_ss13.common.types.primitive.*;
006import swp_compiler_ss13.common.types.derived.*;
007
008import java.util.*;
009
010public class LLVMBackendArrayType extends ArrayType
011{
012        public LLVMBackendArrayType(List<Integer> dimensions, Type type) {
013                super(type, dimensions.get(0));
014                assert((type instanceof PrimitiveType) ||
015                       (type instanceof StructType));
016
017                dimensions.remove(0);
018
019                if(dimensions.size() > 0) {
020                        this.type = new LLVMBackendArrayType(dimensions, type);
021                        this.width = this.type.getWidth() * this.length;
022                }
023        }
024
025        public List<Integer> getDimensions() {
026                List<Integer> dimensions = new LinkedList<Integer>();
027                dimensions.add(this.length);
028
029                if(this.type instanceof LLVMBackendArrayType) {
030                        dimensions.addAll(((LLVMBackendArrayType) this.type).getDimensions());
031                }
032
033                return dimensions;
034        }
035
036        public Type getStorageType() {
037                if(this.type instanceof LLVMBackendArrayType) {
038                        return ((LLVMBackendArrayType) this.type).getStorageType();
039                }
040                else {
041                        return this.type;
042                }
043        }
044}
