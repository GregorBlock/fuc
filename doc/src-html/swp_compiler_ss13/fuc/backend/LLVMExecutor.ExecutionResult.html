001package swp_compiler_ss13.fuc.backend;
002
003import swp_compiler_ss13.common.backend.Backend;
004import swp_compiler_ss13.common.backend.BackendException;
005import swp_compiler_ss13.common.backend.Quadruple;
006
007import java.io.*;
008import java.util.ArrayList;
009import java.util.List;
010import java.util.Map;
011import org.apache.log4j.Logger;
012
013/**
014 * LLVM IR Code Executor.
015 *
016 * This class can just-in-time compile text-style TAC and execute LLVM IR code.
017 *
018 * The execution of LLVM IR Code is done with LLVM's <code>lli</code> command,
019 * which needs to be installed.
020 *
021 * The format for the text-style TAC is:
022 * "Operator|Arg1|Arg2|Res" (without the quotes)
023 *
024 */
025public class LLVMExecutor {
026
027        /**
028         * The logger used to output information.
029         *
030         */
031        private static Logger logger = Logger.getLogger(LLVMExecutor.class);
032
033        /**
034         * Tries to start <code>lli</code> as a process.
035         * @return the process identifier
036         * @throws IOException if <code>lli</code> is not found
037         */
038        private static Process tryToStartLLI() throws IOException {
039                ProcessBuilder pb = new ProcessBuilder("lli", "-jit-enable-eh", "-");
040                pb.redirectErrorStream(true);
041                Process p = null;
042                try {
043                        p = pb.start();
044                } catch (IOException e) {
045                        String errorMsg = "If you have LLVM installed you might need to check your PATH:\n" +
046                                        "Intellij IDEA: Run -> Edit Configurations -> Environment variables\n" +
047                                        "Eclipse: Run Configurations -> Environment\n" +
048                                        "Shell: Check $PATH";
049                        logger.error("No lli (interpreter and dynamic compiler, part of LLVM) found.");
050                        logger.info(errorMsg);
051                        logger.error(e.getStackTrace());
052                        throw e;
053                }
054                return p;
055        }
056
057        /**
058         * Exectues LLVM IR code via LLVM's <code>lli</code> tool and shows the
059         * result of that execution (exit code and output from the programm).
060         *
061         * @param irCode
062         *            an <code>InputStream</code> of LLVM IR Code
063         * @return the LLVM IR Code as String, the output and the exit code of the execution of the LLVM IR code
064         * @exception java.io.IOException
065         *                if an error occurs reading the InputStream or starting <code>lli</code>
066         * @exception InterruptedException
067         *                if an error occurs in the LLVM Backend
068         * @throws swp_compiler_ss13.common.backend.BackendException
069         *             if an error occurs
070         */
071        public static ExecutionResult runIR(InputStream irCode) throws InterruptedException, BackendException, IOException {
072
073                BufferedReader irCodeReader = new BufferedReader(new InputStreamReader(irCode));
074                StringBuilder irCodeStringBuilder = new StringBuilder();
075
076                Process p = tryToStartLLI();
077
078                /* write LLVM IR code to stdin of the lli process */
079                PrintWriter processInputStream = new PrintWriter(p.getOutputStream());
080                String line = null;
081                while ((line = irCodeReader.readLine()) != null) {
082                        irCodeStringBuilder.append(line + "\n");
083                        processInputStream.println(line);
084                }
085                processInputStream.close();
086
087                /* read stdout from lli process */
088                BufferedReader outPutReader = new BufferedReader(new InputStreamReader(p.getInputStream()));
089                StringBuilder executionOutput = new StringBuilder();
090                line = null;
091                while ((line = outPutReader.readLine()) != null) {
092                        executionOutput.append(line + "\n");
093                }
094
095                int executionExitCode = p.waitFor();
096
097                return new ExecutionResult(executionOutput.toString(), executionExitCode, irCodeStringBuilder.toString());
098        }
099
100
101        /**
102         * Reads three address code (as quadruples) from an input stream, Calls the
103         * <code>LLVMBackend</code> with the quadruple list of TAC instructions and
104         * returns the generated LLVM IR code.
105         *
106         * @param tac the TAC instructions as list of quadruples
107         * @return the generated LLVM IR code as <code>InputStream</code>
108         * @throws IOException
109         * @throws BackendException
110         */
111        private static InputStream compileTAC(List<Quadruple> tac) throws IOException, BackendException {
112                Backend b = new LLVMBackend();
113
114                Map.Entry<String, InputStream> entry = b.generateTargetCode("", tac).entrySet().iterator().next();
115                InputStream targetCode = entry.getValue();
116                String name = entry.getKey();
117
118                return targetCode;
119        }
120
121
122        /**
123         * Reads three address code from an input stream
124         * and formats it to the list of <code>Quadruple</code>'s
125         * the <code>Backend</code> requires.
126         *
127         * @param stream the stream containing one TAC operation per line
128         * @return the formatted TAC
129         * @exception IOException if an error occurs
130         */
131        private static List<Quadruple> readTAC(InputStream stream) throws IOException
132        {
133                BufferedReader in = new BufferedReader(new InputStreamReader(stream));
134                List<Quadruple> tac = new ArrayList<Quadruple>();
135
136                String line = null;
137                while((line = in.readLine()) != null)
138                {
139                        String[] tupleFields = line.split("\\|");
140                        Quadruple q = new QuadrupleImpl(Quadruple.Operator.valueOf(tupleFields[0]),
141                                        tupleFields[1],
142                                        tupleFields[2],
143                                        tupleFields[3]);
144                        tac.add(q);
145                }
146
147                return tac;
148        }
149
150        /**
151         * Reads three address in text-style TAC code from an <code>InputStream</code>,
152         * just-in-time compiles it to LLVM IR code, executes the generated code via
153         * LLVM's <code>lli</code> tool show the result of that execution (execution
154         * output and exit code).
155         *
156         * @param stream
157         *            an <code>InputStream</code> of text-style TAC
158         * @return the exit code of the execution of the llvm ir code
159         * @exception IOException
160         * @exception InterruptedException
161         * @throws BackendException
162         */
163        public static ExecutionResult runTAC(InputStream stream) throws IOException, InterruptedException, BackendException {
164                List<Quadruple> quadruples = readTAC(stream);
165                InputStream irCode = compileTAC(quadruples);
166
167                ExecutionResult result = runIR(irCode);
168
169                logger.info("Generated LLVM IR:\n" + result.irCode);
170
171                logger.info("Execution output (stdout,stderr):\n" + result.output);
172
173                logger.info("Exit code of execution: " + String.valueOf(result.exitCode));
174
175                return result;
176        }
177
178        /**
179         * Just-in-time compiles and executes text-style TAC, either:
180         * Command line arguments exist:
181         *   Every argument is a file name and the contents
182         *   will be jitted and executed.
183         * No command line arguments:
184         *   The standard input (stdin) will be read from until
185         *   end-of-file (EOF) is found and everything read will
186         *   be jitted and executed.
187         *
188         *
189         * @param args file names to jit and execute
190         * @exception IOException if an error occurs
191         * @exception InterruptedException if an error occurs
192         */
193        public static void main(String[] args) throws IOException, InterruptedException, BackendException {
194                if (args.length > 0) {
195                        for (String arg : args) {
196                                logger.info("Generating LLVM IR for " + arg);
197                                runTAC(new FileInputStream(arg));
198                        }
199                }
200                else {
201                        logger.info("Generating LLVM IR for stdin, please enter" + " one quadruple per line in the format "
202                                        + "\"Operator|Arg1|Arg2|Res\" (without the quotes):");
203                        runTAC(System.in);
204                }
205        }
206
207
208        /**
209         * The result of executing LLVM IR Code. The result consists of the output
210         * and the exit code of the execution and the IR code that was executed.
211         */
212        public static class ExecutionResult {
213                public String output;
214                public int exitCode;
215                public String irCode;
216
217                public ExecutionResult(String output, int exitCode, String irCode) {
218                        this.output = output;
219                        this.exitCode = exitCode;
220                        this.irCode = irCode;
221                }
222        }
