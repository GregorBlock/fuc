019/**
020 * This class allows for the generation of an LLVM IR module.
021 * Each methode that begins with "add" generates LLVM IR code
022 * and writes it to the <code>PrintWriter</code> <code>out</code>.
023 *
024 */
025public class Function
026{
027        /**
028         * A list of all the string literals that
029         * exist in this module where each element
030         * describes a string literal's number with its
031         * position in the list and that string literal's
032         * length with the value of the element.
033         *
034         */
035        private ArrayList<Integer> stringLiterals;
036
037        /**
038         * This maps a variable's name (not its
039         * identifier, which is %{name}) to the
040         * number of times it has been used in
041         * the module already.
042         * This is necessary because LLVM's IR
043         * allows only for static single assignment
044         * (SSA), wich means we have to store the
045         * variable's actual value as the destination
046         * of a static pointer and generate a new
047         * 'use' variable (%{name}.{use_number})
048         * for every time the variable is read
049         * from or written to and route the pointer
050         * access through that use variable.
051         *
052         */
053        private Map<String,Integer> variableUseCount;
054
055        private StringBuffer code;
056
057        public String getCode() {
058                return this.code.toString();
059        }
060
061        private Map<String,Map.Entry<String,List<Long>>> references;
062
063        private Map<String,DerivedType> derivedTypes;
064
065        /**
066         * Creates a new <code>Module</code> instance.
067         *
068         * to write the LLVM IR
069         */
070        public Function(String name,
071                        Type.Kind returnType,
072                        List<Map.Entry<String,Type.Kind>> arguments)
073        {
074                stringLiterals = new ArrayList<Integer>();
075                variableUseCount = new HashMap<String,Integer>();
076                references = new HashMap<String,Map.Entry<String,List<Long>>>();
077                derivedTypes = new HashMap<String,DerivedType>();
078                code = new StringBuffer();
079
080                /* Add fake temporary variable */
081                variableUseCount.put(".tmp", 0);
082
083                code.append("define ");
084                code.append(Module.getIRType(returnType));
085                code.append(" @");
086                code.append(name);
087                code.append("(");
088
089                for(Map.Entry<String,Type.Kind> argument: arguments) {
090                        code.append(Module.getIRType(argument.getValue()));
091                        code.append(",");
092                }
093
094                if(arguments.size() > 0) {
095                        code.deleteCharAt(code.length() - 1);
096                }
097
098                code.append(") {\n");
099        }
100
101        /**
102         * Append the LLVM IR prefixed with two spaces
103         * to the code <code>StringBuffer</code>
104         *
105         * @param code a <code>String</code> value
106         */
107        private void gen(String code)
108        {
109                this.code.append("  ");
110                this.code.append(code);
111                this.code.append("\n");
112        }

114        /**
115         * Get a string literal's type by its
116         * id (i.e. [i8 * {length}], where the length
117         * is the string literal's length).
118         *
119         * @param id an <code>int</code> value
120         * @return a <code>String</code> value
121         */
122        private String getStringLiteralType(int id)
123        {
124                return "[" + String.valueOf(stringLiterals.get(id)) + " x i8]";
125        }
126
127        /**
128         * Gets a unique use identifier for
129         * a variable; no two calls will return
130         * the same use identifier.
131         *
132         * @param variable the variable's name
133         * @return a free use identifier for the variable
134         * @exception BackendException if an error occurs
135         */
136        private String getUseIdentifierForVariable(String variable) throws BackendException {
137                int ssa_suffix = 0;
138                try {
139                        ssa_suffix = variableUseCount.get(variable);
140                } catch (NullPointerException e) {
141                        throw new BackendException("Use of undeclared variable");
142                }
143                variableUseCount.put(variable, ssa_suffix + 1);
144                return "%" + variable + "." + String.valueOf(ssa_suffix);
145        }
146
147        /**
148         * Generates a new string literal from a <code>String</code>
149         * value and returns its id (i.e. its position in the
150         * list of string literals <code>stringLiterals</code>).
151         *
152         * @param literal the string to use
153         * @return the new string literal's id
154         */
155        private Integer addStringLiteral(String literal) throws BackendException
156        {
157                int length = 0;
158
159                try
160                {
161                        literal = literal.substring(1, literal.length() - 1);
162                        byte[] utf8 = literal.getBytes("UTF-8");
163                        literal = Arrays.toString(utf8).replaceAll("(-?)([0-9]+)(,?)","i8 $1$2$3");
164
165                        if(utf8.length > 0)
166                        {
167                                literal = literal.replace("]",", i8 0]");
168                        }
169                        else
170                        {
171                                literal = "[i8 0]";
172                        }
173
174                        length = utf8.length + 1;
175                }
176                catch(UnsupportedEncodingException e)
177                {
178                        throw new BackendException("LLVM backend cannot handle strings without utf8 support");
179                }
180
181                int id = stringLiterals.size();
182                stringLiterals.add(length);
183
184                String type = getStringLiteralType(id);
185
186                String identifier = Module.getStringLiteralIdentifier(id);
187
188                gen(identifier + " = alloca " + type);
189                gen("store " + type + " " + literal + ", " + type + "* " + identifier);
190
191                return id;
192        }

194        /**
195         * Sets a variable (which must be of the string primitive type)
196         * to contain a pointer to a string literal.
197         *
198         * @param variable the variable's name
199         * @param literalID the string literal's id
200         * @return the used "use identifier" for variable
201         * @exception BackendException if an error occurs
202         */
203        private String addLoadStringLiteral(String variable, int literalID) throws BackendException {
204                String variableIdentifier = "%" + variable;
205                String variableUseIdentifier = getUseIdentifierForVariable(variable);
206                String literalIdentifier = Module.getStringLiteralIdentifier(literalID);
207                String literalType = getStringLiteralType(literalID);
208
209                gen(variableUseIdentifier + " = getelementptr " + literalType + "* " + literalIdentifier + ", i64 0, i64 0");
210                gen("store i8* " + variableUseIdentifier + ", i8** " + variableIdentifier);
211                return variableUseIdentifier;
212        }

214        /**
215         * Adds a new variable and return its identifier.
216         *
217         * @param type the new variable's type
218         * @param variable the new variable's name
219         * @return the variable's identifier
220         */
221        private String addNewVariable(Kind type, String variable)
222        {
223                String irType = Module.getIRType(type);
224                String variableIdentifier = "%" + variable;
225                variableUseCount.put(variable, 0);
226
227                gen(variableIdentifier + " = alloca " + irType);
228                return variableIdentifier;
229        }

231        /**
232         *  Conditionally loads a constant or an identifier. If given a constant, just the leading # character
233         *  is removed. If given a variable (anything without leading # character),
234         *  it is being loaded and the temporary register name is returned.
235         *
236         *      @param constantOrIdentifier Name of a constant or identifier.
237         *  @returns name of usable primitive
238         */
239        private String cdl(String constantOrIdentifier, Kind type) throws BackendException {
240                if(constantOrIdentifier.charAt(0) == '#')
241                        /* constant */
242                        if(type==Kind.STRING)
243                        {
244                                int literalID = addStringLiteral(constantOrIdentifier.substring(1));
245                                String literalIdentifier = Module.getStringLiteralIdentifier(literalID);
246                                String literalType = getStringLiteralType(literalID);
247                                String variableUseIdentifier = getUseIdentifierForVariable(".tmp");
248                                gen(variableUseIdentifier + " = getelementptr " + literalType + "* " + literalIdentifier + ", i64 0, i64 0");
249
250                                return variableUseIdentifier;
251                        }
252                        else
253                                return constantOrIdentifier.substring(1);
254                else {
255                        /* identifier */
256                        String id = getUseIdentifierForVariable(constantOrIdentifier);
257                        gen(id + " = load " + Module.getIRType(type,true) + " %"+constantOrIdentifier);
258                        return id;
259                }
260        }

262        /**
263         * Adds a new variable and optionally sets its
264         * initial value if given.
265         *
266         * @param type the new variable's type
267         * @param variable the new variable's name
268         * @param initializer the new variable's initial value
269         * @exception BackendException if an error occurs
270         */
271        public void addPrimitiveDeclare(Kind type, String variable, String initializer) throws BackendException {
272                addNewVariable(type, variable);
273
274                /* Set default initializer */
275                if(initializer.equals(Quadruple.EmptyArgument)) {
276                        switch(type) {
277                                case LONG:
278                                        initializer = "#0";
279                                        break;
280                                case DOUBLE:
281                                        initializer = "#0.0";
282                                        break;
283                                case BOOLEAN:
284                                        initializer = "#0";
285                                        break;
286                                case STRING:
287                                        initializer = "#\"\"";
288                                        break;
289                        }
290                }
291
292                addPrimitiveAssign(type, variable, initializer);
293        }

295        public String addDerivedDeclare(DerivedType type, String identifier) throws BackendException {
296                String irType = "";
297
298                if(type instanceof LLVMBackendArrayType) {
299                        LLVMBackendArrayType arType = (LLVMBackendArrayType) type;
300                        irType = Module.getIRAType(arType.getStorageType(), arType.getDimensions());
301                }
302                else if(type instanceof LLVMBackendStructType) {
303                        LLVMBackendStructType stType = (LLVMBackendStructType) type;
304                        irType = Module.getIRSType(stType.getMembers());
305                }
306                else {
307                        throw new BackendException("Unexpected type: " + type.getClass().getName());
308                }
309
310                String variableIdentifier = "%" + identifier;
311                variableUseCount.put(identifier, 0);
312                derivedTypes.put(identifier, type);
313                gen(variableIdentifier + " = alloca " + irType);
314                return variableIdentifier;
315        }

317        /**
318         * Assigns either a constant value, or the value of
319         * one variable (<code>src</code>) to another variable
320         * (<code>dst</code>). Only assignments for identical
321         * types are allowed.
322         *
323         * @param type the type of the assignment
324         * @param dst the destination variable's name
325         * @param src the source constant or the source variable's name
326         * @exception BackendException if an error occurs
327         */
328        public void addPrimitiveAssign(Kind type, String dst, String src) throws BackendException {
329                boolean constantSrc = false;
330                if(src.charAt(0) == '#')
331                {
332                        src = src.substring(1);
333                        constantSrc = true;
334                }
335
336                String irType = Module.getIRType(type);
337                String dstIdentifier = "%" + dst;
338
339                if(constantSrc)
340                {
341                        if(type == Kind.STRING)
342                        {
343                                int id = addStringLiteral(src);
344                                addLoadStringLiteral(dst, id);
345                        }
346                        else
347                        {
348                                gen("store " + irType + " " + src + ", " + irType + "* " + dstIdentifier);
349                        }
350                }
351                else
352                {
353                        String srcUseIdentifier = getUseIdentifierForVariable(src);
354                        String srcIdentifier = "%" + src;
355                        gen(srcUseIdentifier + " = load " + irType + "* " + srcIdentifier);
356                        gen("store " + irType + " " + srcUseIdentifier + ", " + irType + "* " + dstIdentifier);
357                }
358        }

360        public void addReferenceDerivedGet(Type.Kind derivedType, String derived, String index, String dst) throws BackendException {
361                String src = derived;
362                List<Long> indices = new LinkedList<Long>();
363
364                while(references.containsKey(src)) {
365                        Map.Entry<String,List<Long>> reference = references.get(src);
366                        indices.addAll(reference.getValue());
367                        src = reference.getKey();
368                }
369
370                switch(derivedType) {
371                        case ARRAY:
372                                indices.add(Long.valueOf(index.substring(1)));
373                                break;
374                        case STRUCT:
375                                DerivedType type = derivedTypes.get(src);
376
377                                if(type instanceof LLVMBackendArrayType) {
378                                        type = (DerivedType) ((LLVMBackendArrayType) type).getStorageType();
379                                }
380
381                                if(type instanceof LLVMBackendStructType) {
382                                        List<Member> members = ((LLVMBackendStructType) type).getMembers();
383                                        for(Member m: members) {
384                                                if(m.getName().equals(index)) {
385                                                        indices.add(Long.valueOf(members.indexOf(m)));
386                                                        break;
387                                                }
388                                        }
389                                }
390                                else {
391                                        throw new BackendException("Expected final type of LLVMBackendStructType, not " + type.getClass().getName());
392                                }
393                                break;
394                }
395
396                references.put(dst,new AbstractMap.SimpleEntry<String,List<Long>>(src, indices));
397        }

399        public void addPrimitiveDerivedSetOrGet(Type.Kind derivedTypeKind,
400                                                String derived,
401                                                String index,
402                                                Kind primitiveType,
403                                                String primitive,
404                                                boolean set) throws BackendException {
405                List<Long> indices = new LinkedList<Long>();
406
407                while(references.containsKey(derived)) {
408                        Map.Entry<String,List<Long>> reference = references.get(derived);
409                        indices.addAll(reference.getValue());
410                        derived = reference.getKey();
411                }
412
413                switch(derivedTypeKind) {
414                        case ARRAY:
415                                indices.add(Long.valueOf(index.substring(1)));
416                                break;
417                        case STRUCT:
418                                DerivedType type = derivedTypes.get(derived);
419
420                                if(type instanceof LLVMBackendArrayType) {
421                                        type = (DerivedType) ((LLVMBackendArrayType) type).getStorageType();
422                                }
423
424                                if(type instanceof LLVMBackendStructType) {
425                                        List<Member> members = ((LLVMBackendStructType) type).getMembers();
426                                        for(Member m: members) {
427                                                if(m.getName().equals(index)) {
428                                                        indices.add(Long.valueOf(members.indexOf(m)));
429                                                        break;
430                                                }
431                                        }
432                                }
433                                else {
434                                        throw new BackendException("Unexpected final derived type: " + type.getClass().getName());
435                                }
436                                break;
437                        default:
438                                throw new BackendException("Unexpected derived type: " + derivedTypeKind.toString());
439                }
440
441                String indexList = "";
442                for(Long i: indices) {
443                        indexList += ", " + i.toString();
444                }
445
446                String primitiveIRType = Module.getIRType(primitiveType);
447
448                DerivedType derivedType = derivedTypes.get(derived);
449                String derivedIRType = "";
450                if(derivedType instanceof LLVMBackendArrayType) {
451                        LLVMBackendArrayType derivedArType = (LLVMBackendArrayType) derivedType;
452                        derivedIRType = Module.getIRAType(derivedArType.getStorageType(), derivedArType.getDimensions());
453                }
454                else if(derivedType instanceof LLVMBackendStructType) {
455                        LLVMBackendStructType derivedStType = (LLVMBackendStructType) derivedType;
456                        derivedIRType = Module.getIRSType(derivedStType.getMembers());
457                }
458                else {
459                        throw new BackendException("Unexpected derived type: " + derivedType.getClass().getName());
460                }
461
462                if(set) {
463                        String derivedUseIdentifier1 = getUseIdentifierForVariable(derived);
464                        String derivedUseIdentifier2 = getUseIdentifierForVariable(derived);
465                        String derivedIdentifier = "%" + derived;
466
467                        primitive = cdl(primitive, primitiveType);
468                        gen(derivedUseIdentifier1 + " = load " + derivedIRType + "* " + derivedIdentifier);
469                        gen(derivedUseIdentifier2 + " = insertvalue " + derivedIRType + " " + derivedUseIdentifier1 +
470                            ", " + primitiveIRType + " " + primitive + indexList);
471                        gen("store " + derivedIRType + " " + derivedUseIdentifier2 + ", " + derivedIRType + "* " + derivedIdentifier);
472                }
473                else {
474                        String primitiveUseIdentifier = getUseIdentifierForVariable(primitive);
475                        String primitiveIdentifier = "%" + primitive;
476                        String derivedUseIdentifier = getUseIdentifierForVariable(derived);
477                        String derivedIdentifier = "%" + derived;
478
479                        gen(derivedUseIdentifier + " = load " + derivedIRType + "* " + derivedIdentifier);
480                        gen(primitiveUseIdentifier + " = extractvalue " + derivedIRType + " " + derivedUseIdentifier + indexList);
481                        gen("store " + primitiveIRType + " " + primitiveUseIdentifier + ", " + primitiveIRType + "* " + primitiveIdentifier);
482                }
483        }

485        /**
486         * Assigns the value of one variable (<code>src</code>)
487         * to another variable (<code>dst</code>), where their types
488         * must only be convertible, not identical.
489         *
490         * @param srcType the source variable's type
491         * @param src the source variable's name
492         * @param dstType the destination variable's type
493         * @param dst the destination variable's name
494         * @exception BackendException if an error occurs
495         */
496        public void addPrimitiveConversion(Kind srcType,
497                                           String src,
498                                           Kind dstType,
499                                           String dst) throws BackendException {
500                String srcUseIdentifier = getUseIdentifierForVariable(src);
501                String dstUseIdentifier = getUseIdentifierForVariable(dst);
502                String srcIdentifier = "%" + src;
503                String dstIdentifier = "%" + dst;
504                String srcIRType = Module.getIRType(srcType);
505                String dstIRType = Module.getIRType(dstType);
506
507                if((srcType == Kind.LONG) && (dstType == Kind.DOUBLE))
508                {
509                        gen(srcUseIdentifier + " = load " + srcIRType + "* " + srcIdentifier);
510                        gen(dstUseIdentifier + " = sitofp " + srcIRType + " " + srcUseIdentifier + " to " + dstIRType);
511                        gen("store " + dstIRType + " " + dstUseIdentifier + ", " + dstIRType + "* " + dstIdentifier);
512                }
513                else if((srcType == Kind.DOUBLE) && (dstType == Kind.LONG))
514                {
515                        gen(srcUseIdentifier + " = load " + srcIRType + "* " + srcIdentifier);
516                        gen(dstUseIdentifier + " = fptosi " + srcIRType + " " + srcUseIdentifier + " to " + dstIRType + "");
517                        gen("store " + dstIRType + " " + dstUseIdentifier + ", " + dstIRType + "* " + dstIdentifier);
518                }
519                else if(dstType == Kind.STRING)
520                {
521                        String conversionFunction = "";
522
523                        switch(srcType) {
524                                case BOOLEAN:
525                                        conversionFunction = "btoa";
526                                        break;
527                                case LONG:
528                                        conversionFunction = "ltoa";
529                                        break;
530                                case DOUBLE:
531                                        conversionFunction = "dtoa";
532                                        break;
533                        }
534
535                        gen(srcUseIdentifier + " = load " + srcIRType + "* " + srcIdentifier);
536                        gen(dstUseIdentifier + " = call i8* (" + srcIRType + ")* @" + conversionFunction + "(" + srcIRType + " " + srcUseIdentifier + ")");
537                        gen("store " + dstIRType + " " + dstUseIdentifier + ", " + dstIRType + "* " + dstIdentifier);
538                }
539        }

541        /**
542         * Adds a generic binary operation of two sources - each can
543         * either be a constant or a variable - the result
544         * of which will be stored in a variable (<code>dst</code>).
545         * All types must be identical.
546         *
547         * @param op the binary operation to add
548         * @param resultType  the result type of the binary operation
549         * @param argumentType the argument type of the binary operation
550         * @param lhs the constant or name of the variable on the left hand side
551         * @param rhs the constant or name of the variable on the right hand side
552         * @param dst the destination variable's name
553         * @exception BackendException if an error occurs
554         */
555        public void addPrimitiveBinaryInstruction(Quadruple.Operator op,
556                                                  Kind resultType,
557                                                  Kind argumentType,
558                                                  String lhs,
559                                                  String rhs,
560                                                  String dst) throws BackendException {
561                String irArgumentType = Module.getIRType(argumentType);
562                String irResultType = Module.getIRType(resultType);
563                String irInst = Module.getIRBinaryInstruction(op);
564
565                lhs = cdl(lhs, argumentType);
566                rhs = cdl(rhs, argumentType);
567
568                String dstUseIdentifier = getUseIdentifierForVariable(dst);
569                String dstIdentifier = "%" + dst;
570
571                gen(dstUseIdentifier + " = " + irInst + " " + irArgumentType + " " + lhs + ", " + rhs);
572                gen("store " + irResultType + " " + dstUseIdentifier + ", " + irResultType + "* " + dstIdentifier);
573        }

575        /**
576         * Adds a call to a argument-homogenous binary functions
577         * where each argument can either be a constant or a variable;
578         * the result of the function will be stored in a variable (<code>dst</code>).
579         * Argument types must both be equal, but may differ
580         * from the result type.
581         *
582         * @param op the binary operation to add
583         * @param resultType the type of the binary operation's result
584         * @param argumentType the type of the binary operation's arguments
585         * @param lhs the constant or name of the variable on the left hand side
586         * @param rhs the constant or name of the variable on the right hand side
587         * @param dst the destination variable's name
588         * @exception BackendException if an error occurs
589         */
590        public void addPrimitiveBinaryCall(Quadruple.Operator op,
591                                           Kind resultType,
592                                           Kind argumentType,
593                                           String lhs,
594                                           String rhs,
595                                           String dst) throws BackendException {
596                String irArgumentType = Module.getIRType(argumentType);
597                String irResultType = Module.getIRType(resultType);
598                String irCall = Module.getIRBinaryCall(op);
599                boolean constantRHS = false;
600
601                if(rhs.charAt(0) == '#')
602                {
603                        constantRHS = true;
604                }
605
606                lhs = cdl(lhs, argumentType);
607                rhs = cdl(rhs, argumentType);
608
609                if(constantRHS) {
610                        if((op == Quadruple.Operator.DIV_LONG) &&
611                           (Long.valueOf(rhs).equals(Long.valueOf(0))))
612                        {
613                                throw new BackendException("Division by zero");
614                        }
615
616                        if((op == Quadruple.Operator.DIV_DOUBLE) &&
617                           ((Double.valueOf(rhs).equals(Double.valueOf(0.0))) ||
618                            (Double.valueOf(rhs).equals(Double.valueOf(-0.0)))))
619                        {
620                                throw new BackendException("Division by zero");
621                        }
622                }
623
624                String dstUseIdentifier = getUseIdentifierForVariable(dst);
625                String dstIdentifier = "%" + dst;
626
627                gen(dstUseIdentifier + " = invoke " + irResultType + " (" + irArgumentType + ", " + irArgumentType + ")* " +
628                    "@" + irCall + "(" + irArgumentType + " " + lhs + ", " + irArgumentType + " " + rhs + ") to label " + dstUseIdentifier + ".ok unwind label %UncaughtException");
629                gen(dstUseIdentifier.substring(1, dstUseIdentifier.length()) + ".ok:");
630                gen("store " + irResultType + " " + dstUseIdentifier + ", " + irResultType + "* " + dstIdentifier);
631        }

633        /**
634         * Adds a binary not operation, where the source
635         * may be either a constant, or a variable, whereas
636         * the result must be a variable.
637         *
638         * @param source the constant or name of the variable to be negated
639         * @param destination the destination variable's name
640         * @exception BackendException if an error occurs
641         */
642        public void addBooleanNot(String source, String destination) throws BackendException {
643
644                String irType = Module.getIRType(Type.Kind.BOOLEAN);
645
646                source = cdl(source, Type.Kind.BOOLEAN);
647                String dstUseIdentifier = getUseIdentifierForVariable(destination);
648                String dstIdentifier = "%" + destination;
649
650                gen(dstUseIdentifier + " = " + "sub " + irType + " 1, " + source);
651                gen("store " + irType + " " + dstUseIdentifier + ", " + irType + "* " + dstIdentifier);
652        }

685        /**
686         * Adds a branching instruction, which may either
687         * be unconditional (e.g. branch immediately) or
688         * depend on a boolean variable 'condition' (e.g. branch to
689         * one of two target labels depening on the boolean
690         * variable's value at runtime).
691         *
692         * @param target1 target label for unconditional branch, target
693         *                for the condition being true for conditional branch
694         * @param target2 <code>Quadruple.EmptyArgument</code> for unconditional
695         *                branch, target for the condition being false for
696         *                the conditional branch
697         * @param condition <code>Quadruple.EmptyArgument</code> for unconditional
698         *                  branch, the condition being tested for conditional branch
699         * @exception BackendException if an error occurs
700         */
701        public void addBranch(String target1, String target2, String condition) throws BackendException {
702                /* conditional branch */
703                if(!target2.equals(Quadruple.EmptyArgument)) {
704                        String irType = Module.getIRType(Kind.BOOLEAN);
705                        String conditionUseIdentifier = getUseIdentifierForVariable(condition);
706                        gen(conditionUseIdentifier + " = load " + irType + "* %" + condition);
707                        gen("br " + irType + " " + conditionUseIdentifier + ", label %" + target1 + ", label %" + target2);
708                }
709                else {
710                        gen("br label %" + target1);
711                }
712        }

714        /**
715         * Adds a call to the printf function for
716         * the value, which may be either a constant
717         * or the name of a variable.
718         *
719         * @param value the value to be printed
720         * @param type the type of the value (boolean, long, double, string)
721         * @exception BackendException if an error occurs
722         */
723        public void addPrint(String value, Kind type) throws BackendException {
724                String irType = Module.getIRType(type);
725                boolean constantSrc = false;
726
727                value = cdl(value, type);
728
729                String temporaryIdentifier = "";
730
731                if(constantSrc)
732                {
733                        temporaryIdentifier = getUseIdentifierForVariable(".tmp");
734                        int id = addStringLiteral(value);
735                        String literalIdentifier = Module.getStringLiteralIdentifier(id);
736                        String literalType = getStringLiteralType(id);
737
738                        gen(temporaryIdentifier + " = getelementptr " + literalType + "* " + literalIdentifier + ", i64 0, i64 0");
739                }
740                else
741                {
742                        temporaryIdentifier = value;
743                }
744
745                gen("call i32 (i8*, ...)* @printf(i8* " + temporaryIdentifier + ")");
746        }
