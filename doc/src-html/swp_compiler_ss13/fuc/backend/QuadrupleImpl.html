001package swp_compiler_ss13.fuc.backend;
002
003import swp_compiler_ss13.common.backend.Quadruple;
004
005/**
006* A bare-bones implementation of the <code>Quadruple</code> interface used in
007* the tests and the <code>TACExecutor</code>
008*
009*/
010public class QuadrupleImpl implements Quadruple
011{
012        private Operator operator;
013        private String argument1;
014        private String argument2;
015        private String result;
016
017        public QuadrupleImpl(Operator o, String a1, String a2, String r)
018        {
019                operator = o;
020                argument1 = a1;
021                argument2 = a2;
022                result = r;
023        }
024
025        public String toString() { return "(" + String.valueOf(operator) + "|" + argument1  + "|" + argument2 + "|" + result + ")"; }
026
027        public Operator getOperator() { return operator; }
028        public String getArgument1() { return argument1; }
029        public String getArgument2() { return argument2; }
030        public String getResult() { return result; }
031}
