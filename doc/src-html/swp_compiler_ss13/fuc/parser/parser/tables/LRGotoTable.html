001package swp_compiler_ss13.fuc.parser.parser.tables;
002
003import java.util.HashMap;
004import java.util.Map;
005
006import swp_compiler_ss13.fuc.parser.grammar.NonTerminal;
007import swp_compiler_ss13.fuc.parser.parser.states.LRErrorState;
008import swp_compiler_ss13.fuc.parser.parser.states.LRParserState;
009
010
011public class LRGotoTable {
012   // --------------------------------------------------------------------------
013   // --- variables and constants ----------------------------------------------
014   // --------------------------------------------------------------------------
015   private final Map<LRTableKey, LRParserState> table = new HashMap<>();
016   
017   // --------------------------------------------------------------------------
018   // --- constructors ---------------------------------------------------------
019   // --------------------------------------------------------------------------
020   public LRGotoTable() {
021      
022   }
023   
024   // --------------------------------------------------------------------------
025   // --- getter/setter --------------------------------------------------------
026   // --------------------------------------------------------------------------
027   public LRParserState get(LRParserState curState, NonTerminal curNonTerminal) {
028      LRParserState result = table.get(new LRTableKey(curState, curNonTerminal));
029      if (result == null) {
030         return new LRErrorState("An Error occurred: No transition from " + curState + " with NonTerminal " + curNonTerminal);
031      } else {
032         return result;
033      }
034   }
035   
036   
037   public void set(LRParserState newState, LRParserState curState, NonTerminal curNonTerminal) throws DoubleEntryException {
038      LRTableKey key = new LRTableKey(curState, curNonTerminal);
039      if (table.containsKey(key)) {
040         LRParserState oldState = table.get(key);
041         throw new DoubleEntryException("There already is a state " + oldState + " for the key " + key + "! (new state: " + newState + ")");
042      }
043      table.put(key, newState);
044   }
045}
