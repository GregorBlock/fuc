001package swp_compiler_ss13.fuc.parser.parser.tables;
002
003import swp_compiler_ss13.fuc.parser.grammar.Symbol;
004import swp_compiler_ss13.fuc.parser.parser.states.LRParserState;
005
006
007/**
008 * Helper class that enables us to use a 1-dimensional map as substitution for the 2-dimensional parsing table. Combines
009 * {@link LRParserState} and {@link Symbol} as key for the {@link LRParsingTable}-maps.
010 * 
011 * @author Gero
012 */
013public class LRTableKey {
014   // --------------------------------------------------------------------------
015   // --- variables and constants ----------------------------------------------
016   // --------------------------------------------------------------------------
017   private final LRParserState state;
018   private final Symbol symbol;
019   
020   // cache
021   private final int hashCode;
022   
023   
024   // --------------------------------------------------------------------------
025   // --- constructors ---------------------------------------------------------
026   // --------------------------------------------------------------------------
027   /**
028         * @see LRTableKey
029         * @param state
030         * @param symbol
031         */
032        public LRTableKey(LRParserState state, Symbol symbol) {
033           if (state == null || symbol == null) {
034                   throw new NullPointerException("LRTableKey components must not be null!");
035           }
036      this.state = state;
037      this.symbol = symbol;
038      
039      // Precompute hash
040      final int prime = 31;
041      int result = 1;
042      result = prime * result + ((state == null) ? 0 : state.hashCode());
043      result = prime * result + ((symbol == null) ? 0 : symbol.hashCode());
044      this.hashCode = result;
045   }
046   
047   
048   // --------------------------------------------------------------------------
049   // --- getter/setter --------------------------------------------------------
050   // --------------------------------------------------------------------------
051   /**
052         * @return The {@link LRParserState}-part of this key
053         */
054   public LRParserState getState() {
055      return state;
056   }
057
058   /**
059         * @return The {@link Symbol}-part of this key
060         */
061   public Symbol getSymbol() {
062      return symbol;
063   }
064   
065   @Override
066   public int hashCode() {
067      return hashCode;
068   }
069   
070   @Override
071   public boolean equals(Object obj) {
072      if (this == obj)
073         return true;
074      if (obj == null)
075         return false;
076      if (getClass() != obj.getClass())
077         return false;
078      LRTableKey other = (LRTableKey) obj;
079      if (!state.equals(other.state))
080         return false;
081      if (!symbol.equals(other.symbol))
082         return false;
083      return true;
084   }
085   
086   @Override
087   public String toString() {
088      return "[Key " + state.getId() + "|" + symbol.toString() + "]";
089   }
090}
