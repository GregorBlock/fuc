001package swp_compiler_ss13.fuc.parser.parser.tables;
002
003import java.util.HashMap;
004import java.util.LinkedList;
005import java.util.List;
006import java.util.Map;
007import java.util.Map.Entry;
008
009import swp_compiler_ss13.fuc.parser.grammar.Terminal;
010import swp_compiler_ss13.fuc.parser.parser.states.LRParserState;
011import swp_compiler_ss13.fuc.parser.parser.tables.actions.ALRAction;
012import swp_compiler_ss13.fuc.parser.parser.tables.actions.Error;
013
014/**
015 * This class stores associations of the form (current {@link LRParserState},
016 * current {@link Terminal}) -> next {@link ALRAction}.
017 */
018public class LRActionTable {
019        // --------------------------------------------------------------------------
020        // --- variables and constants ----------------------------------------------
021        // --------------------------------------------------------------------------
022        private final Map<LRTableKey, ALRAction> table = new HashMap<>();
023
024        // --------------------------------------------------------------------------
025        // --- constructors ---------------------------------------------------------
026        // --------------------------------------------------------------------------
027        /**
028         * @see LRActionTable
029         */
030        public LRActionTable() {
031
032        }
033
034        // --------------------------------------------------------------------------
035        // --- getter/setter --------------------------------------------------------
036        // --------------------------------------------------------------------------
037        /**
038         * @param curState
039         * @param curTerminal
040         * @return The formerly stored {@link ALRAction} associated with the given
041         *         {@link LRParserState} and {@link Terminal}. If there is no entry,
042         *         an {@link Error} is returned.
043         * @see #getWithNull(LRParserState, Terminal)
044         */
045        public ALRAction get(LRParserState curState, Terminal curTerminal) {
046                LRTableKey key = new LRTableKey(curState, curTerminal);
047                ALRAction result = table.get(key);
048                if (result == null) {
049                        return new Error("No entry for " + key.getSymbol() + " in parsetable!");
050                } else {
051                        return result;
052                }
053        }
054
055        /**
056         * @param curState
057         * @param curTerminal
058         * @return The formerly stored {@link ALRAction} associated with the given
059         *         {@link LRParserState} and {@link Terminal}. If there is no entry,
060         *         <code>null</code> is returned!
061         */
062        public ALRAction getWithNull(LRParserState curState, Terminal curTerminal) {
063                LRTableKey key = new LRTableKey(curState, curTerminal);
064                return table.get(key);
065        }
066
067        /**
068         * Sets the given action for the state/terminal combination. Throws a
069         * {@link DoubleEntryException} if there already is a action for this
070         * combination.
071         * 
072         * @param action
073         * @param curState
074         * @param curTerminal
075         * @throws DoubleEntryException
076         */
077        public void set(ALRAction action, LRParserState curState,
078                        Terminal curTerminal) throws DoubleEntryException {
079                LRTableKey key = new LRTableKey(curState, curTerminal);
080                if (table.containsKey(key)) {
081                        ALRAction oldAction = table.get(key);
082                        throw new DoubleEntryException("There already is a state "
083                                        + oldAction + " for the key " + key + "! (new state: "
084                                        + action + ")");
085                }
086                table.put(key, action);
087        }
088
089        /**
090         * Sets the given state for the state/terminal combination. This method
091         * overwrites previous stored actions without further notice.
092         * 
093         * @param action
094         * @param curState
095         * @param curTerminal
096         * @throws DoubleEntryException
097         */
098        public void setHard(ALRAction action, LRParserState curState,
099                        Terminal curTerminal) {
100                LRTableKey key = new LRTableKey(curState, curTerminal);
101                table.put(key, action);
102        }
103        
104        /**
105         * @param state
106         * @return A list of all terminals which this table expects for the given state
107         */
108        public List<Terminal> getPossibleTerminalsFor(LRParserState state) {
109                List<Terminal> possibleTerminals = new LinkedList<>();
110                for (Entry<LRTableKey, ALRAction> entry : table.entrySet()) {
111                        LRTableKey key = entry.getKey();
112                        if (key.getState().equals(state)) {
113                                possibleTerminals.add((Terminal) key.getSymbol());
114                        }
115                }
116                return possibleTerminals;
117        }
118}
