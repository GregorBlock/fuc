001package swp_compiler_ss13.fuc.parser.parser;
002
003import java.util.Arrays;
004import java.util.List;
005import java.util.Stack;
006
007import swp_compiler_ss13.common.report.ReportLog;
008import swp_compiler_ss13.fuc.parser.grammar.Production;
009import swp_compiler_ss13.fuc.parser.grammar.Terminal;
010import swp_compiler_ss13.fuc.parser.grammar.TokenEx;
011import swp_compiler_ss13.fuc.parser.parser.tables.actions.Reduce;
012
013/**
014 * This interface combines all neccessary grammar specific actions that have to
015 * be performed during the parsing process
016 */
017public interface IGrammarImpl {
018        /**
019         * @param prod The {@link Production} which should be performed reduced
020         * @return The {@link ReduceAction} for the given {@link Reduce}
021         */
022        public ReduceAction getReduceAction(Production prod);
023        
024        /**
025         * Tries to put the parser into a state from where it can recover
026         * itself
027         * 
028         * @param possibleTerminals The terminals that the parser expected
029         * @param curToken The Token that the parser received instead
030         * @param lastToken The Token that the parser receiven before
031         * @param valueStack The current value stack
032         * @return A new instance of {@link RecoveryResult} which tells the
033         *              parser how to change its state
034         */
035        public RecoveryResult tryErrorRecovery(List<Terminal> possibleTerminals,
036                        TokenEx curToken, TokenEx lastToken,
037                        Stack<Object> valueStack);
038        
039        /**
040         * @param reportLog The parsers {@link ReportLog}
041         */
042        public void setReportLog(ReportLog reportLog);
043        
044        
045        /**
046         * The result of a recovery attempt. Contains information for the new
047         * state of the parser
048         */
049        public static class RecoveryResult {
050                private final TokenEx newCurToken;
051                private final List<TokenEx> nextTokens;
052                
053                public RecoveryResult(TokenEx newCurToken,
054                                List<TokenEx> nextTokens) {
055                        this.newCurToken = newCurToken;
056                        this.nextTokens = nextTokens;
057                }
058                
059                public RecoveryResult(TokenEx newCurToken,
060                                TokenEx... nextTokens) {
061                        this(newCurToken, Arrays.asList(nextTokens));
062                }
063                
064                public TokenEx getNewCurToken() {
065                        return newCurToken;
066                }
067                
068                public List<TokenEx> getNextTokens() {
069                        return nextTokens;
070                }
071        }
072}
