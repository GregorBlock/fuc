045        public AST parse(LexerWrapper lexer, ReportLog reportLog,
046                        LRParsingTable table) throws ParserException{
047                Stack<LRParserState> parserStack = new Stack<>();
048
049                AST ast = new ASTImpl();
050                Stack<Object> valueStack = new Stack<>();
051
052                // Initialization
053                ALRAction action = null;
054                TokenEx token = lexer.getNextToken();
055                parserStack.add(table.getStartState());
056
057                WHILE: while (true) {
058                        LRParserState state = parserStack.peek();
059                        log.debug("current token: " + token);
060
061                        TokenType tokenType = token.getTokenType();
062                        switch (tokenType) {
063                        case NOT_A_TOKEN:
064                                List<Token> list = new ArrayList<Token>();
065                                list.add(token);
066                                reportLog.reportError(ReportType.UNRECOGNIZED_TOKEN, list,
067                                                "Found undefined token '" + token.getValue() + "'!");
068                                throw new ParserException("Found undefined token");
069
070                        case COMMENT:
071                                // Skip it silently
072                                token = lexer.getNextToken();
073                                continue WHILE;
074                        }
075
076                        Terminal terminal = token.getTerminal();
077                        if (terminal == null)
078                                throw new ParserException("No Terminal associated with token: " + token);
079                        action = table.getActionTable().get(state, terminal);
080                        if (action == null) {
081                                log.error("Error in Parsetable occured!");
082                                throw new ParserException("An Error in Parsetable occured");
083                        }
084
085                        switch (action.getType()) {
086                        case SHIFT: {
087                                // Shift state
088                                Shift shift = (Shift) action;
089                                log.debug(shift.toString());
090                                parserStack.push(shift.getNewState());
091
092                                // Push old token on stack
093                                valueStack.push(token);
094                                token = lexer.getNextToken();
095                        }
096                        break;
097
098                        case REDUCE: {
099                                // pop reduced states from stack
100                                Reduce reduce = (Reduce) action;
101                                for (int i = 1; i <= reduce.getPopCount(); i++) {
102                                        parserStack.pop();
103                                }
104
105                                // +++++++++++++++++++++++++++++++++++
106                                // get action for reduced production
107                                Production prod = reduce.getProduction();
108                                log.debug(reduce.toString() + " " + prod.toString());
109                                ReduceAction reduceAction = null;
110                                
111
112                                reduceAction = ReduceImpl.getReduceAction(prod, reportLog);
113                                
114
115                                // If there is anything to do on the value stack
116                                // (There might be no reduce-action for Productions like unary
117                                // -> factor, e.g.)
118                                if (reduceAction != null) {
119                                        // Pop all values reduced by this production
120                                        int nrOfValuesReduced = reduce.getPopCount();
121                                        LinkedList<Object> valueHandle = new LinkedList<>();
122                                        for (int i = 0; i < nrOfValuesReduced; i++) {
123                                                valueHandle.addFirst(valueStack.pop());
124                                        }
125                                        
126                                        // Execute reduceAction and push onto the stack
127                                        Object newValue = reduceAction.create(arr(valueHandle));
128                                                                                
129                                        if (newValue == null) {
130                                                log.error("Error occurred! newValue is null");
131                                                throw new ParserException("Error occurred! newValue is null");
132                                        }
133                                        valueStack.push(newValue);
134                                }
135
136                                // check where to go-to... and push next state on stack
137                                LRParserState newState = table.getGotoTable().get(parserStack.peek(),
138                                                prod.getLHS());
139                                if (newState.isErrorState()) {
140                                        List<Token> list = new ArrayList<Token>();
141                                        list.add(token);
142                                        reportLog.reportError(ReportType.UNDEFINED, list, "");
143                                        throw new ParserException("Error state occurred");
144                                }
145                                parserStack.push(newState);
146                                
147                                String stackRep = "Stack:";
148                                for (Object obj : valueStack) {
149                                        stackRep += " " + obj.toString();
150                                }
151                                log.debug(stackRep);
152                        }
153                        break;
154
155                        case ACCEPT: {
156                                if (tokenType != TokenType.EOF) {
157                                        List<Token> list = new ArrayList<Token>();
158                                        list.add(token);
159                                        reportLog.reportError(ReportType.UNRECOGNIZED_TOKEN, list,"");
160                                        throw new ParserException("End of File expected!");
161                                } else {
162                                        BlockNode programBlock = (BlockNode) valueStack.pop();
163                                        ast.setRootNode(programBlock);
164                                        return ast;
165                                }
166                        }
167
168                        case ERROR: {
169                                Error error = (Error) action;
170                                List<Token> list = new ArrayList<Token>();
171                                list.add(token);
172                                reportLog.reportError(ReportType.UNDEFINED,list,
173                                                "An error occurred: " + error.getMsg());
174                                                throw new ParserException("Got Error State from Actiontable");
175                        }
176                        }
177                }
178        }
