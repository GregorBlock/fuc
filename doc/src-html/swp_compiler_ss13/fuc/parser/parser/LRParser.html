037        private final Logger log = Logger.getLogger(getClass());
038        
039        /** Used to store Tokens that are (re-)inserted into the token stream */
040        private final LinkedList<TokenEx> errorTokenStream = new LinkedList<>();
041        private TokenEx lastToken = new TokenEx("", TokenType.NOT_A_TOKEN, 0, 0, null);
042

044        // --- constructors
045        // ---------------------------------------------------------

052        @SuppressWarnings("incomplete-switch")
053        public AST parse(LexerWrapper lexer, ReportLog reportLog,
054                        LRParsingTable table, IGrammarImpl grammarImpl) throws ParserException{
055                grammarImpl.setReportLog(reportLog);
056                
057                Stack<LRParserState> parserStack = new Stack<>();
058
059                AST ast = new ASTImpl();
060                Stack<Object> valueStack = new Stack<>();
061
062                // Initialization
063                ALRAction action = null;
064                TokenEx token = getNextToken(lexer);
065                parserStack.add(table.getStartState());
066
067                WHILE: while (true) {
068                        LRParserState state = parserStack.peek();
069                        log.debug("current token: " + token);
070
071                        TokenType tokenType = token.getTokenType();
072                        switch (tokenType) {
073                        case NOT_A_TOKEN:
074                                List<Token> list = new ArrayList<Token>();
075                                list.add(token);
076                                reportLog.reportError(ReportType.UNRECOGNIZED_TOKEN, list,
077                                                "Found undefined token '" + token.getValue() + "'!");
078                                throw new ParserException("Found undefined token");
079
080                        case COMMENT:
081                                // Skip it silently
082                                lastToken = token;
083                                token = getNextToken(lexer);
084                                continue WHILE;
085                        }
086
087                        Terminal terminal = token.getTerminal();
088                        if (terminal == null)
089                                throw new ParserException("No Terminal associated with token: " + token);
090                        action = table.getActionTable().get(state, terminal);
091                        if (action == null) {
092                                log.error("Error in Parsetable occured!");
093                                throw new ParserException("An Error in Parsetable occured");
094                        }
095
096                        switch (action.getType()) {
097                        case SHIFT: {
098                                // Shift state
099                                Shift shift = (Shift) action;
100                                
101                                TokenEx newToken = getNextToken(lexer);
102                                log.debug(shift.toString() + " : " + newToken);
103                                
104                                parserStack.push(shift.getNewState());
105                                valueStack.push(token);
106                                
107                                lastToken = token;
108                                token = newToken;
109                        }
110                        break;
111
112                        case REDUCE: {
113                                // Print current value stack
114                                log.debug(printStack(valueStack));
115                                
116                                // pop reduced states from stack
117                                Reduce reduce = (Reduce) action;
118                                for (int i = 1; i <= reduce.getPopCount(); i++) {
119                                        parserStack.pop();
120                                }
121
122                                // +++++++++++++++++++++++++++++++++++
123                                // get action for reduced production
124                                Production prod = reduce.getProduction();
125                                log.debug(reduce.toString());
126                                
127                                ReduceAction reduceAction = grammarImpl.getReduceAction(prod);
128
129                                // If there is anything to do on the value stack
130                                // (There might be no reduce-action for Productions like unary
131                                // -> factor, e.g.)
132                                if (reduceAction != null) {
133                                        // Pop all values reduced by this production
134                                        int nrOfValuesReduced = reduce.getPopCount();
135                                        LinkedList<Object> valueHandle = new LinkedList<>();
136                                        for (int i = 0; i < nrOfValuesReduced; i++) {
137                                                valueHandle.addFirst(valueStack.pop());
138                                        }
139                                        
140                                        // (Safely) execute reduceAction and push onto the stack
141                                        Object newValue = null;
142                                        try {
143                                                newValue = reduceAction.create(arr(valueHandle));
144                                        } catch (ReduceException err) {
145                                                String errMsg = writeReportError(reportLog, reduce, err);
146                                                throw new ParserException("An error occured during " + reduce + ": " + errMsg, err);
147                                        } catch (ParserException err) {
148                                                throw err;      // Re-Throw
149                                        } catch (Exception err) {
150                                                throw new ParserException("An error occured during " + reduce + ": ", err);
151                                        }
152                                        
153                                        if (newValue == null) {
154                                                log.error("Error occurred! newValue is null");
155                                                throw new ParserException("Error occurred! newValue is null");
156                                        }
157                                        valueStack.push(newValue);
158                                }
159
160                                // check where to go-to... and push next state on stack
161                                LRParserState newState = table.getGotoTable().get(parserStack.peek(),
162                                                prod.getLHS());
163                                if (newState.isErrorState()) {
164                                        List<Token> list = new ArrayList<Token>();
165                                        list.add(token);
166                                        reportLog.reportError(ReportType.UNDEFINED, list, "");
167                                        throw new ParserException("Error state occurred");
168                                }
169                                parserStack.push(newState);
170                        }
171                        break;
172
173                        case ACCEPT: {
174                                if (tokenType != TokenType.EOF) {
175                                        List<Token> list = new ArrayList<Token>();
176                                        list.add(token);
177                                        reportLog.reportError(ReportType.UNRECOGNIZED_TOKEN, list,"");
178                                        throw new ParserException("End of File expected!");
179                                } else {
180                                        BlockNode programBlock = (BlockNode) valueStack.pop();
181                                        ast.setRootNode(programBlock);
182                                        return ast;
183                                }
184                        }
185
186                        case ERROR: {
187                                // Try to recover from misjump
188                                List<Terminal> possibleTerminals = table.getActionTable().getPossibleTerminalsFor(state);
189                                RecoveryResult result = grammarImpl.tryErrorRecovery(possibleTerminals, token, lastToken, valueStack);
190                                if (result != null) {
191                                        // Apply the new state
192                                        token = result.getNewCurToken();
193                                        errorTokenStream.addAll(result.getNextTokens());
194                                        
195                                        // Okay, parser should be in a proper state again... give it a shot!
196                                        continue WHILE;
197                                }
198                                
199                                // No chance. Print 
200//                              Error error = (Error) action;
201                                List<Token> list = new ArrayList<Token>();
202                                list.add(token);
203                                reportLog.reportError(ReportType.WORD_NOT_IN_GRAMMAR,list,
204                                                "An error occurred: Expected one of " + possibleTerminals.toString() + ", but got: '" + token.getTerminal() + "'");
205                                                throw new ParserException("Got Error State from Actiontable");
206                        }
207                        }
208                }
209        }
210        
211        
212        private TokenEx getNextToken(LexerWrapper lexer) {
213                if (!errorTokenStream.isEmpty()) {
214                        return errorTokenStream.poll();
215                } else {
216                        return lexer.getNextToken();
217                }
218        }

222        /**
223         * Gets ReportLog, the Object, thats made some trouble and the message whats expected instead
224         * 
225         * @param reportLog
226         * @param reduce
227         * @param err
228         * @return The error message
229         */
230        private String writeReportError(ReportLog reportLog, Reduce reduce, ReduceException err) {
231                Object obj = err.getObj();
232                Class<?> clazz = err.getClazz();
233                String objStr = obj == null ? "<null>" : obj.getClass().getSimpleName();
234                
235                List<Token> tokens = null;
236                if (obj instanceof ASTNode){
237                        ASTNode node = (ASTNode) obj;
238                        tokens = node.coverage();
239                } else if (obj instanceof Token){
240                        tokens = Arrays.asList((Token) obj);
241                }
242                
243                String result = "Expected an instance of " +
244                                clazz.getSimpleName() + " during " + reduce + ", but got " + objStr + " instead!";
245                reportLog.reportError(ReportType.UNDEFINED, tokens, result);
246                return result;
247        }
248
249        private static Object[] arr(List<Object> objs) {
250                return objs.toArray(new Object[objs.size()]);
251        }
252
253        private static String printStack(Stack<Object> stack) {
254                StringBuilder b = new StringBuilder("Stack:");
255                for (Object obj : stack) {
256                        b.append(" ").append(obj.toString());
257                }
258                return b.toString();
259        }
