001package swp_compiler_ss13.fuc.parser.parser;
002
003import swp_compiler_ss13.common.report.ReportLog;
004
005public interface ReduceAction {
006        
007        /**
008         * Reduce the given Objects in Array of ASTnodes and Tokens to a
009         * new ASTNode and return this
010         * @param objs
011         * @return The result of the reduction
012         * @throws ReduceException If there was an error which has been reported to
013         *              {@link ReportLog} and needs no cancellation
014         * @throws ParserException Iff the reduce aborted anormally and parsing
015         *              should be cancelled.
016         */
017        Object create(Object... objs) throws ReduceException, ParserException;
018        
019
020        
021        public static class ReduceException extends RuntimeException {
022                private static final long serialVersionUID = -1577684340153865593L;
023                
024                private final Class<?> clazz;
025                private final Object obj;
026                
027                public ReduceException(Object obj, Class<?> clazz, Throwable cause) {
028                        super(cause);
029                        this.obj = obj;
030                        this.clazz = clazz;
031                }
032                
033                public Class<?> getClazz() {
034                        return clazz;
035                }
036                
037                public Object getObj() {
038                        return obj;
039                }
040        }
041}
