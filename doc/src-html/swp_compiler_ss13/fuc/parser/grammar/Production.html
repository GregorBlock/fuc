001package swp_compiler_ss13.fuc.parser.grammar;
002
003import java.util.Arrays;
004import java.util.List;
005
006import swp_compiler_ss13.fuc.parser.util.It;
007
008
009
010/**
011 * A production consists of a left hand side {@link NonTerminal} and a list of right hand side {@link Symbol}s.
012 * 
013 * If there are productions with more than one right hand sides
014 * (in parser books separated by "|"), each of these sides
015 * must be represented by its own {@link Production} instance.
016 */
017public final class Production {
018   private int id;
019   private NonTerminal lhs;
020   private List<Symbol> rhs;
021   
022   // cache
023   private Terminal lastTerminal;
024//   private Terminal precTerminal; // save the Terminal with the precedence of this Production
025   private int rhsSizeWithoutEpsilon;
026   private int hashCode;
027   
028   
029   /**
030    * Creates a new {@link Production}.
031    */
032   public Production(int id, NonTerminal lhs, List<Symbol> rhs) {
033      // check rhs
034      if (rhs.size() == 0) {
035         // empty RHS means epsilon
036         rhs = Arrays.asList((Symbol) Terminal.Epsilon);
037         this.rhsSizeWithoutEpsilon = 0;
038      } else if (rhs.size() == 1) {
039         this.rhsSizeWithoutEpsilon = (rhs.get(0) == Terminal.Epsilon ? 0 : 1);
040      } else if (rhs.size() > 1) {
041         // check, that there is no epsilon in a RHS with at least 2 symbols
042         for (Symbol symbol : rhs) {
043            if (symbol == Terminal.Epsilon) {
044               throw new IllegalArgumentException("Epsilon is only allowed as the single symbol of a RHS!");
045            }
046         }
047         this.rhsSizeWithoutEpsilon = rhs.size();
048      }
049      this.id = id;
050      this.lhs = lhs;
051      this.rhs = rhs;
052      
053      // cache
054      // find last terminal
055      Terminal lastTerminal = null;
056      for (int i = rhs.size() - 1; i >= 0; i--) {
057         if (rhs.get(i) instanceof Terminal) {
058            lastTerminal = (Terminal) rhs.get(i);
059            break;
060         }
061      }
062      this.lastTerminal = lastTerminal;
063//      this.precTerminal = (precTerminal != null) ? precTerminal : lastTerminal;
064      
065      // hash code
066      computeHashcode();
067   }
068   
069   
070   public Production(int id, NonTerminal lhs, Symbol... rhs) {
071      this(id, lhs, Arrays.asList(rhs));
072   }
073   
074   
075   public int getID() {
076      return id;
077   }
078   
079   
080   public NonTerminal getLHS() {
081      return lhs;
082   }
083   
084   
085   public List<Symbol> getRHS() {
086      return rhs;
087   }
088   
089   
090   /**
091    * @return The number of right hand side symbols, without epsilons.
092    */
093   public int getRHSSize() {
094      return rhsSizeWithoutEpsilon;
095   }
096   
097   
098   /**
099    * @return Last terminal of RHS or <code>null</code>
100    */
101   public Terminal getLastTerminal() {
102      return lastTerminal;
103   }
104   
105   
106   @Override
107   public boolean equals(Object obj) {
108      if (obj instanceof Production) {
109         Production p = (Production) obj;
110         if (lhs != p.lhs)
111            return false;
112         if (!rhs.equals(p.rhs))
113            return false;
114         return true;
115      }
116      return false;
117   }
118   
119   
120   private int computeHashcode() {
121      int sum = lhs.hashCode(); // default hashCode is the object address
122      for (Symbol s : rhs) {
123         sum = (sum + s.hashCode()) % ((1 << 31) - 1);
124      }
125      this.hashCode = sum;
126      return this.hashCode;
127   }
128   
129   @Override
130   public int hashCode() {
131      return hashCode;
132   }
133   
134   
135   public String getStringRep() {
136      String result = lhs.toString() + " ->";
137      for (Symbol s : rhs) {
138         result += " " + s;
139      }
140      return result;
141   }
142   
143   
144   @Override
145   public String toString() {
146      StringBuilder rhsString = new StringBuilder();
147      for (Symbol s : rhs)
148         rhsString.append(s + " ");
149      return lhs + " → " + rhsString;
150   }
151   
152   
153   public String toString(int dotPosition) {
154      StringBuilder rhsString = new StringBuilder();
155      It<Symbol> rhsSymbols = new It<Symbol>(rhs);
156      for (Symbol s : rhsSymbols) {
157         if (dotPosition == rhsSymbols.getIndex())
158            rhsString.append(" .");
159         rhsString.append(" " + s);
160      }
161      if (dotPosition == rhs.size())
162         rhsString.append(" .");
163      return lhs + " →" + rhsString;
164   }
165}
