001package swp_compiler_ss13.fuc.parser.grammar;
002
003import static swp_compiler_ss13.fuc.parser.grammar.NonTerminal.StartLHS;
004import static swp_compiler_ss13.fuc.parser.grammar.Terminal.EOF;
005
006import java.util.LinkedList;
007import java.util.List;
008
009
010/**
011 * This class represents a context free grammar consisting of
012 * {@link Terminal}s, {@link Production}s and {@link NonTerminal}s.
013 * 
014 * @author Gero
015 */
016public class Grammar {
017   private final List<Terminal> terminals;
018   private final List<NonTerminal> nonTerminals;
019   private final List<Production> productions;
020   
021   
022   public Grammar(List<Terminal> terminals, List<NonTerminal> nonTerminals, List<Production> productions) {
023      this.terminals = terminals;
024      this.nonTerminals = nonTerminals;
025      this.productions = productions;
026   }
027   
028   
029   /**
030    * Adds auxiliary start production "S' â†’ S$" (Using
031    * {@link NonTerminal#StartLHS}). If the grammar already contains a $
032    * terminal, it is returned unmodified.
033    */
034   public Grammar extendByAuxStartProduction() {
035      // Already augmented?
036      if (terminals.contains(EOF)) {
037         // yes. return unmodified grammar
038         return this;
039      } else {
040         // No. Add StartLHS (S') and $ (EndOfInputStream)
041         LinkedList<Terminal> terminals = new LinkedList<Terminal>();
042         terminals.addAll(this.terminals);
043         terminals.add(EOF);
044         
045         // Add StartLHS to the non-terminals list
046         LinkedList<NonTerminal> nonTerminals = new LinkedList<NonTerminal>();
047         nonTerminals.addAll(this.nonTerminals);
048         nonTerminals.add(StartLHS);
049         
050         // Add start production to productions list
051         Production startProduction = new Production(0, StartLHS, productions.get(0).getLHS(), EOF);
052         LinkedList<Production> productions = new LinkedList<Production>();
053         productions.add(startProduction);
054         productions.addAll(this.productions);
055         
056         // Create and return grammar
057         return new Grammar(terminals, nonTerminals, productions);
058      }
059   }
060   
061   
062   public List<Terminal> getTerminals() {
063      return terminals;
064   }
065   
066   
067   public List<NonTerminal> getNonTerminals() {
068      return nonTerminals;
069   }
070   
071   
072   public Production getStartProduction() {
073      return productions.get(0);
074   }
075   
076   
077   public List<Production> getProductions() {
078      return productions;
079   }
080}
