001package swp_compiler_ss13.fuc.parser.generator.automaton;
002
003import swp_compiler_ss13.fuc.parser.generator.items.LR0Item;
004import swp_compiler_ss13.fuc.parser.grammar.Symbol;
005
006/**
007 * A part of the {@link Dfa} connecting two states of type <S> with a
008 * {@link Symbol}.
009 * 
010 * @author Gero
011 * 
012 * @param <S> The type of the states this edge connects with each other
013 */
014public class DfaEdge<S> {
015   // --------------------------------------------------------------------------
016   // --- variables and constants ----------------------------------------------
017   // --------------------------------------------------------------------------
018   private final S src;
019   private final Symbol symbol;
020   private final S dst;
021   private final LR0Item srcItem;
022   
023   // cache
024   private final int hashCode;
025   
026   // --------------------------------------------------------------------------
027   // --- constructors ---------------------------------------------------------
028   // --------------------------------------------------------------------------
029   public DfaEdge(S src, Symbol symbol, S dest, LR0Item srcItem) {
030      this.src = src;
031      this.symbol = symbol;
032      this.dst = dest;
033      this.srcItem = srcItem;
034      
035      // Identity by src x symbol
036      this.hashCode = src.hashCode() * 100 + symbol.hashCode();
037   }
038   
039   public static <S> DfaEdge<S> createAcceptEdge(S src, Symbol symbol)
040   {
041      return new DfaEdge<S>(src, symbol, null, null);
042   }
043   
044   // --------------------------------------------------------------------------
045   // --- getter/setter --------------------------------------------------------
046   // --------------------------------------------------------------------------
047   public S getSrc() {
048      return src;
049   }
050
051   public Symbol getSymbol() {
052      return symbol;
053   }
054
055   public S getDst() {
056      return dst;
057   }
058   
059   public LR0Item getSrcItem() {
060      return srcItem;
061   }
062   
063   public boolean isDestAccepting() {
064      return dst == null;
065   }
066   
067   
068   @Override
069   public int hashCode() {
070      return hashCode;
071   }
072
073   @Override
074   public boolean equals(Object obj) {
075      if (this == obj)
076         return true;
077      if (obj == null)
078         return false;
079      if (getClass() != obj.getClass())
080         return false;
081      DfaEdge<?> other = (DfaEdge<?>) obj;
082      if (src == null) {
083         if (other.src != null)
084            return false;
085      } else if (!src.equals(other.src))
086         return false;
087      if (symbol == null) {
088         if (other.symbol != null)
089            return false;
090      } else if (!symbol.equals(other.symbol))
091         return false;
092      return true;
093   }
094
095   @Override
096   public String toString() {
097      return src + " " + symbol;
098   }
099}
