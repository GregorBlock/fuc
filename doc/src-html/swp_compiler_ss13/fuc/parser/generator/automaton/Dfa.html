001package swp_compiler_ss13.fuc.parser.generator.automaton;
002
003import java.util.LinkedHashSet;
004import java.util.LinkedList;
005import java.util.List;
006import java.util.Set;
007
008import swp_compiler_ss13.fuc.parser.generator.items.Item;
009import swp_compiler_ss13.fuc.parser.generator.states.AState;
010
011
012/**
013 * A deterministic finite automaton (DFA) as a interim step for the generation
014 * of parsing tables. Generators create DFAs first from the grammar, which are
015 * then converted to actions by moving along their edges.
016 * 
017 * @author Gero
018 * 
019 * @param <I> The item type the states consists of 
020 * @param <S> The states type of this automaton
021 */
022public class Dfa<I extends Item, S extends AState<I>> {
023   // --------------------------------------------------------------------------
024   // --- variables and constants ----------------------------------------------
025   // --------------------------------------------------------------------------
026   private final Set<S> states = new LinkedHashSet<>();
027   private final Set<DfaEdge<S>> edges = new LinkedHashSet<>();
028   private final S startState;
029   
030   // --------------------------------------------------------------------------
031   // --- constructors ---------------------------------------------------------
032   // --------------------------------------------------------------------------
033   /**
034    * @see Dfa
035    * @param startState
036    */
037   public Dfa(S startState) {
038      states.add(startState);
039      this.startState = startState;
040   }
041   
042   // --------------------------------------------------------------------------
043   // --- methods --------------------------------------------------------------
044   // --------------------------------------------------------------------------
045   /**
046    * @param state
047    * @return All edges that start form the given state
048    */
049   public List<DfaEdge<S>> getEdgesFrom(S state) {
050      LinkedList<DfaEdge<S>> result = new LinkedList<>();
051      for (DfaEdge<S> edge : edges) {
052         if (edge.getSrc().equals(state)) {
053            result.add(edge);
054         }
055      }
056      return result;
057   }
058
059   /**
060    * @param state
061    * @return All edges that end in the given state
062    */
063   public List<DfaEdge<S>> getEdgesTo(S state) {
064      LinkedList<DfaEdge<S>> result = new LinkedList<>();
065      for (DfaEdge<S> edge : edges) {
066         if (edge.getDst().equals(state)) {
067            result.add(edge);
068         }
069      }
070      return result;
071   }
072   
073   // --------------------------------------------------------------------------
074   // --- getter/setter --------------------------------------------------------
075   // --------------------------------------------------------------------------
076   /**
077    * @return All states this automaton consists of
078    */
079   public Set<S> getStates() {
080      return states;
081   }
082
083   /**
084    * @return All edges this automaton consists of
085    */
086   public Set<DfaEdge<S>> getEdges() {
087      return edges;
088   }
089
090   /**
091    * @return The start state of this automaton
092    */
093   public S getStartState() {
094      return startState;
095   }
096}
