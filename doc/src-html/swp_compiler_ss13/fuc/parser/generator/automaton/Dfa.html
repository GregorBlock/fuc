001package swp_compiler_ss13.fuc.parser.generator.automaton;
002
003import java.util.HashSet;
004import java.util.LinkedList;
005import java.util.List;
006import java.util.Set;
007
008import swp_compiler_ss13.fuc.parser.generator.items.Item;
009import swp_compiler_ss13.fuc.parser.generator.states.AState;
010
011
012/**
013 * A deterministic finite automaton (DFA) as a interim step for the generation
014 * of parsing tables. Generators create DFAs first from the grammar, which are
015 * then converted to actions by moving along their edges.
016 * 
017 * @author Gero
018 * 
019 * @param <I> The item type the states consists of 
020 * @param <S> The states type of this automaton
021 */
022public class Dfa<I extends Item, S extends AState<I>> {
023   // --------------------------------------------------------------------------
024   // --- variables and constants ----------------------------------------------
025   // --------------------------------------------------------------------------
026   private final Set<S> states = new HashSet<>();
027   private final Set<DfaEdge<S>> edges = new HashSet<>();
028   private final S startState;
029   
030   // --------------------------------------------------------------------------
031   // --- constructors ---------------------------------------------------------
032   // --------------------------------------------------------------------------
033   public Dfa(S startState) {
034      states.add(startState);
035      this.startState = startState;
036   }
037   
038   // --------------------------------------------------------------------------
039   // --- methods --------------------------------------------------------------
040   // --------------------------------------------------------------------------
041   public List<DfaEdge<S>> getEdgesFrom(S state) {
042      LinkedList<DfaEdge<S>> result = new LinkedList<>();
043      for (DfaEdge<S> edge : edges) {
044         if (edge.getSrc().equals(state)) {
045            result.add(edge);
046         }
047      }
048      return result;
049   }
050   
051   public List<DfaEdge<S>> getEdgesTo(S state) {
052      LinkedList<DfaEdge<S>> result = new LinkedList<>();
053      for (DfaEdge<S> edge : edges) {
054         if (edge.getDst().equals(state)) {
055            result.add(edge);
056         }
057      }
058      return result;
059   }
060   
061   // --------------------------------------------------------------------------
062   // --- getter/setter --------------------------------------------------------
063   // --------------------------------------------------------------------------
064   public Set<S> getStates() {
065      return states;
066   }
067   
068   public Set<DfaEdge<S>> getEdges() {
069      return edges;
070   }
071   
072   public S getStartState() {
073      return startState;
074   }
075}
