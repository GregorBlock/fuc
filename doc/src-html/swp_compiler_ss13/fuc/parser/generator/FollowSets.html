001package swp_compiler_ss13.fuc.parser.generator;
002
003import java.util.HashMap;
004import java.util.ListIterator;
005import java.util.Map;
006
007import swp_compiler_ss13.fuc.parser.generator.terminals.EfficientTerminalSet;
008import swp_compiler_ss13.fuc.parser.grammar.Grammar;
009import swp_compiler_ss13.fuc.parser.grammar.NonTerminal;
010import swp_compiler_ss13.fuc.parser.grammar.Production;
011import swp_compiler_ss13.fuc.parser.grammar.Symbol;
012import swp_compiler_ss13.fuc.parser.grammar.Terminal;
013
014/**
015 * Calculates and caches the follow sets for the given grammar
016 * 
017 * @author Gero
018 */
019public class FollowSets {
020        // --------------------------------------------------------------------------
021        // --- variables and constants
022        // ----------------------------------------------
023        // --------------------------------------------------------------------------
024        private final Grammar grammar;
025        private final Map<NonTerminal, EfficientTerminalSet> followSets = new HashMap<>();
026
027        // --------------------------------------------------------------------------
028        // --- constructors
029        // ---------------------------------------------------------
030        // --------------------------------------------------------------------------
031        public FollowSets(Grammar grammar, FirstSets firstSets,
032                        NullableSet nullableSet) {
033                this.grammar = grammar;
034
035                // Use fix point iteration to determine all follow sets
036                boolean changed = false;
037                do {
038                        changed = false; // reset
039
040                        for (Production prod : grammar.getProductions()) {
041                                NonTerminal lhs = prod.getLHS();
042
043                                // now iterate through the right side from right to left:
044                                ListIterator<Symbol> itRight = prod.getRHS().listIterator(
045                                                prod.getRHS().size());
046                                EfficientTerminalSet FOLLOWToAdd = getFollowSet(lhs);
047
048                                while (itRight.hasPrevious()) { // right (step through right
049                                                                                                // side of the production)
050                                        Symbol right = itRight.previous();
051                                        if (right.isTerminal()) {
052                                                // TERMINAL: Clear FOLLOW, add it as only element
053                                                FOLLOWToAdd = FOLLOWToAdd.empty()
054                                                                .plus((Terminal) right);
055                                                // FOLLOWToAdd.clear(); FOLLOWToAdd.add((Terminal
056                                                // )right);
057                                        } else {
058                                                // NONTERMINAL:
059                                                NonTerminal varRight = (NonTerminal) right;
060
061                                                // 1. add FOLLOWToAdd to varRight.getFOLLOW():
062                                                // varRight.getFOLLOW().addAll(FOLLOWToAdd);
063                                                EfficientTerminalSet rightFollowOld = getFollowSet(varRight);
064                                                EfficientTerminalSet rightFollowNew = addToFollow(
065                                                                varRight, FOLLOWToAdd);
066                                                // iCard += (varRight.getFOLLOW().size() - oldSize);
067                                                changed |= (!rightFollowNew.equals(rightFollowOld)); // Changed
068                                                                                                                                                                // anything?
069
070                                                // 2. prepare new FOLLOWToAdd:
071                                                if (!nullableSet.contains(varRight)) {
072                                                        FOLLOWToAdd = FOLLOWToAdd.empty();
073                                                } else {
074                                                        // If varRight can produce EPSILON (= isNullable()),
075                                                        // we keep our follow-set and try to add it to
076                                                        // the next (Non?)Terminal on the left
077                                                }
078
079                                                // Set<Terminal> currentFIRST = new
080                                                // HashSet<Terminal>(varRight.getFIRST());
081                                                EfficientTerminalSet curFirst = firstSets.get(varRight);
082                                                // currentFIRST.remove(eps);
083                                                curFirst = curFirst.empty().plusAllExceptEpsilon(
084                                                                curFirst);
085                                                // FOLLOWToAdd.addAll(currentFIRST);
086                                                FOLLOWToAdd = FOLLOWToAdd.plusAll(curFirst);
087                                                // if (varRight.getFIRST().contains(eps)) {
088                                                // FOLLOWToAdd.addAll(varRight.getFOLLOW());
089                                                // } ==> see above!
090                                        }
091                                }
092                        }
093                } while (changed);
094        }
095
096        private EfficientTerminalSet addToFollow(NonTerminal key,
097                        EfficientTerminalSet terminalSet) {
098                EfficientTerminalSet set = getFollowSet(key);
099                set = set.plusAll(terminalSet);
100                followSets.put(key, set); // Needed, as TerminalSets are immutable and
101                                                                        // plus(..) creates new instance
102                return set;
103        }
104
105        private EfficientTerminalSet getFollowSet(NonTerminal key) {
106                EfficientTerminalSet set = followSets.get(key);
107                if (set == null) {
108                        set = new EfficientTerminalSet(grammar.getTerminals());
109                }
110                return set;
111        }
112
113        // --------------------------------------------------------------------------
114        // --- getter/setter
115        // --------------------------------------------------------
116        // --------------------------------------------------------------------------
117        /**
118         * Gets the FOLLOW set for the given symbol.
119         */
120        public EfficientTerminalSet get(NonTerminal key) {
121                return followSets.get(key);
122        }
123}
