001package swp_compiler_ss13.fuc.parser.generator.states;
002
003import java.util.Collection;
004import java.util.Iterator;
005
006import swp_compiler_ss13.fuc.parser.generator.GrammarInfo;
007import swp_compiler_ss13.fuc.parser.grammar.Symbol;
008
009/**
010 * The base class for all types of LR-states. Simply extends the {@link AState}
011 * with methods for goto- and closure- calculation.
012 * 
013 * @author Gero
014 * 
015 * @param <I>
016 */
017public abstract class ALRState<I> extends AState<I> {
018
019        /**
020         * @see ALRState
021         * @param items Added by the ordering of the collections {@link Iterator}!
022         */
023        public ALRState(Collection<I> items) {
024                super(items);
025        }
026
027        @SafeVarargs
028        public ALRState(I... items) {
029                super(items);
030        }
031
032        /**
033         * Moves the position if the items one step further when the given symbol
034         * follows and returns them as a new state (only the kernel, without
035         * closure).
036         */
037        public abstract ALRState<I> goTo(Symbol symbol);
038
039        /**
040         * Returns the closure of this {@link LR0State} as another {@link LR0State}.
041         * See dragon book, p. 271
042         */
043        public abstract ALRState<I> closure(GrammarInfo grammarInfo);
044}
