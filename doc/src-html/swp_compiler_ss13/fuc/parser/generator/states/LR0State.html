001package swp_compiler_ss13.fuc.parser.generator.states;
002
003import java.util.Collection;
004import java.util.LinkedHashSet;
005import java.util.LinkedList;
006
007import swp_compiler_ss13.fuc.parser.generator.GrammarInfo;
008import swp_compiler_ss13.fuc.parser.generator.items.LR0Item;
009import swp_compiler_ss13.fuc.parser.grammar.NonTerminal;
010import swp_compiler_ss13.fuc.parser.grammar.Production;
011import swp_compiler_ss13.fuc.parser.grammar.Symbol;
012
013/**
014 * Implements {@link #goTo(Symbol)} and {@link #closure(GrammarInfo)} from
015 * {@link ALRState} for LR0 grammar generation.
016 * 
017 * @author Gero
018 */
019public class LR0State extends ALRState<LR0Item> {
020        // --------------------------------------------------------------------------
021        // --- variables and constants
022        // ----------------------------------------------
023        // --------------------------------------------------------------------------
024        // cache
025        private LR0State closureCache = null;
026
027        // --------------------------------------------------------------------------
028        // --- constructors
029        // ---------------------------------------------------------
030        // --------------------------------------------------------------------------
031        /**
032         * @see LR0State
033         * @param items
034         */
035        public LR0State(Collection<LR0Item> items) {
036                super(items);
037        }
038
039        /**
040         * @see LR0State
041         * @param items
042         */
043        public LR0State(LR0Item... items) {
044                super(items);
045        }
046
047        // --------------------------------------------------------------------------
048        // --- methods
049        // --------------------------------------------------------------
050        // --------------------------------------------------------------------------
051        @Override
052        public LR0State goTo(Symbol symbol) {
053                LinkedHashSet<LR0Item> ret = new LinkedHashSet<LR0Item>();
054                // Find all items where the given symbol follows and add them shifted
055                for (LR0Item item : getItems()) {
056                        if (item.getNextSymbol() == symbol) {
057                                ret.add(item.shift());
058                        }
059                }
060                return new LR0State(ret);
061        }
062
063        @Override
064        public LR0State closure(GrammarInfo grammarInfo) {
065                if (closureCache == null) {
066                        closureCache = calcClosure(grammarInfo);
067                }
068                return closureCache;
069        }
070
071        /**
072         * @param grammarInfo
073         * @return The closure for this state as a new {@link LR0State}
074         */
075        private LR0State calcClosure(GrammarInfo grammarInfo) {
076                // The closure contains all items of the source...
077                LinkedHashSet<LR0Item> result = new LinkedHashSet<LR0Item>();
078                result.addAll(items);
079
080                // ... and the following ones (for any item "A → α.Xβ" and any "X → γ"
081                // add "X → .γ")
082                LinkedList<LR0Item> queue = new LinkedList<LR0Item>();
083                queue.addAll(items);
084                while (!queue.isEmpty()) {
085                        LR0Item it = queue.removeFirst();
086                        Symbol nextSymbol = it.getNextSymbol();
087                        if (it.isShiftable() && nextSymbol.isNonTerminal()) {
088                                for (Production p : grammarInfo
089                                                .getProductionsFrom((NonTerminal) nextSymbol)) {
090                                        LR0Item newItem = new LR0Item(p, 0);
091                                        if (!result.contains(newItem)) {
092                                                result.add(newItem);
093                                                queue.addLast(newItem);
094                                        }
095                                }
096                        }
097                }
098                return new LR0State(result);
099        }
100
101        // --------------------------------------------------------------------------
102        // --- getter/setter
103        // --------------------------------------------------------
104        // --------------------------------------------------------------------------
105}
