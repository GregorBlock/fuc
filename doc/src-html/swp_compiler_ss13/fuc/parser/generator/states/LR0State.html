001package swp_compiler_ss13.fuc.parser.generator.states;
002
003import java.util.Collection;
004import java.util.HashSet;
005import java.util.LinkedList;
006
007import swp_compiler_ss13.fuc.parser.generator.GrammarInfo;
008import swp_compiler_ss13.fuc.parser.generator.items.LR0Item;
009import swp_compiler_ss13.fuc.parser.grammar.NonTerminal;
010import swp_compiler_ss13.fuc.parser.grammar.Production;
011import swp_compiler_ss13.fuc.parser.grammar.Symbol;
012
013
014/**
015 * Implements {@link #goTo(Symbol)} and {@link #closure(GrammarInfo)} from
016 * {@link ALRState} for LR0 grammar generation.
017 * 
018 * @author Gero
019 */
020public class LR0State extends ALRState<LR0Item> {
021   // --------------------------------------------------------------------------
022   // --- variables and constants ----------------------------------------------
023   // --------------------------------------------------------------------------
024   // cache
025   private LR0State closureCache = null;
026   
027   // --------------------------------------------------------------------------
028   // --- constructors ---------------------------------------------------------
029   // --------------------------------------------------------------------------
030   public LR0State(Collection<LR0Item> items) {
031      super(items);
032   }
033   
034   public LR0State(LR0Item... items) {
035      super(items);
036   }
037   
038   
039   // --------------------------------------------------------------------------
040   // --- methods --------------------------------------------------------------
041   // --------------------------------------------------------------------------
042   @Override
043   public LR0State goTo(Symbol symbol) {
044      HashSet<LR0Item> ret = new HashSet<LR0Item>();
045      // Find all items where the given symbol follows and add them shifted
046      for (LR0Item item : getItems()) {
047         if (item.getNextSymbol() == symbol) {
048            ret.add(item.shift());
049         }
050      }
051      return new LR0State(ret);
052   }
053   
054   
055   @Override
056   public LR0State closure(GrammarInfo grammarInfo) {
057      if (closureCache == null) {
058         closureCache = calcClosure(grammarInfo);
059      }
060      return closureCache;
061   }
062   
063   private LR0State calcClosure(GrammarInfo grammarInfo) {
064      // The closure contains all items of the source...
065      HashSet<LR0Item> result = new HashSet<LR0Item>();
066      result.addAll(items);
067      
068      // ... and the following ones (for any item "A → α.Xβ" and any "X → γ" add "X → .γ")
069      LinkedList<LR0Item> queue = new LinkedList<LR0Item>();
070      queue.addAll(items);
071      while (!queue.isEmpty()) {
072         LR0Item it = queue.removeFirst();
073         Symbol nextSymbol = it.getNextSymbol();
074         if (it.isShiftable() && nextSymbol.isNonTerminal()) {
075            for (Production p : grammarInfo.getProductionsFrom((NonTerminal) nextSymbol)) {
076               LR0Item newItem = new LR0Item(p, 0);
077               if (!result.contains(newItem)) {
078                  result.add(newItem);
079                  queue.addLast(newItem);
080               }
081            }
082         }
083      }
084      return new LR0State(result);
085   }
086   
087   // --------------------------------------------------------------------------
088   // --- getter/setter --------------------------------------------------------
089   // --------------------------------------------------------------------------
090}
