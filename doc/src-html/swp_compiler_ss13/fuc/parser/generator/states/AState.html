001package swp_compiler_ss13.fuc.parser.generator.states;
002
003import java.util.Collection;
004import java.util.Iterator;
005import java.util.LinkedHashSet;
006
007import swp_compiler_ss13.fuc.parser.generator.items.Item;
008import swp_compiler_ss13.fuc.parser.util.It;
009
010/**
011 * This is the base class for generator-created parser states. It simply
012 * consists of a fixed set of {@link Item}s
013 * 
014 * @author Gero
015 * 
016 * @param <I>
017 */
018public class AState<I> {
019        // --------------------------------------------------------------------------
020        // --- variables and constants
021        // ----------------------------------------------
022        // --------------------------------------------------------------------------
023        protected final LinkedHashSet<I> items;
024
025        // cache
026        private final int hashCode;
027
028        // --------------------------------------------------------------------------
029        // --- constructors
030        // ---------------------------------------------------------
031        // --------------------------------------------------------------------------
032        /**
033         * @see AState
034         * @param items
035         *            Added by the ordering of the collections {@link Iterator}!
036         */
037        public AState(Collection<I> items) {
038                this.items = new LinkedHashSet<>(items);
039                this.hashCode = calcHashCode();
040        }
041
042        @SafeVarargs
043        public AState(I... items) {
044                this.items = new LinkedHashSet<>();
045                for (I i : items) {
046                        this.items.add(i);
047                }
048                this.hashCode = calcHashCode();
049        }
050
051        // --------------------------------------------------------------------------
052        // --- getter/setter
053        // --------------------------------------------------------
054        // --------------------------------------------------------------------------
055        /**
056         * @return An {@link It}erator over the items of this state. Order of the
057         *         given items is maintained!
058         */
059        public It<I> getItems() {
060                return new It<>(items);
061        }
062
063        /**
064         * @return The first of the items
065         */
066        public I getFirstItem() {
067                return items.iterator().next();
068        }
069
070        /**
071         * @return The number of items defining this state
072         */
073        public int getItemsCount() {
074                return items.size();
075        }
076
077        private int calcHashCode() {
078                return items.hashCode();
079        }
080
081        @Override
082        public int hashCode() {
083                return hashCode;
084        }
085
086        @Override
087        public boolean equals(Object obj) {
088                if (this == obj)
089                        return true;
090                if (obj == null)
091                        return false;
092                if (!(obj instanceof AState))
093                        return false;
094                AState<?> other = (AState<?>) obj;
095                if (items == null) {
096                        if (other.items != null)
097                                return false;
098                } else if (!items.equals(other.items))
099                        return false;
100                return true;
101        }
102
103        @Override
104        public String toString() {
105                return items.toString();
106        }
107}
