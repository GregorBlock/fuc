001package swp_compiler_ss13.fuc.parser.generator;
002
003import java.util.Collections;
004import java.util.HashMap;
005import java.util.LinkedList;
006import java.util.List;
007import java.util.Map;
008
009import swp_compiler_ss13.fuc.parser.generator.terminals.EfficientTerminalSet;
010import swp_compiler_ss13.fuc.parser.grammar.Grammar;
011import swp_compiler_ss13.fuc.parser.grammar.NonTerminal;
012import swp_compiler_ss13.fuc.parser.grammar.Production;
013import swp_compiler_ss13.fuc.parser.grammar.Terminal;
014
015
016/**
017 * This class contains a {@link Grammar} extended by more information,
018 * like the set of nullable nonterminals and the FIRST sets.
019 * 
020 * These values are cached for performance reasons
021 * 
022 * @author Gero
023 */
024public class GrammarInfo {
025   // --------------------------------------------------------------------------
026   // --- variables and constants ----------------------------------------------
027   // --------------------------------------------------------------------------
028   private final Grammar grammar;
029   
030   private final NullableSet nullableSet;
031   private final FirstSets firstSets;
032   private final FollowSets followSets;
033   private final HashMap<NonTerminal, List<Production>> productionsStartingWith;
034   private final Map<NonTerminal, List<Production>> constProductionsStartingWith;
035   
036   private final EfficientTerminalSet emptyTerminalsSet;
037   
038   
039   // --------------------------------------------------------------------------
040   // --- constructors ---------------------------------------------------------
041   // --------------------------------------------------------------------------
042   /**
043    * Creates a {@link GrammarInfo} from the given {@link Grammar}.
044    */
045   public GrammarInfo(Grammar grammar) {
046      this.grammar = grammar;
047      
048      // Compute additional information about the grammar
049      this.nullableSet = new NullableSet(grammar.getProductions());
050      this.firstSets = new FirstSets(grammar, this.nullableSet);
051      this.followSets = new FollowSets(grammar, firstSets, nullableSet);
052      this.productionsStartingWith = new HashMap<NonTerminal, List<Production>>();
053      for (NonTerminal nonTerminal : grammar.getNonTerminals()) {
054         List<Production> list = new LinkedList<Production>();
055         for (Production p : grammar.getProductions()) {
056            if (p.getLHS() == nonTerminal)
057               list.add(p);
058         }
059         this.productionsStartingWith.put(nonTerminal, Collections.unmodifiableList(list));
060      }
061      this.constProductionsStartingWith = Collections.unmodifiableMap(this.productionsStartingWith);
062      
063      // Empty set of terminals
064      this.emptyTerminalsSet = new EfficientTerminalSet(grammar.getTerminals());
065   }
066   
067   // --------------------------------------------------------------------------
068   // --- methods --------------------------------------------------------------
069   // --------------------------------------------------------------------------
070   
071   
072   // --------------------------------------------------------------------------
073   // --- getter/setter --------------------------------------------------------
074   // --------------------------------------------------------------------------
075   public Grammar getGrammar() {
076      return grammar;
077   }
078   
079   
080   /**
081    * @return all productions beginning with the given {@link NonTerminal}
082    */
083   public List<Production> getProductionsFrom(NonTerminal lhs) {
084      return productionsStartingWith.get(lhs);
085   }
086   
087   
088   /**
089    * @return all productions beginning with the given {@link NonTerminal}
090    */
091   public Map<NonTerminal, List<Production>> getProductionsFrom() {
092      return constProductionsStartingWith;
093   }
094   
095   
096   /**
097    * Gets the set of all nullable non-terminals, that means the
098    * non-terminals that can derive the empty string.
099    */
100   public NullableSet getNullableSet() {
101      return nullableSet;
102   }
103   
104   
105   public FirstSets getFirstSets() {
106      return firstSets;
107   }
108   
109   public FollowSets getFollowSets() {
110      return followSets;
111   }
112   
113   
114   public EfficientTerminalSet getEmptyTerminalSet() {
115      return emptyTerminalsSet;
116   }
117   
118   
119   public EfficientTerminalSet getTerminalSet(Terminal terminal) {
120      return emptyTerminalsSet.plus(terminal);
121   }
122}
