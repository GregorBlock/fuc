001package swp_compiler_ss13.fuc.parser.generator.terminals;
002
003import java.util.Set;
004
005import swp_compiler_ss13.fuc.parser.grammar.Terminal;
006
007/**
008 * The interface for all implementations of sets of terminals
009 * 
010 * @author Gero
011 */
012public interface ITerminalSet {
013        /**
014         * @return New, empty {@link ITerminalSet} with indexes for the same
015         * grammar
016         */
017        public ITerminalSet empty();
018
019        /**
020         * @param terminal
021         * @return New instance of {@link ITerminalSet} with added terminal
022         */
023        public ITerminalSet plus(Terminal terminal);
024
025        /**
026         * @param terminals
027         * @return New instance of {@link ITerminalSet} plus the given terminal set.
028         */
029        public ITerminalSet plusAll(ITerminalSet terminals);
030
031        /**
032         * @param terminals
033         * @return New instance of {@link ITerminalSet} plus all given terminal -
034         *         except {@link Terminal#Epsilon}.
035         */
036        public ITerminalSet plusAllExceptEpsilon(ITerminalSet terminals);
037
038        public boolean contains(Terminal terminal);
039
040        public Set<Terminal> getTerminals();
041}
