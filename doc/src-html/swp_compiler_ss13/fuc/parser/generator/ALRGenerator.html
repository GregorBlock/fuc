029/**
030 * Base class for LR(0) and LR(1) parser generators. It takes the given
031 * {@link Grammar},creates the according {@link Dfa} and finally iterates it to
032 * create SHIFT, ACCEPT and REDUCE actions. The latter one is delegated to
033 * implementations via
034 * {@link #createReduceAction(LRActionTable, Item, LRParserState)}
035 * 
036 * @author Gero
037 */
038public abstract class ALRGenerator<I extends Item, S extends ALRState<I>> {
039        // --------------------------------------------------------------------------
040        // --- variables and constants
041        // ----------------------------------------------
042        // --------------------------------------------------------------------------
043        protected final GrammarInfo grammarInfo;
044        private final Dfa<I, S> dfa;
045        private final LRParsingTable table;
046
047        // --------------------------------------------------------------------------
048        // --- constructors
049        // ---------------------------------------------------------
050        // --------------------------------------------------------------------------
051        public ALRGenerator(Grammar grammar) throws RuntimeException {
052                // Compute NULLABLE, FIRST and FOLLOW sets
053                grammar = grammar.extendByAuxStartProduction();
054                grammarInfo = new GrammarInfo(grammar);
055
056                // Create automaton
057                dfa = createDFA();
058                S startState = dfa.getStartState();
059
060                // Create a supportive mapping between generator state -> parser state
061                Map<S, LRParserState> statesMap = new HashMap<>();
062                // Gets stored for debug/testing in the table!
063                LinkedHashMap<LRParserState, AState<?>> parserStatesMap = new LinkedHashMap<>();
064                LRParserState parserStartState = new LRParserState(0);
065                statesMap.put(startState, parserStartState);
066                parserStatesMap.put(parserStartState, startState);
067                int i = 1;
068                for (S dfaState : dfa.getStates()) {
069                        if (!dfaState.equals(startState)) {
070                                LRParserState parserState = new LRParserState(i);
071                                statesMap.put(dfaState, parserState);
072                                parserStatesMap.put(parserState, dfaState);
073                                i++;
074                        }
075                }
076
077                // Construct parsing table (pass association 'parser state' ->
078                // 'generator state' for debugging)
079                table = new LRParsingTable(parserStatesMap);
080
081                // Traverse edges and generate shift, goto and accept
082                for (DfaEdge<S> edge : dfa.getEdges()) {
083                        LRParserState src = statesMap.get(edge.getSrc());
084                        if (edge.getSymbol().isTerminal()) {
085                                // Terminal
086                                Terminal terminal = (Terminal) edge.getSymbol();
087                                if (edge.isDestAccepting()) {
088                                        // ACCEPT
089                                        try {
090                                                table.getActionTable().set(new Accept(), src, terminal);
091                                        } catch (DoubleEntryException err) {
092                                                throw new RuntimeException(err); // TODO Really the
093                                                                                                                        // right way...?
094                                        }
095                                } else {
096                                        // SHIFT
097                                        LRParserState dstState = statesMap.get(edge.getDst());
098                                        try {
099                                                // Item item = edge.getSrcItem().shift();
100                                                table.getActionTable().set(
101                                                                new Shift(dstState/* , item */), src, terminal);
102                                        } catch (DoubleEntryException err) {
103                                                throw new RuntimeException(err); // TODO Really the
104                                                                                                                        // right way...?
105                                        }
106                                }
107                        } else {
108                                // NonTerminal
109                                NonTerminal nonTerminal = (NonTerminal) edge.getSymbol();
110                                LRParserState dstState = statesMap.get(edge.getDst());
111                                try {
112                                        table.getGotoTable().set(dstState, src, nonTerminal);
113                                } catch (DoubleEntryException err) {
114                                        throw new RuntimeException(err); // TODO Really the right
115                                                                                                                // way...?
116                                }
117                        }
118                }
119
120                // Find reduction targets
121                // Ufff... what to do here? :-o
122                for (S kernel : statesMap.keySet()) {
123                        ALRState<I> state = kernel.closure(grammarInfo); // unpack
124                        for (I item : state.getItems()) {
125                                if (item.isComplete()) {
126                                        LRParserState fromState = statesMap.get(kernel);
127                                        try {
128                                                createReduceAction(table.getActionTable(), item,
129                                                                fromState);
130                                        } catch (GeneratorException err) {
131                                                throw new RuntimeException(err); // TODO Really the
132                                                // right way...?
133                                        }
134                                }
135                        }
136                }
137        }
138
139        /**
140         * The implementing generator has to set its reduce actions here
141         * 
142         * @param table
143         *            The table to set the actions to
144         * @param item
145         *            The item that gets reduced
146         * @param fromState
147         *            The state the parser is currently in
148         * @throws GeneratorException
149         */
150        protected abstract void createReduceAction(LRActionTable table, I item,
151                        LRParserState fromState) throws GeneratorException;
152
153        /**
154         * Tries to set the given {@link Reduce} to the given {@link LRActionTable}.
155         * If there's a conflict, it tries to resolve it by precedence.
156         * 
157         * @param table
158         * @param reduce
159         * @throws DoubleEntryException
160         */
161        protected void setReduceAction(LRActionTable table, Reduce reduce,
162                        LRParserState curState, Terminal curTerminal)
163                        throws GeneratorException {
164                ALRAction action = table.getWithNull(curState, curTerminal);
165                if (action == null) {
166                        // Free cell
167                        table.set(reduce, curState, curTerminal);
168                } else {
169                        // Try to resolve conflict...
170                        if (action.getType() == SHIFT) {
171                                // Shift-Reduce-Conflict. Try to resolve by precedence for
172                                // production or terminal..
173                                Shift shift = (Shift) action;
174                                if (curTerminal.getTokenTypes().next() == TokenType.ELSE) {
175                                        // Fake precedences for Dangling Else: Prefer SHIFT! :-P
176                                        // TODO Introduce Associativity/Precedences...
177                                } else {
178                                        throw new ShiftReduceConflict(shift, reduce, curTerminal);
179                                }
180                        } else if (action.getType() == REDUCE) {
181                                // Reduce-Reduce-Conflict.
182                                Reduce reduce1 = (Reduce) action;
183                                throw new ReduceReduceConflict(reduce1, reduce);
184                        }
185                }
186        }

188        // --------------------------------------------------------------------------
189        // --- methods
190        // --------------------------------------------------------------
191        // --------------------------------------------------------------------------
192        public Dfa<I, S> createDFA() {
193                // Init
194                S startState = createStartState();
195                Dfa<I, S> dfa = createLrDFA(startState);
196
197                LinkedList<S> todo = new LinkedList<>();
198                todo.add(startState);
199
200                // For all state, find edges to (possibly) new states
201                while (!todo.isEmpty()) {
202                        S kernel = todo.removeFirst();
203                        ALRState<I> state = kernel.closure(grammarInfo); // unpack!
204
205                        for (I item : state.getItems()) {
206                                if (item.isShiftable()) {
207                                        Symbol symbol = item.getNextSymbol();
208                                        if (symbol.equals(Terminal.EOF)) {
209                                                // $: We accept...? TODO Is the simplest solution,
210                                                // but... sufficient?
211                                                dfa.getEdges().add(
212                                                                DfaEdge.createAcceptEdge(kernel, symbol));
213                                        } else {
214                                                @SuppressWarnings("unchecked")
215                                                S nextState = (S) state.goTo(symbol);
216
217                                                // Is it new?
218                                                S oldState = contains(dfa, nextState);
219                                                if (oldState == null) {
220                                                        dfa.getStates().add(nextState);
221                                                        todo.add(nextState);
222                                                } else {
223                                                        // Re-use old instance and drop temporarely
224                                                        // generated 'nextState'..
225                                                        nextState = oldState;
226                                                }
227
228                                                // Add edge to automaton
229                                                DfaEdge<S> edge = new DfaEdge<S>(kernel, symbol,
230                                                                nextState, item.getLR0Kernel());
231                                                dfa.getEdges().add(edge);
232                                        }
233                                }
234                        }
235                }
236
237                return dfa;
238        }
239
240        /**
241         * @return The start state of type <S> determined by the implementation
242         */
243        protected abstract S createStartState();
244
245        /**
246         * @param startState
247         * @return An instance of {@link Dfa} with the correct types <I, S>
248         */
249        protected abstract Dfa<I, S> createLrDFA(S startState);
250
251        /**
252         * @return The equivalent {@link LR0State} if there is one contained in the
253         *         DFA; else <code>null</code>
254         */
255        private S contains(Dfa<I, S> dfa, S newState) {
256                for (S oldState : dfa.getStates()) {
257                        if (oldState.equals(newState)) {
258                                return oldState;
259                        }
260                }
261                return null;
262        }
263
264        // --------------------------------------------------------------------------
265        // --- getter/setter
266        // --------------------------------------------------------
267        // --------------------------------------------------------------------------
268        public GrammarInfo getGrammarInfo() {
269                return grammarInfo;
270        }
271
272        public LRParsingTable getParsingTable() {
273                return table;
274        }
