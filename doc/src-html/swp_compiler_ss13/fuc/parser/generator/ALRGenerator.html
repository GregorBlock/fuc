029/**
030 * Base class for LR(0) and LR(1) parser generators.
031 * 
032 * @author Gero
033 */
034public abstract class ALRGenerator<I extends Item, S extends ALRState<I>> {
035        // --------------------------------------------------------------------------
036        // --- variables and constants
037        // ----------------------------------------------
038        // --------------------------------------------------------------------------
039        protected final GrammarInfo grammarInfo;
040        private final Dfa<I, S> dfa;
041        private final LRParsingTable table;
042
043        // --------------------------------------------------------------------------
044        // --- constructors
045        // ---------------------------------------------------------
046        // --------------------------------------------------------------------------
047        public ALRGenerator(Grammar grammar) throws RuntimeException {
048                // Compute NULLABLE, FIRST and FOLLOW sets
049                grammar = grammar.extendByAuxStartProduction();
050                grammarInfo = new GrammarInfo(grammar);
051
052                // Create automaton
053                dfa = createDFA();
054                S startState = dfa.getStartState();
055
056                // Create a supportive mapping between generator state -> parser state
057                Map<S, LRParserState> statesMap = new HashMap<>();
058                // Gets stored for debug/testing in the table!
059                LinkedHashMap<LRParserState, AState<?>> parserStatesMap = new LinkedHashMap<>();
060                LRParserState parserStartState = new LRParserState(0);
061                statesMap.put(startState, parserStartState);
062                parserStatesMap.put(parserStartState, startState);
063                int i = 1;
064                for (S dfaState : dfa.getStates()) {
065                        if (!dfaState.equals(startState)) {
066                                LRParserState parserState = new LRParserState(i);
067                                statesMap.put(dfaState, parserState);
068                                parserStatesMap.put(parserState, dfaState);
069                                i++;
070                        }
071                }
072
073                // Construct parsing table (pass association 'parser state' -> 'generator state' for debugging)
074                table = new LRParsingTable(parserStatesMap);
075
076                // Traverse edges and generate shift, goto and accept
077                for (DfaEdge<S> edge : dfa.getEdges()) {
078                        LRParserState src = statesMap.get(edge.getSrc());
079                        if (edge.getSymbol().isTerminal()) {
080                                // Terminal
081                                Terminal terminal = (Terminal) edge.getSymbol();
082                                if (edge.isDestAccepting()) {
083                                        // ACCEPT
084                                        try {
085                                                table.getActionTable().set(new Accept(), src, terminal);
086                                        } catch (DoubleEntryException err) {
087                                                throw new RuntimeException(err); // TODO Really the
088                                                                                                                        // right way...?
089                                        }
090                                } else {
091                                        // SHIFT
092                                        LRParserState dstState = statesMap.get(edge.getDst());
093                                        try {
094//                                              Item item = edge.getSrcItem().shift();
095                                                table.getActionTable().set(new Shift(dstState/*, item*/), src,
096                                                                terminal);
097                                        } catch (DoubleEntryException err) {
098                                                throw new RuntimeException(err); // TODO Really the
099                                                                                                                        // right way...?
100                                        }
101                                }
102                        } else {
103                                // NonTerminal
104                                NonTerminal nonTerminal = (NonTerminal) edge.getSymbol();
105                                LRParserState dstState = statesMap.get(edge.getDst());
106                                try {
107                                        table.getGotoTable().set(dstState, src, nonTerminal);
108                                } catch (DoubleEntryException err) {
109                                        throw new RuntimeException(err); // TODO Really the right
110                                                                                                                // way...?
111                                }
112                        }
113                }
114
115                // Find reduction targets
116                // Ufff... what to do here? :-o
117                for (S kernel : statesMap.keySet()) {
118                        ALRState<I> state = kernel.closure(grammarInfo); // unpack
119                        for (I item : state.getItems()) {
120                                if (item.isComplete()) {
121                                        LRParserState fromState = statesMap.get(kernel);
122                                        try {
123                                                createReduceAction(table.getActionTable(), item, fromState);
124                                        } catch (GeneratorException err) {
125                                                throw new RuntimeException(err); // TODO Really the
126                                                // right way...?
127                                        }
128                                }
129                        }
130                }
131        }
132        
133        protected abstract void createReduceAction(LRActionTable table, I item, LRParserState fromState) throws GeneratorException;
134        
135        /**
136         * Tries to set the given {@link Reduce} to the given {@link LRActionTable}. If there's a conflict, it tries to resolve it by precedence.
137         * 
138         * @param table
139         * @param reduce
140         * @throws DoubleEntryException 
141         */
142        protected void setReduceAction(LRActionTable table, Reduce reduce, LRParserState curState, Terminal curTerminal) throws GeneratorException {
143                ALRAction action = table.getWithNull(curState, curTerminal);
144                if (action == null) {
145                        // Free cell
146                        table.set(reduce, curState, curTerminal);
147                } else {
148                        // Try to resolve conflict...
149                        if (action.getType() == SHIFT) {
150                                // Shift-Reduce-Conflict. Try to resolve by precedence for production or terminal..
151                                Shift shift = (Shift) action;
152                                if (curTerminal.getTokenTypes().next() == TokenType.ELSE) {
153                                        // Fake precedences for Dangling Else: Prefer SHIFT! :-P
154                                        // TODO Introduce Associativity/Precedences...
155                                } else {
156                                        throw new ShiftReduceConflict(shift, reduce, curTerminal);
157                                }
158                        } else if (action.getType() == REDUCE) {
159                                // Reduce-Reduce-Conflict.
160                                Reduce reduce1 = (Reduce) action;
161                                throw new ReduceReduceConflict(reduce1, reduce);
162                        }
163                }
164        }

167        // --------------------------------------------------------------------------
168        // --- methods
169        // --------------------------------------------------------------
170        // --------------------------------------------------------------------------
171        public Dfa<I, S> createDFA() {
172                // Init
173                S startState = createStartState();
174                Dfa<I, S> dfa = createLrDFA(startState);
175
176                LinkedList<S> todo = new LinkedList<>();
177                todo.add(startState);
178
179                // For all state, find edges to (possibly) new states
180                while (!todo.isEmpty()) {
181                        S kernel = todo.removeFirst();
182                        ALRState<I> state = kernel.closure(grammarInfo); // unpack!
183
184                        for (I item : state.getItems()) {
185                                if (item.isShiftable()) {
186                                        Symbol symbol = item.getNextSymbol();
187                                        if (symbol.equals(Terminal.EOF)) {
188                                                // $: We accept...? TODO Is the simplest solution,
189                                                // but... sufficient?
190                                                dfa.getEdges().add(
191                                                                DfaEdge.createAcceptEdge(kernel, symbol));
192                                        } else {
193                                                @SuppressWarnings("unchecked")
194                                                S nextState = (S) state.goTo(symbol);
195
196                                                // Is it new?
197                                                S oldState = contains(dfa, nextState);
198                                                if (oldState == null) {
199                                                        dfa.getStates().add(nextState);
200                                                        todo.add(nextState);
201                                                } else {
202                                                        // Re-use old instance and drop temporarely
203                                                        // generated 'nextState'..
204                                                        nextState = oldState;
205                                                }
206
207                                                // Add edge to automaton
208                                                DfaEdge<S> edge = new DfaEdge<S>(kernel, symbol,
209                                                                nextState, item.getLR0Kernel());
210                                                dfa.getEdges().add(edge);
211                                        }
212                                }
213                        }
214                }
215
216                return dfa;
217        }
218        
219        protected abstract S createStartState();
220        protected abstract Dfa<I, S> createLrDFA(S startState);
221        
222        /**
223         * @return The equivalent {@link LR0State} if there is one contained in the
224         *         DFA; else <code>null</code>
225         */
226        private S contains(Dfa<I, S> dfa, S newState) {
227                for (S oldState : dfa.getStates()) {
228                        if (oldState.equals(newState)) {
229                                return oldState;
230                        }
231                }
232                return null;
233        }

235        // --------------------------------------------------------------------------
236        // --- getter/setter
237        // --------------------------------------------------------
238        // --------------------------------------------------------------------------
239        public GrammarInfo getGrammarInfo() {
240                return grammarInfo;
241        }
242
243        public LRParsingTable getParsingTable() {
244                return table;
245        }
