001package swp_compiler_ss13.fuc.parser.generator;
002
003import static swp_compiler_ss13.fuc.parser.grammar.Terminal.Epsilon;
004
005import java.util.HashSet;
006import java.util.Iterator;
007import java.util.List;
008import java.util.Set;
009
010import swp_compiler_ss13.fuc.parser.grammar.Grammar;
011import swp_compiler_ss13.fuc.parser.grammar.NonTerminal;
012import swp_compiler_ss13.fuc.parser.grammar.Production;
013import swp_compiler_ss13.fuc.parser.grammar.Symbol;
014import swp_compiler_ss13.fuc.parser.grammar.Terminal;
015import swp_compiler_ss13.fuc.parser.util.It;
016
017
018/**
019 * All {@link Terminal}s of a {@link Grammar} that can derive {@link Terminal#Epsilon}
020 * 
021 * @author Gero
022 */
023public class NullableSet implements Iterable<NonTerminal> {
024   // --------------------------------------------------------------------------
025   // --- variables and constants ----------------------------------------------
026   // --------------------------------------------------------------------------
027   private final Set<NonTerminal> set = new HashSet<>();
028   
029   // --------------------------------------------------------------------------
030   // --- constructors ---------------------------------------------------------
031   // --------------------------------------------------------------------------
032   /**
033    * Computes the non-terminals that can derive the empty string.
034    */
035   public NullableSet(List<Production> productions) {
036      // iterate as long as we get new nullable productions.
037      // in the first step, we only get the trivial ones (X → ɛɛɛ...),
038      // then the ones using X (e.g. Y -> XɛX), and so on
039      boolean changed;
040      do {
041         changed = false;
042         for (Production production : productions) {
043            NonTerminal lhs = production.getLHS();
044            // only check non-terminals which are not nullable already
045            if (!set.contains(lhs)) {
046               // all rhs symbols nullable?
047               boolean nullable = true;
048               for (Symbol symbol : production.getRHS()) {
049                  if (!(symbol == Epsilon || set.contains(symbol))) {
050                     nullable = false;
051                     break;
052                  }
053               }
054               // then remember it as a nullable terminal
055               if (nullable) {
056                  set.add(lhs);
057                  changed = true;
058               }
059            }
060         }
061      } while (changed);
062   }
063   
064   
065   // --------------------------------------------------------------------------
066   // --- getter/setter --------------------------------------------------------
067   // --------------------------------------------------------------------------
068   @Override
069   public Iterator<NonTerminal> iterator() {
070      return new It<>(set);
071   }
072   
073   
074   public boolean contains(NonTerminal nonTerminal) {
075      return set.contains(nonTerminal);
076   }
077   
078   /**
079    * Convenience method
080    */
081   public boolean contains(Symbol symbol) {
082      if (!symbol.isNonTerminal())
083         return false;
084      return contains((NonTerminal) symbol);
085   }
086   
087   @Override
088        public String toString() {
089                return "NullableSet: " + set.toString();
090        }
091}
