001package swp_compiler_ss13.fuc.parser.generator.items;
002
003import static swp_compiler_ss13.fuc.parser.grammar.Terminal.Epsilon;
004
005import java.util.List;
006
007import swp_compiler_ss13.fuc.parser.generator.FirstSets;
008import swp_compiler_ss13.fuc.parser.generator.NullableSet;
009import swp_compiler_ss13.fuc.parser.generator.terminals.ITerminalSet;
010import swp_compiler_ss13.fuc.parser.grammar.Production;
011import swp_compiler_ss13.fuc.parser.grammar.Symbol;
012
013/**
014 * This class extends the standard LR(0)-Item by a lookahead for LR1-grammars
015 * /generators
016 * 
017 * @author Gero
018 */
019public class LR1Item implements Item {
020        // --------------------------------------------------------------------------
021        // --- variables and constants
022        // ----------------------------------------------
023        // --------------------------------------------------------------------------
024        private final LR0Item kernel;
025        private final ITerminalSet lookaheads;
026
027        // cache
028        private ITerminalSet nextLookaheads = null;
029        private final int hashCode;
030
031        // --------------------------------------------------------------------------
032        // --- constructors
033        // ---------------------------------------------------------
034        // --------------------------------------------------------------------------
035        public LR1Item(Production production, int position, ITerminalSet lookaheads) {
036                this.kernel = new LR0Item(production, position);
037                this.lookaheads = lookaheads;
038
039                // Compute hashcode
040                this.hashCode = this.kernel.hashCode() + lookaheads.hashCode();
041        }
042
043        public LR1Item(LR0Item kernel, ITerminalSet lookaheads) {
044                this.kernel = kernel;
045                this.lookaheads = lookaheads;
046
047                // Compute hashcode
048                this.hashCode = this.kernel.hashCode() + lookaheads.hashCode();
049        }
050
051        // --------------------------------------------------------------------------
052        // --- methods
053        // --------------------------------------------------------------
054        // --------------------------------------------------------------------------
055        /**
056         * Returns this item with the position shifted one symbol further.
057         */
058        public LR1Item shift() {
059                if (!kernel.isShiftable())
060                        throw new RuntimeException(
061                                        "Shifting not possible: Item already closed: "
062                                                        + kernel.production.toString(kernel.position));
063                return new LR1Item(kernel.production, kernel.position + 1, lookaheads);
064        }
065
066        /**
067         * Gets the FIRST set of βa (where this item has the following form:
068         * "A → α.Xβ with lookahead a" with X being the next symbol), using the
069         * given precomputed nullable set and FIRST sets for all symbols.
070         */
071        public ITerminalSet getNextLookaheads(FirstSets firstSets,
072                        NullableSet nullableSet) {
073                if (nextLookaheads == null) {
074                        nextLookaheads = computeNextLookaheadSymbols(firstSets, nullableSet);
075                }
076                return nextLookaheads;
077        }
078
079        private ITerminalSet computeNextLookaheadSymbols(FirstSets firstSets,
080                        NullableSet nullableSet) {
081                ITerminalSet result = lookaheads.empty();
082
083                // While the symbols of β (if any) are nullable, collect their FIRST
084                // sets. When not nullable, collect the symbols and stop.
085                List<Symbol> rhs = kernel.getProduction().getRHS();
086                for (int i = kernel.position + 1; i < rhs.size(); i++) {
087                        Symbol symbol = rhs.get(i);
088                        result = result.plusAll(firstSets.get(symbol));
089                        if (!(symbol == Epsilon || nullableSet.contains(symbol))) {
090                                return result;
091                        }
092                }
093
094                // Since we still did not return, all symbols of β (if any) were
095                // nullable,
096                // so we add all our lookaheads to the result
097                result = result.plusAll(lookaheads);
098                return result;
099        }
100
101        /**
102         * Merges this item with the given one and returns the result.
103         */
104        public LR1Item merge(LR1Item item) {
105                if (!kernel.equals(item.kernel))
106                        throw new IllegalArgumentException(
107                                        "Only items with equal LR(0) kernel can be merged!");
108                return new LR1Item(kernel.production, kernel.position,
109                                lookaheads.plusAll(item.lookaheads));
110        }
111
112        // --------------------------------------------------------------------------
113        // --- getter/setter
114        // --------------------------------------------------------
115        // --------------------------------------------------------------------------
116        public Production getProduction() {
117                return kernel.getProduction();
118        }
119
120        public int getPosition() {
121                return kernel.getPosition();
122        }
123
124        /**
125         * Gets the {@link Symbol} right behind the current position, or
126         * <code>null</code> if there is none left.
127         */
128        public Symbol getNextSymbol() {
129                return kernel.getNextSymbol();
130        }
131
132        public boolean isShiftable() {
133                return kernel.isShiftable();
134        }
135
136        public boolean isComplete() {
137                return kernel.isComplete();
138        }
139
140        @Override
141        public LR0Item getLR0Kernel() {
142                return kernel;
143        }
144
145        public ITerminalSet getLookaheads() {
146                return lookaheads;
147        }
148
149        @Override
150        public String toString() {
151                StringBuilder s = new StringBuilder(
152                                kernel.production.toString(kernel.position));
153                s.append(" " + lookaheads.toString());
154                return s.toString();
155        }
156
157        @Override
158        public boolean equals(Object obj) {
159                if (obj instanceof LR1Item) {
160                        LR1Item l = (LR1Item) obj;
161                        if (!kernel.equals(l.kernel))
162                                return false;
163                        if (!lookaheads.equals(l.lookaheads))
164                                return false;
165                        return true;
166                }
167                return false;
168        }
169
170        @Override
171        public int hashCode() {
172                return hashCode;
173        }
174}
