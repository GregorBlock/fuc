001package swp_compiler_ss13.fuc.parser.generator.items;
002
003import swp_compiler_ss13.fuc.parser.grammar.Production;
004import swp_compiler_ss13.fuc.parser.grammar.Symbol;
005
006/**
007 * Basic interface for a {@link Production} associated with a position ("dot").
008 * 
009 * @author Gero
010 */
011public interface Item {
012        public Production getProduction();
013
014        public int getPosition();
015
016        /**
017         * @return The next {@link Symbol} (after the dot) in the {@link Production}
018         *         . <code>null</code> if {@link #isComplete()}
019         */
020        public Symbol getNextSymbol();
021
022        /**
023         * @return If there are still {@link Symbol}s left in the prodution. Reverse of {@link #isComplete()}
024         */
025        public boolean isShiftable();
026
027        /**
028         * @return Reverse of {@link #isShiftable()}
029         */
030        public boolean isComplete();
031
032        /**
033         * @return Creates a new instance with the position = position + 1
034         */
035        public Item shift();
036        
037        public LR0Item getLR0Kernel();
038}
