001package swp_compiler_ss13.fuc.parser.generator.items;
002
003import java.util.List;
004
005import swp_compiler_ss13.fuc.parser.grammar.Production;
006import swp_compiler_ss13.fuc.parser.grammar.Symbol;
007import swp_compiler_ss13.fuc.parser.grammar.Terminal;
008
009/**
010 * An LR(0)-Item is a {@link Production} combined with a position in its right
011 * hand side. E.g.: "E â†’ (.B)", where the dot denotes the position.
012 */
013public class LR0Item implements Item {
014        // --------------------------------------------------------------------------
015        // --- variables and constants
016        // ----------------------------------------------
017        // --------------------------------------------------------------------------
018        protected final Production production;
019        protected final int position;
020
021        // cache
022        protected final Symbol nextSymbol;
023        protected final int hashCode;
024        private final boolean shiftable;
025
026        // --------------------------------------------------------------------------
027        // --- constructors
028        // ---------------------------------------------------------
029        // --------------------------------------------------------------------------
030        public LR0Item(Production production, int position) {
031                this.production = production;
032
033                // correct position if epsilons are following
034                List<Symbol> rhs = production.getRHS();
035                int correctedPosition = position;
036                if (position == 0 && rhs.get(0) == Terminal.Epsilon) {
037                        correctedPosition = 1; // immediately shift over epsilon
038                }
039                this.position = correctedPosition;
040
041                // shiftable
042                this.shiftable = (this.position < rhs.size());
043
044                // next symbol
045                if (this.shiftable)
046                        this.nextSymbol = rhs.get(this.position);
047                else
048                        this.nextSymbol = null;
049
050                // hash code
051                this.hashCode = production.hashCode() + this.position;
052        }
053
054        // --------------------------------------------------------------------------
055        // --- methods
056        // --------------------------------------------------------------
057        // --------------------------------------------------------------------------
058        @Override
059        public Production getProduction() {
060                return production;
061        }
062
063        @Override
064        public int getPosition() {
065                return position;
066        }
067
068        @Override
069        public Symbol getNextSymbol() {
070                return nextSymbol;
071        }
072
073        @Override
074        public boolean isShiftable() {
075                return shiftable;
076        }
077
078        @Override
079        public boolean isComplete() {
080                return !shiftable;
081        }
082        
083        @Override
084        public LR0Item getLR0Kernel() {
085                return this;
086        }
087
088        @Override
089        public LR0Item shift() {
090                if (!isShiftable())
091                        throw new RuntimeException(
092                                        "Shifting not possible: Item already closed: " + this);
093
094                // return shifted item
095                return new LR0Item(production, position + 1);
096        }
097
098        /**
099         * @return A new {@link LR0Item} with the same production but shifted until
100         *         it is complete
101         */
102        public LR0Item complete() {
103                if (isComplete()) {
104                        return this;
105                } else {
106                        return new LR0Item(production, production.getRHS().size());
107                }
108        }
109
110        // --------------------------------------------------------------------------
111        // --- getter/setter
112        // --------------------------------------------------------
113        // --------------------------------------------------------------------------
114        @Override
115        public String toString() {
116                return production.toString(position);
117        }
118
119        @Override
120        public boolean equals(Object obj) {
121                if (obj instanceof LR0Item) {
122                        LR0Item l = (LR0Item) obj;
123                        if (!production.equals(l.production))
124                                return false;
125                        if (position != l.position)
126                                return false;
127                        return true;
128                }
129                return false;
130        }
131
132        @Override
133        public int hashCode() {
134                return hashCode;
135        }
136}
