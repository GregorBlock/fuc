001package swp_compiler_ss13.fuc.parser;
002
003import swp_compiler_ss13.common.ast.AST;
004import swp_compiler_ss13.common.lexer.Lexer;
005import swp_compiler_ss13.common.parser.Parser;
006import swp_compiler_ss13.common.report.ReportLog;
007import swp_compiler_ss13.fuc.parser.generator.ALRGenerator;
008import swp_compiler_ss13.fuc.parser.generator.LR1Generator;
009import swp_compiler_ss13.fuc.parser.generator.items.LR1Item;
010import swp_compiler_ss13.fuc.parser.generator.states.LR1State;
011import swp_compiler_ss13.fuc.parser.grammar.Grammar;
012import swp_compiler_ss13.fuc.parser.grammar.ProjectGrammar;
013import swp_compiler_ss13.fuc.parser.parser.DoubleIdentifierException;
014import swp_compiler_ss13.fuc.parser.parser.LRParser;
015import swp_compiler_ss13.fuc.parser.parser.LexerWrapper;
016import swp_compiler_ss13.fuc.parser.parser.ParserException;
017import swp_compiler_ss13.fuc.parser.parser.tables.LRParsingTable;
018
019public class ParserImpl implements Parser {
020        // --------------------------------------------------------------------------
021        // --- variables and constants
022        // ----------------------------------------------
023        // --------------------------------------------------------------------------
024        private Lexer lexer = null;
025        private ReportLog reportLog = null;
026
027        // --------------------------------------------------------------------------
028        // --- constructors
029        // ---------------------------------------------------------
030        // --------------------------------------------------------------------------
031        public ParserImpl() {
032
033        }
034
035        // --------------------------------------------------------------------------
036        // --- methods
037        // --------------------------------------------------------------
038        // --------------------------------------------------------------------------
039        @Override
040        public AST getParsedAST() {
041                // Generate parsing table
042                Grammar grammar = new ProjectGrammar.Complete().getGrammar();
043                ALRGenerator<LR1Item, LR1State> generator = new LR1Generator(grammar);
044                LRParsingTable table = generator.getParsingTable();
045
046                // Run LR-parser with table
047                LRParser lrParser = new LRParser();
048                LexerWrapper lexWrapper = new LexerWrapper(this.lexer, grammar);
049                AST ast = null;
050                
051                if (reportLog == null) {
052                        throw new NullPointerException("reportLog is not set");
053                }
054                
055                try     {
056                        ast = lrParser.parse(lexWrapper, this.reportLog, table);
057                } catch (DoubleIdentifierException e) {
058                        return null;
059                } catch (ParserException e) {
060                        return null;
061                }
062
063                return ast;
064        }
065
066        // --------------------------------------------------------------------------
067        // --- getter/setter
068        // --------------------------------------------------------
069        // --------------------------------------------------------------------------
070        @Override
071        public void setLexer(Lexer lexer) {
072                this.lexer = lexer;
073        }
074
075        @Override
076        public void setReportLog(ReportLog reportLog) {
077                this.reportLog = reportLog;
078        }
079}
