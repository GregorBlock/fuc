001package swp_compiler_ss13.fuc.parser.util;
002
003import java.util.Collection;
004import java.util.Iterator;
005import java.util.LinkedList;
006import java.util.NoSuchElementException;
007
008/**
009 * This class is an {@link Iterable} {@link Iterator}, which is quite useful in
010 * for-each loops (if you need to know your current index, e.g.: {@link #getIndex()})
011 * 
012 * @param <T>
013 */
014public final class It<T> implements Iterator<T>, Iterable<T> {
015
016        private final Iterator<T> iterator; // first option: iterator, used for
017                                                                                // collections
018        private final T[] array; // second option: array
019        private int currentIndex = -1;
020
021        /**
022         * Creates a new {@link It} for the given {@link Collection}. If null is
023         * given, a valid iterator with no elements is returned.
024         */
025        public It(Collection<T> collection) {
026                if (collection != null)
027                        this.iterator = collection.iterator();
028                else
029                        this.iterator = new LinkedList<T>().iterator();
030                this.array = null;
031        }
032
033        /**
034         * Creates a new {@link It} for the given array. If null is given, a valid
035         * iterator with no elements is returned.
036         */
037        public It(T[] array) {
038                if (array != null) {
039                        this.array = array;
040                        this.iterator = null;
041                } else {
042                        this.array = null;
043                        this.iterator = new LinkedList<T>().iterator();
044                }
045        }
046
047        @Override
048        public boolean hasNext() {
049                if (iterator != null)
050                        return iterator.hasNext();
051                else
052                        return currentIndex + 1 < array.length;
053        }
054
055        @Override
056        public T next() throws NoSuchElementException {
057                currentIndex++;
058                if (iterator != null)
059                        return iterator.next();
060                else
061                        return array[currentIndex];
062        }
063
064        @Override
065        public void remove() {
066                throw new UnsupportedOperationException();
067        }
068
069        /**
070         * @return this
071         * @see {@link Iterable#iterator()}
072         */
073        @Override
074        public Iterator<T> iterator() {
075                return this;
076        }
077
078        /**
079         * @return The current index of this iterator
080         */
081        public int getIndex() {
082                return currentIndex;
083        }
084}
