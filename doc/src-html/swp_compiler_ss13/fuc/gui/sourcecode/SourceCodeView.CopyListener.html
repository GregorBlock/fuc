187        protected void setLineNumbers(int amount) {
188                StringBuilder builder = new StringBuilder("1");
189                for (int i = 2; i <= amount; i++) {
190                        builder.append('\n');
191                        builder.append(i);
192                }
193                lineNumberField.setText(builder.toString());
194        }
195
196        protected void paintErrors() {
197                int actualLine = 1;
198                int line;
199                int textIndex = 0;
200                Document document = markErrorField.getDocument();
201                try {
202                        document.remove(0, document.getLength());
203                        for (int transferredLine : errorLines) {
204                                line = Math.abs(transferredLine);
205                                for (; actualLine < line; actualLine++) {
206                                        document.insertString(textIndex++, "\n", defaultAttributes);
207                                }
208                                document.insertString(textIndex++, "I",
209                                                line == transferredLine ? markErrorAttributes : markWarningAttributes);
210                        }
211                        if (textIndex == 0) {
212                                document.insertString(0, " ", defaultAttributes);
213                        }
214                } catch (BadLocationException e) {
215                        LOG.error("error on marking errors", e);
216                }
217        }

219        private class CopyListener implements ActionListener, DocumentListener {
220
221                private final IDE ide;
222                private int modCount = 0;
223                private final Lock lock = new ReentrantLock();
224
225                public CopyListener(IDE ide) {
226                        this.ide = ide;
227                }
228
229                private void statusChanged() {
230                        LOG.trace("wrote source code");
231                        final int expectedModCount;
232                        final String text;
233                        lock.lock();
234                        try {
235                                expectedModCount = ++modCount;
236                                text = sourceCodeField.getText();
237                        } finally {
238                                lock.unlock();
239                        }
240                        SwingUtilities.invokeLater(new Runnable() {
241
242                                @Override
243                                public void run() {
244                                        int caretPosition = sourceCodeField.getCaretPosition();
245                                        Document document = sourceCodeField.getDocument();
246                                        document.removeDocumentListener(CopyListener.this);
247                                        try {
248                                                {
249                                                        int newLineAmount = 1;
250                                                        char newLine = '\n';
251                                                        for (char c : text.toCharArray()) {
252                                                                if (newLine == c) {
253                                                                        newLineAmount++;
254                                                                }
255                                                        }
256                                                        setLineNumbers(newLineAmount);
257                                                }
258                                                String[] lines = text.split("\n");
259                                                int[] lineOffset = new int[lines.length];
260                                                lineOffset[0] = 0;
261                                                for (int i = 0; i < lines.length - 1; i++) {
262                                                        lineOffset[i + 1] = lineOffset[i] + 1 + lines[i].length();
263                                                }
264                                                List<Token> tokens = lexerCheck(document, text, lineOffset,
265                                                                expectedModCount);
266                                                if (tokens != null) {
267                                                        AST ast = parserCheck(document, tokens, lineOffset, expectedModCount);
268                                                        if (ast != null) {
269                                                                ast = semanticCheck(document, ast, lineOffset, expectedModCount);
270                                                                sourceCodeField.setCaretPosition(caretPosition);
271                                                        } else {
272                                                                sourceCodeField.setCaretPosition(caretPosition);
273                                                        }
274                                                }
275
276                                        } catch (Exception e) {
277                                                LOG.warn("error durring highlighting", e);
278                                        } finally {
279                                                document.addDocumentListener(CopyListener.this);
280                                        }
281                                        sourceCodeLock.lock();
282                                        try {
283                                                boolean oldValue = setSourceCode;
284                                                setSourceCode = false;
285                                                LOG.info("send sourcecode");
286                                                ide.setSourceCode(text);
287                                                setSourceCode = oldValue;
288                                        } finally {
289                                                sourceCodeLock.unlock();
290                                        }
291                                }
292
293                                private List<Token> lexerCheck(Document document, String text, int[] lineLength,
294                                                int expectedModCount) throws BadLocationException {
295                                        SimpleAttributeSet attrs;
296                                        int tokenstart;
297                                        errorLines.clear();
298                                        List<Token> tokens = ide.runLexer(text, true);
299                                        lock.lock();
300                                        try {
301                                                if (expectedModCount == modCount) {
302                                                        document.remove(0, document.getLength());
303                                                        document.insertString(0, text, defaultAttributes);
304                                                        for (Token token : tokens) {
305                                                                attrs = attributes.get(token.getTokenType());
306                                                                if (attrs != null) {
307                                                                        if (attrs == errorAttributes) {
308                                                                                errorLines.add(token.getLine());
309                                                                        }
310                                                                        tokenstart = lineLength[token.getLine() - 1]
311                                                                                        + token.getColumn() - 1;
312                                                                        document.remove(tokenstart, token.getValue().length());
313                                                                        document.insertString(tokenstart, token.getValue(), attrs);
314                                                                }
315                                                        }
316                                                        paintErrors();
317                                                } else {
318                                                        return null;
319                                                }
320                                        } finally {
321                                                lock.unlock();
322                                        }
323                                        return tokens;
324                                }
325
326                                private AST parserCheck(Document document, List<Token> tokens, int[] lineLength,
327                                                int expectedModCount) throws BadLocationException {
328                                        AST ast = ide.runParser(tokens, false);
329                                        return printAstErrors(document, lineLength, expectedModCount, ast);
330                                }
331
332                                private AST semanticCheck(Document document, AST ast, int[] lineLength,
333                                                int expectedModCount) throws BadLocationException {
334                                        ast = ide.runSemanticAnalysis(ast, false);
335                                        return printAstErrors(document, lineLength, expectedModCount, ast);
336                                }
337
338                                private AST printAstErrors(Document document, int[] lineLength,
339                                                int expectedModCount, AST ast) throws BadLocationException {
340                                        ReportLogImpl reportLog = ide.getReportLog();
341                                        lock.lock();
342                                        try {
343                                                if (expectedModCount == modCount && reportLog != null) {
344                                                        // errors from Lexer can be removed, because they
345                                                        // are repeated in the parser
346                                                        errorLines.clear();
347                                                        int tokenstart;
348                                                        boolean isError;
349                                                        List<Token> tokens;
350                                                        for (LogEntry entry : reportLog.getEntries()) {
351                                                                isError = entry.getLogType() == Type.ERROR;
352                                                                tokens = entry.getTokens();
353                                                                errorLines.add(tokens.get(0).getLine() * (isError ? 1 : -1));
354                                                                for (Token token : tokens) {
355                                                                        if (!token.getTokenType().equals(TokenType.EOF)) {
356                                                                                tokenstart = lineLength[token.getLine() - 1]
357                                                                                                + token.getColumn() - 1;
358                                                                                document.remove(tokenstart, token.getValue().length());
359                                                                                document.insertString(tokenstart, token.getValue(),
360                                                                                                isError ? errorAttributes : warningAttributes);
361                                                                        }
362                                                                }
363                                                                paintErrors();
364                                                        }
365                                                } else {
366                                                        return null;
367                                                }
368                                        } finally {
369                                                lock.unlock();
370                                        }
371                                        return ast;
372                                }
373
374                        });
375                }

399        private class FileListener implements ActionListener {
400                private static final String SAFE_NEW_FILE = "safe new";
401                private static final String SAFE_OLD_FILE = "safe old";
402                private static final String LOAD_FILE = "load";
403
404                private final JFileChooser chooser;
405                private final IDE ide;
406
407                private File file = null;
408
409                public FileListener(IDE ide) {
410                        this.ide = ide;
411                        this.chooser = new JFileChooser();
412                        this.chooser.setFileFilter(new FileNameExtensionFilter("choose a filename", "prog"));
413                }
414
415                @Override
416                public void actionPerformed(ActionEvent e) {
417                        final String text = sourceCodeField.getText();
418                        final boolean safeNew = SAFE_NEW_FILE.equals(e.getActionCommand());
419                        final boolean load = LOAD_FILE.equals(e.getActionCommand());
420                        if (safeNew || load || SAFE_OLD_FILE.equals(e.getActionCommand())) {
421                                SwingUtilities.invokeLater(new Runnable() {
422
423                                        @Override
424                                        public void run() {
425                                                File newFile = file;
426                                                if (load || safeNew || newFile == null) {
427                                                        int showOpenDialog;
428                                                        if (load) {
429                                                                showOpenDialog = chooser.showOpenDialog(component);
430                                                        } else {
431                                                                showOpenDialog = chooser.showSaveDialog(component);
432                                                        }
433                                                        switch (showOpenDialog) {
434                                                        case JFileChooser.APPROVE_OPTION:
435                                                                newFile = new File(chooser.getSelectedFile().getPath()
436                                                                                .replaceFirst("(|\\.prog)\\Z", ".prog"));
437                                                                if (!newFile.exists()) {
438                                                                        try {
439                                                                                newFile.createNewFile();
440                                                                        } catch (IOException e) {
441                                                                                LOG.error("couldn't create file", e);
442                                                                                return;
443                                                                        }
444                                                                } else if (load) {
445                                                                        BufferedReader reader = null;
446                                                                        StringBuilder sourceCode = new StringBuilder();
447                                                                        String line;
448                                                                        try {
449                                                                                reader = new BufferedReader(new InputStreamReader(
450                                                                                                new FileInputStream(newFile)));
451                                                                                while ((line = reader.readLine()) != null) {
452                                                                                        sourceCode.append(line);
453                                                                                        sourceCode.append('\n');
454                                                                                }
455                                                                                if (sourceCode.length() != 0) {
456                                                                                        sourceCode.setLength(sourceCode.length() - 1);
457                                                                                }
458                                                                                sourceCodeLock.lock();
459                                                                                try {
460                                                                                        boolean oldValue = setSourceCode;
461                                                                                        setSourceCode = true;
462                                                                                        ide.setSourceCode(sourceCode.toString());
463                                                                                        setSourceCode = oldValue;
464                                                                                } finally {
465                                                                                        sourceCodeLock.unlock();
466                                                                                }
467                                                                                file = newFile;
468                                                                        } catch (FileNotFoundException e1) {
469                                                                                LOG.error("choosen file doesn't exist", e1);
470                                                                        } catch (IOException e1) {
471                                                                                LOG.error("error while reading from file", e1);
472                                                                        } finally {
473                                                                                if (reader != null) {
474                                                                                        try {
475                                                                                                reader.close();
476                                                                                        } catch (IOException e1) {
477                                                                                                LOG.error("error while closing file", e1);
478                                                                                        }
479                                                                                }
480                                                                        }
481                                                                        return;
482                                                                }
483                                                                file = newFile;
484                                                                break;
485                                                        case JFileChooser.CANCEL_OPTION:
486                                                        case JFileChooser.ERROR_OPTION:
487                                                                return;
488                                                        default:
489                                                                LOG.warn("unknown result while choosing the load file");
490                                                                return;
491                                                        }
492                                                }
493                                                PrintWriter writer = null;
494                                                try {
495                                                        writer = new PrintWriter(newFile);
496                                                        writer.write(text);
497                                                } catch (FileNotFoundException e1) {
498                                                        LOG.error("choosen file doesn't exist", e1);
499                                                } finally {
500                                                        if (writer != null) {
501                                                                writer.close();
502                                                        }
503                                                }
504                                                LOG.info("saved file");
505                                        }
506                                });
507                        }
508                }
509        }

511        private static class MyList implements Iterable<Integer> {
512                private List<Integer> list = new ArrayList<>();
513
514                public void clear() {
515                        list.clear();
516                }
517
518                public void add(int i) {
519                        int index = list.size();
520                        int element;
521                        for (ListIterator<Integer> iterator = list.listIterator(list.size()); iterator
522                                        .hasPrevious();) {
523                                element = iterator.previous();
524                                if (element == i || element == -i) {
525                                        index = iterator.nextIndex();
526                                        break;
527                                }
528                        }
529                        list.add(index, i);
530                }
531
532                @Override
533                public Iterator<Integer> iterator() {
534                        return list.iterator();
535                }
536        }
