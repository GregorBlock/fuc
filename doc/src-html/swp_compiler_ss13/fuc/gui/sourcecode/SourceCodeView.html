084        public SourceCodeView(SourceCodeController controller) {
085                this.controller = controller;
086                sourceCodeField = new JTextPane();
087                sourceCodeField.setEditable(true);
088                sourceCodeField.setEditorKit(new UnderlineStyledEditorKit());
089                lineNumberField = new JTextPane();
090                lineNumberField.setText("1");
091                lineNumberField.setEditable(false);
092                lineNumberField.setBackground(new Color(224, 224, 224));
093                markErrorField = new JTextPane();
094                markErrorField.setText(" ");
095                markErrorField.setEditable(false);
096                markErrorField.setBackground(new Color(224, 224, 224));
097                markErrorField.setForeground(Color.RED);
098                errorLines = new LinkedList<>();
099                JPanel viewPort = new JPanel(new BorderLayout(2, 1));
100                viewPort.add(lineNumberField, BorderLayout.WEST);
101                viewPort.add(sourceCodeField, BorderLayout.CENTER);
102                viewPort.add(markErrorField, BorderLayout.EAST);
103                component = new JScrollPane(viewPort);
104                StyleConstants.setBold(keywordAttributes, true);
105                StyleConstants.setForeground(keywordAttributes, new Color(127, 0, 85));
106                StyleConstants.setForeground(stringAttributes, new Color(42, 0, 255));
107                StyleConstants.setItalic(commentAttributes, true);
108                StyleConstants.setForeground(commentAttributes, Color.GRAY);
109                errorAttributes.addAttribute(UnderlineStyledEditorKit.WAVY_LINE, true);
110                errorAttributes.addAttribute(UnderlineStyledEditorKit.UNDERLINE_COLOR, Color.RED);
111                attributes = new HashMap<>();
112                for (TokenType type : highlightedKeywords) {
113                        attributes.put(type, keywordAttributes);
114                }
115                attributes.put(highlightedString, stringAttributes);
116                attributes.put(highlightedComment, commentAttributes);
117                attributes.put(highlightedError, errorAttributes);
118        }

167        protected void setLineNumbers(int amount) {
168                StringBuilder builder = new StringBuilder("1");
169                for (int i = 2; i <= amount; i++) {
170                        builder.append('\n');
171                        builder.append(i);
172                }
173                lineNumberField.setText(builder.toString());
174        }
175
176        protected void paintErrors() {
177                int actualLine = 0;
178                StringBuilder builder = new StringBuilder("");
179                for (int line : errorLines) {
180                        for (; actualLine < line; actualLine++) {
181                                builder.append('\n');
182                        }
183                        builder.append('I');
184                }
185                if (builder.length() == 0) {
186                        builder.append(' ');
187                }
188                markErrorField.setText(builder.toString());
189        }

191        private class CopyListener implements ActionListener, DocumentListener {
192
193                private final IDE ide;
194
195                public CopyListener(IDE ide) {
196                        this.ide = ide;
197                }
198
199                private void statusChanged() {
200                        LOG.trace("wrote source code");
201                        SwingUtilities.invokeLater(new Runnable() {
202
203                                @Override
204                                public void run() {
205                                        String text = sourceCodeField.getText();
206                                        int caretPosition = sourceCodeField.getCaretPosition();
207                                        Document document = sourceCodeField.getDocument();
208                                        document.removeDocumentListener(CopyListener.this);
209                                        try {
210                                                document.remove(0, document.getLength());
211                                                document.insertString(0, text, defaultAttributes);
212                                                {
213                                                        int newLineAmount = 1;
214                                                        char newLine = '\n';
215                                                        for (char c : text.toCharArray()) {
216                                                                if (newLine == c) {
217                                                                        newLineAmount++;
218                                                                }
219                                                        }
220                                                        setLineNumbers(newLineAmount);
221                                                }
222
223                                                String[] lines = text.split("\n");
224                                                int[] lineLength = new int[lines.length];
225                                                lineLength[0] = 0;
226                                                for (int i = 0; i < lines.length - 1; i++) {
227                                                        lineLength[i + 1] = lineLength[i] + 1 + lines[i].length();
228                                                }
229                                                // List<Token> tokens =
230                                                lexerCheck(document, text, lineLength);
231                                                // parserCheck(document, tokens, lineLength);
232                                                sourceCodeField.setCaretPosition(caretPosition);
233                                        } catch (Exception e) {
234                                                LOG.warn("error durring highlighting", e);
235                                        } finally {
236                                                document.addDocumentListener(CopyListener.this);
237                                        }
238                                        sourceCodeLock.lock();
239                                        try {
240                                                boolean oldValue = setSourceCode;
241                                                setSourceCode = false;
242                                                LOG.info("send sourcecode");
243                                                ide.setSourceCode(text);
244                                                setSourceCode = oldValue;
245                                        } finally {
246                                                sourceCodeLock.unlock();
247                                        }
248                                }
249
250                                private List<Token> lexerCheck(Document document, String text, int[] lineLength)
251                                                throws BadLocationException {
252                                        SimpleAttributeSet attrs;
253                                        int tokenstart;
254                                        errorLines.clear();
255                                        List<Token> tokens = ide.runLexer(text, true);
256                                        for (Token token : tokens) {
257                                                attrs = attributes.get(token.getTokenType());
258                                                if (attrs != null) {
259                                                        if (attrs == errorAttributes) {
260                                                                errorLines.add(token.getLine() - 1);
261                                                                LOG.info("Found Error in line " + (token.getLine() - 1));
262                                                        }
263                                                        tokenstart = lineLength[token.getLine() - 1] + token.getColumn() - 1;
264                                                        document.remove(tokenstart, token.getValue().length());
265                                                        document.insertString(tokenstart, token.getValue(), attrs);
266                                                }
267                                        }
268                                        paintErrors();
269                                        return tokens;
270                                }
271
272                                private void parserCheck(Document document, List<Token> tokens, int[] lineLength)
273                                                throws BadLocationException {
274                                        SimpleAttributeSet attrs;
275                                        AST ast = ide.runParser(tokens, true);
276                                        /*
277                                         * TODO get reportLog
278                                         */
279                                        int tokenstart;
280                                        for (Token token : tokens) {
281                                                attrs = attributes.get(token.getTokenType());
282                                                if (attrs != null) {
283                                                        tokenstart = lineLength[token.getLine() - 1] + token.getColumn() - 1;
284                                                        document.remove(tokenstart, token.getValue().length());
285                                                        document.insertString(tokenstart, token.getValue(), attrs);
286                                                }
287                                        }
288                                }
289                        });
290                }

314        private class LoadFileListener implements ActionListener {
315
316                private final IDE ide;
317                private final SaveFileListener listener;
318
319                public LoadFileListener(IDE ide, SaveFileListener listener) {
320                        this.ide = ide;
321                        this.listener = listener;
322                }
323
324                @Override
325                public void actionPerformed(ActionEvent e) {
326                        SwingUtilities.invokeLater(new Runnable() {
327
328                                @Override
329                                public void run() {
330                                        JFileChooser chooser = new JFileChooser();
331                                        chooser.setFileFilter(new FileNameExtensionFilter("choose a prog file", "prog"));
332                                        int showOpenDialog = chooser.showOpenDialog(component);
333                                        switch (showOpenDialog) {
334                                        case JFileChooser.APPROVE_OPTION:
335                                                File file = chooser.getSelectedFile();
336                                                BufferedReader reader = null;
337                                                StringBuilder sourceCode = new StringBuilder();
338                                                String line;
339                                                try {
340                                                        reader = new BufferedReader(new InputStreamReader(new FileInputStream(
341                                                                        file)));
342                                                        while ((line = reader.readLine()) != null) {
343                                                                sourceCode.append(line);
344                                                                sourceCode.append('\n');
345                                                        }
346                                                        sourceCodeLock.lock();
347                                                        try {
348                                                                boolean oldValue = setSourceCode;
349                                                                setSourceCode = true;
350                                                                ide.setSourceCode(sourceCode.substring(0, sourceCode.length() - 1));
351                                                                setSourceCode = oldValue;
352                                                        } finally {
353                                                                sourceCodeLock.unlock();
354                                                        }
355                                                        if (listener != null) {
356                                                                listener.file = file;
357                                                        }
358                                                } catch (FileNotFoundException e1) {
359                                                        LOG.error("choosen file doesn't exist", e1);
360                                                } catch (IOException e1) {
361                                                        LOG.error("error while reading from file", e1);
362                                                } finally {
363                                                        if (reader != null) {
364                                                                try {
365                                                                        reader.close();
366                                                                } catch (IOException e1) {
367                                                                        LOG.error("error while closing file", e1);
368                                                                }
369                                                        }
370                                                }
371                                        case JFileChooser.CANCEL_OPTION:
372                                        case JFileChooser.ERROR_OPTION:
373                                                break;
374                                        default:
375                                                LOG.warn("unknown result while choosing the load file");
376                                                break;
377                                        }
378                                }
379                        });
380                }
381        }

383        private class SaveFileListener implements ActionListener {
384                private File file = null;
385
386                private static final String SAFE_NEW_FILE = "safe new";
387                private static final String SAFE_OLD_FILE = "safe old";
388
389                @Override
390                public void actionPerformed(ActionEvent e) {
391                        final String text = sourceCodeField.getText();
392                        final boolean safeNew = SAFE_NEW_FILE.equals(e.getActionCommand());
393                        if (safeNew || SAFE_OLD_FILE.equals(e.getActionCommand())) {
394                                SwingUtilities.invokeLater(new Runnable() {
395
396                                        @Override
397                                        public void run() {
398                                                File newFile = file;
399                                                if (safeNew || newFile == null) {
400                                                        JFileChooser chooser = new JFileChooser();
401                                                        chooser.setFileFilter(new FileNameExtensionFilter("choose a filename",
402                                                                        "prog"));
403                                                        int showOpenDialog = chooser.showSaveDialog(component);
404                                                        switch (showOpenDialog) {
405                                                        case JFileChooser.APPROVE_OPTION:
406                                                                newFile = chooser.getSelectedFile();
407                                                                if (!newFile.exists()) {
408                                                                        try {
409                                                                                newFile.createNewFile();
410                                                                        } catch (IOException e) {
411                                                                                LOG.error("couldn't create file", e);
412                                                                                return;
413                                                                        }
414                                                                }
415                                                                file = newFile;
416                                                                break;
417                                                        case JFileChooser.CANCEL_OPTION:
418                                                        case JFileChooser.ERROR_OPTION:
419                                                                return;
420                                                        default:
421                                                                LOG.warn("unknown result while choosing the load file");
422                                                                return;
423                                                        }
424                                                }
425                                                PrintWriter writer = null;
426                                                try {
427                                                        writer = new PrintWriter(newFile);
428                                                        writer.write(text);
429                                                } catch (FileNotFoundException e1) {
430                                                        LOG.error("choosen file doesn't exist", e1);
431                                                } finally {
432                                                        if (writer != null) {
433                                                                writer.close();
434                                                        }
435                                                }
436                                                LOG.info("saved file");
437                                        }
438                                });
439                        }
440                }
