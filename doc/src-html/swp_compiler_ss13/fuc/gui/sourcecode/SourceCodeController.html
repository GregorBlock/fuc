001package swp_compiler_ss13.fuc.gui.sourcecode;
002
003import swp_compiler_ss13.fuc.gui.ide.mvc.Controller;
004import swp_compiler_ss13.fuc.gui.ide.mvc.IDE;
005import swp_compiler_ss13.fuc.gui.ide.mvc.Model;
006import swp_compiler_ss13.fuc.gui.ide.mvc.View;
007
008public class SourceCodeController implements Controller {
009
010        private final SourceCodeView view;
011        private final SourceCodeModell model;
012
013        public SourceCodeController() {
014                this.view = new SourceCodeView(this);
015                this.model = new SourceCodeModell(this);
016        }
017
018        @Override
019        public View getView() {
020                return view;
021        }
022
023        @Override
024        public Model getModel() {
025                return model;
026        }
027
028        @Override
029        public void notifyModelChanged() {
030                view.setSourceCode(model.getSourceCode());
031        }
032
033        @Override
034        public void init(IDE ide) {
035                view.initComponents(ide);
036        }
037
038}
