001package swp_compiler_ss13.fuc.gui.text;
002
003import java.awt.BorderLayout;
004import java.util.List;
005
006import javax.swing.JComponent;
007import javax.swing.JPanel;
008import javax.swing.JScrollPane;
009import javax.swing.JTextPane;
010import javax.swing.text.BadLocationException;
011import javax.swing.text.Document;
012import javax.swing.text.SimpleAttributeSet;
013import javax.swing.text.StyleConstants;
014
015import org.apache.log4j.Logger;
016
017import swp_compiler_ss13.fuc.gui.ide.mvc.Controller;
018import swp_compiler_ss13.fuc.gui.ide.mvc.IDE;
019import swp_compiler_ss13.fuc.gui.ide.mvc.Position;
020import swp_compiler_ss13.fuc.gui.ide.mvc.View;
021
022/**
023 * 
024 * View of {@link IDE} to display the elements converted from {@link Text_Model}
025 * 
026 * @author "Eduard Wolf"
027 * 
028 */
029public class Text_View implements View {
030
031        private static final Logger LOG = Logger.getLogger(Text_View.class);
032
033        private final Position position;
034        private final Controller controller;
035        private final JScrollPane panel;
036        private final JPanel contentPanel;
037        private final JTextPane area;
038
039        private final Document document;
040
041        public Text_View(Controller controller, Position position) {
042                this.controller = controller;
043                this.position = position;
044                this.contentPanel = new JPanel(new BorderLayout());
045                this.panel = new JScrollPane(this.contentPanel, JScrollPane.VERTICAL_SCROLLBAR_AS_NEEDED,
046                                JScrollPane.HORIZONTAL_SCROLLBAR_AS_NEEDED);
047                area = new JTextPane();
048                area.setEditable(false);
049                document = area.getDocument();
050                contentPanel.add(area);
051        }
052
053        /**
054         * {@inheritDoc}
055         */
056        @Override
057        public JComponent getComponent() {
058                return this.panel;
059        }
060
061        /**
062         * {@inheritDoc}
063         */
064        @Override
065        public String getName() {
066                return this.position.name();
067        }
068
069        /**
070         * {@inheritDoc}
071         */
072        @Override
073        public Position getPosition() {
074                return this.position;
075        }
076
077        /**
078         * {@inheritDoc}
079         */
080        @Override
081        public Controller getController() {
082                return this.controller;
083        }
084
085        public void setViewInformation(List<StringColourPair> viewInformation) {
086                try {
087                        document.remove(0, document.getLength());
088                } catch (BadLocationException e) {
089                        LOG.error("Error while removing text to View component", e);
090                }
091                SimpleAttributeSet attrs;
092                for (StringColourPair pair : viewInformation) {
093                        attrs = new SimpleAttributeSet();
094                        StyleConstants.setForeground(attrs, pair.getColor().getColor());
095                        try {
096                                document.insertString(document.getLength(), pair.getText(), attrs);
097                        } catch (BadLocationException e) {
098                                LOG.error("Error while adding text to View component", e);
099                        }
100                }
101                panel.validate();
102                area.setCaretPosition(0);
103        }
104
105        /**
106         * {@inheritDoc}
107         */
108        @Override
109        public void initComponents(IDE ide) {}
110
111}
