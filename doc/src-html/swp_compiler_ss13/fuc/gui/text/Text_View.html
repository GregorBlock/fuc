001package swp_compiler_ss13.fuc.gui.text;
002
003import java.util.List;
004
005import javax.swing.JComponent;
006import javax.swing.JPanel;
007import javax.swing.JScrollPane;
008import javax.swing.JTextPane;
009import javax.swing.text.BadLocationException;
010import javax.swing.text.Document;
011import javax.swing.text.SimpleAttributeSet;
012import javax.swing.text.StyleConstants;
013
014import org.apache.log4j.Logger;
015
016import swp_compiler_ss13.fuc.gui.ide.mvc.Controller;
017import swp_compiler_ss13.fuc.gui.ide.mvc.IDE;
018import swp_compiler_ss13.fuc.gui.ide.mvc.Position;
019import swp_compiler_ss13.fuc.gui.ide.mvc.View;
020
021/**
022 * 
023 * View of {@link IDE} to display the elements converted from {@link Text_Model}
024 * 
025 * @author "Eduard Wolf"
026 * 
027 */
028public class Text_View implements View {
029
030        private Logger log = Logger.getLogger(Text_View.class);
031
032        private final Position position;
033        private final Controller controller;
034        private final JScrollPane panel;
035        private final JPanel contentPanel;
036
037        public Text_View(Controller controller, Position position) {
038                this.controller = controller;
039                this.position = position;
040                this.contentPanel = new JPanel();
041                this.panel = new JScrollPane(this.contentPanel, JScrollPane.VERTICAL_SCROLLBAR_AS_NEEDED,
042                                JScrollPane.HORIZONTAL_SCROLLBAR_AS_NEEDED);
043        }
044
045        /**
046         * {@inheritDoc}
047         */
048        @Override
049        public JComponent getComponent() {
050                return this.panel;
051        }
052
053        /**
054         * {@inheritDoc}
055         */
056        @Override
057        public String getName() {
058                return this.position.name();
059        }
060
061        /**
062         * {@inheritDoc}
063         */
064        @Override
065        public Position getPosition() {
066                return this.position;
067        }
068
069        /**
070         * {@inheritDoc}
071         */
072        @Override
073        public Controller getController() {
074                return this.controller;
075        }
076
077        public void setViewInformation(List<StringColourPair> viewInformation) {
078                this.contentPanel.removeAll();
079                JTextPane area = new JTextPane();
080                SimpleAttributeSet attrs;
081                Document document = area.getDocument();
082                for (StringColourPair pair : viewInformation) {
083                        attrs = new SimpleAttributeSet();
084                        StyleConstants.setForeground(attrs, pair.getColor().getColor());
085                        try {
086                                document.insertString(document.getLength(), pair.getText(), attrs);
087                        } catch (BadLocationException e) {
088                                this.log.error("Error while adding text to View component", e);
089                        }
090                }
091                this.contentPanel.add(area);
092        }
093
094        /**
095         * {@inheritDoc}
096         */
097        @Override
098        public void initComponents(IDE ide) {
099        }
100
101}
