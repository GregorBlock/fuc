001package swp_compiler_ss13.fuc.gui.text.token;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.HashMap;
006import java.util.List;
007import java.util.Map;
008
009import swp_compiler_ss13.common.lexer.Token;
010import swp_compiler_ss13.common.lexer.TokenType;
011import swp_compiler_ss13.fuc.gui.text.ColorWrapper;
012import swp_compiler_ss13.fuc.gui.text.StringColourPair;
013import swp_compiler_ss13.fuc.gui.text.StringColourPair.DefaultColorWrapper;
014import swp_compiler_ss13.fuc.gui.text.Text_Controller;
015import swp_compiler_ss13.fuc.gui.text.Text_Model;
016
017/**
018 * @author "Eduard Wolf"
019 * 
020 */
021public class TextGUITokenVisualizationModel extends Text_Model {
022
023        private final Map<TokenType, ColorWrapper> tokenColor;
024        private boolean showLineColumnInformation = true;
025        private List<Token> lastToken;
026
027        public TextGUITokenVisualizationModel(Text_Controller controller) {
028                super(controller, ModelType.TOKEN);
029                tokenColor = new HashMap<>();
030                ColorWrapper wrapper;
031                for (TokenType type : TokenType.values()) {
032                        switch (type) {
033                        case AND:
034                        case OR:
035                        case NOT:
036                        case NOT_EQUALS:
037                        case EQUALS:
038                        case GREATER:
039                        case GREATER_EQUAL:
040                        case LESS:
041                        case LESS_OR_EQUAL:
042                                wrapper = DefaultColorWrapper.BLUE;
043                                break;
044                        case ASSIGNOP:
045                                wrapper = DefaultColorWrapper.GREEN;
046                                break;
047                        case BOOL_SYMBOL:
048                        case DOUBLE_SYMBOL:
049                        case RECORD_SYMBOL:
050                        case LONG_SYMBOL:
051                                wrapper = DefaultColorWrapper.ORANGE;
052                                break;
053                        case BREAK:
054                        case WHILE:
055                        case DO:
056                        case IF:
057                        case ELSE:
058                        case RETURN:
059                        case PRINT:
060                                wrapper = DefaultColorWrapper.YELLOW;
061                                break;
062                        case NUM:
063                        case REAL:
064                        case STRING:
065                        case TRUE:
066                        case FALSE:
067                                wrapper = DefaultColorWrapper.RED;
068                                break;
069                        case COMMENT:
070                        case EOF:
071                        case NOT_A_TOKEN:
072                                wrapper = DefaultColorWrapper.GRAY;
073                                break;
074                        case DIVIDE:
075                        case TIMES:
076                        case PLUS:
077                        case MINUS:
078                                wrapper = DefaultColorWrapper.CYAN;
079                                break;
080                        case ID:
081                        case DOT:
082                                wrapper = DefaultColorWrapper.ORANGE;
083                                break;
084                        case LEFT_BRACE:
085                        case LEFT_BRACKET:
086                        case RIGHT_BRACE:
087                        case RIGHT_BRACKET:
088                        case LEFT_PARAN:
089                        case RIGHT_PARAN:
090                        case SEMICOLON:
091                        default:
092                                wrapper = DefaultColorWrapper.BLACK;
093                                break;
094                        }
095                        tokenColor.put(type, wrapper);
096                }
097        }
098
099        @Override
100        protected List<StringColourPair> tokenToViewInformation(List<Token> tokens) {
101                lastToken = tokens;
102                if (tokens == null) {
103                        return new ArrayList<>(Arrays.asList(new StringColourPair()));
104                }
105                List<StringColourPair> result = new ArrayList<>();
106                Integer line = tokens.get(0).getLine();
107                StringBuilder text;
108                for (Token token : tokens) {
109                        text = new StringBuilder();
110                        text.append(token.getLine().equals(line) ? "" : "\n");
111                        text.append('<');
112                        line = token.getLine();
113                        if (token.getTokenType() == TokenType.NUM || token.getTokenType() == TokenType.REAL
114                                        || token.getTokenType() == TokenType.ID) {
115                                text.append(token.getTokenType());
116                                text.append(", ");
117                        } else {}
118                        text.append(token.getValue());
119                        if (showLineColumnInformation) {
120                                text.append(", ");
121                                text.append(token.getLine());
122                                text.append(", ");
123                                text.append(token.getColumn());
124                        }
125                        text.append('>');
126                        result.add(new StringColourPair().setText(text.toString()).setColor(
127                                        tokenColor.get(token.getTokenType())));
128                }
129                return result;
130        }
131
132        public void toggleLineColumnInfo() {
133                showLineColumnInformation = !showLineColumnInformation;
134                setTokens(lastToken);
135        }
136
137}
