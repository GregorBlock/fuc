001package swp_compiler_ss13.fuc.gui.text.tac;
002
003import java.util.HashMap;
004import java.util.Map;
005
006import swp_compiler_ss13.common.backend.Quadruple;
007import swp_compiler_ss13.common.backend.Quadruple.Operator;
008import swp_compiler_ss13.fuc.gui.text.ColorWrapper;
009import swp_compiler_ss13.fuc.gui.text.StringColourPair;
010import swp_compiler_ss13.fuc.gui.text.StringColourPair.DefaultColorWrapper;
011import swp_compiler_ss13.fuc.gui.text.Text_Controller;
012import swp_compiler_ss13.fuc.gui.text.Text_Model;
013
014/**
015 * @author "Eduard Wolf"
016 * 
017 */
018public class TextGUITacVisualizationModel extends Text_Model {
019
020        private final Map<Operator, ColorWrapper> tacColor;
021
022        public TextGUITacVisualizationModel(Text_Controller controller) {
023                super(controller, ModelType.TAC);
024                tacColor = new HashMap<>();
025                ColorWrapper wrapper;
026                for (Operator type : Operator.values()) {
027                        switch (type) {
028                        case ADD_DOUBLE:
029                        case ADD_LONG:
030                        case SUB_DOUBLE:
031                        case SUB_LONG:
032                        case MUL_DOUBLE:
033                        case MUL_LONG:
034                        case DIV_DOUBLE:
035                        case DIV_LONG:
036                                wrapper = DefaultColorWrapper.CYAN;
037                                break;
038                        case AND_BOOLEAN:
039                        case OR_BOOLEAN:
040                        case NOT_BOOLEAN:
041                        case COMPARE_DOUBLE_E:
042                        case COMPARE_DOUBLE_G:
043                        case COMPARE_DOUBLE_GE:
044                        case COMPARE_DOUBLE_L:
045                        case COMPARE_DOUBLE_LE:
046                        case COMPARE_LONG_E:
047                        case COMPARE_LONG_G:
048                        case COMPARE_LONG_GE:
049                        case COMPARE_LONG_L:
050                        case COMPARE_LONG_LE:
051                                wrapper = DefaultColorWrapper.YELLOW;
052                                break;
053                        case ASSIGN_BOOLEAN:
054                        case ASSIGN_DOUBLE:
055                        case ASSIGN_LONG:
056                        case ASSIGN_STRING:
057                                wrapper = DefaultColorWrapper.GREEN;
058                                break;
059                        case ARRAY_GET_BOOLEAN:
060                        case ARRAY_GET_DOUBLE:
061                        case ARRAY_GET_LONG:
062                        case ARRAY_GET_REFERENCE:
063                        case ARRAY_GET_STRING:
064                                wrapper = DefaultColorWrapper.ORANGE;
065                                break;
066                        case ARRAY_SET_BOOLEAN:
067                        case ARRAY_SET_DOUBLE:
068                        case ARRAY_SET_LONG:
069                        case ARRAY_SET_STRING:
070                                wrapper = DefaultColorWrapper.ORANGE;
071                                break;
072                        case DECLARE_ARRAY:
073                        case DECLARE_BOOLEAN:
074                        case DECLARE_DOUBLE:
075                        case DECLARE_LONG:
076                        case DECLARE_REFERENCE:
077                        case DECLARE_STRING:
078                                wrapper = DefaultColorWrapper.BLUE;
079                                break;
080                        case LABEL:
081                        case BRANCH:
082                        case RETURN:
083                                wrapper = DefaultColorWrapper.RED;
084                                break;
085
086                        case DOUBLE_TO_LONG:
087                        case LONG_TO_DOUBLE:
088                                wrapper = DefaultColorWrapper.GRAY;
089                                break;
090                        case PRINT_STRING:
091                        default:
092                                wrapper = DefaultColorWrapper.BLACK;
093                                break;
094                        }
095                        tacColor.put(type, wrapper);
096                }
097        }
098
099        @Override
100        protected StringColourPair tacToViewInformation(Quadruple quadruple) {
101                if (quadruple == null) {
102                        return new StringColourPair();
103                }
104                String format = "(%s | %s | %s | %s)\n";
105                Operator operator = quadruple.getOperator();
106                String text = String.format(format, operator.toString(), quadruple.getArgument1(),
107                                quadruple.getArgument2(), quadruple.getResult());
108                return new StringColourPair().setText(text).setColor(tacColor.get(operator));
109
110        }
111
112}
