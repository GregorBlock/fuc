001package swp_compiler_ss13.fuc.gui.text.target;
002
003import java.io.BufferedReader;
004import java.io.ByteArrayInputStream;
005import java.io.IOException;
006import java.io.InputStream;
007import java.io.InputStreamReader;
008import java.util.LinkedList;
009import java.util.List;
010import java.util.Map;
011import java.util.Map.Entry;
012
013import org.apache.log4j.Logger;
014
015import swp_compiler_ss13.fuc.gui.text.StringColourPair;
016import swp_compiler_ss13.fuc.gui.text.StringColourPair.DefaultColorWrapper;
017import swp_compiler_ss13.fuc.gui.text.Text_Controller;
018import swp_compiler_ss13.fuc.gui.text.Text_Model;
019
020/**
021 * @author "Eduard Wolf"
022 * 
023 */
024public class TextGUITargetVisualizationModel extends Text_Model {
025
026        private static final Logger LOG = Logger.getLogger(TextGUITargetVisualizationModel.class);
027
028        private Map<String, InputStream> lastTarget;
029        private boolean showFullCode = true;
030
031        public TextGUITargetVisualizationModel(Text_Controller controller) {
032                super(controller, ModelType.TARGET);
033        }
034
035        @Override
036        protected List<StringColourPair> targetToViewInformation(Map<String, InputStream> target) {
037                lastTarget = target;
038                List<StringColourPair> result = new LinkedList<>();
039                if (target != null) {
040                        BufferedReader reader;
041                        String line;
042                        for (Entry<String, InputStream> entry : target.entrySet()) {
043                                if ("main.ll".equals(entry.getKey())) {
044                                        StringBuilder code = new StringBuilder();
045                                        boolean inMain = false;
046                                        String codeStart = "define i64 @main() {";
047                                        result.add(new StringColourPair().setText(entry.getKey() + ":\n").setColor(
048                                                        DefaultColorWrapper.BLUE));
049                                        reader = new BufferedReader(new InputStreamReader(entry.getValue()));
050                                        try {
051                                                while ((line = reader.readLine()) != null) {
052                                                        code.append(line);
053                                                        code.append('\n');
054                                                        if (codeStart.equals(line)) {
055                                                                inMain = true;
056                                                        } else if (line.isEmpty()) {
057                                                                if (inMain) {}
058                                                                inMain = false;
059                                                        }
060                                                        LOG.debug("line: " + line);
061                                                        if (showFullCode || inMain) {
062                                                                if (line.startsWith(";")) {
063                                                                        result.add(new StringColourPair().setText(line + "\n")
064                                                                                        .setColor(DefaultColorWrapper.GRAY));
065                                                                } else {
066                                                                        result.add(new StringColourPair().setText(line + "\n"));
067                                                                }
068                                                        }
069                                                }
070                                                if (code.length() != 0) {
071                                                        code.setLength(code.length() - 1);
072                                                }
073                                        } catch (IOException e) {
074                                                LOG.error("bad inputstream in from target code generator", e);
075                                        } finally {
076                                                try {
077                                                        reader.close();
078                                                } catch (IOException e) {}
079                                        }
080                                        entry.setValue(new ByteArrayInputStream(code.toString().getBytes()));
081                                        // only one file in llvm backend
082                                        break;
083                                } else {
084                                        // TODO java code
085                                }
086                        }
087                }
088                return result;
089        }
090
091        public void toogleCodeAmount() {
092                showFullCode = !showFullCode;
093                setTargetCode(lastTarget);
094        }
095}
