001package swp_compiler_ss13.fuc.gui.ide;
002
003import java.awt.Component;
004import java.io.PrintWriter;
005import java.io.StringWriter;
006import java.io.Writer;
007
008import javax.swing.JOptionPane;
009
010/**
011 * Displays a critical error on the view
012 * 
013 * @author "Frank Zechert"
014 * 
015 */
016public class FucIdeCriticalError {
017        private boolean recoverable;
018        private String errormsg;
019        private Component parent;
020
021        /**
022         * Initialize and show the alert dialog
023         * 
024         * @param parent
025         *            The parent component
026         * @param errormsg
027         *            The error message
028         * @param recoverable
029         *            Whether the error is recoverable
030         */
031        public FucIdeCriticalError(Component parent, String errormsg, boolean recoverable) {
032                this.parent = parent;
033                this.errormsg = errormsg;
034                this.recoverable = recoverable;
035                this.show();
036        }
037
038        public FucIdeCriticalError(Component parent, Throwable e, boolean recoverable) {
039                this.parent = parent;
040                this.recoverable = recoverable;
041                StringBuilder message = new StringBuilder();
042                message.append("An Exception occurred!\n\n");
043                message.append(e.toString() + "\n");
044                message.append(e.getMessage() + "\n");
045                Throwable t = e;
046                while (t.getCause() != null && message.length() < 400)
047                {
048                        message.append("caused by: " + t.toString() + "\n");
049                }
050                message.append("\n");
051                message.append("Stacktrace (truncated):\n");
052                final Writer result = new StringWriter();
053                final PrintWriter printWriter = new PrintWriter(result);
054                t.printStackTrace(printWriter);
055                message.append(result.toString().substring(0, 600) + "...");
056                this.errormsg = message.toString();
057                this.show();
058
059        }
060
061        private void show() {
062                if (FucIdeCriticalError.this.recoverable) {
063                        FucIdeCriticalError.this.errormsg += "\n\nPress OK to quit. Press Cancel to continue anyway.";
064                        int r = JOptionPane.showConfirmDialog(FucIdeCriticalError.this.parent,
065                                        FucIdeCriticalError.this.errormsg, "Recoverable Error",
066                                        JOptionPane.OK_CANCEL_OPTION,
067                                        JOptionPane.ERROR_MESSAGE);
068                        if (r == JOptionPane.OK_OPTION) {
069                                System.exit(1);
070                        }
071                }
072                else {
073                        JOptionPane.showMessageDialog(FucIdeCriticalError.this.parent,
074                                        FucIdeCriticalError.this.errormsg, "Critical Error",
075                                        JOptionPane.ERROR_MESSAGE);
076                        System.exit(1);
077                }
078
079        }
080}
