001package swp_compiler_ss13.fuc.gui.ide.java;
002
003import java.io.BufferedReader;
004import java.io.File;
005import java.io.FileOutputStream;
006import java.io.IOException;
007import java.io.InputStream;
008import java.io.InputStreamReader;
009
010public class JavaExecuter {
011        private final static boolean NO_VERIFY_NEEDED = false;
012        private final static String NO_VERIFY = "-noverify";
013
014        Process p;
015
016        public JavaExecuter(File classFile) throws IOException {
017                if (classFile == null) throw new NullPointerException();
018
019                if (!classFile.exists())
020                        throw new JavaClassProcessRuntimeException(
021                                        "JavaClassProcess can not be started. File does not exists.");
022
023                if (classFile.isDirectory())
024                        throw new JavaClassProcessRuntimeException(
025                                        "JavaClassProcess can not be started. Need a file not a directory.");
026
027                if (!classFile.getName().endsWith(".class"))
028                        throw new JavaClassProcessRuntimeException(
029                                        "JavaClassProcess can not be started. File has not expected extension.");
030
031                String classPath = classFile.getParent();
032                String className = classFile.getName();
033                className = className.substring(0, className.lastIndexOf('.'));
034
035                String javaExecutablePath = System.getProperty("java.home") + File.separator + "bin"
036                                + File.separator + "java";
037                ProcessBuilder processBuilder;
038                if (NO_VERIFY_NEEDED) {
039                        processBuilder = new ProcessBuilder(javaExecutablePath, "-Dfile.encoding=UTF-8", "-cp",
040                                        classPath, NO_VERIFY, className);
041                } else {
042                        processBuilder = new ProcessBuilder(javaExecutablePath, "-Dfile.encoding=UTF-8", "-cp",
043                                        classPath, className);
044                }
045                try {
046                        p = processBuilder.
047                                        redirectErrorStream(true).
048                                        start();
049                        p.waitFor();
050                } catch (IOException | InterruptedException e) {
051                        throw new JavaClassProcessRuntimeException(e.getMessage(), e);
052                }
053        }
054
055        public Integer getReturnValue() {
056                return p.exitValue();
057        }
058
059        public String getProcessOutput() {
060                StringBuilder writer = new StringBuilder();
061                try {
062                        BufferedReader reader = new BufferedReader(new InputStreamReader(p.getInputStream()));
063                        String line;
064                        while ((line = reader.readLine()) != null) {
065                                writer.append(line);
066                                writer.append('\n');
067                        }
068                        if (writer.length() > 0) {
069                                writer.setLength(writer.length() - 1);
070                        }
071                } catch (IOException e) {
072                        e.printStackTrace();
073                }
074                return writer.toString();
075        }
076
077        public InputStream getInputstream() {
078                return p.getInputStream();
079        }
080
081        public static File cloneInputStream(String name, InputStream input) throws IOException {
082                File file = new File("build",name);
083                if(!file.exists()){
084                        (new File("build")).mkdirs();
085                        file.createNewFile();
086                }
087                FileOutputStream bout = new FileOutputStream(file);
088                byte buffer[] = new byte[1024];
089                int len;
090                while ((len = input.read(buffer)) > 0) {
091                        bout.write(buffer, 0, len);
092                }
093                bout.flush();
094                bout.close();
095                
096                return file;
097        }
098}
