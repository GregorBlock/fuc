350        public void addComponentRadioMenuItem(final Lexer lexer) {
351                String name = lexer.getClass().getName();
352                JRadioButtonMenuItem jrb = new JRadioButtonMenuItem(name);
353                this.lexerMenus.put(jrb, lexer);
354                this.lexerMenu.add(jrb);
355                this.lexerGroup.add(jrb);
356                if (this.lexerMenus.size() == 1) {
357                        jrb.setSelected(true);
358                        this.controller.onLexerSelected(lexer);
359                }
360                jrb.addActionListener(new ActionListener() {
361
362                        @Override
363                        public void actionPerformed(ActionEvent e) {
364                                FucIdeView.this.controller.onLexerSelected(lexer);
365                        }
366                });
367        }

474        public void clearButtons() {
475                for (JButton b : this.customButtons) {
476                        this.logger.info("Removing the button " + b.getText());
477                        this.buttonPanel.remove(b);
478                        b.revalidate();
479                }
480                this.customButtons.clear();
481                this.buttonPanel.revalidate();
482                this.buttonPanel.repaint();
483        }

509        public boolean isFirstTab(FucIdeTab t) {
510                try {
511                        return this.componentTabs.getComponentAt(0) == t.getComponent();
512                } catch (IndexOutOfBoundsException e) {
513                        return false;
514                }
515        }
516
517        public boolean isCurrentTab(FucIdeTab t) {
518                return this.componentTabs.getSelectedComponent() == t.getComponent();
519        }
520
521        public void showFirstTab() {
522                this.componentTabs.setSelectedIndex(0);
523        }

525        public void updateTextPane(String valueOf) {
526                Document doc = this.consoleOutput.getDocument();
527                try {
528                        doc.insertString(doc.getLength(), valueOf, null);
529                } catch (BadLocationException e) {
530                        throw new RuntimeException(e);
531                }
532                this.consoleOutput.setCaretPosition(doc.getLength() - 1);
533        }
