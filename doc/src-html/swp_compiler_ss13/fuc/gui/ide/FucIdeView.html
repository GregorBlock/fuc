269        public void addComponentRadioMenuItem(final Lexer lexer) {
270                String name = lexer.getClass().getName();
271                JRadioButtonMenuItem jrb = new JRadioButtonMenuItem(name);
272                this.lexerMenus.put(jrb, lexer);
273                this.lexerMenu.add(jrb);
274                this.lexerGroup.add(jrb);
275                if (this.lexerMenus.size() == 1) {
276                        jrb.setSelected(true);
277                        this.controller.onLexerSelected(lexer);
278                }
279                jrb.addActionListener(new ActionListener() {
280
281                        @Override
282                        public void actionPerformed(ActionEvent e) {
283                                FucIdeView.this.controller.onLexerSelected(lexer);
284                        }
285                });
286        }

377        public void clearMenus() {
378                for (JMenu m : this.customMenus) {
379                        this.menuBar.remove(m);
380                }
381                this.customMenus.clear();
382                this.menuBar.revalidate();
383                this.menuBar.repaint();
384        }

393        public void clearButtons() {
394                for (JButton b : this.customButtons) {
395                        this.logger.info("Removing the button " + b.getText());
396                        this.buttonPanel.remove(b);
397                        b.revalidate();
398                }
399                this.customButtons.clear();
400                this.buttonPanel.revalidate();
401                this.buttonPanel.repaint();
402        }

428        public boolean isFirstTab(FucIdeTab t) {
429                try {
430                        return this.componentTabs.getComponentAt(0) == t.getComponent();
431                } catch (IndexOutOfBoundsException e) {
432                        return false;
433                }
434        }
435
436        public boolean isCurrentTab(FucIdeTab t) {
437                return this.componentTabs.getSelectedComponent() == t.getComponent();
438        }
439
440        public void showFirstTab() {
441                this.componentTabs.setSelectedIndex(0);
442        }

444        public void updateTextPane(String valueOf) {
445                Document doc = this.consoleOutput.getDocument();
446                try {
447                        doc.insertString(doc.getLength(), valueOf, null);
448                } catch (BadLocationException e) {
449                        throw new RuntimeException(e);
450                }
451                this.consoleOutput.setCaretPosition(doc.getLength() - 1);
452        }

454        public void clearErrorLog() {
455                this.errorReportList.setModel(new DefaultListModel<String>());
456                this.errorReportList.revalidate();
457        }

459        public void addErrorLog(String msg) {
460                ((DefaultListModel<String>) this.errorReportList.getModel()).addElement(msg);
461                this.errorReportList.revalidate();
462        }
