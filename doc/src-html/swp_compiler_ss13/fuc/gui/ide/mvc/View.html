001package swp_compiler_ss13.fuc.gui.ide.mvc;
002
003import javax.swing.JComponent;
004
005
006/**
007 * View interface for the MVC pattern
008 * 
009 * @author "Frank Zechert"
010 * 
011 */
012public interface View {
013        /**
014         * Get the component to display this view
015         * 
016         * @return The component containing the gui component
017         */
018        public JComponent getComponent();
019
020        /**
021         * Get the name of this component
022         * 
023         * @return The name of this component
024         */
025        public String getName();
026
027        /**
028         * Get the position of this component
029         * 
030         * @return the position of this component
031         */
032        public Position getPosition();
033
034        /**
035         * Get the controller associated with this view
036         * 
037         * @return the controller
038         */
039        public Controller getController();
040
041        /**
042         * Called to initialize the gui component
043         * 
044         * @param ide
045         *            The ide the component is displayed in
046         */
047        public void initComponents(IDE ide);
048}
