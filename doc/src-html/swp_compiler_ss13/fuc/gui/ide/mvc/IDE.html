001package swp_compiler_ss13.fuc.gui.ide.mvc;
002
003import java.io.InputStream;
004import java.util.List;
005import java.util.Map;
006
007import javax.swing.JButton;
008import javax.swing.JLabel;
009import javax.swing.JMenu;
010
011import swp_compiler_ss13.common.ast.AST;
012import swp_compiler_ss13.common.backend.Quadruple;
013import swp_compiler_ss13.common.lexer.Token;
014import swp_compiler_ss13.fuc.errorLog.ReportLogImpl;
015
016/**
017 * Interface for access to the GUI panels.
018 * 
019 * @author "Frank Zechert"
020 * 
021 */
022public interface IDE {
023        /**
024         * Add your own menu to the menu bar
025         * 
026         * @param menu
027         *            The menu to add
028         * @param position
029         *            The position of the new menu
030         * @param displayAlways
031         *            if true the menu will be displayed all the time. if false the
032         *            menu will only be shown, when the component is currently
033         *            visible (the tab is actvie)
034         */
035        public void addMenu(JMenu menu, Position position, boolean displayAlways);
036
037        /**
038         * Add your own button to the button bar
039         * 
040         * @param button
041         *            The button to add
042         * @param position
043         *            The position of the new button
044         * @param displayAlways
045         *            if true the button will be displayed all the time. if false
046         *            the button will only be shown, when the component is currently
047         *            visible (the tab is actvie)
048         */
049        public void addButton(JButton button, Position position, boolean displayAlways);
050
051        /**
052         * Add your own label to the status bar
053         * 
054         * @param label
055         *            The label to add
056         * @param position
057         *            The position of the new label
058         * @param displayAlways
059         *            if true the label will be displayed all the time. if false the
060         *            label will only be shown, when the component is currently
061         *            visible (the tab is actvie)
062         */
063        public void addStatusLabel(JLabel label, Position position, boolean displayAlways);
064
065        /**
066         * Set the source code that is used for the compiler
067         * 
068         * @param sourceCode
069         *            the source code that should be compiled
070         */
071        public void setSourceCode(String sourceCode);
072
073        /**
074         * Run the compiler (all components) and update all gui components with new
075         * data
076         * 
077         * @param silent
078         *            if set to true no output of the compiler is shown in the gui
079         *            (no errorLog and no exceptions shown).
080         */
081        public void run(boolean silent);
082
083        /**
084         * Run the currently selected lexer component
085         * 
086         * @param sourceCode
087         *            The source code to lex
088         * @param silent
089         *            if set to true no output of the compiler is shown in the gui
090         *            (no errorLog and no exceptions shown).
091         * @return The list of tokens returned by the lexer, returns null if an
092         *         error occured
093         */
094        public List<Token> runLexer(String sourceCode, boolean silent);
095
096        /**
097         * Run the currently selected parser component
098         * 
099         * @param tokens
100         *            The token list to parse
101         * @param silent
102         *            if set to true no output of the compiler is shown in the gui
103         *            (no errorLog and no exceptions shown).
104         * @return The AST generated by the parser, returns null if an error occured
105         */
106        public AST runParser(List<Token> tokens, boolean silent);
107
108        /**
109         * Run the currently selected semantic analyser
110         * 
111         * @param ast
112         *            The ast to analyse
113         * @param silent
114         *            if set to true no output of the compiler is shown in the gui
115         *            (no errorLog and no exceptions shown).
116         * @return The ast checked by the semantic analyser, returns null if an
117         *         error occured
118         */
119        public AST runSemanticAnalysis(AST ast, boolean silent);
120
121        /**
122         * Run the currently selected intermediate code generator
123         * 
124         * @param ast
125         *            The ast to generate tac for
126         * @param silent
127         *            if set to true no output of the compiler is shown in the gui
128         *            (no errorLog and no exceptions shown).
129         * @return The list of tac generated for the ast, returns null if an error
130         *         occured
131         */
132        public List<Quadruple> runIntermediateCodeGenerator(AST ast, boolean silent);
133
134        /**
135         * Run the currently selected backend
136         * 
137         * @param tac
138         *            The tac to generated target code for
139         * @param silent
140         *            if set to true no output of the compiler is shown in the gui
141         *            (no errorLog and no exceptions shown).
142         * @return The target code, returns null if an error occured
143         */
144        public Map<String, InputStream> runBackend(List<Quadruple> tac, boolean silent);
145
146        /**
147         * Run the given Program
148         * 
149         * @param program
150         *            The program to execute
151         * @param silent
152         *            if set to true no output of the compiler is shown in the gui
153         *            (no errorLog and no exceptions shown).
154         * @return The output of the executed program
155         */
156        public String runProgram(Map<String, InputStream> program, boolean silent);
157
158        /**
159         * Show the tab for the given controller (select it, make it active)
160         * 
161         * @param controller
162         *            The controller to show the tabe for
163         */
164        public void showTab(Controller controller);
165
166        /**
167         * Reports the last used report log. Never returns null but an empty report
168         * log instead
169         * 
170         * @return the last used report log
171         */
172        public ReportLogImpl getReportLog();
173}
