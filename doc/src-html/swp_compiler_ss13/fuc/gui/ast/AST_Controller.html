001package swp_compiler_ss13.fuc.gui.ast;
002
003import org.apache.log4j.Logger;
004
005import swp_compiler_ss13.common.ast.ASTNode;
006import swp_compiler_ss13.fuc.gui.ide.mvc.Controller;
007import swp_compiler_ss13.fuc.gui.ide.mvc.IDE;
008import swp_compiler_ss13.fuc.gui.ide.mvc.Model;
009import swp_compiler_ss13.fuc.gui.ide.mvc.View;
010
011/**
012 * 
013 * @author "Eduard Wolf"
014 * 
015 */
016public class AST_Controller implements Controller {
017
018        private static final Logger LOG = Logger.getLogger(AST_Controller.class);
019
020        private final AST_Model model;
021        private final AST_View view;
022
023        private final AST_Controller root;
024
025        private final boolean checkAST;
026
027        public AST_Controller() {
028                this(false);
029        }
030
031        public AST_Controller(boolean checkedAST) {
032                this(null, checkedAST);
033        }
034
035        AST_Controller(AST_Controller root, boolean checkAST) {
036                this.checkAST = checkAST;
037                this.root = root == null ? this : root;
038                this.model = new AST_Model(this);
039                this.view = new AST_View(this, root == null, getAstGuiElementFactory(), checkAST);
040        }
041
042        protected AstGuiElementFactory getAstGuiElementFactory() {
043                return AstGuiElementFactory.DEFAULT_FACTORY;
044        }
045
046        @Override
047        public View getView() {
048                return this.view;
049        }
050
051        @Override
052        public Model getModel() {
053                return this.model;
054        }
055
056        @Override
057        public void notifyModelChanged() {
058                this.notifyModelChangedWithoutLayoutChange();
059                this.root.view.recalculateLayout();
060        }
061
062        protected void notifyModelChangedWithoutLayoutChange() {
063                LOG.trace("node: " + this.nodeToString(this.model.getNode()));
064                this.view.setNode(this.model.getNode());
065                AST_Controller ast_Controller;
066                for (ASTNode node : this.model.getChildren()) {
067                        LOG.trace("childnode: " + this.nodeToString(node));
068                        ast_Controller = getAstGuiElementFactory().getController(this.root, checkAST);
069                        ast_Controller.model.setNode(node);
070                        ast_Controller.notifyModelChangedWithoutLayoutChange();
071                        this.view.addChild(ast_Controller.view);
072                }
073        }
074
075        private String nodeToString(ASTNode node) {
076                if (node == null) {
077                        return "null";
078                }
079                return "type: " + node.getNodeType().name() + ", amount of children: "
080                                + node.getChildren().size();
081        }
082
083        @Override
084        public void init(IDE ide) {
085                this.view.initComponents(ide);
086        }
087
088        public void viewStateChanged() {
089                this.view.changeChildState();
090                this.root.view.recalculateLayout();
091        }
092
093        public void toggleNodeSize() {
094                this.view.toggleNodeSize();
095        }
096
097}
