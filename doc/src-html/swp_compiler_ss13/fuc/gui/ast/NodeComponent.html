001package swp_compiler_ss13.fuc.gui.ast;
002
003import java.awt.BorderLayout;
004import java.awt.Image;
005import java.awt.Toolkit;
006import java.util.HashMap;
007import java.util.Map;
008
009import javax.swing.BoxLayout;
010import javax.swing.Icon;
011import javax.swing.ImageIcon;
012import javax.swing.JButton;
013import javax.swing.JComponent;
014import javax.swing.JLabel;
015import javax.swing.JPanel;
016
017import org.apache.log4j.Logger;
018
019import swp_compiler_ss13.common.ast.ASTNode;
020import swp_compiler_ss13.common.ast.ASTNode.ASTNodeType;
021import swp_compiler_ss13.common.ast.nodes.binary.BinaryExpressionNode;
022import swp_compiler_ss13.common.ast.nodes.binary.BinaryExpressionNode.BinaryOperator;
023import swp_compiler_ss13.common.ast.nodes.leaf.BasicIdentifierNode;
024import swp_compiler_ss13.common.ast.nodes.leaf.LiteralNode;
025import swp_compiler_ss13.common.ast.nodes.marynary.BlockNode;
026import swp_compiler_ss13.common.ast.nodes.unary.ArrayIdentifierNode;
027import swp_compiler_ss13.common.ast.nodes.unary.DeclarationNode;
028import swp_compiler_ss13.common.ast.nodes.unary.StructIdentifierNode;
029import swp_compiler_ss13.common.ast.nodes.unary.UnaryExpressionNode;
030
031public class NodeComponent {
032
033        private static final Logger LOG = Logger.getLogger(NodeComponent.class);
034        private static final Map<ASTNodeType, ImageIcon> IMAGE_TYPE_MAP = new HashMap<>();
035        private static final Map<BinaryOperator, ImageIcon> IMAGE_OPERATOR_MAP = new HashMap<>();
036        private static final String dir = "/swp_compiler_ss13/fuc/gui/ast/assets/";
037        private static final int LOGO_WIDTH = 30;
038        private static final int LOGO_HEIGHT = 30;
039
040        private final JComponent component;
041
042        public NodeComponent(ASTNode node) {
043                if (node.getChildren().isEmpty()) {
044                        component = new JPanel();
045                } else {
046                        component = new JButton();
047                }
048                getComponent().setLayout(new BorderLayout(3, 3));
049                if (IMAGE_TYPE_MAP.isEmpty()) {
050                        IMAGE_TYPE_MAP.put(null, getIconFromString(dir + "default.png"));
051                        IMAGE_OPERATOR_MAP.put(null, getIconFromString(dir + "default.png"));
052                }
053                getComponent().add(new JLabel(getIcon(node)), BorderLayout.WEST);
054                getComponent().add(new JLabel(node.getNodeType().name()), BorderLayout.NORTH);
055                getComponent().add(getInfoComponent(node), BorderLayout.CENTER);
056        }
057
058        private JComponent getInfoComponent(ASTNode node) {
059                JPanel result = new JPanel();
060                result.setLayout(new BoxLayout(result, BoxLayout.Y_AXIS));
061                switch (node.getNodeType()) {
062                case LiteralNode:
063                        LiteralNode literalNode = (LiteralNode) node;
064                        result.add(new JLabel("type: " + literalNode.getLiteralType()));
065                        result.add(new JLabel("name: " + literalNode.getLiteral()));
066                        break;
067                case ArithmeticBinaryExpressionNode:
068                case LogicBinaryExpressionNode:
069                case RelationExpressionNode:
070                        BinaryExpressionNode binaryNode = (BinaryExpressionNode) node;
071                        result.add(new JLabel("operator: " + binaryNode.getOperator()));
072                        break;
073                case ArithmeticUnaryExpressionNode:
074                case LogicUnaryExpressionNode:
075                        UnaryExpressionNode unaryNode = (UnaryExpressionNode) node;
076                        result.add(new JLabel("operator: " + unaryNode.getOperator()));
077                        break;
078                case ArrayIdentifierNode:
079                        ArrayIdentifierNode arrayNode = (ArrayIdentifierNode) node;
080                        result.add(new JLabel("index: " + arrayNode.getIndex()));
081                        break;
082                case BasicIdentifierNode:
083                        BasicIdentifierNode basicNode = (BasicIdentifierNode) node;
084                        result.add(new JLabel("variable: " + basicNode.getIdentifier()));
085                        break;
086                case BlockNode:
087                        BlockNode blockNode = (BlockNode) node;
088                        result.add(new JLabel("amount of declarations: " + blockNode.getNumberOfDeclarations()));
089                        result.add(new JLabel("amount of statements: " + blockNode.getNumberOfStatements()));
090                        break;
091                case DeclarationNode:
092                        DeclarationNode declNode = (DeclarationNode) node;
093                        result.add(new JLabel("type: " + declNode.getType()));
094                        result.add(new JLabel("name: " + declNode.getIdentifier()));
095                        break;
096                case StructIdentifierNode:
097                        StructIdentifierNode structNode = (StructIdentifierNode) node;
098                        result.add(new JLabel("field: " + structNode.getFieldName()));
099                case AssignmentNode:
100                case BranchNode:
101                case BreakNode:
102                case DoWhileNode:
103                case PrintNode:
104                case WhileNode:
105                default:
106                        break;
107                }
108                return result;
109        }
110
111        private Icon getIcon(ASTNode node) {
112                ImageIcon imageIcon = IMAGE_TYPE_MAP.get(node.getNodeType());
113                if (imageIcon != null) {
114                        return imageIcon;
115                }
116                switch (node.getNodeType()) {
117                case ArithmeticBinaryExpressionNode:
118                case LogicBinaryExpressionNode:
119                case RelationExpressionNode:
120                        BinaryOperator operator = ((BinaryExpressionNode) node).getOperator();
121                        imageIcon = IMAGE_OPERATOR_MAP.get(operator);
122                        if (imageIcon != null) {
123                                return imageIcon;
124                        }
125                        switch (operator) {
126                        case ADDITION:
127                                imageIcon = getIconFromString(dir + "add.png");
128                                IMAGE_OPERATOR_MAP.put(operator, imageIcon);
129                                return imageIcon;
130                        case SUBSTRACTION:
131                                imageIcon = getIconFromString(dir + "minus.png");
132                                IMAGE_OPERATOR_MAP.put(operator, imageIcon);
133                                return imageIcon;
134                        case MULTIPLICATION:
135                                imageIcon = getIconFromString(dir + "mult.png");
136                                IMAGE_OPERATOR_MAP.put(operator, imageIcon);
137                                return imageIcon;
138                        case DIVISION:
139                                imageIcon = getIconFromString(dir + "div.png");
140                                IMAGE_OPERATOR_MAP.put(operator, imageIcon);
141                                return imageIcon;
142                        case EQUAL:
143                                imageIcon = getIconFromString(dir + "equal.png");
144                                IMAGE_OPERATOR_MAP.put(operator, imageIcon);
145                                return imageIcon;
146                        default:
147                                return IMAGE_OPERATOR_MAP.get(null);
148                        }
149                case LogicUnaryExpressionNode:
150                        imageIcon = getIconFromString(dir + "not.png");
151                        IMAGE_TYPE_MAP.put(node.getNodeType(), imageIcon);
152                        return imageIcon;
153                case BlockNode:
154                        imageIcon = getIconFromString(dir + "braces.png");
155                        IMAGE_TYPE_MAP.put(node.getNodeType(), imageIcon);
156                        return imageIcon;
157                case AssignmentNode:
158                        imageIcon = getIconFromString(dir + "equal.png");
159                        IMAGE_TYPE_MAP.put(node.getNodeType(), imageIcon);
160                        return imageIcon;
161                default:
162                        return IMAGE_TYPE_MAP.get(null);
163                }
164        }
165
166        private ImageIcon getIconFromString(String resource) {
167                Image image = Toolkit.getDefaultToolkit().getImage(
168                                NodeComponent.class.getResource(resource));
169                if (image == null) {
170                        LOG.warn("didn't found image: " + resource);
171                        return null;
172                }
173                return new ImageIcon(image.getScaledInstance(LOGO_WIDTH, LOGO_HEIGHT, Image.SCALE_SMOOTH));
174        }
175
176        public JComponent getComponent() {
177                return component;
178        }
179
180}
