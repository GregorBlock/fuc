001package swp_compiler_ss13.fuc.gui.ast;
002
003import java.awt.BorderLayout;
004import java.awt.Image;
005import java.awt.Toolkit;
006import java.util.HashMap;
007import java.util.Map;
008
009import javax.swing.BoxLayout;
010import javax.swing.Icon;
011import javax.swing.ImageIcon;
012import javax.swing.JButton;
013import javax.swing.JComponent;
014import javax.swing.JLabel;
015import javax.swing.JPanel;
016
017import org.apache.log4j.Logger;
018
019import swp_compiler_ss13.common.ast.ASTNode;
020import swp_compiler_ss13.common.ast.ASTNode.ASTNodeType;
021import swp_compiler_ss13.common.ast.nodes.binary.BinaryExpressionNode;
022import swp_compiler_ss13.common.ast.nodes.binary.BinaryExpressionNode.BinaryOperator;
023import swp_compiler_ss13.common.ast.nodes.leaf.BasicIdentifierNode;
024import swp_compiler_ss13.common.ast.nodes.leaf.LiteralNode;
025import swp_compiler_ss13.common.ast.nodes.marynary.BlockNode;
026import swp_compiler_ss13.common.ast.nodes.unary.DeclarationNode;
027import swp_compiler_ss13.common.ast.nodes.unary.StructIdentifierNode;
028import swp_compiler_ss13.common.ast.nodes.unary.UnaryExpressionNode;
029
030public class NodeComponent {
031
032        private static final Logger LOG = Logger.getLogger(NodeComponent.class);
033        private static final Map<ASTNodeType, ImageIcon> IMAGE_TYPE_MAP = new HashMap<>();
034        private static final Map<BinaryOperator, ImageIcon> IMAGE_OPERATOR_MAP = new HashMap<>();
035        private static final String dir = "/swp_compiler_ss13/fuc/gui/ast/assets/";
036        private static final int LOGO_WIDTH = 30;
037        private static final int LOGO_HEIGHT = 30;
038
039        private final JComponent component;
040        private final JComponent iconComponent;
041        private final JComponent nameComponent;
042        private final JComponent infoComponent;
043
044        private boolean large = true;
045
046        public NodeComponent(ASTNode node) {
047                if (node.getChildren().isEmpty()) {
048                        this.component = new JPanel();
049                } else {
050                        this.component = new JButton();
051                }
052                this.getComponent().setLayout(new BorderLayout(3, 3));
053                if (IMAGE_TYPE_MAP.isEmpty()) {
054                        IMAGE_TYPE_MAP.put(null, this.getIconFromString(dir + "default.png"));
055                        IMAGE_OPERATOR_MAP.put(null, this.getIconFromString(dir + "default.png"));
056                }
057                this.iconComponent = new JLabel(this.getIcon(node));
058                this.getComponent().add(this.iconComponent, BorderLayout.WEST);
059                this.nameComponent = new JLabel(node.getNodeType().name());
060                this.getComponent().add(this.nameComponent, BorderLayout.NORTH);
061                this.infoComponent = this.getInfoComponent(node);
062                this.getComponent().add(this.infoComponent, BorderLayout.CENTER);
063                this.getComponent().setVisible(true);
064        }
065
066        void toggleSize() {
067                this.large = !this.large;
068                this.nameComponent.setVisible(this.large);
069                this.infoComponent.setVisible(this.large);
070        }
071
072        protected JComponent getInfoComponent(ASTNode node) {
073                JPanel result = new JPanel();
074                result.setLayout(new BoxLayout(result, BoxLayout.Y_AXIS));
075                switch (node.getNodeType()) {
076                case LiteralNode:
077                        LiteralNode literalNode = (LiteralNode) node;
078                        result.add(new JLabel("type: " + literalNode.getLiteralType()));
079                        result.add(new JLabel("name: " + literalNode.getLiteral()));
080                        break;
081                case ArithmeticBinaryExpressionNode:
082                case LogicBinaryExpressionNode:
083                case RelationExpressionNode:
084                        BinaryExpressionNode binaryNode = (BinaryExpressionNode) node;
085                        result.add(new JLabel("operator: " + binaryNode.getOperator()));
086                        break;
087                case ArithmeticUnaryExpressionNode:
088                case LogicUnaryExpressionNode:
089                        UnaryExpressionNode unaryNode = (UnaryExpressionNode) node;
090                        result.add(new JLabel("operator: " + unaryNode.getOperator()));
091                        break;
092                case ArrayIdentifierNode:
093                        //ArrayIdentifierNode arrayNode = (ArrayIdentifierNode) node;
094                        //result.add(new JLabel("index: " + arrayNode.getIndex()));
095                        break;
096                case BasicIdentifierNode:
097                        BasicIdentifierNode basicNode = (BasicIdentifierNode) node;
098                        result.add(new JLabel("variable: " + basicNode.getIdentifier()));
099                        break;
100                case BlockNode:
101                        BlockNode blockNode = (BlockNode) node;
102                        result.add(new JLabel("amount of declarations: " + blockNode.getNumberOfDeclarations()));
103                        result.add(new JLabel("amount of statements: " + blockNode.getNumberOfStatements()));
104                        break;
105                case DeclarationNode:
106                        DeclarationNode declNode = (DeclarationNode) node;
107                        result.add(new JLabel("type: " + declNode.getType()));
108                        result.add(new JLabel("name: " + declNode.getIdentifier()));
109                        break;
110                case StructIdentifierNode:
111                        StructIdentifierNode structNode = (StructIdentifierNode) node;
112                        result.add(new JLabel("field: " + structNode.getFieldName()));
113                case AssignmentNode:
114                case BranchNode:
115                case BreakNode:
116                case DoWhileNode:
117                case PrintNode:
118                case WhileNode:
119                default:
120                        break;
121                }
122                return result;
123        }
124
125        private Icon getIcon(ASTNode node) {
126                ImageIcon imageIcon = IMAGE_TYPE_MAP.get(node.getNodeType());
127                if (imageIcon != null) {
128                        return imageIcon;
129                }
130                String imageName = null;
131                switch (node.getNodeType()) {
132                case ArithmeticBinaryExpressionNode:
133                case LogicBinaryExpressionNode:
134                case RelationExpressionNode:
135                        BinaryOperator operator = ((BinaryExpressionNode) node).getOperator();
136                        imageIcon = IMAGE_OPERATOR_MAP.get(operator);
137                        if (imageIcon != null) {
138                                return imageIcon;
139                        }
140                        switch (operator) {
141                        case ADDITION:
142                                imageName = "add";
143                                break;
144                        case SUBSTRACTION:
145                                imageName = "minus";
146                                break;
147                        case MULTIPLICATION:
148                                imageName = "mult";
149                                break;
150                        case DIVISION:
151                                imageName = "div";
152                                break;
153                        case EQUAL:
154                                imageName = "equal";
155                                break;
156                        case GREATERTHAN:
157                                imageName = "greater";
158                                break;
159                        case GREATERTHANEQUAL:
160                                imageName = "greater_equal";
161                                break;
162                        case INEQUAL:
163                                imageName = "inequal";
164                                break;
165                        case LESSTHAN:
166                                imageName = "less";
167                                break;
168                        case LESSTHANEQUAL:
169                                imageName = "less_equal";
170                                break;
171                        case LOGICAL_AND:
172                                imageName = "ampersand";
173                                break;
174                        case LOGICAL_OR:
175                                imageName = "or";
176                                break;
177                        default:
178                                return IMAGE_OPERATOR_MAP.get(null);
179                        }
180                        imageIcon = this.getIconFromString(dir + imageName + ".png");
181                        IMAGE_OPERATOR_MAP.put(operator, imageIcon);
182                        return imageIcon;
183                case LogicUnaryExpressionNode:
184                        imageName = "not";
185                        break;
186                case ArithmeticUnaryExpressionNode:
187                        imageName = "minus";
188                        break;
189                case BlockNode:
190                        imageName = "braces";
191                        break;
192                case AssignmentNode:
193                        imageName = "assign";
194                        break;
195                case ArrayIdentifierNode:
196                        imageName = "array";
197                        break;
198                case BasicIdentifierNode:
199                        imageName = "basicIdentifier";
200                        break;
201                case BranchNode:
202                        imageName = "if_else";
203                        break;
204                case DeclarationNode:
205                        imageName = "decl";
206                        break;
207                case DoWhileNode:
208                case WhileNode:
209                        imageName = "loop";
210                        break;
211                case StructIdentifierNode:
212                        imageName = "struct";
213                        break;
214                case PrintNode:
215                        imageName = "print";
216                        break;
217                case BreakNode:
218                        imageName = "break";
219                        break;
220                case ReturnNode:
221                        imageName = "return";
222                        break;
223                case LiteralNode:
224                        imageName = "literal";
225                        break;
226                default:
227                        return IMAGE_TYPE_MAP.get(null);
228                }
229                imageIcon = this.getIconFromString(dir + imageName + ".png");
230                IMAGE_TYPE_MAP.put(node.getNodeType(), imageIcon);
231                return imageIcon;
232        }
233
234        private ImageIcon getIconFromString(String resource) {
235                Image image = Toolkit.getDefaultToolkit().getImage(
236                                NodeComponent.class.getResource(resource));
237                if (image == null) {
238                        LOG.warn("didn't found image: " + resource);
239                        return null;
240                }
241                return new ImageIcon(image.getScaledInstance(LOGO_WIDTH, LOGO_HEIGHT, Image.SCALE_SMOOTH));
242        }
243
244        public JComponent getComponent() {
245                return this.component;
246        }
