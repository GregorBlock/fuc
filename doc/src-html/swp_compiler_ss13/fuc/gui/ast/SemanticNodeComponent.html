001package swp_compiler_ss13.fuc.gui.ast;
002
003import java.awt.ComponentOrientation;
004import java.awt.GridLayout;
005import java.util.Map;
006import java.util.Map.Entry;
007
008import javax.swing.JComponent;
009import javax.swing.JLabel;
010import javax.swing.JPanel;
011
012import org.apache.log4j.Logger;
013
014import swp_compiler_ss13.common.ast.ASTNode;
015
016public class SemanticNodeComponent extends NodeComponent {
017
018        private static final Logger LOG = Logger.getLogger(SemanticNodeComponent.class);
019
020        public SemanticNodeComponent(ASTNode node) {
021                super(node);
022        }
023
024        protected JComponent getInfoComponent(ASTNode node) {
025                JPanel result = new JPanel();
026                result.setComponentOrientation(ComponentOrientation.LEFT_TO_RIGHT);
027                Map<?, ?> attributeValues = node.getAttributeValues();
028                result.setLayout(new GridLayout(attributeValues.size(), 2, 2, 2));
029                for (Entry<?, ?> entry : attributeValues.entrySet()) {
030                        result.add(new JLabel(entry.getKey().toString()));
031                        result.add(new JLabel(entry.getValue().toString()));
032                        LOG.debug(String.format("found attribute %s with value %s", entry.getKey(), entry
033                                        .getValue()));
034                }
035                return result;
036        }
037
038}
