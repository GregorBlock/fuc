001package swp_compiler_ss13.fuc.gui.ast;
002
003import java.awt.Color;
004import java.awt.ComponentOrientation;
005import java.awt.GridBagConstraints;
006import java.awt.GridBagLayout;
007import java.awt.Insets;
008import java.util.Map;
009import java.util.Map.Entry;
010
011import javax.swing.BorderFactory;
012import javax.swing.JComponent;
013import javax.swing.JLabel;
014import javax.swing.JPanel;
015import javax.swing.JSeparator;
016import javax.swing.SwingConstants;
017
018import org.apache.log4j.Logger;
019
020import sun.swing.SwingAccessor;
021import swp_compiler_ss13.common.ast.ASTNode;
022
023public class SemanticNodeComponent extends NodeComponent {
024
025        private static final Logger LOG = Logger.getLogger(SemanticNodeComponent.class);
026
027        public SemanticNodeComponent(ASTNode node) {
028                super(node);
029        }
030
031        protected JComponent getInfoComponent(ASTNode node) {
032                JPanel result = new JPanel();
033                result.setComponentOrientation(ComponentOrientation.LEFT_TO_RIGHT);
034                Map<?, ?> attributeValues = node.getAttributeValues();
035                result.setLayout(new GridBagLayout());
036                GridBagConstraints c = new GridBagConstraints();
037                Insets elementInsets = new Insets(2, 2, 2, 2);
038                Insets seperatorInsets = new Insets(0, 0, 0, 0);
039                c.gridy = -1;
040                LOG.debug("checking " + (attributeValues.isEmpty() ? "empty" : "full") + " attribute map");
041                for (Entry<?, ?> entry : attributeValues.entrySet()) {
042                        c.gridy++;
043                        c.gridx = 0;
044                        c.gridwidth = 5;
045                        c.insets = seperatorInsets;
046                        result.add(new JSeparator(SwingConstants.HORIZONTAL), c);
047                        c.gridy++;
048                        c.gridx++;
049                        c.gridwidth = 1;
050                        c.insets = elementInsets;
051                        result.add(new JLabel(entry.getKey().toString()), c);
052                        c.gridx += 2;
053                        result.add(new JLabel(entry.getValue().toString()), c);
054                        LOG.debug(String.format("found attribute %s with value %s", entry.getKey(), entry
055                                        .getValue()));
056                }
057                c.gridy++;
058                c.gridx = 0;
059                c.gridwidth = 5;
060                c.insets = seperatorInsets;
061                result.add(new JSeparator(SwingConstants.HORIZONTAL), c);
062                c.gridheight = c.gridy;
063                c.gridy = 0;
064                c.gridwidth = 1;
065                result.add(new JSeparator(SwingConstants.VERTICAL), c);
066                c.gridx += 2;
067                result.add(new JSeparator(SwingConstants.VERTICAL), c);
068                c.gridx += 2;
069                result.add(new JSeparator(SwingConstants.VERTICAL), c);
070                return result;
071        }
072
073}
