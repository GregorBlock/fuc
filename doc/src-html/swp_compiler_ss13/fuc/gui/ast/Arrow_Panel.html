001package swp_compiler_ss13.fuc.gui.ast;
002
003import java.awt.Color;
004import java.awt.Graphics;
005import java.awt.Graphics2D;
006import java.awt.geom.AffineTransform;
007import java.util.ArrayList;
008import java.util.List;
009
010import javax.swing.JLabel;
011
012/**
013 * Component to draw arrows.<br/>
014 * The points can be set by {@link #addArrow}.
015 * 
016 * @author "Eduard Wolf"
017 * 
018 */
019class Arrow_Panel extends JLabel {
020
021        private static final long serialVersionUID = 2703529835276986317L;
022
023        private final int ARR_SIZE = 4;
024        private final List<int[]> points = new ArrayList<>();
025
026        @Override
027        protected void paintComponent(Graphics g) {
028                if (g instanceof Graphics2D) {
029                        Graphics2D graphics = (Graphics2D) g;
030                        graphics.setColor(Color.BLACK);
031                        AffineTransform transform = graphics.getTransform();
032                        for (int[] point : points) {
033                                int x1 = point[0];
034                                int y1 = point[1];
035                                int x2 = point[2];
036                                int y2 = point[3];
037                                double dx = x2 - x1, dy = y2 - y1;
038                                double angle = Math.atan2(dy, dx);
039                                int len = (int) Math.sqrt(dx * dx + dy * dy);
040                                AffineTransform at = AffineTransform.getTranslateInstance(x1, y1);
041                                at.concatenate(AffineTransform.getRotateInstance(angle));
042                                graphics.transform(at);
043
044                                g.drawLine(0, 0, len, 0);
045                                g.fillPolygon(new int[] { len,len - ARR_SIZE,len - ARR_SIZE,len }, new int[] { 0,
046                                                -ARR_SIZE,ARR_SIZE,0 }, 4);
047                                graphics.setTransform(transform);
048                        }
049                }
050        }
051
052        public void addArrow(int x1, int y1, int x2, int y2) {
053                points.add(new int[] { x1,y1,x2,y2 });
054        }
055
056        public void deletePoints() {
057                points.clear();
058        }
059
060}
