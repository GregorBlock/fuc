001package swp_compiler_ss13.fuc.gui.ast;
002
003import java.util.Collections;
004import java.util.List;
005
006import swp_compiler_ss13.common.ast.AST;
007import swp_compiler_ss13.common.ast.ASTNode;
008import swp_compiler_ss13.fuc.gui.ide.mvc.AbstractModel;
009import swp_compiler_ss13.fuc.gui.ide.mvc.Controller;
010
011public class AST_Model extends AbstractModel {
012
013        private final AST_Controller controller;
014        private ASTNode node;
015
016        public AST_Model(AST_Controller controller) {
017                this.controller = controller;
018        }
019
020        @Override
021        public Controller getController() {
022                return controller;
023        }
024
025        @Override
026        public boolean setAST(AST ast) {
027                setNode(ast == null ? null : ast.getRootNode());
028                return true;
029        }
030
031        public void setNode(ASTNode node) {
032                this.node = node;
033        }
034
035        public ASTNode getNode() {
036                return node;
037        }
038
039        public List<ASTNode> getChildren() {
040                return node == null ? Collections.<ASTNode> emptyList() : node.getChildren();
041        }
042
043}
