001package swp_compiler_ss13.fuc.gui.ast;
002
003import java.awt.event.MouseAdapter;
004import java.awt.event.MouseEvent;
005import java.util.ArrayList;
006import java.util.LinkedList;
007import java.util.List;
008
009import javax.swing.JComponent;
010import javax.swing.JLayeredPane;
011import javax.swing.JScrollPane;
012
013import org.apache.log4j.Logger;
014
015import swp_compiler_ss13.common.ast.ASTNode;
016
017/**
018 * @author "Eduard Wolf"
019 * 
020 */
021public class AST_Component {
022
023        private static final Logger LOG = Logger.getLogger(AST_Component.class);
024
025        private static final int BUTTON_LAYER = 2;
026        private static final int CHILD_LAYER = 1;
027        private static final int ARROW_LAYER = 0;
028
029        private final List<AST_Component> astChildren;
030        private final List<JComponent> children;
031        private final Arrow_Panel arrows;
032        private final AST_Controller controller;
033        private final JComponent wrapper;
034        private final JComponent component;
035
036        private NodeComponent nodeComponent;
037
038        private final boolean isRoot;
039
040        private final AstGuiElementFactory nodeComponentFactory;
041
042        public AST_Component(AST_Controller controller, boolean isRoot,
043                        AstGuiElementFactory nodeComponentFactory) {
044                this.controller = controller;
045                this.isRoot = isRoot;
046                this.nodeComponentFactory = nodeComponentFactory;
047                astChildren = new LinkedList<>();
048                wrapper = new JLayeredPane();
049                component = isRoot ? new JScrollPane(wrapper) : wrapper;
050                children = new ArrayList<>();
051                wrapper.setLayout(new AST_LayoutManager());
052                arrows = new Arrow_Panel();
053                arrows.setVisible(false);
054                wrapper.add(arrows, AST_LayoutManager.ARROWS, ARROW_LAYER);
055        }
056
057        public void setNode(ASTNode node) {
058                wrapper.removeAll();
059                if (node != null) {
060                        nodeComponent = nodeComponentFactory.getNodeComponent(node);
061                        JComponent label = nodeComponent.getComponent();
062                        label.setToolTipText(node.getNodeType().name());
063                        wrapper.add(label, AST_LayoutManager.BUTTON, BUTTON_LAYER);
064                        if (!node.getChildren().isEmpty()) {
065                                label.addMouseListener(new ClickListener());
066                        }
067                }
068                wrapper.add(arrows, AST_LayoutManager.ARROWS, ARROW_LAYER);
069
070                arrows.setVisible(false);
071        }
072
073        void toggleSize() {
074                if (nodeComponent != null) {
075                        nodeComponent.toggleSize();
076                }
077                for (AST_Component child : astChildren) {
078                        child.toggleSize();
079                }
080        }
081
082        public boolean isRoot() {
083                return isRoot;
084        }
085
086        public void addChild(AST_Component child) {
087                astChildren.add(child);
088                children.add(child.component);
089                child.component.setVisible(false);
090                wrapper.add(child.component, AST_LayoutManager.CHILDREN, CHILD_LAYER);
091        }
092
093        public void changeChildState() {
094                boolean visibleState = !arrows.isVisible();
095                arrows.setVisible(visibleState);
096                for (JComponent child : children) {
097                        child.setVisible(visibleState);
098                }
099        }
100
101        public void recalculateLayout() {
102                LOG.debug("recalculate layout");
103                wrapper.getParent().validate();
104        }
105
106        JComponent getComponent() {
107                return component;
108        }
109
110        private class ClickListener extends MouseAdapter {
111                @Override
112                public void mouseClicked(MouseEvent e) {
113                        LOG.debug("button clicked");
114                        controller.viewStateChanged();
115                }
116        }
117}
