001package swp_compiler_ss13.fuc.ast;
002
003import java.util.Iterator;
004import java.util.LinkedHashMap;
005import java.util.LinkedList;
006import java.util.List;
007import java.util.Map;
008
009import swp_compiler_ss13.common.ast.ASTNode;
010import swp_compiler_ss13.common.lexer.Token;
011
012/**
013 * AST Node implementation
014 * 
015 * @author "Frank Zechert, Danny Maasch"
016 * @version 1
017 */
018public abstract class ASTNodeImpl implements ASTNode {
019
020        /**
021         * parent node
022         */
023        private ASTNode parent;
024
025        private Map<Object, Object> attributes = new LinkedHashMap<>();
026
027        /**
028         * coverage List
029         */
030        private List<Token> coverage = new LinkedList<Token>();
031
032        @Override
033        public ASTNode getParentNode() {
034                return this.parent;
035        }
036
037        @Override
038        public void setParentNode(ASTNode node) {
039                this.parent = node;
040        }
041
042        @Override
043        public Iterator<ASTNode> getDFSLTRNodeIterator() {
044                final List<ASTNode> children = this.getChildren();
045                final Iterator<ASTNode> childIterator = children.iterator();
046                return new DFSLTRIterator(childIterator);
047        }
048
049        @Override
050        public List<Token> coverage() {
051                return this.coverage;
052        }
053
054        /**
055         * Gets an array of Token and set the coverage. Only used in parser module,
056         * not implemented in ast interface.
057         * 
058         * @param token
059         */
060        public void setCoverage(Token... token) {
061                for (Token toke : token) {
062                        this.coverage.add(toke);
063                }
064        }
065
066        /**
067         * Gets a list of Token and set the coverage. Only used in parser module,
068         * not implemented in ast interface.
069         * 
070         * @param token
071         */
072        public void setCoverage(List<Token> tokenList) {
073                this.coverage.addAll(tokenList);
074        }
075
076        /**
077         * Gets an array of Token and set the coverage in front of the coverage
078         * list. Only used in parser module, not implemented in ast interface.
079         * 
080         * @param token
081         */
082        public void setCoverageAtFront(Token... token) {
083                for (int i = token.length - 1; i >= 0; i--) {
084                        this.coverage.add(0, token[i]);
085                }
086        }
087
088
089        /**
090         * {@inheritDoc}
091         */
092        @Override
093        public void setAttributeValue(Object attribute, Object value) {
094                this.attributes.put(attribute, value);
095        }
096
097        /**
098         * {@inheritDoc}
099         */
100        @Override
101        public Object getAttributeValue(Object attribute) {
102                return this.attributes.get(attribute);
103        }
104
105        /**
106         * {@inheritDoc}
107         */
108        @Override
109        public Map<?, ?> getAttributeValues() {
110                return this.attributes;
111        }
112        
113        @Override
114        public Integer getNumberOfNodes() {
115           int nodes = 1;
116           for (ASTNode node : getChildren()) {
117              nodes += node.getNumberOfNodes();
118           }
119           return nodes;
120        }
121}
