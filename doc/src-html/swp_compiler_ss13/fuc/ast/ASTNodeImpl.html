001package swp_compiler_ss13.fuc.ast;
002
003import java.util.Iterator;
004import java.util.LinkedList;
005import java.util.List;
006
007import swp_compiler_ss13.common.ast.ASTNode;
008import swp_compiler_ss13.common.lexer.Token;
009
010/**
011 * AST Node implementation
012 * 
013 * @author "Frank Zechert, Danny Maasch"
014 * @version 1
015 */
016public abstract class ASTNodeImpl implements ASTNode {
017
018        /**
019         * parent node
020         */
021        private ASTNode parent;
022        
023        /**
024         * coverage List
025         */
026        private List<Token> coverage = new LinkedList<Token>();
027
028        @Override
029        public ASTNode getParentNode() {
030                return this.parent;
031        }
032
033        @Override
034        public void setParentNode(ASTNode node) {
035                this.parent = node;
036        }
037
038        @Override
039        public Iterator<ASTNode> getDFSLTRNodeIterator() {
040                final List<ASTNode> children = this.getChildren();
041                final Iterator<ASTNode> childIterator = children.iterator();
042                return new DFSLTRIterator(childIterator);
043        }
044
045        @Override
046        public List<Token> coverage() {
047                return coverage;
048        }
049        
050        /**
051         * Gets an array of Token and set the coverage.
052         * Only used in parser module, not implemented in 
053         * ast interface.
054         * @param token
055         */
056        public void setCoverage(Token... token){
057                for(Token toke: token){
058                        coverage.add(toke);
059                }
060        }
061        
062        /**
063         * Gets a list of Token and set the coverage.
064         * Only used in parser module, not implemented in 
065         * ast interface.
066         * @param token
067         */
068        public void setCoverage(List<Token> tokenList){
069                coverage.addAll(tokenList);
070        }
071        
072        /**
073         * Gets an array of Token and set the coverage in front
074         * of the coverage list.
075         * Only used in parser module, not implemented in 
076         * ast interface.
077         * @param token
078         */
079        public void setCoverageAtFront(Token... token){
080                for(int i = token.length-1; i >= 0; i--){
081                        coverage.add(0,token[i]);
082                }
083        }
084        
085        
086}
