001package swp_compiler_ss13.fuc.ast.visualization;
002
003import java.io.PrintStream;
004import java.util.List;
005
006import swp_compiler_ss13.common.ast.AST;
007import swp_compiler_ss13.common.ast.ASTNode;
008import swp_compiler_ss13.common.ast.ASTNode.ASTNodeType;
009import swp_compiler_ss13.common.ast.nodes.binary.ArithmeticBinaryExpressionNode;
010import swp_compiler_ss13.common.ast.nodes.binary.LogicBinaryExpressionNode;
011import swp_compiler_ss13.common.ast.nodes.binary.RelationExpressionNode;
012import swp_compiler_ss13.common.ast.nodes.leaf.BasicIdentifierNode;
013import swp_compiler_ss13.common.ast.nodes.leaf.LiteralNode;
014import swp_compiler_ss13.common.ast.nodes.unary.ArithmeticUnaryExpressionNode;
015import swp_compiler_ss13.common.ast.nodes.unary.DeclarationNode;
016import swp_compiler_ss13.common.ast.nodes.unary.LogicUnaryExpressionNode;
017import swp_compiler_ss13.common.visualization.ASTVisualization;
018
019/**
020 * @author Manuel
021 * 
022 */
023public class ASTXMLVisualization implements ASTVisualization {
024
025        /**
026         * {@inheritDoc}
027         */
028        @Override
029        public void visualizeAST(AST ast) {
030                PrintStream out = System.out;
031
032                out.println("<!-- Printing the AST as XML -->");
033                out.println("<AST>");
034                this.visualize(ast.getRootNode(), "  ", out);
035                out.println("</AST>");
036        }
037
038        /**
039         * Recursively prints all Nodes
040         * 
041         * @param node
042         *            The current node to be displayed
043         * @param indent
044         *            A String of spaces for the current indentation level
045         * @param out
046         *            the stream in which the XML gets written
047         */
048        protected void visualize(ASTNode node, String indent, PrintStream out) {
049                String attributes = "";
050                String value = "";
051
052                ASTNodeType nodeType = node.getNodeType();
053                String type = nodeType.toString().replaceFirst("Node$", "");
054                List<ASTNode> children = node.getChildren();
055
056                switch (nodeType) {
057                case ArithmeticBinaryExpressionNode:
058                        attributes = " operator=\"" + ((ArithmeticBinaryExpressionNode) node).getOperator().toString() + "\"";
059                        break;
060                case ArithmeticUnaryExpressionNode:
061                        attributes = " operator=\"" + ((ArithmeticUnaryExpressionNode) node).getOperator().toString() + "\"";
062                        break;
063                case ArrayIdentifierNode:
064                        // FIXME fix this!
065                        // attributes = " index=\"" + ((ArrayIdentifierNode)
066                        // node).getIndex().toString() + "\"";
067                        break;
068                case AssignmentNode:
069                        // nothing to do here
070                        break;
071                case BasicIdentifierNode:
072                        value = ((BasicIdentifierNode) node).getIdentifier();
073                        break;
074                case BlockNode:
075                        // nothing to do here
076                        break;
077                case BranchNode:
078                        // nothing to do here
079                        break;
080                case BreakNode:
081                        // nothing to do here
082                        break;
083                case DeclarationNode:
084                        attributes = " identifier=\"" + ((DeclarationNode) node).getIdentifier().toString() + "\"";
085                        value = ((DeclarationNode) node).getType().toString().replaceFirst("Type($|\\[)", "$1");
086                        break;
087                case DoWhileNode:
088                        // TODO
089                        break;
090                case LiteralNode:
091                        value = ((LiteralNode) node).getLiteral();
092                        break;
093                case LogicBinaryExpressionNode:
094                        attributes = " operator=\""
095                                        + ((LogicBinaryExpressionNode) node).getOperator().toString().replaceFirst("LOGICAL_", "") + "\"";
096                        break;
097                case LogicUnaryExpressionNode:
098                        attributes = " operator=\""
099                                        + ((LogicUnaryExpressionNode) node).getOperator().toString().replaceFirst("LOGICAL_", "") + "\"";
100                        break;
101                case PrintNode:
102                        // nothing to do here
103                        break;
104                case RelationExpressionNode:
105                        attributes = " operator=\"" + ((RelationExpressionNode) node).getOperator().toString() + "\"";
106                        break;
107                case ReturnNode:
108                        // nothing to do here
109                        break;
110                case StructIdentifierNode:
111                        // TODO
112                        break;
113                case WhileNode:
114                        // TODO
115                        break;
116                }
117
118                out.print(indent + "<" + type + attributes + ">");
119
120                if (children.isEmpty()) {
121                        out.print(value);
122                } else {
123                        assert (value == "");
124
125                        out.println();
126                        for (ASTNode child : children) {
127                                this.visualize(child, indent + "  ", out);
128                        }
129                        out.print(indent);
130                }
131
132                out.println("</" + type + ">");
133        }
134}
