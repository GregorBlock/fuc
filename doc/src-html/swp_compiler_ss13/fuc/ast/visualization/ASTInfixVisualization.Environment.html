001package swp_compiler_ss13.fuc.ast.visualization;
002
003import java.io.PrintStream;
004
005import swp_compiler_ss13.common.ast.AST;
006import swp_compiler_ss13.common.ast.ASTNode;
007import swp_compiler_ss13.common.ast.nodes.binary.ArithmeticBinaryExpressionNode;
008import swp_compiler_ss13.common.ast.nodes.binary.AssignmentNode;
009import swp_compiler_ss13.common.ast.nodes.binary.BinaryExpressionNode.BinaryOperator;
010import swp_compiler_ss13.common.ast.nodes.leaf.BasicIdentifierNode;
011import swp_compiler_ss13.common.ast.nodes.leaf.LiteralNode;
012import swp_compiler_ss13.common.ast.nodes.unary.ArithmeticUnaryExpressionNode;
013import swp_compiler_ss13.common.ast.nodes.unary.DeclarationNode;
014import swp_compiler_ss13.common.ast.nodes.unary.UnaryExpressionNode.UnaryOperator;
015import swp_compiler_ss13.common.visualization.ASTVisualization;
016
017/**
018 * @author Manuel
019 * 
020 */
021public class ASTInfixVisualization implements ASTVisualization {
022
023        /**
024         * Represents the current environment of the node. It changes the output
025         * e.g. nested assignments don't have semicolons
026         */
027        protected enum Environment {
028                /**
029                 * use newlines and semicolons
030                 */
031                BLOCK, /**
032                 * print everything in one line
033                 */
034                ASSIGNMENT
035        };
036
037        /**
038         * {@inheritDoc}
039         */
040        @Override
041        public void visualizeAST(AST ast) {
042                PrintStream out = System.out;
043
044                out.println("# Printing the AST in Infix Notation");
045                this.visualizeChildren(ast.getRootNode(), "", out, Environment.BLOCK);
046        }
047
048        /**
049         * Recursively visualizes all Nodes
050         * 
051         * @param node
052         *            The current node to be displayed
053         * @param indent
054         *            A String of spaces for the current indentation level
055         * @param out
056         *            The stream to which the output gets written
057         * @param env
058         *            the environment of the current node
059         */
060        protected void visualize(ASTNode node, String indent, PrintStream out, Environment env) {
061                switch (node.getNodeType()) {
062                case ArithmeticBinaryExpressionNode:
063                        out.print("(");
064                        this.visualize(((ArithmeticBinaryExpressionNode) node).getLeftValue(), indent + "  ", out, env);
065                        out.print(this.getBinaryOperatorSign((ArithmeticBinaryExpressionNode) node));
066                        this.visualize(((ArithmeticBinaryExpressionNode) node).getRightValue(), indent + "  ", out, env);
067                        out.print(")");
068                        break;
069                case ArithmeticUnaryExpressionNode:
070                        out.print("(");
071                        out.print(this.getUnaryOperatorSign((ArithmeticUnaryExpressionNode) node));
072                        this.visualizeChildren(node, indent + "  ", out, env);
073                        out.print(")");
074                        break;
075                case ArrayIdentifierNode:
076                        // TODO
077                        break;
078                case AssignmentNode:
079                        if (env == Environment.ASSIGNMENT) {
080                                out.print("(");
081                        } else {
082                                out.print(indent);
083                        }
084                        this.visualize(((AssignmentNode) node).getLeftValue(), indent + "  ", out, Environment.ASSIGNMENT);
085                        out.print(" = ");
086                        this.visualize(((AssignmentNode) node).getRightValue(), indent + "  ", out, Environment.ASSIGNMENT);
087                        if (env == Environment.ASSIGNMENT) {
088                                out.print(")");
089                        } else {
090                                out.println(";");
091                        }
092                        break;
093                case BasicIdentifierNode:
094                        out.print(((BasicIdentifierNode) node).getIdentifier());
095                        break;
096                case BlockNode:
097                        assert (env != Environment.ASSIGNMENT);
098                        out.print(indent);
099                        out.println("{");
100                        this.visualizeChildren(node, indent + "  ", out, env);
101                        out.print(indent);
102                        out.println("}");
103                        break;
104                case BranchNode:
105                        // TODO
106                        break;
107                case BreakNode:
108                        // TODO
109                        break;
110                case DeclarationNode:
111                        assert (env != Environment.ASSIGNMENT);
112                        out.print(indent);
113                        out.print(((DeclarationNode) node).getType().toString().replaceFirst("Type$", ""));
114                        out.print(" ");
115                        out.print(((DeclarationNode) node).getIdentifier().toString());
116                        out.println(";");
117                        break;
118                case DoWhileNode:
119                        // TODO
120                        break;
121                case LiteralNode:
122                        out.print(((LiteralNode) node).getLiteral());
123                        break;
124                case LogicBinaryExpressionNode:
125                        // TODO
126                        break;
127                case LogicUnaryExpressionNode:
128                        // TODO
129                        break;
130                case PrintNode:
131                        // TODO
132                        break;
133                case RelationExpressionNode:
134                        // TODO
135                        break;
136                case ReturnNode:
137                        assert (env != Environment.ASSIGNMENT);
138                        out.print(indent);
139                        out.print("return ");
140                        this.visualizeChildren(node, indent + "  ", out, env);
141                        out.println(";");
142                        break;
143                case StructIdentifierNode:
144                        // TODO
145                        break;
146                case WhileNode:
147                        // TODO
148                        break;
149                }
150        }
151
152        /**
153         * Recursively prints all Child Nodes
154         * 
155         * @param node
156         *            The node whose children should be visualized
157         * @param indent
158         *            A String of spaces for the current indentation level
159         * @param out
160         *            The stream to which the output gets written
161         * @param env
162         *            the environment of the current node
163         */
164        protected void visualizeChildren(ASTNode node, String indent, PrintStream out, Environment env) {
165                for (ASTNode child : node.getChildren()) {
166                        this.visualize(child, indent, out, env);
167                }
168        }
169
170        /**
171         * Convert the operator of the given node into a string that represents the
172         * operator.
173         * 
174         * @param node
175         *            The node from which to get the operator
176         * @return a string representing the operator
177         */
178        protected String getBinaryOperatorSign(ArithmeticBinaryExpressionNode node) {
179                String operator;
180                BinaryOperator binaryOperator = node.getOperator();
181
182                switch (binaryOperator) {
183                case ADDITION:
184                        operator = "+";
185                        break;
186                case DIVISION:
187                        operator = "/";
188                        break;
189                case EQUAL:
190                        operator = "==";
191                        break;
192                case GREATERTHAN:
193                        operator = ">";
194                        break;
195                case GREATERTHANEQUAL:
196                        operator = ">=";
197                        break;
198                case INEQUAL:
199                        operator = "!=";
200                        break;
201                case LESSTHAN:
202                        operator = "<";
203                        break;
204                case LESSTHANEQUAL:
205                        operator = "<=";
206                        break;
207                case LOGICAL_AND:
208                        operator = "&&";
209                        break;
210                case LOGICAL_OR:
211                        operator = "||";
212                        break;
213                case MULTIPLICATION:
214                        operator = "*";
215                        break;
216                case SUBSTRACTION:
217                        operator = "-";
218                        break;
219                default:
220                        operator = "[invalid ArithmeticBinaryExpressionNode operator: " + binaryOperator.toString() + "]";
221                        break;
222                }
223
224                return operator;
225        }
226
227        /**
228         * Convert the operator of the given node into a string that represents the
229         * operator.
230         * 
231         * @param node
232         *            The node from which to get the operator
233         * @return a string representing the operator
234         */
235        protected String getUnaryOperatorSign(ArithmeticUnaryExpressionNode node) {
236                String operator;
237                UnaryOperator unaryOperator = node.getOperator();
238
239                switch (unaryOperator) {
240                case LOGICAL_NEGATE:
241                        operator = "!";
242                        break;
243                case MINUS:
244                        operator = "-";
245                        break;
246                default:
247                        operator = "[invalid ArithmeticUnaryExpressionNode operator: " + unaryOperator.toString() + "]";
248                        break;
249                }
250
251                return operator;
252        }
253}
