001package swp_compiler_ss13.fuc.ast;
002
003import java.util.LinkedList;
004import java.util.List;
005
006import org.apache.log4j.Logger;
007
008import swp_compiler_ss13.common.ast.ASTNode;
009import swp_compiler_ss13.common.ast.nodes.IdentifierNode;
010import swp_compiler_ss13.common.ast.nodes.unary.StructIdentifierNode;
011
012/**
013 * StructIdentifierNode implementation
014 * 
015 * @author "Frank Zechert, Danny Maasch"
016 * @version 1
017 */
018public class StructIdentifierNodeImpl extends ASTNodeImpl implements
019                StructIdentifierNode {
020
021        /**
022         * The logger
023         */
024        private static Logger logger = Logger.getLogger(DeclarationNodeImpl.class);
025
026        /**
027         * The identifier name
028         */
029        private IdentifierNode identifier;
030
031        /**
032         * the struct field
033         */
034        private String field;
035
036        @Override
037        public ASTNodeType getNodeType() {
038                return ASTNodeType.StructIdentifierNode;
039        }
040
041        @Override
042        public Integer getNumberOfNodes() {
043                if (this.identifier == null) {
044                        return 1;
045                }
046                return 1 + this.identifier.getNumberOfNodes();
047        }
048
049        @Override
050        public List<ASTNode> getChildren() {
051                List<ASTNode> children = new LinkedList<>();
052                if (this.identifier != null) {
053                        children.add(this.identifier);
054                }
055                return children;
056        }
057
058        @Override
059        public void setFieldName(String field) {
060                if (field == null) {
061                        logger.error("The given field can not be null");
062                        throw new IllegalArgumentException(
063                                        "The given field can not be null");
064                }
065                if (field.length() == 0) {
066                        logger.error("The given field can not be the empty string");
067                        throw new IllegalArgumentException(
068                                        "The given field can not be the empty string");
069                }
070                this.field = field;
071        }
072
073        @Override
074        public String getFieldName() {
075                if (this.field == null) {
076                        logger.warn("Returning null as field name.");
077                }
078                return this.field;
079        }
080
081        @Override
082        public void setIdentifierNode(IdentifierNode node) {
083                if (node == null) {
084                        logger.error("The given node can not be null");
085                        throw new IllegalArgumentException("The given node can not be null");
086                }
087                this.identifier = node;
088        }
089
090        @Override
091        public IdentifierNode getIdentifierNode() {
092                if (this.field == null) {
093                        logger.warn("Returning null as identifier node.");
094                }
095                return this.identifier;
096        }
097
098}
