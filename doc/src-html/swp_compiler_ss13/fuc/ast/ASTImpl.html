004

029/**
030 * Implementation of the AST interface
031 * 
032 * @author "Frank Zechert, Danny Maasch"
033 * @version 1
034 * 
035 */
036public class ASTImpl implements AST {
037
038        /**
039         * The root node of the AST
040         */
041        private BlockNode rootNode;
042
043        /**
044         * The logger
045         */
046        private static Logger logger = Logger.getLogger(ASTImpl.class);
047
048        /**
049         * Create a new empty AST.
050         */
051        public ASTImpl() {
052                logger.debug("Creating a new empty AST");
053                this.rootNode = new BlockNodeImpl();
054        }
055
056        /**
057         * Create a new AST with the given root node.
058         * 
059         * @param rootNode
060         *            The root node of the ast.
061         */
062        public ASTImpl(BlockNode rootNode) {
063                logger.debug(String.format("Creating AST with root node: %s", rootNode));
064                if (rootNode == null) {
065                        logger.error("The parameter rootNode can not be null!");
066                        throw new IllegalArgumentException("The parameter rootNode can not be null!");
067                }
068                this.rootNode = rootNode;
069        }
070
071        @Override
072        public BlockNode getRootNode() {
073                return this.rootNode;
074        }
075
076        @Override
077        public void setRootNode(BlockNode rootNode) {
078                if (rootNode == null) {
079                        logger.error("The parameter rootNode can not be null!");
080                        throw new IllegalArgumentException("The parameter rootNode can not be null!");
081                }
082                this.rootNode = rootNode;
083        }
084
085        @Override
086        public Integer getNumberOfNodes() {
087                logger.debug(String.format("Numer of nodes in AST: %d", this.rootNode.getNumberOfNodes()));
088                return this.rootNode.getNumberOfNodes();
089        }
090
091        @Override
092        public Iterator<ASTNode> getDFSLTRIterator() {
093                return new Iterator<ASTNode>() {
094                        private boolean outputRoot = false;
095                        private Iterator<ASTNode> innerIt;
096
097                        @Override
098                        public boolean hasNext() {
099                                if (!this.outputRoot) {
100                                        return true;
101                                }
102                                if (this.innerIt == null) {
103                                        this.innerIt = ASTImpl.this.getRootNode().getDFSLTRNodeIterator();
104                                }
105                                return this.innerIt.hasNext();
106                        }
107
108                        @Override
109                        public ASTNode next() {
110                                if (!this.outputRoot) {
111                                        this.outputRoot = true;
112                                        return ASTImpl.this.rootNode;
113                                }
114                                return this.innerIt.next();
115                        }
116
117                        @Override
118                        public void remove() {
119                                throw new UnsupportedOperationException();
120                        }
121                };
122        }
123
124        @Override
125        public SymbolTable getRootSymbolTable() {
126                return this.rootNode.getSymbolTable();
127        }
128
129        @Override
130        public String toString() {
131                StringBuilder b = new StringBuilder();
132                String indentation = "";
133                toString(b, indentation, this.rootNode);
134                return b.toString();
135        }
136
137        public static void toString(StringBuilder b, String indentation, ASTNode node) {
138                if (node == null) {
139                        return;
140                }
141                b.append(indentation);
142                switch (node.getNodeType()) {
143                case ArithmeticBinaryExpressionNode:
144                        b.append("ArithmeticBinaryExpression(");
145                        toString(b, "", ((ArithmeticBinaryExpressionNode) node).getLeftValue());
146                        b.append(" ").append(((ArithmeticBinaryExpressionNode) node).getOperator().toString()).append(" ");
147                        toString(b, "", ((ArithmeticBinaryExpressionNode) node).getRightValue());
148                        b.append(")");
149                        break;
150                case ArithmeticUnaryExpressionNode:
151                        b.append("ArithmeticUnaryExpressionNode(");
152                        b.append(" ").append(((ArithmeticUnaryExpressionNode) node).getOperator().toString()).append(" ");
153                        toString(b, "", ((ArithmeticUnaryExpressionNode) node).getRightValue());
154                        b.append(")");
155                        break;
156                case ArrayIdentifierNode:
157                        b.append("ArrayIdentifierNode(");
158                        toString(b, "", ((ArrayIdentifierNode) node).getIdentifierNode());
159                        b.append(" [");
160                        toString(b, "", ((ArrayIdentifierNode) node).getIndexNode());
161                        b.append("])");
162                        break;
163                case AssignmentNode:
164                        b.append("AssignmentNode(");
165                        toString(b, "", ((AssignmentNode) node).getLeftValue());
166                        b.append(" = ");
167                        toString(b, "", ((AssignmentNode) node).getRightValue());
168                        b.append(")\n");
169                        break;
170                case BasicIdentifierNode:
171                        b.append("BasicIdentifierNode(");
172                        b.append(((BasicIdentifierNode) node).getIdentifier());
173                        b.append(")");
174                        break;
175                case BlockNode:
176                        b.append("{\n");
177                        String newindentation = indentation + "\t";
178                        for (DeclarationNode d : ((BlockNode) node).getDeclarationList()) {
179                                toString(b, newindentation, d);
180                        }
181                        for (StatementNode s : ((BlockNode) node).getStatementList()) {
182                                toString(b, newindentation, s);
183                        }
184                        b.append(indentation).append("}");
185                        b.append("\n");
186                        break;
187                case BranchNode:
188                        b.append("BranchNode(if(");
189                        toString(b, "", (((BranchNode) node).getCondition()));
190                        b.append(")\n");
191                        toString(b, indentation + "\t", (((BranchNode) node).getStatementNodeOnTrue()));
192                        if (((BranchNode) node).getStatementNodeOnFalse() != null) {
193                                b.append(indentation + "\nelse\n");
194                                toString(b, indentation + "\t", (((BranchNode) node).getStatementNodeOnFalse()));
195                        }
196                        b.append(indentation).append(")\n");
197                        break;
198                case BreakNode:
199                        b.append("BreakNode\n");
200                        break;
201                case DeclarationNode:
202                        b.append("DeclarationNode(");
203                        b.append(((DeclarationNode) node).getType().toString()).append(" ");
204                        b.append(((DeclarationNode) node).getIdentifier());
205                        b.append(")\n");
206                        break;
207                case DoWhileNode:
208                        b.append("DoWhileNode(do(\n");
209                        toString(b, indentation + "\t", (((DoWhileNode) node).getLoopBody()));
210                        b.append(indentation).append(")\n");
211                        b.append(indentation).append("while(");
212                        toString(b, "", (((DoWhileNode) node).getCondition()));
213                        b.append(")\n");
214                        break;
215                case LiteralNode:
216                        b.append("LiteralNode(");
217                        b.append(((LiteralNode) node).getLiteralType().toString());
218                        b.append(" ");
219                        b.append(((LiteralNode) node).getLiteral());
220                        b.append(")");
221                        break;
222                case LogicBinaryExpressionNode:
223                        b.append("LogicBinaryExpressionNode(");
224                        toString(b, "", ((LogicBinaryExpressionNode) node).getLeftValue());
225                        b.append(" ").append(((LogicBinaryExpressionNode) node).getOperator().toString()).append(" ");
226                        toString(b, "", ((LogicBinaryExpressionNode) node).getRightValue());
227                        b.append(")");
228                        break;
229                case LogicUnaryExpressionNode:
230                        b.append("LogicUnaryExpressionNode(");
231                        b.append(" ").append(((LogicUnaryExpressionNode) node).getOperator().toString()).append(" ");
232                        toString(b, "", ((LogicUnaryExpressionNode) node).getRightValue());
233                        b.append(")");
234                        break;
235                case PrintNode:
236                        b.append("PrintNode(");
237                        toString(b, "", ((PrintNode) node).getRightValue());
238                        b.append(")\n");
239                        break;
240                case RelationExpressionNode:
241                        b.append("RelationExpressionNode(");
242                        toString(b, "", ((RelationExpressionNode) node).getLeftValue());
243                        b.append(" ").append(((RelationExpressionNode) node).getOperator().toString()).append(" ");
244                        toString(b, "", ((RelationExpressionNode) node).getRightValue());
245                        b.append(")");
246                        break;
247                case ReturnNode:
248                        b.append("ReturnNode(");
249                        toString(b, "", ((ReturnNode) node).getRightValue());
250                        b.append(")\n");
251                        break;
252                case StructIdentifierNode:
253                        b.append("StructIdentifierNode(");
254                        toString(b, "", ((StructIdentifierNode) node).getIdentifierNode());
255                        b.append(".").append(((StructIdentifierNode) node).getFieldName()).append(")");
256                        break;
257                case WhileNode:
258                        b.append("WhileNode(\n");
259                        b.append(indentation).append("\t").append("while(");
260                        toString(b, "", (((WhileNode) node).getCondition()));
261                        b.append(")\n");
262                        toString(b, indentation + "\t", (((WhileNode) node).getLoopBody()));
263                        b.append(indentation).append(")\n");
264                        break;
265                default:
266                        break;
267                }
268        }
269}
