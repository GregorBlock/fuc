001package swp_compiler_ss13.fuc.ast;
002
003import java.util.Iterator;
004
005import org.apache.log4j.Logger;
006
007import swp_compiler_ss13.common.ast.AST;
008import swp_compiler_ss13.common.ast.ASTNode;
009import swp_compiler_ss13.common.ast.nodes.marynary.BlockNode;
010import swp_compiler_ss13.common.parser.SymbolTable;
011
012/**
013 * Implementation of the AST interface
014 * 
015 * @author "Frank Zechert, Danny Maasch"
016 * @version 1
017 * 
018 */
019public class ASTImpl implements AST {
020
021        /**
022         * The root node of the AST
023         */
024        private BlockNode rootNode;
025
026        /**
027         * The logger
028         */
029        private static Logger logger = Logger.getLogger(ASTImpl.class);
030
031        /**
032         * Create a new empty AST.
033         */
034        public ASTImpl() {
035                logger.debug("Creating a new empty AST");
036                this.rootNode = new BlockNodeImpl();
037        }
038
039        /**
040         * Create a new AST with the given root node.
041         * 
042         * @param rootNode
043         *            The root node of the ast.
044         */
045        public ASTImpl(BlockNode rootNode) {
046                logger.debug(String.format("Creating AST with root node: %s", rootNode));
047                if (rootNode == null) {
048                        logger.error("The parameter rootNode can not be null!");
049                        throw new IllegalArgumentException("The parameter rootNode can not be null!");
050                }
051                this.rootNode = rootNode;
052        }
053
054        @Override
055        public BlockNode getRootNode() {
056                return this.rootNode;
057        }
058
059        @Override
060        public void setRootNode(BlockNode rootNode) {
061                if (rootNode == null) {
062                        logger.error("The parameter rootNode can not be null!");
063                        throw new IllegalArgumentException("The parameter rootNode can not be null!");
064                }
065                this.rootNode = rootNode;
066        }
067
068        @Override
069        public Integer getNumberOfNodes() {
070                logger.debug(String.format("Numer of nodes in AST: %d", this.rootNode.getNumberOfNodes()));
071                return this.rootNode.getNumberOfNodes();
072        }
073
074        @Override
075        public Iterator<ASTNode> getDFSLTRIterator() {
076                return new Iterator<ASTNode>() {
077                        private boolean outputRoot = false;
078                        private Iterator<ASTNode> innerIt;
079
080                        @Override
081                        public boolean hasNext() {
082                                if (!this.outputRoot) {
083                                        return true;
084                                }
085                                if (this.innerIt == null) {
086                                        this.innerIt = ASTImpl.this.getRootNode().getDFSLTRNodeIterator();
087                                }
088                                return this.innerIt.hasNext();
089                        }
090
091                        @Override
092                        public ASTNode next() {
093                                if (!this.outputRoot) {
094                                        this.outputRoot = true;
095                                        return ASTImpl.this.rootNode;
096                                }
097                                return this.innerIt.next();
098                        }
099
100                        @Override
101                        public void remove() {
102                                throw new UnsupportedOperationException();
103                        }
104                };
105        }
106
107        @Override
108        public SymbolTable getRootSymbolTable() {
109                return this.rootNode.getSymbolTable();
110        }
111
112}
