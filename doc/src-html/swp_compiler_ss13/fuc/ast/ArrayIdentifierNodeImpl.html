001package swp_compiler_ss13.fuc.ast;
002
003import java.util.LinkedList;
004import java.util.List;
005
006import org.apache.log4j.Logger;
007
008import swp_compiler_ss13.common.ast.ASTNode;
009import swp_compiler_ss13.common.ast.nodes.ExpressionNode;
010import swp_compiler_ss13.common.ast.nodes.IdentifierNode;
011import swp_compiler_ss13.common.ast.nodes.unary.ArrayIdentifierNode;
012
013/**
014 * ArrayIdentifierNode implementation
015 * 
016 * @author "Frank Zechert, Danny Maasch"
017 * @version 1
018 */
019public class ArrayIdentifierNodeImpl extends ASTNodeImpl implements ArrayIdentifierNode {
020
021        /**
022         * The logger
023         */
024        private static Logger logger = Logger.getLogger(DeclarationNodeImpl.class);
025
026        /**
027         * The identifier name
028         */
029        private IdentifierNode identifier;
030
031        /**
032         * the array index
033         */
034        private ExpressionNode index;
035
036        @Override
037        public ASTNodeType getNodeType() {
038                return ASTNodeType.ArrayIdentifierNode;
039        }
040
041        @Override
042        public List<ASTNode> getChildren() {
043                List<ASTNode> children = new LinkedList<>();
044                if (this.identifier != null) {
045                        children.add(this.identifier);
046                }
047                if (this.index != null) {
048                   children.add(this.index);
049                }
050                return children;
051        }
052
053        @Override
054        public void setIndexNode(ExpressionNode index) {
055                if (index == null) {
056                        logger.error("The given index can not be null");
057                        throw new IllegalArgumentException("The given index can not be null");
058                }
059
060                this.index = index;
061        }
062
063        @Override
064        public ExpressionNode getIndexNode() {
065                if (this.index == null) {
066                        logger.warn("Returning null as array index.");
067                }
068                return this.index;
069        }
070
071        @Override
072        public void setIdentifierNode(IdentifierNode node) {
073                if (node == null) {
074                        logger.error("The given node can not be null");
075                        throw new IllegalArgumentException("The given node can not be null");
076                }
077                this.identifier = node;
078        }
079
080        @Override
081        public IdentifierNode getIdentifierNode() {
082                if (this.index == null) {
083                        logger.warn("Returning null as identifier node.");
084                }
085                return this.identifier;
086        }
087
088
089}
