035        /**
036         * Stores the constructed AST
037         */
038        AST ast;
039
040        /**
041         * Stores the current node (is always a StatementNode which can have
042         * StatementNode children)
043         */
044        ASTNode node;
045
046        /**
047         * A Factory to create an AST without needing to write a million lines of
048         * code.
049         * 
050         * This is achieved by employing methods that act as "constructors" that get
051         * the child nodes directly and thus save the need to call the setters.
052         * 
053         * furthermore is the last StatementNode that can have additional
054         * StatementNodes as children remembered which allows easy appending to
055         * BlockNodes
056         */
057        public ASTFactory() {
058                this.ast = new ASTImpl();
059                BlockNode program = new BlockNodeImpl();
060                program.setSymbolTable(new SymbolTableImpl());
061                this.ast.setRootNode(program);
062                this.node = program;
063        }
064
065        /**
066         * returns the constructed ast
067         * 
068         * @return the constructed ast
069         */
070        public AST getAST() {
071                return this.ast;
072        }
073
074        /**
075         * returns the current node (is always a StatementNode which can have
076         * StatementNode children)
077         * 
078         * @return return the current node
079         */
080        public ASTNode getNode() {
081                return this.node;
082        }
083
084        /**
085         * moves the construction to the parent
086         * 
087         * @return the new current node
088         */
089        public ASTNode goToParent() {
090                ASTNode parent = this.node.getParentNode();
091                assert (parent != null);
092                this.node = parent;
093                return this.node;
094        }
095        
096        /**
097         * Adds the given node to the current node and handles different node
098         * types like {@link BranchNode}, {@link LoopNode} and
099         * {@link BlockNode}
100         * 
101         * @param stmt The {@link StatementNode} which should be added to the
102         * current {@link #node}
103         */
104        private void addStatement(StatementNode stmt) {
105             switch (this.node.getNodeType()) {
106                case BranchNode:
107                   BranchNode branch = ((BranchNode) this.node);
108                   if (branch.getStatementNodeOnTrue() == null) {
109                      branch.setStatementNodeOnTrue(stmt);
110                   } else if ((branch.getStatementNodeOnFalse() == null)) {
111                      branch.setStatementNodeOnFalse(stmt);
112                   } else {
113                      assert (false);
114                   }
115                   break;
116                case DoWhileNode:
117                case WhileNode:
118                   ((LoopNode) this.node).setLoopBody(stmt);
119                   break;
120                case BlockNode:
121                   ((BlockNode) this.node).addStatement(stmt);
122                   break;
123                default:
124                   assert (false);
125                   break;
126                }
127        }
128
129        /**
130         * adds a declaration to the current node
131         * 
132         * @param identifier
133         *            the identifier to declare
134         * @param type
135         *            the type to declare
136         * @return the created DeclarationNode
137         */
138        public DeclarationNode addDeclaration(String identifier, Type type) {
139                DeclarationNode declaration = new DeclarationNodeImpl();
140                BlockNode block = (BlockNode) this.node;
141                block.addDeclaration(declaration);
142                block.getSymbolTable().insert(identifier, type);
143                declaration.setIdentifier(identifier);
144                declaration.setType(type);
145                declaration.setParentNode(this.node);
146                return declaration;
147        }
148
149        /**
150         * creates a new BlockNode, adds it as child to the current node, and sets
151         * it as the new current node
152         * 
153         * @return the created BlockNode
154         */
155        public BlockNode addBlock() {
156                BlockNode block = new BlockNodeImpl();
157                block.setParentNode(this.node);
158
159                ASTNode parentBlock = this.node;
160                while (!(parentBlock instanceof BlockNode)) {
161                        parentBlock = parentBlock.getParentNode();
162                }
163
164                block.setSymbolTable(new SymbolTableImpl(((BlockNode) parentBlock).getSymbolTable()));
165                addStatement(block);
166                this.node = block;
167                return block;
168        }
169
170        /**
171         * creates a new ReturnNode and adds it as child to the current node
172         * 
173         * @param identifier
174         *            the identifier to return
175         * @return the created ReturnNode
176         */
177        public ReturnNode addReturn(IdentifierNode identifier) {
178                ReturnNode ret = new ReturnNodeImpl();
179                addStatement(ret);
180                ret.setParentNode(this.node);
181                ret.setRightValue(identifier);
182                if (identifier != null) {
183                        identifier.setParentNode(ret);
184                }
185                return ret;
186        }

188        /**
189         * creates a new BranchNode, adds it as child to the current node, and sets
190         * it as the new current node
191         * 
192         * the next addBlock() will add the TrueBlock. goToParent(); addBlock() will
193         * add the FalseBlock
194         * 
195         * @param condition
196         *            the condition which decides which childBlock gets executed
197         * @return the created BranchNode
198         */
199        public BranchNode addBranch(ExpressionNode condition) {
200                BranchNode branch = new BranchNodeImpl();
201                addStatement(branch);
202                branch.setParentNode(this.node);
203                branch.setCondition(condition);
204                condition.setParentNode(branch);
205                this.node = branch;
206                return branch;
207        }

209        /**
210         * creates a new LoopNode, adds it as child to the current node, and sets it
211         * as the new current node
212         * 
213         * (needs 2 calls to goToParent() to get to the original node)
214         * 
215         * @param condition
216         *            the condition to end the loop
217         * @param loop
218         *            the type of loop this is (while, dowhile)
219         * @return the created LoopNode
220         */
221        protected LoopNode addLoop(ExpressionNode condition, LoopNode loop) {
222                addStatement(loop);
223                loop.setParentNode(this.node);
224                loop.setCondition(condition);
225                condition.setParentNode(loop);
226                this.node = loop;
227                return loop;
228        }

299        /**
300         * creates a new BinaryExpressionNode and sets the relations to the children
301         * 
302         * @param operator
303         *            the BinaryOperator
304         * @param leftExpression
305         *            the left expression
306         * @param rightExpression
307         *            the right expression
308         * @return the created BinaryExpressionNode
309         */
310        public BinaryExpressionNode newBinaryExpression(BinaryOperator operator, ExpressionNode leftExpression,
311                        ExpressionNode rightExpression) {
312                BinaryExpressionNode binaryExpression = null;
313                switch (operator) {
314                case ADDITION:
315                case DIVISION:
316                case MULTIPLICATION:
317                case SUBSTRACTION:
318                        binaryExpression = new ArithmeticBinaryExpressionNodeImpl();
319                        break;
320                case EQUAL:
321                case GREATERTHAN:
322                case GREATERTHANEQUAL:
323                case INEQUAL:
324                case LESSTHAN:
325                case LESSTHANEQUAL:
326                        binaryExpression = new RelationExpressionNodeImpl();
327                        break;
328                case LOGICAL_AND:
329                case LOGICAL_OR:
330                        binaryExpression = new LogicBinaryExpressionNodeImpl();
331                        break;
332                default:
333                        assert (false);
334                        break;
335                }
336                binaryExpression.setOperator(operator);
337                binaryExpression.setLeftValue(leftExpression);
338                leftExpression.setParentNode(binaryExpression);
339                binaryExpression.setRightValue(rightExpression);
340                rightExpression.setParentNode(binaryExpression);
341                return binaryExpression;
342        }

344        /**
345         * Creates a new {@link PrintNode} with the given {@link IdentifierNode} as
346         * right value.
347         * 
348         * @param id
349         *            May be <code>null</code>
350         * @return
351         */
352        public PrintNode newPrint(IdentifierNode id) {
353                PrintNode print = new PrintNodeImpl();
354                if (id == null) {
355                        print.setRightValue(null);
356                } else {
357                        id.setParentNode(print);
358                        print.setRightValue(id);
359                }
360                return print;
361        }

363        /**
364         * creates a new UnaryExpressionNode and sets the relations to the children
365         * 
366         * @param operator
367         *            the UnaryOperator
368         * @param expression
369         *            the expression
370         * @return the created UnaryExpressionNode
371         */
372        public UnaryExpressionNode newUnaryExpression(UnaryOperator operator, ExpressionNode expression) {
373                UnaryExpressionNode unaryExpression = null;
374                switch (operator) {
375                case LOGICAL_NEGATE:
376                        unaryExpression = new LogicUnaryExpressionNodeImpl();
377                        break;
378                case MINUS:
379                        unaryExpression = new ArithmeticUnaryExpressionNodeImpl();
380                        break;
381                default:
382                        assert (false);
383                        break;
384                }
385                unaryExpression.setOperator(operator);
386                unaryExpression.setRightValue(expression);
387                expression.setParentNode(unaryExpression);
388                return unaryExpression;
389        }
