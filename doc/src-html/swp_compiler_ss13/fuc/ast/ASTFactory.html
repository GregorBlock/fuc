034        /**
035         * Stores the constructed AST
036         */
037        AST ast;
038
039        /**
040         * Stores the current node (is always a StatementNode which can have
041         * StatementNode children)
042         */
043        ASTNode node;
044
045        /**
046         * A Factory to create an AST without needing to write a million lines of
047         * code.
048         * 
049         * This is achieved by employing methods that act as "constructors" that get
050         * the child nodes directly and thus save the need to call the setters.
051         * 
052         * furthermore is the last StatementNode that can have additional
053         * StatementNodes as children remembered which allows easy appending to
054         * BlockNodes
055         */
056        public ASTFactory() {
057                this.ast = new ASTImpl();
058                BlockNode program = new BlockNodeImpl();
059                program.setSymbolTable(new SymbolTableImpl());
060                this.ast.setRootNode(program);
061                this.node = program;
062        }
063
064        /**
065         * returns the constructed ast
066         * 
067         * @return the constructed ast
068         */
069        public AST getAST() {
070                return this.ast;
071        }

073        /**
074         * returns the current node (is always a StatementNode which can have
075         * StatementNode children)
076         * 
077         * @return return the current node
078         */
079        public ASTNode getNode() {
080                return this.node;
081        }
082
083        /**
084         * moves the construction to the parent
085         * 
086         * @return the new current node
087         */
088        public ASTNode goToParent() {
089                ASTNode parent = this.node.getParentNode();
090                assert (parent != null);
091                this.node = parent;
092                return this.node;
093        }
094
095        /**
096         * adds a declaration to the current node
097         * 
098         * @param identifier
099         *            the identifier to declare
100         * @param type
101         *            the type to declare
102         * @return the created DeclarationNode
103         */
104        public DeclarationNode addDeclaration(String identifier, Type type) {
105                DeclarationNode declaration = new DeclarationNodeImpl();
106                BlockNode block = (BlockNode) this.node;
107                block.addDeclaration(declaration);
108                block.getSymbolTable().insert(identifier, type);
109                declaration.setIdentifier(identifier);
110                declaration.setType(type);
111                declaration.setParentNode(this.node);
112                return declaration;
113        }

115        /**
116         * creates a new BlockNode, adds it as child to the current node, and sets
117         * it as the new current node
118         * 
119         * @return the created BlockNode
120         */
121        public BlockNode addBlock() {
122                BlockNode block = new BlockNodeImpl();
123                block.setParentNode(this.node);
124
125                ASTNode parentBlock = this.node;
126                while (!(parentBlock instanceof BlockNode)) {
127                        parentBlock = parentBlock.getParentNode();
128                }
129
130                block.setSymbolTable(new SymbolTableImpl(((BlockNode) parentBlock).getSymbolTable()));
131                switch (this.node.getNodeType()) {
132                case BranchNode:
133                        BranchNode branch = ((BranchNode) this.node);
134                        if (branch.getStatementNodeOnTrue() == null) {
135                                branch.setStatementNodeOnTrue(block);
136                        } else if ((branch.getStatementNodeOnFalse() == null)) {
137                                branch.setStatementNodeOnFalse(block);
138                        } else {
139                                assert (false);
140                        }
141                        break;
142                case DoWhileNode:
143                case WhileNode:
144                        ((LoopNode) this.node).setLoopBody(block);
145                        break;
146                case BlockNode:
147                        ((BlockNode) this.node).addStatement(block);
148                        break;
149                default:
150                        assert (false);
151                        break;
152                }
153                this.node = block;
154                return block;
155        }
156
157        /**
158         * creates a new ReturnNode and adds it as child to the current node
159         * 
160         * @param identifier
161         *            the identifier to return
162         * @return the created ReturnNode
163         */
164        public ReturnNode addReturn(IdentifierNode identifier) {
165                ReturnNode ret = new ReturnNodeImpl();
166                ((BlockNode) this.node).addStatement(ret);
167                ret.setParentNode(this.node);
168                ret.setRightValue(identifier);
169                if (identifier != null) {
170                        identifier.setParentNode(ret);
171                }
172                return ret;
173        }

175        /**
176         * creates a new BranchNode, adds it as child to the current node, and sets
177         * it as the new current node
178         * 
179         * the next addBlock() will add the TrueBlock. goToParent(); addBlock() will
180         * add the FalseBlock
181         * 
182         * @param condition
183         *            the condition which decides which childBlock gets executed
184         * @return the created BranchNode
185         */
186        public BranchNode addBranch(ExpressionNode condition) {
187                BranchNode branch = new BranchNodeImpl();
188                ((BlockNode) this.node).addStatement(branch);
189                branch.setParentNode(this.node);
190                branch.setCondition(condition);
191                condition.setParentNode(branch);
192                this.node = branch;
193                return branch;
194        }
195
196        /**
197         * creates a new LoopNode, adds it as child to the current node, and sets it
198         * as the new current node
199         * 
200         * (needs 2 calls to goToParent() to get to the original node)
201         * 
202         * @param condition
203         *            the condition to end the loop
204         * @param loop
205         *            the type of loop this is (while, dowhile)
206         * @return the created LoopNode
207         */
208        protected LoopNode addLoop(ExpressionNode condition, LoopNode loop) {
209                ((BlockNode) this.node).addStatement(loop);
210                loop.setParentNode(this.node);
211                loop.setCondition(condition);
212                condition.setParentNode(loop);
213                this.node = loop;
214                this.addBlock();
215                return loop;
216        }

225         *            the condition to end the loop
226         * @return the created DoWhileNode

228        public DoWhileNode addDoWhile(ExpressionNode condition) {

287        /**
288         * creates a new BinaryExpressionNode and sets the relations to the children
289         * 
290         * @param operator
291         *            the BinaryOperator
292         * @param leftExpression
293         *            the left expression
294         * @param rightExpression
295         *            the right expression
296         * @return the created BinaryExpressionNode
297         */
298        public BinaryExpressionNode newBinaryExpression(BinaryOperator operator, ExpressionNode leftExpression,
299                        ExpressionNode rightExpression) {
300                BinaryExpressionNode binaryExpression = null;
301                switch (operator) {
302                case ADDITION:
303                case DIVISION:
304                case MULTIPLICATION:
305                case SUBSTRACTION:
306                        binaryExpression = new ArithmeticBinaryExpressionNodeImpl();
307                        break;
308                case EQUAL:
309                case GREATERTHAN:
310                case GREATERTHANEQUAL:
311                case INEQUAL:
312                case LESSTHAN:
313                case LESSTHANEQUAL:
314                        binaryExpression = new RelationExpressionNodeImpl();
315                        break;
316                case LOGICAL_AND:
317                case LOGICAL_OR:
318                        binaryExpression = new LogicBinaryExpressionNodeImpl();
319                        break;
320                default:
321                        assert (false);
322                        break;
323                }
324                binaryExpression.setOperator(operator);
325                binaryExpression.setLeftValue(leftExpression);
326                leftExpression.setParentNode(binaryExpression);
327                binaryExpression.setRightValue(rightExpression);
328                rightExpression.setParentNode(binaryExpression);
329                return binaryExpression;
330        }

332        /**
333         * Creates a new {@link PrintNode} with the given {@link IdentifierNode} as right value.
334         * 
335         * @param id May be <code>null</code>
336         * @return
337         */
338        public PrintNode newPrint(IdentifierNode id) {
339           PrintNode print = new PrintNodeImpl();
340           if (id == null) {
341              print.setRightValue(null);
342           } else {
343              id.setParentNode(print);
344         print.setRightValue(id);
345           }
346           return print;
347        }

350         * creates a new UnaryExpressionNode and sets the relations to the children

352         * @param operator
353         *            the UnaryOperator
354         * @param expression
355         *            the expression
356         * @return the created UnaryExpressionNode
357         */
358        public UnaryExpressionNode newUnaryExpression(UnaryOperator operator, ExpressionNode expression) {
359                UnaryExpressionNode unaryExpression = null;
360                switch (operator) {
361                case LOGICAL_NEGATE:
362                        unaryExpression = new LogicUnaryExpressionNodeImpl();
363                        break;
364                case MINUS:
365                        unaryExpression = new ArithmeticUnaryExpressionNodeImpl();
366                        break;
367                default:
368                        assert (false);
369                        break;
370                }
371                unaryExpression.setOperator(operator);
372                unaryExpression.setRightValue(expression);
373                expression.setParentNode(unaryExpression);
374                return unaryExpression;
375        }
